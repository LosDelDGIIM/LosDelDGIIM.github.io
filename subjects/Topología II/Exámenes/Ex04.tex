    \begin{ejercicio}
        Razona si son verdaderas o falsas las siguientes afirmaciones:
        \begin{enumerate}[label=\alph*)]
            \item Existe una aplicación recubridora $p:\mathbb{R}^2\to \mathbb{R}\bb{P}^2$.
            \item Existe una aplicacición recubridora $p:\bb{S}^2\times \mathbb{R}\to \bb{S}^1\times \bb{S}^1$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Prueba uno de los dos siguientes resultados:
        \begin{enumerate}[label=\alph*)]
            \item Sean $p:R\to B$ una aplicación recubridora y $p(r_0) = b_0$. Demuestra que $H_0=p_\ast(\pi_1(R,r_0))$ es un subgrupo normal de $\pi_1(B,b_0)$ si, y solo si, para cada dos puntos $r_1,r_2\in p^{-1}(\{b_0\})$ existe un isomorfismo de recubridores $\varphi:(R,p)\to (R,p)$ tal que $\varphi(r_1)=r_2$.
            \item Demuestra que toda aplicación $f:\mathbb{R}\bb{P}P^2\to \bb{S}^1$ es homotópicamente nula (es decir, $f$ es homotópica a una aplicación constante).
        \end{enumerate}
    \end{ejercicio}
