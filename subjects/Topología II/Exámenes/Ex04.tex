    \begin{ejercicio}
        Razona si son verdaderas o falsas las siguientes afirmaciones:
        \begin{enumerate}[label=\alph*)]
            \item Existe una aplicación recubridora $p:\mathbb{R}^2\to \mathbb{R}\bb{P}^2$.
            \item Existe una aplicacición recubridora $p:\bb{S}^2\times \mathbb{R}\to \bb{S}^1\times \bb{S}^1$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Prueba uno de los dos siguientes resultados:
        \begin{enumerate}[label=\alph*)]
            \item Sean $p:R\to B$ una aplicación recubridora y $p(r_0) = b_0$. Demuestra que $H_0=p_\ast(\pi_1(R,r_0))$ es un subgrupo normal de $\pi_1(B,b_0)$ si, y solo si, para cada dos puntos $r_1,r_2\in p^{-1}(\{b_0\})$ existe un isomorfismo de recubridores \newline $\varphi:(R,p)\to (R,p)$ tal que $\varphi(r_1)=r_2$.
            \item Demuestra que toda aplicación continua $f:\mathbb{R}\bb{P}^2\to \bb{S}^1$ es homotópicamente nula (es decir, $f$ es homotópica a una aplicación constante).
        \end{enumerate}
    \end{ejercicio}

    \textbf{Solución.}

    \begin{ejercicio}
        Razona si son verdaderas o falsas las siguientes afirmaciones:
        \begin{enumerate}[label=\alph*)]
            \item Existe una aplicación recubridora $p:\mathbb{R}^2\to \mathbb{R}\bb{P}^2$.

                Es falsa: por reducción al absurdo, supongamos que existe una aplicación recubridora $p:\mathbb{R}^2\to \mathbb{R}\bb{P}^2$. Sabemos que $\mathbb{S}^2$ es el recubridor universal de $\mathbb{R}\bb{P}^2$, gracias a la aplicación que proyecta en el cociente. Como $\mathbb{R}^2$ también recubre a $\mathbb{R}\bb{P}^2$ y también es simplemente conexo tenemos entonces que debe existir un homeomorfismo entre $\mathbb{R}^2$ y $\mathbb{S}^2$. Sin embargo, esto es una contradicción, ya que $\mathbb{S}^2$ es compacto y $\mathbb{R}^2$ no.
            \item Existe una aplicacición recubridora $p:\bb{S}^2\times \mathbb{R}\to \bb{S}^1\times \bb{S}^1$.

                Es falsa: por reducción al absurdo, supongamos la existencia de una aplicación recubridora $p:\mathbb{S}^2\times\mathbb{R}\to \mathbb{S}^1\times\mathbb{S}^1$, como $\mathbb{R}^2$ es el recubridor universal de $\mathbb{S}^1\times\mathbb{S}^1$ y $\mathbb{S}^2\times \mathbb{R}$ es simplemente conexo (como producto de simplemente conexos) tenemos entonces que ha de existir un homeomorfismo entre $\mathbb{S}^2\times\mathbb{R}$ y $\mathbb{R}^2$, lo que lleva a una contradicción:

                Todo punto de $\mathbb{S}^2\times\mathbb{R}$ admite un entorno abierto homeomorfo a un abierto de $\mathbb{R}^3$, ya que $\mathbb{S}^2$ es localmente homeomorfo a $\mathbb{R}^2$, y la imagen de dicho entorno por el homeomorfismo con $\mathbb{R}^2$ sería un abierto de $\mathbb{R}^2$, en la relación 1 vimos que ningún abierto de $\mathbb{R}^2$ puede ser homeomorfo a un abierto de $\mathbb{R}^3$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Prueba uno de los dos siguientes resultados:
        \begin{enumerate}[label=\alph*)]
            \item Sean $p:R\to B$ una aplicación recubridora y $p(r_0) = b_0$. Demuestra que $H_0=p_\ast(\pi_1(R,r_0))$ es un subgrupo normal de $\pi_1(B,b_0)$ si, y solo si, para cada dos puntos $r_1,r_2\in p^{-1}(\{b_0\})$ existe un isomorfismo de recubridores \newline $\varphi:(R,p)\to (R,p)$ tal que $\varphi(r_1)=r_2$.\\

                Por doble implicación:
                \begin{description}
                    \item [$\Longrightarrow )$] Dados $r_1,r_2\in p^{-1}(\{b_0\})$, por ser $R$ arcoconexo tenemos que existen arcos $\alpha_1,\alpha_2:[0,1]\to R$ de forma que:
                        \begin{equation*}
                            \alpha_1(0) = r_0 = \alpha_2(0), \qquad \alpha_1(1) = r_1, \qquad \alpha_2(1) = r_2
                        \end{equation*}
                        En dicho caso, hemos visto en una Proposición de teoría que entonces:
                        \begin{align*}
                            p_\ast(\pi_1(R,r_1)) &= {[p\circ \alpha_1]}^{-1}\ast p_\ast(\pi_1(R,r_0))\ast [p\circ \alpha_1] \\
                            p_\ast(\pi_1(R,r_2)) &= {[p\circ \alpha_2]}^{-1}\ast p_\ast(\pi_1(R,r_0))\ast [p\circ \alpha_2] 
                        \end{align*}
                        Pero $[p\circ \alpha_1],[p\circ \alpha_2]\in \pi_1(B,b_0)$ y $H_0=p_\ast(\pi_1(R,r_0))$ es normal en $\pi_1(B,b_0)$, por lo que tenemos:
                        \begin{equation*}
                            p_\ast(\pi_1(R,r_1)) = H_0 = p_\ast(\pi_1(R,r_2))
                        \end{equation*}
                        y en teoría hemos visto que bajo estas condiciones existe entonces un isomorfismo de recubridores $\varphi:(R,p)\to (R,p)$ de forma que $\varphi(r_1) = r_2$.
                    \item [$\Longleftarrow )$] Sea $g\in \pi_1(B,b_0)$, vimos en una Proposición de teoría que para el subgrupo $gH_0g^{-1}$ conjugado de $H_0 = p_\ast(\pi_1(R,r_0))$ debía existir $r_1\in R$ de forma que $p_\ast(\pi_1(R,r_1)) = gH_0g^{-1}$. Ahora, para $r_0$ y $r_1$ tenemos que existe un isomorfismo de recubridores $\varphi:(R,p)\to (R,p)$ con $\varphi(r_0) = r_1$, de donde debemos tener:
                        \begin{equation*}
                            gH_0g^{-1} = p_\ast(\pi_1(R,r_1)) = p_\ast(\pi_1(R,r_0)) = H_0
                        \end{equation*}
                        La arbitrariedad de $g$ nos permite concluir que $H_0\lhd \pi_1(B,b_0)$.
                \end{description}
            \item Demuestra que toda aplicación continua $f:\mathbb{R}\bb{P}^2\to \bb{S}^1$ es homotópicamente nula (es decir, $f$ es homotópica a una aplicación constante).\\

                Sea $f:\mathbb{R}\bb{P}^2\to \mathbb{S}^1$ una aplicación continua, si consideramos el recubridor $(\mathbb{R},p)$ de $\mathbb{S}^1$ donde $p$ es la aplicación recubridora estándar, estamos en la situación:
                \begin{figure}[H]
                    \centering
                    \shorthandoff{""}
                    \begin{tikzcd}
                                                              & \mathbb{R} \arrow[d, "p"] \\
                        \mathbb{R}\mathbb{P}^2 \arrow[r, "f"] & \mathbb{S}^1             
                    \end{tikzcd}
                    \shorthandon{""}
                \end{figure}
                \noindent
                Fijado $x_0 \in \mathbb{R}\bb{P}^2$, si consideramos $b_0 = f(x_0)$ observemos que siempre tenemos que $f_\ast(\pi_1(\mathbb{R}\bb{P}^2,x_0))$ es un subgrupo de $\pi_1(\mathbb{S}^1,b_0)$. En este caso tenemos que $\pi_1(\mathbb{S}^1,b_0)\cong \mathbb{Z}$ y que $\pi_1(\mathbb{R}\mathbb{P}^2, x_0)\cong \mathbb{Z}_2$ que es un grupo finito, por lo que su imagen por $f_\ast$ seguirá siendo un grupo finito, luego tiene que ser:
                \begin{equation*}
                    f_\ast(\pi_1(\mathbb{R}\mathbb{P}^2,x_0)) = \{[\varepsilon_{b_0}]\}
                \end{equation*}
                Por tanto, podemos levantar la aplicación $f$, ya que fijado $r_0\in p^{-1}(\{b_0\})$  tenemos:
                \begin{equation*}
                    f_\ast(\pi_1(\mathbb{R}\mathbb{P}^2,x_0)) \subseteq p_\ast(\pi_1(\mathbb{R},r_0)) 
                \end{equation*}
                tenemos entonces que existe $\hat{f}:\mathbb{R}\mathbb{P}^2\to \mathbb{R}$ levantamiento de $f$ con $\hat{f}(r_0) = x_0$. Como $\hat{f}$ llega a $\mathbb{R}$ (que es contráctil) tenemos que $\hat{f}$ es homotópicamente nula, es decir, existe $z_0 \in \mathbb{R}$ y una homotopía $H:\mathbb{R}\mathbb{P}^2\times [0,1]\to \mathbb{R}$ de forma que:
                \begin{equation*}
                    H(x,0) = \hat{f}(x), \qquad H(x,1) = z_0 \qquad \forall x\in \mathbb{R}\mathbb{P}^2
                \end{equation*}
                Si consideramos ahora la homotopía $p\circ H:\mathbb{R}\mathbb{P}^2\times [0,1]\to \mathbb{S}^1$ tenemos que:
                \begin{equation*}
                    (p\circ H)(x,0) = p(\hat{f}(x)) = f(x), \qquad (p\circ H)(x,1)=p(z_0) \qquad \forall x\in \mathbb{R}\mathbb{P}^2
                \end{equation*}
                Por lo que $f$ es homotópica a la aplicación constantemente igual a $p(z_0)$, como queríamos probar.
        \end{enumerate}
    \end{ejercicio}
