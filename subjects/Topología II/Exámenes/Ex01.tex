    \begin{ejercicio}[7.5 puntos]
        Sea $X_n = \{x\in \mathbb{R}^{n+1} : \|x\| \geq 2\}$ con la topología usual inducida.
        \begin{enumerate}[label=\alph*)]
            \item Encontrar un retracto de deformación de $X_n$ distinto de $X_n$ (detallar la retracción y la homotopía).

                Si consideramos:
                \begin{equation*}
                    S = \{x\in \mathbb{R}^{n+1} : \|x\| = 2\}
                \end{equation*}
                y definimos $H:X_n\times [0,1]\to X_n$ dada por:
                \begin{equation*}
                    H(x,t) = (1-t)x + 2t\frac{x}{\|x\|}
                \end{equation*}
                observemos que $\|x\|\neq 0$ para todo $x\in X_n$ así como que:
                \begin{align*}
                    \|H(x,t)\| &= \left\|(1-t)x + 2t\frac{x}{\|x\|}\right\| = \|x\| \left(1-t+\frac{2t}{\|x\|}\right) = \|x\|(1-t) + 2t  \\
                               &\geq 2(1-t) + 2t = 2 \qquad \forall x\in X_n, \quad \forall t\in [0,1]
                \end{align*}
                Por lo que $H(x,t)\in X_n$ para todo $(x,t)\in X_n\times [0,1]$, por lo que $H$ está bien definida. Es claro también que $H$ es continua, y observamos además que:
                \begin{itemize}
                    \item $H(x,0) = x\quad \forall x\in X_n$.
                    \item $H(x,1) = \dfrac{2x}{\|x\|}\in S \quad \forall x\in X_n$.
                    \item Si $y\in S$:
                        \begin{equation*}
                            H(y,1) = 2\frac{y}{\|y\|} = y 
                        \end{equation*}
                \end{itemize}
                En definitiva, si consideramos $r:X_n\to S$ dada por:
                \begin{equation*}
                    r(x) = H(x,1) = 2\frac{x}{\|x\|}
                \end{equation*}
                tenemos que $H$ es una homotopía entre $Id_{X_n}$ y $i\circ r$ (con $i:S\hookrightarrow X_n$), por lo que $S$ es un retracto de deformación de $X_n$.
            \item Estudiar si $X_1 \cong X_n$ cuando $n\geq 2$.

                Para cada $m\geq 1$ tenemos que $S_m=\{x\in \mathbb{R}^{m+1}:\|x\|=2\}$ es un retracto de deformación de $X_m$. Observamos además que la aplicación $f:\mathbb{S}^m\to S_m$ dada por:
                \begin{equation*}
                    f(x) = 2x
                \end{equation*}
                es un homeomorfismo, de donde deducimos que:
                \begin{equation*}
                    \pi_1(X_m) = \pi_1(S_m) \cong \pi_1(\mathbb{S}^m) \qquad \forall m\geq 1
                \end{equation*}
                Ahora, si $n\geq 2$ tenemos por un lado que:
                \begin{equation*}
                    \pi_1(X_1) \cong \pi_1(\mathbb{S}^1) \cong \mathbb{Z}
                \end{equation*}
                y por otro lado que:
                \begin{equation*}
                    \pi_1(X_n)\cong \pi_1(\mathbb{S}^n) = \{1\}
                \end{equation*}
                Por lo que $X_1$ y $X_n$ no pueden ser homeomorfos.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[2.5 puntos]
        Sea $X$ un conjunto con $|X|\geq 2$. Fijado $x_0\in X$ se considera la topología
        \begin{equation*}
            T = \{U\subseteq X : x_0\notin U\}\cup \{X\}
        \end{equation*}
        ¿Es $(X,T)$ contráctil?\\

        \noindent
        Sí es contráctil, pues si consideramos $H:X\times [0,1]\to X$ dada por:
        \begin{equation*}
            H(x,t) = \left\{\begin{array}{ll}
                x & \text{si\ } t\in \left[0,\nicefrac{1}{2}\right[ \\
                    x_0 & \text{si\ } t\in [\nicefrac{1}{2},1]
            \end{array}\right. 
        \end{equation*}
        Si tomamos $U\in T$:
        \begin{itemize}
            \item Si $U=X$ tenemos trivialmente que $H^{-1}(X) = X\times [0,1]$, que es un conjunto abierto.
            \item Si $U\neq X$ tenemos entonces que $x_0\notin U$, y:
                \begin{align*}
                    H^{-1}(U) &= \{(x,t)\in X\times [0,1] : H(x,t) \in U\} \\
                              &= \{(x,t)\in X\times [0,1]: (t\geq \nicefrac{1}{2}\land x_0 \in U) \lor (t<\nicefrac{1}{2}\land x\in U)\} \\
                              &=\{(x,t)\in X\times [0,1]: t<\nicefrac{1}{2}\land x\in U\} = U\times \left[0,\nicefrac{1}{2}\right[
                \end{align*}
                Con $U\times \left[0,\nicefrac{1}{2}\right[$ un conjunto abierto, como producto de dos conjuntos abiertos.
        \end{itemize}
        En definitiva, tenemos que $H$ es continua, así como que:
        \begin{itemize}
            \item $H(x,0) = x \quad \forall x\in X$.
            \item $H(x,1) = x_0\quad \forall x\in X$.
        \end{itemize}
        Por lo que $H$ es una homotopía entre $Id_X$ y la función constantemente igual a $x_0$, de donde $\{x_0\}$ es un retracto de deformación de $(X,T)$, por lo que este espacio topológico es contráctil.
    \end{ejercicio}
