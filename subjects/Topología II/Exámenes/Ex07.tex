    \begin{ejercicio}[5 puntos]
        Elija una pregunta de las siguientes:
        \begin{enumerate}[label=\alph*)]
            \item Sea $T$ un toro de $\mathbb{R}$. Prueba que no existe una retracción $r:\mathbb{R}^3\to T$.
            \item Sea $f:\mathbb{R}^2\to \mathbb{R}^2$ una aplicación continua y acotada. Demuestra que existe un punto $x_0\in \mathbb{R}^2$ tal que $f(x_0) = x_0$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Sean $R_1$ y $R_2$ las rectas de $\mathbb{R}^3$ con ecuaciones paramétricas respectivas $(\lm,0,-1)$ y $(\mu,0,1)$, $\lm,\mu\in \mathbb{R}$. Calcula el grupo fundamental de
        \begin{equation*}
            X = (\mathbb{S}^1\times \mathbb{R}) \cup R_1 \cup R_2
        \end{equation*}
        en el punto $p_0 = (1,0,0)$. Determina lazos basados en $p_0$ cuyas clases de equivalencia generen $\pi_1(X,p_0)$.
    \end{ejercicio}
