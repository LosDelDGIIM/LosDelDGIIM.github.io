    \begin{ejercicio}[5 puntos]
        Elija una pregunta de las siguientes:
        \begin{enumerate}[label=\alph*)]
            \item Sea $T$ un toro de $\mathbb{R}^3$. Prueba que no existe una retracción $r:\mathbb{R}^3\to T$.
            \item Sea $f:\mathbb{R}^2\to \mathbb{R}^2$ una aplicación continua y acotada. Demuestra que existe un punto $x_0\in \mathbb{R}^2$ tal que $f(x_0) = x_0$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Sean $R_1$ y $R_2$ las rectas de $\mathbb{R}^3$ con ecuaciones paramétricas respectivas $(\lm,0,-1)$ y $(\mu,0,1)$, $\lm,\mu\in \mathbb{R}$. Calcula el grupo fundamental de
        \begin{equation*}
            X = (\mathbb{S}^1\times \mathbb{R}) \cup R_1 \cup R_2
        \end{equation*}
        en el punto $p_0 = (1,0,0)$. Determina lazos basados en $p_0$ cuyas clases de equivalencia generen $\pi_1(X,p_0)$.
    \end{ejercicio}

    \textbf{Solución.}

    \begin{ejercicio}[5 puntos]
        Elija una pregunta de las siguientes:
        \begin{enumerate}[label=\alph*)]
            \item Sea $T$ un toro de $\mathbb{R}^3$. Prueba que no existe una retracción $r:\mathbb{R}^3\to T$.

                Por reducción al absurdo, si existiera una retracción $r:\mathbb{R}^3\to T$ tendríamos entonces que fijado $x\in T$ la inclusión $i:T\to \mathbb{R}^3$ induce un homomorfismo de grupos inyectivo $i_\ast:\pi_1(T,x)\to \pi_1(\mathbb{R}^3,x)$, pero tenemos que $\mathbb{R}^3$ es simplemente conexo, con lo que $\pi_1(\mathbb{R}^3,x) = \{[\varepsilon_x]\}$ y sabemos además que:
                \begin{equation*}
                    \pi_1(T,x) \cong \mathbb{Z}\times \mathbb{Z}
                \end{equation*}
                Por lo que tenemos un homomorfismo de grupos $\mathbb{Z}\times \mathbb{Z}\to \{1\}$ inyectivo, hemos llegado a una contradicción.
            \item Sea $f:\mathbb{R}^2\to \mathbb{R}^2$ una aplicación continua y acotada. Demuestra que existe un punto $x_0\in \mathbb{R}^2$ tal que $f(x_0) = x_0$.

                Tenemos que existe $R\in \mathbb{R}^+$ de forma que:
                \begin{equation*}
                    f(\mathbb{R}^2)\subseteq \overline{B}(0,R)
                \end{equation*}
                de esta forma, tenemos que la aplicación $\overline{f}:\overline{B}(0,R)\to \overline{B}(0,R)$ es continua. Como $\overline{B}(0,R)$ es homeomorfo a $\overline{\mathbb{D}}$, el Teorema del punto fijo de Brouwer nos dice que existe $x_0\in \overline{B}(0,R)$ de forma que:
                \begin{equation*}
                    x_0 = \overline{f}(x_0) = f(x_0)
                \end{equation*}
        \end{enumerate}
    \end{ejercicio}
