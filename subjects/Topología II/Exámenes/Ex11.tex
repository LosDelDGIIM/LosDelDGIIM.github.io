    \begin{ejercicio}
        Consideremos la circunferencia de $\mathbb{R}^2$ dada por
        \begin{equation*}
            C_r = \{x\in \mathbb{R}^2 : \|x\| = r\}
        \end{equation*}
        con $r>0$. Si $f:\mathbb{R}^2\to \mathbb{R}^2$ es una aplicación continua tal que $f\big|_{C_r}:C_r\to \mathbb{R}^2\setminus \{(0,0)\}$ no es homotópicamente nula, demuestra que existe un punto $x_0\in \mathbb{R}^2$ tal que $f(x_0) = (0,0)$.
    \end{ejercicio}

    \begin{ejercicio}
        Consideremos la esfera $\mathbb{S}^n$ con $n\geq 2$ y $p:\mathbb{S}^n\to B$ una aplicación recubridora sobre un espacio topológico $B$. Demuestra que $\pi_1(B,b)$ es finito para cada $b\in B$.
    \end{ejercicio}

    \begin{ejercicio}
        Sean $R$ y $B$ dos espacios topológicos donde $B$ es un espacio métrico compacto. Demuestra que si $p:R\to B$ es una aplicación recubridora donde cada punto $b\in B$ tiene el mismo número finito de preimágenes $n_0\in \mathbb{N}$ entonces $R$ es compacto.
    \end{ejercicio}

    \noindent
    \textbf{Solución.}

    \begin{ejercicio}
        Demuestra que no existe una aplicación continua e inyectiva desde la esfera $\mathbb{S}^2$ en el toro $T$.\\

        \noindent
        Por reducción al absurdo, supuesto que existe una aplicación $f:\mathbb{S}^2\to T$ continua e inyectiva, consideramos una aplicación recubridora $p:\mathbb{R}^2\to T$ (que puede ser por ejemplo $p = p_0\times p_0$ donde $p_0$ es la aplicación recubridora estándar de $\mathbb{S}^1$). Fijado $x_0\in \mathbb{S}^2$, $y_0 = f(x_0)$ y tomando $r_0 \in p^{-1}(y_0)$ tenemos que $\pi_1(\mathbb{S}^2,x_0)$ es trivial, con lo que la aplicación $f$ se puede levantar a una aplicación continua $\hat{f}:\mathbb{S}^2\to \mathbb{R}^2$. Por el Teorema de Borsuk-Ulam tenemos que existe $z_0\in \mathbb{S}^2$ de forma que:
        \begin{equation*}
            \hat{f}(z_0) = \hat{f}(-z_0)
        \end{equation*}
        Por lo que tenemos que:
        \begin{equation*}
            f(z_0) = p(\hat{f}(z_0)) = p(\hat{f}(-z_0)) = f(-z_0)
        \end{equation*}
        con $z_0\neq -z_0$, hemos llegado a una contradicción con la condición de que $f$ es inyectiva.
    \end{ejercicio}
