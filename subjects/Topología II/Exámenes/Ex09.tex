    \begin{ejercicio}[5 puntos]
        Razona si son verdaderas o falsas las siguientes afirmaciones:
        \begin{enumerate}[label=\alph*)]
            \item Existe una aplicación recubridora $p:\mathbb{R}\bb{P}^3\to \mathbb{S}^2\times\mathbb{S}^1$.
            \item Si $X$ es un espacio topológico conexo y localmente arcoconexo tal que toda aplicación continua $f:X\to \mathbb{S}^1$ se puede levantar a una aplicación continua $\hat{f}:X\to \mathbb{R}$ (para el recubridor usual $p:\mathbb{R}\to\mathbb{S}^1$) entonces $X$ es simplemente conexo.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[5 puntos]
        Consideremos la aplicación $p:\mathbb{S}^1\times\mathbb{S}^1\to\mathbb{S}^1\times \mathbb{S}^1$ dada por
        \begin{equation*}
            p(x_1,x_2,x_3,x_4) = \left(x_1x_3 - x_2x_4, x_1x_4 + x_2x_3, x_1(x^2_3-x^2_4)-2x_2x_3x_4,2x_1x_3x_4 + x_2(x^2_3 - x^2_4)\right)
        \end{equation*}
        o equivalentemente,
        \begin{equation*}
            p(\cos\theta, \sen\theta, \cos\varphi, \sen\varphi) = (\cos(\theta+\varphi), \sen(\theta+\varphi), \cos(\theta+2\varphi), \sen(\theta+2\varphi))
        \end{equation*}
        Demuestra que $p$ es una aplicación recubridora (3.5 puntos). Además, para el punto $x_0 = (1,0,1,0)$ calcula $p_\ast(\pi_1(\mathbb{S}^1\times\mathbb{S}^1,x_0))$ (1.5 puntos).
    \end{ejercicio}

    \textbf{Solución.}

    \begin{ejercicio}[5 puntos]
        Razona si son verdaderas o falsas las siguientes afirmaciones:
        \begin{enumerate}[label=\alph*)]
            \item Existe una aplicación recubridora $p:\mathbb{R}\bb{P}^3\to \mathbb{S}^2\times\mathbb{S}^1$.

                Es falsa: por reducción al absurdo, supuesto que existe $p:\mathbb{R}\mathbb{P}^3\to \mathbb{S}^2\times \mathbb{S}^1$ aplicación recubridora como $\mathbb{S}^2\times \mathbb{R}$ es recubridor universal de $\mathbb{S}^2\times \mathbb{S}^1$ por la aplicación $Id_{\mathbb{S}^2}\times p$ donde $p:\mathbb{R}\to\mathbb{S}^1$ es la aplicación usual:
                \begin{equation*}
                    p(x) = (\cos(2\pi x), \sen(2\pi x))
                \end{equation*}
                Tenemos entonces que $\mathbb{S}^2\times \mathbb{S}^1$ recubre a $\mathbb{R}\mathbb{P}^3$, y $\mathbb{S}^3$ es el recubridor universal de $\mathbb{R}\mathbb{P}^3$ (basta considerar la proyección al cociente), por lo que tendremos entonces que $\mathbb{S}^2\times \mathbb{R}$ es homeomorfo a $\mathbb{S}^3$, pero esto es una contradicción, ya que $\mathbb{S}^3$ es compacto y $\mathbb{S}^2\times \mathbb{R}$ no lo es.
            \item Si $X$ es un espacio topológico conexo y localmente arcoconexo tal que toda aplicación continua $f:X\to \mathbb{S}^1$ se puede levantar a una aplicación continua $\hat{f}:X\to \mathbb{R}$ (para el recubridor usual $p:\mathbb{R}\to\mathbb{S}^1$) entonces $X$ es simplemente conexo.

                Es falsa: podemos tomar tomar $X=\mathbb{R}\mathbb{P}^2$ y sea $f:X\to \mathbb{S}^1$ cualquier aplicación continua, si tomamos $x_0\in X$, $b_0 = f(x_0)$ y $r_0 \in p^{-1}(b_0)$ tenemos entonces que $\pi_1(X,x_0)\cong \mathbb{Z}_2$, grupo finito, con lo que $f_\ast(\pi_1(X,x_0))$ también ha de ser un subgupo finito de $\pi_1(B,b_0)\cong \mathbb{Z}$. Como el único subgrupo finito de $\mathbb{Z}$ es $\{0\}$ ha de ser:
                \begin{equation*}
                    f_\ast(\pi_1(X,x_0)) = \{[\varepsilon_{b_0}]\}
                \end{equation*}
                Tenemos así que $f_\ast(\pi_1(X,x_0))\subseteq p_\ast(\pi_1(\mathbb{R},r_0))$, por lo que $f$ se puede levantar. Vemos además que $X$ no es simplemente conexo, pues $\pi_1(X,x_0)\cong \mathbb{Z}_2$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[5 puntos]
        Consideremos la aplicación $p:\mathbb{S}^1\times\mathbb{S}^1\to\mathbb{S}^1\times \mathbb{S}^1$ dada por
        \begin{equation*}
            p(x_1,x_2,x_3,x_4) = \left(x_1x_3 - x_2x_4, x_1x_4 + x_2x_3, x_1(x^2_3-x^2_4)-2x_2x_3x_4,2x_1x_3x_4 + x_2(x^2_3 - x^2_4)\right)
        \end{equation*}
        o equivalentemente,
        \begin{equation*}
            p(\cos\theta, \sen\theta, \cos\varphi, \sen\varphi) = (\cos(\theta+\varphi), \sen(\theta+\varphi), \cos(\theta+2\varphi), \sen(\theta+2\varphi))
        \end{equation*}
        Demuestra que $p$ es una aplicación recubridora (3.5 puntos). Además, para el punto $x_0 = (1,0,1,0)$ calcula $p_\ast(\pi_1(\mathbb{S}^1\times\mathbb{S}^1,x_0))$ (1.5 puntos).\\

        \noindent
        Vemos que $p$ es claramente una aplicación continua. Definimos ahora la aplicación $q:\mathbb{S}^1\times \mathbb{S}^1\to \mathbb{S}^1\times \mathbb{S}^1 $ dada por:
        \begin{equation*}
            q(\cos\theta,\sen\theta,\cos\varphi,\sen\varphi) = (\cos(2\theta-\varphi),\sen(2\theta-\varphi), \cos(\varphi-\theta), \sen(\varphi-\theta))
        \end{equation*}
        vemos que es una aplicación continua, y que:
        \begin{align*}
            &p(q(\cos\theta,\sen\theta,\cos\varphi,\sen\varphi)) = p(\cos(2\theta-\varphi),\sen(2\theta-\varphi), \cos(\varphi-\theta), \sen(\varphi-\theta)) \\
                                                                &= (\cos(2\theta-\varphi+\varphi-\theta),\sen(2\theta-\varphi+\varphi-\theta),\cos(2\theta-\varphi+2\varphi-2\theta),\sen(2\theta-\varphi+2\varphi-2\theta)) \\
                                                                &= (\cos(\theta),\sen(\theta),\cos(\varphi),\sen(\varphi)) \\
                                                                &q(p(\cos(\theta),\sen(\theta),\cos(\varphi),\sen(\varphi))) = q(\cos(\theta+\varphi), \sen(\theta+\varphi), \cos(\theta+2\varphi), \sen(\theta+2\varphi)) \\
                                                                &= (\cos(2(\theta+\varphi)-(\theta+2\varphi)), \sen(2(\theta+\varphi)-(\theta+2\varphi)), \\ &\qquad \qquad \cos(\theta+2\varphi-(\theta+\varphi)), \sen(\theta+2\varphi-(\theta+\varphi))) \\
                                                                &= (\cos(\theta),\sen(\theta),\cos(\varphi),\sen(\varphi))
        \end{align*}
        Por lo que $q$ es la aplicación inversa de $p$, con lo que $p$ es un homeomorfismo, luego una aplicación recubridora.\\

        \noindent
        Para $x_0 = (1,0,1,0)$, tenemos que $\pi_1(\mathbb{S}^1\times \mathbb{S}^1,x_0)$ está generador por $[\alpha],[\beta]$, donde $\alpha,\beta:[0,1]\to \mathbb{S}^1\times\mathbb{S}^1$ vienen dadas por:
        \begin{equation*}
            \alpha(t) = (\cos(2\pi t),\sen(2\pi t), 1, 0), \qquad \beta(t) = (1,0,\cos(2\pi t),\sen(2\pi t))
        \end{equation*}
        Si calculamos su imagen mediante $p$ obtenemos que:
        \begin{align*}
            p(\alpha(t)) &= (\cos(2\pi t),\sen(2\pi t),\cos(2\pi t),\sen(2\pi t))\\
            p(\beta(t)) &= (\cos(2\pi t),\sen(2\pi t),\cos(4\pi t),\sen(4\pi t)) 
        \end{align*}
        Vemos así que $[p(\alpha)] = [\alpha]\ast [\beta]$ y que $[p(\beta)] = [\alpha]\ast [\beta]\ast [\beta]$, como tenemos la imagen de los generadores de $\pi_1(\mathbb{S}^1\times\mathbb{S}^1,x_0)$ tenemos pues que:
        \begin{equation*}
            p_\ast(\mathbb{S}^1\times\mathbb{S}^1,x_0) = \langle [\alpha]\ast[\beta],[\alpha]\ast[\beta]\ast[\beta] \rangle 
        \end{equation*}
        Podemos además observar que:
        \begin{equation*}
            [\beta] = ([\alpha]\ast[\beta]\ast[\beta])\ast([\alpha]^{-1}\ast[\beta]^{-1}) 
        \end{equation*}
        de donde $[\beta]$ también está en dicho grupo y por tanto:
        \begin{equation*}
            [\alpha] = [\alpha] \ast [\beta] \ast [\beta]^{-1}
        \end{equation*}
        En definitiva, tenemos que:
        \begin{equation*}
        \begin{equation*}
            p_\ast(\mathbb{S}^1\times\mathbb{S}^1,x_0) = \langle [\alpha]\ast[\beta],[\alpha]\ast[\beta]\ast[\beta] \rangle  = \pi_1(\mathbb{S}^1\times \mathbb{S}^1,p(x_0))
        \end{equation*}
    \end{ejercicio}
