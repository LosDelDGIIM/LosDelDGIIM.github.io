\chapter{Ecuaciones clásicas de la Física Matemática}
\begin{description}
    \item [Ecuación de ondas.] Versión en dimensión 1 en espacio:
        \begin{equation*}
            \frac{\partial^2 u}{\partial t^2} - c^2 \frac{\partial^2 u}{\partial x^2} = 0
        \end{equation*}
    \item [Ecuación del calor.] Versión en dimensión 1 en espacio:
        \begin{equation*}
            \frac{\partial u}{\partial t} - D\frac{\partial^2 u}{\partial x^2} = 0
        \end{equation*}
    \item [Ecuación de Laplace.] Versión en dimensión 2 en el espacio:
        \begin{equation*}
            \frac{\partial^2 u}{\partial x^2} + \frac{\partial^2 u}{\partial y^2} = 0
        \end{equation*}
\end{description}
Dado un campo vectorial $F:\mathbb{R}^d\to \mathbb{R}^d$, podemos escribir $F=(F_1,\ldots,F_d)$ con cada $F_i:\mathbb{R}^d\to \mathbb{R}$.

\begin{definicion}
    Para $F:\mathbb{R}^d\to \mathbb{R}^d$ definimos la divergencia de $F$ como la aplicación $div F:\mathbb{R}^d\to \mathbb{R}$ dada por:
    \begin{equation*}
        div F = \sum_{i=1}^{d} \frac{\partial F_i}{\partial x_i}
    \end{equation*}
    A veces se le denota por $\nabla\cdot F$, ya que formalmente tenemos:
    \begin{equation*}
        \left(\frac{\partial }{\partial x_1}, \ldots, \frac{\partial }{\partial x_d}\right) \cdot (F_1,\ldots,F_d)
    \end{equation*}
\end{definicion}

\noindent
Si pensamos que $F$ es un campo de fuerzas, $div F$ nos dice cómo varían los volúmenes en dicho campo de fuerzas.

\begin{ejemplo}
    Si $x\in \mathbb{R}^d$ se escribe como $x=(x_1,x_2,\ldots,x_d)$, podemos considerar $F(x)=x$ y tendríamos entonces que:
    \begin{equation*}
        div F = \frac{\partial x_1}{\partial x_1} + \frac{\partial x_2}{\partial x_2} + \ldots + \frac{\partial x_d}{\partial x_d} = d
    \end{equation*}
    Dada $f:\mathbb{R}^d\to \mathbb{R}$, podemos calcular $\nabla f:\mathbb{R}^d\to \mathbb{R}^d$ y:
    \begin{equation*}
        div(\nabla f) = \sum_{i=1}^{d} \frac{\partial }{\partial x_i}\left(\frac{\partial f}{\partial x_i}\right) = \sum_{i=1}^{d} \frac{\partial^2 f}{\partial x_i^2} = tr(D^2 f)
    \end{equation*}
    A esta cantidad se le suele denotar por $\triangle f$, o como $\nabla^2 f$, el ``Laplaciano de $f$''.
\end{ejemplo}

\noindent
En dimensiones arbitrarias tenemos:
\begin{description}
    \item [Ecuación del calor.] 
        \begin{equation*}
            \frac{\partial u}{\partial t} = D\triangle u
        \end{equation*}
    \item [Ecuación de ondas.] 
        \begin{equation*}
            \frac{\partial^2 u}{\partial t^2} = c^2 \triangle u
        \end{equation*}
    \item [Ecuación de Laplace.]
        \begin{equation*}
            \triangle u = 0
        \end{equation*}
        Ecuación de Poisson para $\rho$ dada:
        \begin{equation*}
            \triangle u = \rho(x)
        \end{equation*}
\end{description}
Aquí tenemos el convenio de que para $u = u(t,x_1,\ldots,x_d)$:
\begin{equation*}
    \triangle u = \sum_{i=1}^d \frac{\partial^2 u}{\partial x_i^2}
\end{equation*}
Es decir, sólo afecta a las variables espaciales.\\

\noindent
La ecuación de ondas y de Laplace son variacionales\footnote{Esto se debe a que en la ecuación de calor hay disipación de la energía.}, es decir, son la ecuación de Euler-Lagrange de determinado lagrangiano.\\

\noindent
Dada $u:\Omega\to \mathbb{R}$ con $\Omega\subset \mathbb{R}^d$ abierto, consideramos un funcional de la forma
\begin{equation*}
    \cc{F}(u) = \int_\Omega F(x,u,\nabla u)~dx \qquad (\text{donde\ } x=(x_1,\ldots,x_d))
\end{equation*}
La ecuación de Euler-Lagrange asociada es en $\Omega$:
\begin{equation}\label{eq:abreviar}
    F_u - \frac{d}{dx_1}(F_{p_1}) - \frac{d}{dx_2}(F_{p_2}) - \ldots - \frac{d}{dx_d}(F_{p_d})
\end{equation}
Donde hemos escrito:
\begin{equation*}
    F_u = \frac{\partial F}{\partial u}, \qquad F_{p_i} = \frac{\partial F}{\partial p_i} 
\end{equation*}
Con el convenio de que $F=F(x,u,p_1,\ldots,p_d)$, donde $p_i = \frac{\partial u}{\partial x_i}$. Observamos que $F:\Omega\times\mathbb{R}\times\mathbb{R}^d\to \mathbb{R}$.\\

\noindent
En vista de la definición de la divergencia, podemos resumir~\eqref{eq:abreviar} como:
\begin{equation*}
    F_u - div_x(F_p) = 0
\end{equation*}
Donde entendemos que $F_p = (F_{p_1}, \ldots, F_{p_d})$.

\noindent
Para $d=2$ variables esto funciona porque (variable $(x,y)$):
\begin{align*}
    \delta_\varphi \cc{F}(u) &= \int_\Omega F_u\varphi + F_{p_x}\frac{\partial \varphi}{\partial x} + F_{p_y} \frac{\partial \varphi}{\partial y}~dxdy = 0 \\
                             &= \int_\Omega F_u\varphi - \frac{d}{dx}(F_{p_x})\varphi - \frac{d}{dy}(F_{p_y})\varphi~dxdy = 0 \\
                             &= \int_\Omega(F_u - \frac{d}{dx}(F_{p_x}) - \frac{d}{dy}(F_{p_y}))\varphi~dxdy = 0
\end{align*}

\begin{ejemplo} % // TODO: Desplazamientos de un membrana elástica en prado
    Sea $S$ una superficie descrita mediante la gráfica de una función $u:\Omega\subset\mathbb{R}^2\to \mathbb{R}$. 
    La ecuación de Euler-Lagrange es:
    \begin{equation*}
        F_u - \frac{d}{dx}(F_{p_x}) - \frac{d}{dy}(F_{p_y}) = 0 \qquad \text{en\ } \Omega
    \end{equation*}
    Donde:
    \begin{equation*}
        F_{p_x} = \frac{\partial F}{\partial p_x}, \quad F_{p_y} = \frac{\partial F}{\partial p_y}, \quad p_x = \frac{\partial u}{\partial x},\quad p_y = \frac{\partial u}{\partial y}
    \end{equation*}
    Tenemos:
    \begin{equation*}
        F = F(x,y,u,p_x,p_y) = \sqrt{1+{(p_x)}^{2} + {(p_y)}^{2}}
    \end{equation*}
    Así:
    \begin{equation*}
        F_{p_x} = \frac{p_x}{\sqrt{1+{(p_x)}^{2}+{(p_y)}^{2}}}, \quad F_{p_y} = \frac{p_y}{\sqrt{1+{(p_x)}^{2}+{(p_y)}^{2}}}
    \end{equation*}
    La ecuación queda:
    \begin{equation*}
        -\frac{\partial }{\partial x}\left(\frac{\frac{\partial u}{\partial x}}{\sqrt{1+{\left(\frac{\partial u}{\partial x}\right)}^{2}+{\left(\frac{\partial u}{\partial y}\right)}^{2}}}\right) - \frac{\partial }{\partial y}\left(\frac{\frac{\partial u}{\partial y}}{\sqrt{1+{(\frac{\partial u}{\partial x})}^{2}+{\left(\frac{\partial u}{\partial y}\right)}^{2}}}\right) = 0
    \end{equation*}
    Es una ``ecuación de las superficies mínimas''. Podemos abreviarla como:
    \begin{equation*}
        div\left(\frac{\nabla u}{\sqrt{1+|\nabla u|^2}}\right) = 0
    \end{equation*}
\end{ejemplo}

\section{Ecuación de ondas}
\noindent
Anteriormente dijimos que era varacional. Se puede obtener a partir de la acción:
\begin{equation*}
    A(u) = \int_{0}^{T} \left(\int_{\Omega} {\left(\frac{\partial u}{\partial t}\right)}^{2}~dx - \int_\Omega c^2|\nabla u|^2~dx\right) ~dt 
\end{equation*}
Donde $\Omega\subset \mathbb{R}^d$ es un abierto y $c>0$ es la ``velocidad del sonido'' (realmente es la velocidad a la que se propagan las interacciones, pero se le llama indistintamente velocidad del sonido). Tenemos $u:[0,T]\times \Omega\to \mathbb{R}$ la coordenada de desplazamiento respecto al equilibrio. Calculamos su ecuación de Ecuación-Lagrange:
\begin{gather*}
    F_u - div_{(t,x)}(F_p) = 0 \\
    -\frac{d}{dt}\left(2\frac{\partial u}{\partial t}\right) +\sum_{i=1}^d \frac{d}{dx_i}\left(2c^2\frac{\partial u}{\partial x_i}\right) = 0
\end{gather*}
En notación compacta obtenemos la ecuación de ondas ya enunciada:
\begin{equation*}
    \frac{\partial^2 u}{\partial t^2} - c^2 \triangle u = 0 
\end{equation*}
Nos centraremos ahora en dimensión espacial $d=1$ por falta de tiempo, con lo que:
\begin{equation*}
    \frac{\partial^2 u}{\partial t^2} - c^2 \frac{\partial^2 u}{\partial x^2} = 0
\end{equation*}
Observemos que la ecuación de ondas es lineal, es decir, dadas $u_1,u_2$ soluciones de la ecuación, entonces $c_1u_1 + c_2u_2$ es también solución, para cualesquiera $c_1,c_2\in \mathbb{R}$.\\

\noindent
No nos interesará conocer todas las soluciones de dicha ecuación. Así, nos centraremos en:
\begin{description}
    \item [Problemas de Cauchy.] Tenemos la ecuación:
        \begin{equation*}
            u_{tt} - c^2 u_{xx} = 0 \qquad \text{en\ } \left[0,\infty\right[\times \mathbb{R}
        \end{equation*}
        Junto con la condición inicial (para ciertas $\varphi,\psi$ dadas, tenemos dos derivadas de tiempo, por lo que tendremos que dar dos condiciones):
        \begin{equation*}
            \left\{\begin{array}{l}
                u(t=0,x) = \varphi(x) \quad \text{en\ } \mathbb{R} \\
                u_t(t=0,x) = \psi(x) \quad \text{en\ } \mathbb{R}
            \end{array}\right.
        \end{equation*}
    \item [Problemas de valores en la frontera con extremos fijos.] Aquí:
        \begin{equation*}
            u_{tt} - c^2 u_{xx} = 0 \qquad \text{en\ } \left[0,\infty\right[\times [0,L]
        \end{equation*}
        Junto con la condición inicial (para ciertas $\varphi,\psi$):
        \begin{equation*}
            \left\{\begin{array}{l}
                u(t=0,x) = \varphi(x) \quad \text{en\ } [0,L] \\
                u_t(t=0,x) = \psi(x) \quad \text{en\ } [0,L]
            \end{array}\right.
        \end{equation*}
        Y para que este problema esté bien planteado, hemos de añadir las condiciones:
        \begin{equation*}
            \left\{\begin{array}{l}
                u(t,x=0) = 0 \\
                u(t,x=L) = 0 
            \end{array}\right.
        \end{equation*}
\end{description}

\subsection{Conservación de la energía}
\noindent
En el problema de valores en la frontera con extremos fijos, la energía asociada es:
\begin{equation*}
    E(u) = \frac{1}{2}\int_{0}^{L} {(u_t)}^{2}+c^2{(u_x)}^{2}~dx 
\end{equation*}
Dada $u$ solución del problema, vamos que $E(u)$ no depende del tiempo. Para ello:
\begin{equation*}
    2\frac{d}{dt}E(u) = \int_{0}^{L} \frac{d}{dt}{(u_t)}^{2}+c^2 \frac{d}{dt}{(u_x)}^{2}~dx  = \int_{0}^{L} 2u_tu_{tt} + 2c^2u_xu_{tx}~dx 
\end{equation*}
Integrando por partes el segundo sumando de la última integral:
\begin{align*}
    2\frac{d}{dt}E(u) &= \int_{0}^{L} 2u_tu_{tt} - 2c^2u_{xx}u_t~dx + [2c^2u_xu_t]_{x=0}^{x=L} \\
                      &= 2\int_{0}^{L} u_t\cancelto{0}{(u_{tt}-c^2u_{xx})}~dx + 2c^2 u_x(t,x=L)u_t(t,x=L) \\ &\qquad \qquad \qquad \qquad \qquad \qquad\quad  - 2c^2 u_x(t,x=0)u_t(t,x=0)
\end{align*}
Ahora, observamos que $u_t(t,x=L) = 0 = u_t(t,x=0)$, pues tenemos siempre que $u(t,x=0) = 0 = u(t,x=L)$ para todo $t\geq 0$. En definitiva, $\frac{d}{dt}E(u) = 0$, por lo que $E(u)$ no depende del tiempo, es decir, la energía se mantiene constante en todo momento, es decir, el modelo no disipa energía.\\

\noindent
Para la versión del Problema de Cauchy podríamos hacer algo parecido, pero sustituyendo $x=0$ y $x=L$ por $x\to -\infty$ y $x\to +\infty$, haciendo necesario que el argumento sea más técnico.

\subsubsection{Unicidad de la solución}
\noindent
Esta \textbf{conservación de la energía} nos dice que el problema de valores en la frontera con extremos fijos \textbf{tiene a lo mucho una solución}, pues la ecuación es lineal:
\begin{proof}
    por reducción al absurdo, si tenemos $u_1,u_2$ dos soluciones distintas del mismo problema tenemos entonces que: 
    \begin{equation*}
        \left\{\begin{array}{l}
            u_{1_{tt}} - c^2 u_{1_{xx}} = 0 \\
            u_1(t=0,x) = \varphi(x) \\
            u_{1_t}(t=0,x) = \psi(x) \\
            u_1(t,x=0) = 0 = u_1(t,x=L)
        \end{array}\right.\quad 
        \left\{\begin{array}{l}
            u_{2_{tt}} - c^2 u_{2_{xx}} = 0 \\
            u_2(t=0,x) = \varphi(x) \\
            u_{2_t}(t=0,x) = \psi(x) \\
            u_2(t,x=0) = 0 = u_2(t,x=L)
        \end{array}\right.
    \end{equation*}
    Como la ecuación es lineal, si tomamos $v(t,x) = u_1(t,x) - u_2(t,x)$ tenemos entonces que:
    \begin{equation*}
        \left\{\begin{array}{l}
            v_{tt} - c^2 v_{xx} = 0 \\
            v(t=0,x) = 0 = v_t(t=0,x) \quad \text{en\ } [0,L] \\
            v(t,x=0) = 0 = v_t(t,x=0) \quad t\geq 0
        \end{array}\right.
    \end{equation*}
    Es decir, tenemos una posición inicial de 0 y una velocidad inicial nula, por lo que intuitivamente la única solución es $0$, de donde $v=0$. Para verlo de forma matemática tenemos que usar que la energía se conserva, es decir, que $E(v)$ no depende del tiempo, por lo que podemos calcular la energía del sistema para $v$ en el instante inicial:
    \begin{equation*}
        E(v)(t=0) = \frac{1}{2}\int_{0}^{L} {(v_t)}^{2}+c^2 {(v_x)}^{2}~dx  = \frac{1}{2}\int_{0}^{L} 0^2 + c^2 0^2~dx  = 0 
    \end{equation*}
    Por tanto, tenemos que:
    \begin{equation*}
        0 = \int_{0}^{L} {\left(\frac{\partial v}{\partial t}\right)}^{2}+c^2 {\left(\frac{\partial v}{\partial x}\right)}^{2}~dx  \quad \forall t\geq 0 \quad\Longrightarrow\quad \frac{\partial v}{\partial t} \equiv 0 \equiv \frac{\partial v}{\partial x}
    \end{equation*}
    Así, $v$ es constante y sabemos que $v(t=0,x) = 0$ para $x\in [0,L]$, por lo que $v\equiv 0$, de donde $u_1 \equiv u_2$.
\end{proof}

\noindent
Para el caso de Problemas de Cauchy tenemos una situación análoga, pero más técnica.

\subsubsection{Existencia de soluciones}
\noindent
Para el problema de Cauchy conocemos una fórmula\footnote{No la demostraremos, pero se hace el cambio de variable $y=x+ct$, $z=x-ct$ y la ecuación se simplifica.} para la solución (fórmula de d'Alembert):
\begin{enumerate}[label=\alph*)]
    \item Existen $\Phi_+,\Phi_-:\mathbb{R}\to\mathbb{R}$ tales que
        \begin{equation*}
            u(t,x) = \Phi_+(x+ct) + \Phi_-(x-ct)
        \end{equation*}
    \item Y tenemos:
        \begin{equation*}
            u(t,x) = \frac{\varphi(x+ct)+\varphi(x-ct)}{2} + \frac{1}{2c} \int_{x-ct}^{x+ct} \psi(s)~ds 
        \end{equation*}
\end{enumerate}
Son relevantes los dominios de influencia y los de dependencia.

\subsection{Problemas de valores en la frontera con extremos fijos}
\noindent
Este problema también recibe el nombre de \textbf{problema mixto} par ala ecuación de ondas.\\

\noindent
Tenemos la ecuación:
\begin{equation*}
    u_{tt} - c^2 u_{xx} = 0 \qquad \text{en} \quad \left[0,+\infty\right[\times [0,L]
\end{equation*}
donde $c>0$ es la velocidad del sonido y $L>0$ la longitud del medio. Buscamos soluciones $u:\left[0,\infty\right[\times[0,L]\to \mathbb{R}$, conociendo:
\begin{align*}
    u(t=0,x) &= \varphi(x) \quad \text{en} \quad [0,L] \\
    u_t(t=0,x) &= \psi(x) \quad \text{en} \quad [0,L] \\
    u(t,x=0) &= 0 \quad \text{para} \quad t > 0 \\
    u(t,x=L) &= 0  \quad \text{para} \quad  t > 0 
\end{align*}
El nombre de ``problema con extremos fijos'' proviene de estas dos últimas condiciones, donde pueden modelar, por ejemplo, que una cuerda está atada en sus extremos.\\

\noindent
Para resolverlo, una primera observación es que $u(t,x) = 0$ es una solución de la ecuación para $\varphi\equiv 0\equiv \psi$. Buscamos ahora soluciones en variables separadas, es decir tras un cambio de variable $x\pm ct$ obtenemos:
\begin{equation*}
    u(t,x) = T(t)X(x)
\end{equation*}
para $T:\left[0,\infty\right[\to \mathbb{R}$ y $X:[0,L]\to \mathbb{R}$.\\

\noindent
Sea $u$ una solución de dicha ecuación, tenemos:
\begin{align*}
    u_t &= T'(t) X(x), \qquad u_{tt} = T''(t)X(x) \\
    u_x &= T(t) X'(x), \qquad u_{xx} = T(t)X''(x)
\end{align*}
Sustituimos en la ecuación de ondas, para averiguar qué funciones $T$ y $X$ nos sirven: 
\begin{equation*}
    T''X - c^2 TX'' = 0 \quad\Longrightarrow\quad T''X = c^2 TX''
\end{equation*}
formalmente:
\begin{equation*}
    \frac{T''}{T}(t) = c^2 \frac{X''}{X}(x)
\end{equation*}
Tenemos dos funciones iguales que dependen de variables distintas, por lo que\footnote{También se puede ver derivando y viendo que obtenemos derivada nula.} la única salida es que cada una de las funciones sea igual a una determinada constante. Así, para cierto $\lm \in \mathbb{R}$ tenemos que (tomamos como constante $-\lm$):
\begin{equation*}
    \frac{T''}{T}(t) = -\lm = c^2 \frac{X''}{X}(x)
\end{equation*}
Tenemos ahora dos problemas:
\begin{equation*}
    T'' + \lm T = 0, \qquad X'' + \frac{\lm}{c^2} X = 0
\end{equation*}
Observando las condiciones iniciales que debe verificar una solución $u$ para intentar resolver el problema, tenemos:
\begin{align*}
    u(t=0,x) &= T(0)X(x) \stackrel{?}{=} \varphi(x) \qquad \text{similar con\ } \psi \\
    u(t,x=0) &= T(t)X(0) = 0 \\
    u(t,x=L) &= T(t)X(L) = 0
\end{align*}
Tenemos ahora dos opciones:
\begin{itemize}
    \item Bien $T(t) = 0$, de donde $u(t,x) = 0$.
    \item Bien $X(0) = 0 = X(L)$.
\end{itemize}
Pasamos así a resolver el problema:
\begin{gather*}
    X'' + \frac{\lm}{c^2} X = 0 \quad \text{en\ } [0,L] \\
    X(0) = 0 = X(L)
\end{gather*}
Es un problema de Sturm-Liouville, donde obtendríamos: % // TODO: A partir de aqui me pierdo, dice que ya lo hemos visto
\begin{equation*}
    c_1\sen\left(\frac{\sqrt{\lm}}{c}x\right) + c_2\cos\left(\frac{\sqrt{\lm}}{c}x\right)
\end{equation*}
Valores propios:
\begin{equation*}
    \lm_n = {\left(\frac{n\pi c}{L}\right)}^{2}, \qquad n\geq 1
\end{equation*}
Funciones propias:
\begin{equation*}
    X_n(x) = c_n \sen\left(\frac{n\pi x}{L}\right), \qquad c_n \in \mathbb{R}
\end{equation*}
Con estos valores $\lm_n$ resolvemos:
\begin{equation*}
    T_n'' + \lm_n T_n = 0
\end{equation*}
obteniendo:
\begin{equation*}
    T_n(t) = a_n \sen\left(\frac{n\pi c}{L}t\right) + b_n \cos\left(\frac{n\pi c}{L}\right), \quad a_n,b_n\in \mathbb{R}, \quad n\geq 1
\end{equation*}
Juntándolo todo obtenemos la siguiente familia de soluciones:
\begin{align*}
    u_n(t,x) &= \left(a_n\sen\left(\sqrt{\lm_n}t\right) + b_n\cos\left(\sqrt{\lm_n}t\right)\right)c_n\sen\left(\frac{\sqrt{\lm_n}}{c}x\right) \\
             &= \left(A_n\sen\left(\frac{n\pi c}{L}t\right) + B_n\cos\left(\frac{n\pi c}{L}t\right)\right)\sen\left(\frac{n\pi x}{L}\right)
\end{align*}
Ahora, ¿cumplimos las condiciones iniciales $\varphi$ y $\psi$? Observamos que:
\begin{align*}
    u(t=0,x) &= B_n\sen\left(\frac{n\pi x}{L}\right) \stackrel{?}{=} \varphi(x) \\
    \frac{\partial u_m}{\partial t}\big|_{t=0}(x) &= A_n\sqrt{\lm_n}\sen(\frac{n\pi x}{L}) \stackrel{?}{=} \psi(x)
\end{align*}
Si $\varphi,\psi$ son múltiplos de $\sen\left(\frac{n\pi x}{L}\right)$ entonces sabemos resolver el problema. Si $\varphi,\psi$ son combinaciones lineales de senos con frecuencias naturales, por linalidad también podemos calcular la solución.\\

\noindent
Dada $\varphi(x)$ arbitrario, hasta qué punto podemos aproximarla con expresiones del tipo:
\begin{equation*}
    \sum_{k=1}^{n}A_k\sen\left(\frac{k\pi x}{L}\right)
\end{equation*}
para coeficientes $A_k\in \mathbb{R}$ ordenados.\\

\noindent
En definitiva, sabemos resolver la ecuación de ondas
\begin{equation*}
    u_{tt} - c^2 u_{xx} = 0
\end{equation*}
con condiciones Dirichlet cero en $[0,L]$ para datos iniciales que sean combinaciones lineales de $\sen\left(\frac{n\pi x}{L}\right)$. En definitiva, nos interesa conocer qué tal funciona una combinación del tipo: 
\begin{equation*}
    \frac{a_0}{2} + \sum_{n=1}^\infty a_n\cos(nx) + b_n\sen(nx)
\end{equation*}
para aproximar cualquier otra función. Es decir, una \textbf{serie de Fourier}. Tendremos también:
\begin{itemize}
    \item Desarrollo en base de senos:
        \begin{equation*}
            \sum_{n=1}^\infty b_n \sen\left(\frac{nx}{2}\right)
        \end{equation*}
    \item Desarrollo en base de cosenos:
        \begin{equation*}
            \frac{a_0}{2} + \sum_{n=1}^\infty a_n \cos\left(\frac{nx}{2}\right)
        \end{equation*}
\end{itemize}
Resulta que $\{1, \cos(mx), \sen(mx)\}$ es una base ortogonal, variando $m$ y podemos aplicar la teoría de aproximación de Métodos Numéricos I.\\

\section{Series de Fourier}
\noindent
Dada $f\in L^2(I)$, definimos las sumas parciales de la serie de Fourier (generalizada) asociada a $f$ mediante:
\begin{equation*}
    S_n = \sum_{k=1}^{n} c_k f_k, \qquad c_k = \langle f,f_k \rangle 
\end{equation*}
¿Cuál es la bondad de esta aproximación?
\begin{equation*}
    \|f-S_n\|_2^2 = \langle f-S_n,f-S_n \rangle  = \langle f,f \rangle -2\langle f,S_n \rangle  + \langle S_n,S_n \rangle 
\end{equation*}
Así:
\begin{align*}
    \langle f,f \rangle &= \|f\|_2^2 \\
    \langle S_n,S_n \rangle  &= \left\langle \sum_{k=1}^{n}c_kf_k,\sum_{k=1}^{n}c_kf_k \right\rangle  = \sum_{k=1}^{n} c_k^2 \langle f_k,f_k \rangle  = \sum_{k=1}^{n} c_k^2 \\
    \langle f,S_n \rangle  &= \langle f,\sum_{k=1}^{n}c_kf_k \rangle = \sum_{k=1}^{n}c_k \langle f, f_k \rangle = \sum_{k=1}^{n} c_k^2
\end{align*}
Con lo que:
\begin{equation*}
    \|f-S_n\|_2^2 = \langle f,f \rangle - 2\langle f,S_n \rangle  + \langle S_n,S_n \rangle  =\|f\|_2^2 - \sum_{k=1}^{n}c_k^2 \geq 0
\end{equation*}
En definitiva:
\begin{equation*}
    \sum_{k=1}^{n}c_k^2 \leq \|f\|_2^2 \qquad \forall n\geq 1
\end{equation*}
Por lo que $\left\{\sum\limits_{k=1}^n c_k^2\right\}$ es una serie convergente, tomando límite obtenemos la \textbf{desigualdad de Bessel}:
\begin{equation*}
    \sum_{k=1}^{\infty}c_k^2 \leq \|f\|_2^2
\end{equation*}

\noindent
Se ve que:
\begin{equation*}
    \left\{\begin{array}{l}
        f_0(x) = 1 \\
        f_{2n-1}(x) = \cos(nx) \\
        f_{2n}(x) = \sen(nx)
    \end{array}\right.
\end{equation*}
Es un sistema ortogonal y completo (haría falta probarlo, pero es demansiado complicado para este curso) en $L^2(0,2\pi)$. Nos sale que el sistema no es ortonormal, puesto que por ejemplo:
\begin{equation*}
    \|f_{2n-1}\|_2^2 = \pi = \|f_{2n}\|^2_2
\end{equation*}
Por lo que ajustando con constantes, el sistema:
\begin{equation*}
    \left\{\begin{array}{l}
        f_0(x) = \frac{1}{\sqrt{2\pi}} \\
        f_{2n-1}(x) = \frac{\cos(nx)}{\sqrt{\pi}} \\
        f_{2n}(x) = \frac{\sen(nx)}{\sqrt{\pi}} \\
    \end{array}\right.
\end{equation*}
Sí que es un sistema ortonormal completo de $L^2(0,2\pi)$. Para obtener el sistema en $L^2(0, L)$ se obtienen distintas fórmulas, donde aparece $L$.

\subsection{Problema mixto para la ecuación de ondas}
\noindent
Tenemos el problema:
\begin{equation*}
    \left\{\begin{array}{ll}
            u_{tt} - c^2 u_{xx} = 0 , & \text{en\ } \left[0,\infty\right[\times [0,L] \\
                u(0,x) = \varphi(x) , &\text{en\ } [0,L] \\
                u_t(0,x) = \psi(x) , &\text{en\ } [0,L] \\
                u(t,0) = 0 = u(t,L) , & \text{en\ } \left[0,\infty\right[
    \end{array}\right.
\end{equation*}
Diremos que $u$ es solución de:
\begin{equation*}
    u_{tt}-c^2u_{xx} = 0 \quad \text{en\ } \left[0,\infty\right[\times[0,L]
\end{equation*}
si $u\in C^2(\left[0,\infty\right[\times [0,L])$ y cumple que $u_{tt}-c^2u_{xx}=0$ para todo $(t,x)\in \left]0,\infty\right[\times [0,L]$. Ahora:
\begin{itemize}
    \item Como $u(t=0,x)$ es de clase $C^2$, entonces $\varphi$ debe ser de clase $C^2$.
    \item Como $u_t(t=0,x)$ es de clase $C^1$, entonces $\psi$ debe ser de clase $C^1$.
\end{itemize}
Ver diapositivas para las condiciones (problema mixto para la ecuación de ondas).

\begin{ejemplo} % // TODO: Ejercicio 6 de la relación de Series de Fourier % // TODO: COpiar enunciado
    Calcule la solución del siguiente problema mixto para la ecuación de ondas
    \begin{equation*}
        \left\{\begin{array}{ll}
                u_{tt} - 4u_{xx} = 0 , & \text{en\ } \left[0,\infty\right[\times [0,1] \\
                    u(0,x) = \varphi(x), &\text{en\ } [0,1] \\
                    u_t(0,x) = \psi(x), &\text{en\ } [0,1] \\
                    u(t,0) = u(t,1) = 0, &\text{en\ } \left[0,\infty\right[
        \end{array}\right.
    \end{equation*}
    en los siguientes casos:
    \begin{enumerate}[label=\alph*)]
        \item $\varphi(x) = \sen(\pi x) + \sen(2\pi x), \psi(x) = -\sen(\pi x) + \sen(2\pi x)$.
        \item $\varphi(x) = 0, \psi(x) = x(1-x)$.
    \end{enumerate}~\\

    \noindent
    Planteamos la ecuación de ondas:
    \begin{equation*}
        \left\{\begin{array}{ll}
                u_{tt} - 4u_{xx} = 0, &\text{en\ } \left[0,\infty\right[\times [0,1] \\
                    u(t,x=0) = u(t,x=1)=0 \qquad &\text{para\ } t=0
        \end{array}\right.
    \end{equation*}
    Buscamos primero soluciones en variables separadas:
    \begin{equation*}
        u(t,x) = T(t)X(x)
    \end{equation*}
    Resulta:
    \begin{equation*}
        T''X - 4TX'' = 0
    \end{equation*}
    ahora, tenemos dos opciones:
    \begin{itemize}
        \item Bien $T''X = 4TX''$.
        \item Bien $\frac{T''}{4T}(t) = \frac{X''}{X}(x)$, de donde cada una debe ser igual a una constante, que llamaremos $-\lm$ y estará por determinar.
    \end{itemize}
    Pasamos ahora a estudiar:
    \begin{equation*}
        \left\{\begin{array}{l}
            X'' + \lm X = 0 \qquad \text{en\ } [0,1] \\
            X(0) = 0  \\
            X(1) = 0 
        \end{array}\right.
    \end{equation*}
    Que es un problema de tipo Sturm-Liouville, con solución general para $\lm>0$:
    \begin{equation*}
        X(x) = c_1\cos(\sqrt{\lm}x) + c_2\cos(\sqrt{\lm}x), \qquad c_1,c_2\in \mathbb{R}
    \end{equation*}
    Imponiendo $X(0) = 0 = X(1)$ deducimos que $c_1 = 0$ y o bien $c_2=0$ o bien $\sen\left(\sqrt{\lm}\right)=1$, por lo que consideramos el conjunto discreto de $\lm$ que nos dan información no trivial:
    \begin{equation*}
        \lm_n = {(n\pi)}^{2}, \qquad n\in \mathbb{N}
    \end{equation*}
    Así, obtenemos soluciones $X_n(x) = C^{(n)}\sen(n\pi x)$, para $C^{(n)}\in \mathbb{R}$. Resolvemos ahora $T_n'' + 4\lm_nT_n = 0$ para cada $n\in \mathbb{N}$, obteniendo que:
    \begin{equation*}
        T_n(t) = a^{(n)}\cos\left(2\sqrt{\lm_n}t\right) + b^{(n)}\sen\left(2\sqrt{\lm_n}t\right),\qquad a^{(n)},b^{(n)}\in \mathbb{R}
    \end{equation*}
    Obtenemos las siguientes soluciones (juntando constantes):
    \begin{equation*}
        u_n(t,x) = T_n(t)X_n(t) = \left(a_n \cos(2n\pi t) + b_n\sen(2n\pi t)\right)\sen(n\pi x), \qquad n\in \mathbb{N},\quad a_n,b_n\in \mathbb{R}
    \end{equation*}
    Resolvemos ahora cada apartado:
    \begin{enumerate}[label=\alph*)]
        \item Para $\varphi(x) = \sen(\pi x)+\sen(2\pi x)$, $\psi(x) = -\sen(\pi x)+\sen(2\pi x)$.

            Vemos que los primeros sumandos de $\varphi(x),\psi(x)$ corresponden a $\lm_1$ y los segundos a $\lm_2$. Una opción es separar las frecuencias, resolver y usar que la ecuación es lineal:

            Así, para $\lm_1$ tenemos:
            \begin{align*}
                u_n(t=0,x) &= a_n\sen(\pi x) \\
                \frac{\partial u_n}{\partial t}(t,x) &= (-2n\pi a_n\sen(2n\pi t) + 2n\pi b_n\cos(2n\pi t)) \sen(n\pi x) \\
                \frac{\partial u_n}{\partial t}(t=0,x) &=  2n\pi b_n  \sen(n\pi x)
            \end{align*}
            Así, para $n=1$ y $n=2$ despejamos los valores de $a_n$ y $b_n$ ($n=1,2$), obteniendo:
            \begin{gather*}
                a_1 = 1, \quad 2\pi b_1 = -1 \quad\Longrightarrow\quad b_1 = \frac{-1}{2\pi} \\
                a_2 = 1, \quad 4\pi b_2 = 1 \quad\Longrightarrow\quad b_2 = \frac{1}{4\pi}
            \end{gather*}
            Juntándolo, la solución es:
            \begin{equation*}
                u(t,x) = \left(\cos(2\pi t) - \frac{1}{2\pi}\sen(2\pi t)\right)\sen(\pi x) + \left(\cos(4\pi t) + \frac{1}{4\pi}\sen(4\pi t)\right)\sen(2\pi x)
            \end{equation*}
        \item $\varphi(x) = 0$, $\psi(x) = x(1-x)$.

            En este caso no tenemos una cantidad finita de frecuencias, por lo que tenemos que tratar de descomponerlo en suma de frecuencias. Haciendo el desarrollo en serie de Fourier de $x(x-1)$ se obtendría:
            \begin{equation}\label{eq:ejm_psi}
                \psi(x) = x(x-1) = \sum_{n=1}^{\infty} \frac{8}{{((2n-1)\pi)}^{3}}\sen((2n-1)\pi x)
            \end{equation}
            En primer lugar, buscamos poner $\varphi(x)=0$ como combinación de funciones del tipo $u_n(t=0,x) = a_n\sen(\pi x)$:
            \begin{equation*}
                \varphi(x) = 0 = \sum_{n=1}^{\infty}a_n \sen(n\pi x)
            \end{equation*}
            se cumple tomando $a_n = 0 \quad \forall n\in \mathbb{N}$. Nos queda ahora determinar $b_n$, tratando de expresar los coeficientes de~\eqref{eq:ejm_psi} como $2n\pi b_n\sen(n\pi x)$. Para ello, es sencillo si primero reescribimos~\eqref{eq:ejm_psi}:
            \begin{equation*}
                \psi(x) = \sum_{n=1}^{\infty} \frac{8}{{((2n-1)\pi)}^{3}}\sen((2n-1)\pi x) = \sum_{n\geq 1 \text{\ impar}} \frac{8}{{(n\pi)}^{3}}\sen(n\pi x)
            \end{equation*}
            Por lo que igualando coeficientes:
            \begin{equation*}
                \left\{\begin{array}{ll}
                     \frac{8}{{(n\pi)}^{3}} = 2n\pi b_n & \text{si\ } n \text{\ impar} \\
                     0 = 2n\pi b_n & \text{si\ } n \text{\ par} 
                \end{array}\right. 
            \end{equation*}
            Así, tomaremos $b_n = 0$ para $n$ par y $b_n = \frac{4}{{(n\pi)}^{4}}$ si $n$ impar. En definitiva y en vista de la fórmula de $u_n(t,x)$, la solución para el apartado $b)$ es:
            \begin{equation*}
                \boxed{u(t,x) = \sum_{n\geq 1 \text{\ impar}} \frac{4}{{(n\pi)}^{4}} \sen(2n\pi t)\sen(n\pi x)}
            \end{equation*}
            Ahora, ¿cómo hemos obtenido la identidad~\eqref{eq:ejm_psi}? Revisando los apuntes de series de Fourier para la ecuación de ondas, para una función $f\in C([0,L])$ con $f(0) = 0 = f(L)$ podemos escribir:
            \begin{equation*}
                f(x) = \sum_{n=1}^{\infty} B_n \sen\left(\frac{n\pi x}{L}\right), \qquad B_n = \frac{2}{L}\int_{0}^{L} f(x)\sen\left(\frac{n\pi x}{L}\right)~dx 
            \end{equation*}
            En este caso particular:
            \begin{align*}
                B_n &= 2\int_{0}^{1} x(1-x)\sen(n\pi x)~dx  \\
                    &= 2\int_{0}^{1} x\sen(n\pi x)~dx  - 2\int_{0}^{1} x^2\sen(n\pi x)~dx 
            \end{align*}
            Y después de un rato se obtienen los coeficientes de la identidad~\eqref{eq:ejm_psi}.
    \end{enumerate}
\end{ejemplo}

\begin{ejemplo} % // TODO: Ejercicio 7
    Calcule la solución del siguiente problema mixto para la ecuación de ondas
    \begin{equation*}
        \left\{\begin{array}{ll}
                u_{tt} - 4u_{xx} = 0  & \text{en\ } \left[0,\infty\right[\times[0,1] \\
                    u(0,x) = \varphi(x) &\text{en\ } [0,1] \\
                    u_t(0,x) = \psi(x) &\text{en\ } [0,1] \\
                    u_x(t,0) = u_x(t,1) = 0 & \text{en\ } \left[0,\infty\right[
        \end{array}\right.
    \end{equation*}
    en los siguientes casos:
    \begin{enumerate}[label=\alph*)]
        \item $\varphi(x) = 1+\cos(2\pi x)$, $\psi(x) = 1-\cos(\pi x)$.
        \item $\varphi(x) = \cos(2\pi x)$, $\psi(x) = 2\pi x- \sen(2\pi x)$.
    \end{enumerate}~\\

    \noindent
    La estrategia es primero buscar soluciones en variables separadas de la ecuación:
    \begin{gather*}
        u_{tt} - 4 u_{xx} = 0 \quad \text{en\ } \left[0,\infty\right[\times [0,1] \\
            u_x(t,x=0) = 0 = u_x(t,x=1)
    \end{gather*}
    Tenemos:
    \begin{equation*}
        u(t,x) = T(t)X(x)
    \end{equation*}
    Al sustituir $u$ en la ecuación:
    \begin{equation*}
        T''X - 4TX'' = 0 \quad\Longrightarrow\quad \frac{T''}{4T}(t) = \frac{X''}{X}(x) = -\lm, \quad \lm \in \mathbb{R}
    \end{equation*}
    Nos topamos con el problema auxiliar ($u_x = T(t)X'(x)$):
    \begin{gather*}
        X'' + \lm X = 0 \quad \text{en\ } [0,1], \quad \lm \in \mathbb{R} \\
        X'(0) = 0 = X'(1)
    \end{gather*}
    Que es un problema de Sturm-Lliouville, por lo que sabemos ya que los $\lm$ que nos dan soluciones es un conjunto discreto. Buscamos $X\neq 0$  que nos resuelvan el problema para ciertos $\lm$, que serán las frecuencias fundamentales con las que trabajar. Hay que distinguir casos:
    \begin{itemize}
        \item Si $\lm < 0$ obtenemos combinación de exponenciales y al formar las dos condiciones obtenemos que ambos coeficientes son iguales a cero, con lo que queda la solución $X\equiv 0$, y esta solución ya la conocíamos.
        \item Si $\lm = 0$ resolvemos $X_0'' = 0$, luego $X_0(x) = ax+b$, para $a,b\in \mathbb{R}$ y obtenemos imponiendo las condiciones iniciales que $X_0(x) = b\in \mathbb{R}$. 

            Esto no pasaba en problemas tipo Dirichlet, pero ahora estamos con condiciones tipo Neumann.
        \item Para $\lm > 0$ tenemos que:
            \begin{equation*}
               X(x) = a\cos\left(\sqrt{\lm}x\right)  + b\sen\left(\sqrt{\lm}x\right), \qquad a,b\in \mathbb{R}
            \end{equation*}
            Calculamos $X'$ para imponer las condiciones:
            \begin{equation*}
                X'(x) = -a\sqrt{\lm}\sen\left(\sqrt{\lm}x\right) + b\sqrt{\lm}\cos\left(\sqrt{\lm}x\right)
            \end{equation*}
            Imponiendo las condiciones:
            \begin{align*}
                0 = X'(0) = b\sqrt{\lm} &\quad\Longrightarrow\quad b = 0 \\
                0 = X'(1) = -a\sqrt{\lm}\sen\left(\sqrt{\lm}\right) &\quad\Longrightarrow\quad \left\{\begin{array}{l}
                    \text{bien\ } a = 0 \text{\ luego\ } X\equiv 0 \text{,\ ya discutido}\\
                    \text{bien\ } \sen(\sqrt{\lm}) = 0 
                \end{array}\right.
            \end{align*}
    \end{itemize}
    Como necesitamos $\sen(\sqrt{\lm}) = 0$, definimos $\lm_n = {(n\pi)}^{2}$ para $n\geq 0$. Así:
    \begin{equation*}
        X_n(x) = a_n\cos(\sqrt{\lm_n}x), \qquad a_n \in \mathbb{R}
    \end{equation*}
    Tendremos así soluciones de la forma:
    \begin{equation*}
        u_n(t,x) = T_n(t)X_n(x), \qquad n\geq 0
    \end{equation*}
    donde $T_n''(t) + 4\lm_n T_n(t) = 0$.\\

    \noindent
    Para $n=0$ tenemos que $T_0(t) = a_0t + b_0,\quad a_0,b_0\in \mathbb{R}$. En otro caso:
    \begin{equation*}
        T_n(t) = a_n\cos\left(2\sqrt{\lm_n}t\right) + b_n\sen\left(2\sqrt{\lm_n}t\right), \quad a_n,b_n\in \mathbb{R}
    \end{equation*}
    Juntándolo con $X_n$ tenemos que (para otras constantes libres $a_n,b_n$):
    \begin{align*}
        u_0(t,x) &= a_0 + b_0 t \\
        u_n(t,x) &= (a_n\cos(2n\pi t) + b_n\sen(2n\pi t))\cos(n\pi x), \quad n \geq 1
    \end{align*}
    Así, la solución más general posible (puesto que la ecuación es lineal) es:
    \begin{equation*}
        u(t,x) = a_0 + b_0 t + \sum_{n=1}^{\infty} \left[(a_n\cos(2n\pi t) + b_n\sen(2n\pi t))\cos(n\pi x)\right]
    \end{equation*}
    En vista de exigir las condiciones para $\varphi(x)$ y $\psi(x)$:
    \begin{align*}
        \varphi(x) = u(t=0,x) &= a_0 + \sum_{n=1}^{\infty} a_n\cos(n\pi x) \\
        u_t(t,x) &= b_0 + \sum_{n=1}^{\infty} [(-a_n2n\pi \sen(2n\pi t) + b_n2n\pi\cos(2n\pi t))\cos(n\pi x)] \\
        \psi(x) = u_t(t=0,x) &= b_0 + \sum_{n=1}^{\infty} 2n\pi b_n \cos(n\pi x)
    \end{align*}
    Ahora, hay que escribir las condiciones iniciales $\varphi(x)$ y $\psi(x)$ en base de cosenos para frecuencias naturales y comparar término a término:
    \begin{enumerate}[label=\alph*)]
        \item Para $\varphi(x) = 1+\cos(2\pi x)$, $\psi(x) = 1-\cos(\pi x)$.

            Vemos que tomando:
            \begin{gather*}
                a_0 = 1, \quad b_0 = 1  \\
                a_1 = 0, \quad b_1 = \frac{-1}{2\pi} \\
                a_2 = 1, \quad b_2 = 0 \\
                a_n = 0 = b_n, \quad n \geq 3
            \end{gather*}
            obtenemos la solución para este caso:
            \begin{equation*}
                u(t,x) = 1 + t + \frac{-1}{2\pi}\sen(2\pi t)\cos(\pi x) + \cos(4\pi t)\cos(2\pi x) 
            \end{equation*}
        \item Para $\varphi(x) = \cos(2\pi x)$, $\psi(x) = 2\pi x- \sen(2\pi x)$.

            Para la condición de $\varphi(x)$ tomamos $a_2 = 1$, $a_n = 0 \quad n\neq 2$.

            Tenemos ahora la dificultad de que el seno no se puede expresar como suma finita en base de cosenos, obtendremos una suma infinita. Realizando el desarrollo en base de cosenos obtenemos (tras varias integrales):
            \begin{equation*}
                B_0 = 2\pi, \quad B_n = \left\{\begin{array}{ll}
                    0 & \text{si\ } n \text{\ es par} \\
                     \frac{-32}{\pi n^2(4-n^2)}& \text{si\ } n \text{\ es impar}
                \end{array}\right. 
            \end{equation*}
            de donde:
            \begin{equation*}
                \psi(x) = \frac{B_0}{2} + \sum_{n=1}^{\infty} B_n \cos(n\pi x)
            \end{equation*}
            por lo que tenemos:
            \begin{equation*}
                2n\pi b_n = B_n, \quad n\geq 1,\qquad  b_0 = \frac{B_0}{2}
            \end{equation*}
            despejando los valores de $b_n$ obtenemos la solución para este caso.
    \end{enumerate}
\end{ejemplo}
