\chapter{Derivación en sentido débil}
\noindent
Como motivacion para estudiar la derivada débil, si tenemos la ecuación de ondas:
\begin{equation*}
    \left\{\begin{array}{ll}
            u_{tt} - c^2 u_{xx} = 0 , &\text{en\ } \left[0,\infty\right[\times \mathbb{R} \\
                u(t=0,x) = \varphi(x), &\text{en\ } \mathbb{R} \\
                u_t(t=0,x) = 0 , & \text{en\ } \mathbb{R}
    \end{array}\right.
\end{equation*}
La solución se escribe así:
\begin{equation*}
    u(t,x) = \frac{\varphi(x+ct)+\varphi(x-ct)}{2}
\end{equation*}
Y esta puede presentar ``picos'' en su gráfica, es decir, puntos aislados en los que no es derivable.

\section{Definición de derivada débil}
\noindent
Sea $\left]a,b\right[$ un intervalo acotado, sabemos que para $f\in C^1(\left]a,b\right[)$ tenemos:
\begin{equation}\label{eq:int_partes}
    \int_{a}^{b} f'(x)\varphi(x)~dx = -\int_{a}^{b} f(x)\varphi'(x)~dx 
\end{equation}
Para $\varphi \in C^1_c\left(\left]a,b\right[\right)$, luego $\varphi(a) = 0 = \varphi(b)$, ya que $\sop \varphi \subsetneq \left]a,b\right[$ por ser $\sop\varphi$ compacto\footnote{Recordamos que $\sop\varphi = \overline{\{x\in \left]a,b\right[ : \varphi(x)\neq 0\}}$.}. También es claro que $\sop\varphi' \subset \sop\varphi$.\\

\noindent
Para que la fórmula~\eqref{eq:int_partes} tenga sentido, necesitamos exigir:
\begin{equation*}
    f'\in L^1(\sop\varphi), \qquad f\in L^1(\sop\varphi')
\end{equation*}
Así, bastaría con $f,f' \in L^1_{\text{loc}}(\left]a,b\right[)$, pues tanto $\sop\varphi$ como $\sop\varphi'$ son compactos.

\begin{definicion}
    Sea $f\in L^1_\text{loc}(\left]a,b\right[)$, diremos que admite derivada débil si existe $g\in L^1_\text{loc}(\left]a,b\right[)$ tal que:
    \begin{equation*}
        \int_{a}^{b} f(x)\varphi'(x)~dx  = -\int_{a}^{b} g(x)\varphi(x)~dx  \qquad \forall \varphi \in C_c^1(\left]a,b\right[)
    \end{equation*}
    En cuyo caso, diremos que $g$ es la derivada débil de $f$, notado por $g = f'$.
\end{definicion}

\noindent
En primer lugar tenemos que ver que la derivada débil está bien definida, es decir, que supuesto que $f\in L^1_\text{loc}(\left]a,b\right[)$ tiene derivada débil, esta ha de ser única, salvo un conjunto de medida nula:
\begin{proof}
    Para $f\in L^1_\text{loc}(\left]a,b\right[)$, por reducción al absurdo, suponemos que tenemos $g_1,g_2\in L^1_\text{loc}(\left]a,b\right[)$ tales que:
    \begin{multline*}
        \int_{a}^{b} f(x)\varphi'(x)~dx  = -\int_{a}^{b} g_1(x)\varphi(x)~dx, \quad 
        \int_{a}^{b} f(x)\varphi'(x)~dx  = -\int_{a}^{b} g_2(x)\varphi(x)~dx  \\ \forall \varphi \in C_c^1(\left]a,b\right[) 
    \end{multline*}
    De aquí deducimos que:
    \begin{equation*}
        \int_{a}^{b} (g_1(x)-g_2(x))\varphi(x)~dx = 0 \qquad \varphi \in C_c^1(\left]a,b\right[)
    \end{equation*}
    Por una versión del Lema Fundamental del Cálculo de Variaciones concluimos que $g_1-g_2\equiv 0$ casi por doquier, por lo que $g_1 = g_2$ casi por doquier.
\end{proof}

\begin{observacion}
    Observamos que si $f\in C^1(\left]a,b\right[)$, entonces $f$ es débilmente derivable, y su derivada débil coincide con su derivada habitual, salvo un conjunto de medida nula.\\

    \noindent
    Normalmente diremos que la derivada débil coincide con la habitual, puesto que lo que sí tenemos es que la derivada débil pertenece a la misma clase de equivalencia que la derivada usual en $L^1_\text{loc}(\left]a,b\right[)$, y de esa clase podemos tomar la derivada usual como representante canónico de cualquier elemento de la clase. Así, a partir de ahora omitiremos el ``salvo un conjunto de medida nula'', entiendo que trabajamos con funciones en $L^1_\text{loc}(\left]a,b\right[)$.
\end{observacion}

\begin{prop}
    Si $f_1,f_2\in L^1_\text{loc}(\left]a,b\right[)$ son débilmente derivables y $\lm_1,\lm_2\in \mathbb{R}$, entonces:
    \begin{equation*}
        (\lm_1f_2 + \lm_2f_2)' = \lm_1f_1' + \lm_2f_2'
    \end{equation*}
    \begin{proof}
        Se deduce directamente de la linealidad de la integral, puesto que la derivada débil se define a partir de una función que verifica una igualdad integral.
    \end{proof}
\end{prop}

\begin{observacion}
    La derivada en sentido débil no siempre satisface la regla del producto ni la regla de la cadena, puesto que puede que el producto o la composición de funciones en $L^1_\text{loc}(\left]a,b\right[)$ no esté en $L^1_\text{loc}(\left]a,b\right[)$. Sin embargo, en situaciones buenas sí que se cumplen las mismas reglas.
\end{observacion}

\begin{ejemplo}
    Vamos a calcular la derivada débil de la función $f:\left]-1,1\right[\to \mathbb{R}$ valor absoluto, $f(x) = |x|$. Guiándonos por su definición, para $\varphi \in C_c^1(\left]-1,1\right[)$ escribimos:
    \begin{align*}
        \int_{-1}^{1} |x|\varphi'(x)~dx  &= -\int_{-1}^{0} x\varphi'(x)~dx + \int_{0}^{1} x\varphi(x)~dx  \\
                                         &\AstIg \int_{-1}^{0} \varphi(x)~dx - \left[x\varphi(x)\right]_{-1}^0 - \int_{0}^{1} \varphi(x)~dx + [x\varphi(x)]_0^1 \\
                                         &= \int_{-1}^{0} \varphi(x)~dx -\int_{0}^{1} \varphi(x)~dx  -\cancelto{0}{\varphi(-1)} + \cancelto{0}{\varphi(1)} = -\int_{-1}^{1} sgn(x)\varphi(x)~dx 
    \end{align*}
    donde en $(\ast)$ hemos usado integración por partes, algo que no podíamos haber aplicado a $|x|\varphi'(x)$ por no ser derivable en $0$, pero sí que lo podemos aplicar en los intervalos $\left]-1,0\right[$ y $\left]0,1\right[$, como lo hemos hecho. Hemos considerado:
    \begin{equation*}
        sgn(x) = \left\{\begin{array}{ll}
            -1 & \text{si\ } x<0 \\
            0 & \text{si\ } x=0 \\
            1 & \text{si\ } x>0
        \end{array}\right. 
    \end{equation*}
    Aunque en realidad podríamos haber considerado cualquier función igual a esta casi por doquier.\\

    \noindent
    Si hubiéramos sido algo más espabilados, podríamos haber dicho que $f\big|_{\left]-1,0\right[}$ y $f\big|_{\left]0,1\right[}$ son derivables y conocemos sus derivadas, con lo que podríamos haber obtenido la función signo (o cualquiera igual casi por doquier) como derivada débil directamente.
\end{ejemplo}

\noindent
A partir del ejemplo anterior, no es difícil darse cuenta de que cualquier función con un número de ``picos'' contenido en un conjunto de medida nula es débilmente derivable.

\begin{ejemplo}
    Tratemos de calcular la derivada débil de $sgn(x)$ en $\left]-1,1\right[$ según la definición hecha en el ejemplo anterior:
    \begin{align*}
        \int_{-1}^{1} sgn(x)\varphi'(x)~dx &= \int_{-1}^{0} -\varphi'(x)~dx + \int_{0}^{1} \varphi'(x)~dx \\
                                           &= \left[-\varphi(0) + \cancelto{0}{\varphi(-1)}\right] + \left[\cancelto{0}{\varphi(1)} - \varphi(0)\right] = -2\varphi(0) \quad \forall \varphi \in C_c^1(\left]-1,1\right[)
    \end{align*}
    Ahora, buscamos $g$ para tener:
    \begin{equation*}
        -2\varphi(0) = -\int_{-1}^{1} g(x)\varphi(x)~dx 
    \end{equation*}
    Sin embargo, no existe ninguna función $g\in L^1_\text{loc}(\left]-1,1\right[)$ que lo verifique, por lo que $sgn(x)$ no es débilmente derivable en $\left]-1,1\right[$.\\

    \noindent
    Veamos que no existe dicha $g\in L^1_\text{loc}(\left]a,b\right[)$, por reducción al absurdo, supongamos que existe $g\in L^1_\text{loc}(\left]a,b\right[)$ con:
    \begin{equation*}
        2\varphi(0) = \int_{-1}^{1} g(x)\varphi(x)~dx  \qquad \forall \varphi \in C_c^1(\left]-1,1\right[)
    \end{equation*}
    Así, tenemos que:
    \begin{align*}
        \int_{-1}^{0} g(x)\varphi(x)~dx &= \int_{-1}^{1} g(x)\varphi(x)~dx = -2\varphi(0) = 0 \qquad \forall \varphi \in C_c^1(\left]-1,0\right[) \\
        \int_{0}^{1} g(x)\varphi(x)~dx &= \int_{-1}^{1} g(x)\varphi(x)~dx = -2\varphi(0) = 0 \qquad \forall \varphi \in C_c^1(\left]0,1\right[)
    \end{align*}
    Usando el Lema Fundamental del Cálculo de Variaciones, de la primera línea concluimos que $g\equiv 0$ casi por doquier en $\left]0,1\right[$ y de la segunda que $g\equiv 0$ casi por doquier en $\left]-1,0\right[$, de donde $g\equiv 0$ casi por doquier en $\left]-1,1\right[$.\\

    \noindent
    Si tomamos ahora $\varphi \in C_c^1(\left]a,b\right[)$ con $\varphi(0) = 1$ obtenemos que:
    \begin{equation*}
        0 = \int_{-1}^{1} g(x)\varphi(x)~dx  = 2\varphi(0) = 2
    \end{equation*}
    Hemos llegado a \underline{contradicción}.
\end{ejemplo}

\noindent
Introducimos ahora la función de Heaviside (o ``función escalón'', \textit{step function}) en $L^1_\text{loc}(\left]-1,1\right[)$:
\begin{equation*}
    H(x) = \left\{\begin{array}{ll}
        1 & \text{si\ } x>0 \\
        0 & \text{si\ } x<0
    \end{array}\right. 
\end{equation*}
Obtendremos que:
\begin{equation}\label{eq:heaviside}
    \int_{-1}^{1} H(x)\varphi'(x)~dx = -\varphi(0) \qquad \forall \varphi \in C_c^1(\left]-1,1\right[)
\end{equation}
Por lo que de manera análoga al ejemplo anterior, llegaremos a que $H$ no es débilmente derivable.\\

\noindent
Sin embargo, existe una entidad matemática (no es una función), la delta de Dirac, $\delta(x)$, tal que formalmente:
\begin{equation*}
    \int_{-1}^{1} \delta(x)\varphi(x)~dx = \varphi(0) \qquad \forall \varphi \in C(\left]a,b\right[)
\end{equation*}
Para interpretar rigurosamente esta formalidad necesitamos:
\begin{itemize}
    \item Bien interpretar que $\delta$ es un funcional lineal sobre $\varphi$: $L_\delta(\varphi) = \varphi(0)$.

        Que además es continuo, pues un cálculo formal nos dice:
        \begin{equation*}
            |L_\delta(\varphi_1) - L_\delta(\varphi_2)| = |\varphi_1(0) - \varphi_2(0)| \leq \|\varphi_1-\varphi_2\|_\infty
        \end{equation*}
        Y habrá que fijar previamente hipótesis para saber dónde estamos trabajando y que el cálculo funcione bien, obteniendo por ejemplo que $L_\delta:C_c(\mathbb{R})\to\mathbb{R}$ es lineal y continua, luego $L_\delta \in C_c(\mathbb{R})^\ast$.

        Se tiene que $C_c(\mathbb{R})^\ast$ es el espacio de ``medidas sobre $\mathbb{R}$'', como la medida de Lebesgue.

        Volviendo a la ecuación~\eqref{eq:heaviside}, asociado a $H$ podemos deducir el funcional
        \begin{equation*}
            L_H(\varphi) = \int_\mathbb{R} H(x)\varphi(x)~dx
        \end{equation*}
        que es lineal y continuo, puesto que:
        \begin{equation*}
            |L_H(\varphi_1) - L_H(\varphi_2)| \leq \int_{\mathbb{R}^+}|\varphi_1(x) - \varphi_2(x)|~dx \stackrel{(\ast)}{\leq} C\|\varphi_1-\varphi_2\|_\infty
        \end{equation*}
        Pero el paso $(\ast)$ es delicado, aunque no hay ningún problema siempre y cuando tengamos $\varphi_1,\varphi_2\in C_c(\mathbb{R})$ podemos escribir:
        \begin{align*}
            |L_H(\varphi_1) - L_H(\varphi_2)| &\leq \int_{\mathbb{R}^+}|\varphi_1(x) - \varphi_2(x)|~dx \\ 
            &\leq \int_{\sop\varphi_1\cup\sop\varphi_2}|\varphi_1-\varphi_2|~dx \leq |\sop\varphi_1\cup\sop\varphi_2|\cdot  \|\varphi_1-\varphi_2\|_\infty
        \end{align*}
        Por lo que si definimos $L_H:C_c(-1,1)\to \mathbb{R}$ sí que obtenemos un funcional lineal y continuo.

        Volviendo ahora a la fórmula~\eqref{eq:heaviside}, podemos reescribirla como:
        \begin{equation*}
            L_H(\varphi') = -L_\delta(\varphi) \qquad \forall \varphi \in C_c(-1,1)
        \end{equation*}
        o con la notación que usábamos en Análisis Funcional:
        \begin{equation*}
            \langle H,\varphi' \rangle = -\langle \delta,\varphi \rangle 
        \end{equation*}
        Conforme a la ``teoría de distribuciones'', está fórmula dice que $H' = \delta$.

        Así, la derivada débil no nos permite derivar funciones de salto, pero existe una teoría mucho más profunda en la que no vamos a entrar que sí lo permite.
    \item Bien cambiar la integral de Lebesgue por una integral con otra medida.

        Podemos aproximar la función $H(x)$ de la siguiente forma:
        \begin{figure}[H]
            \centering
        \begin{tikzpicture}
        \begin{axis}[
            axis lines=middle,
            xlabel={$t$},
            ylabel={$y$},
            xmin=-1.5, xmax=3,
            ymin=-0.2, ymax=1.2,
            samples=100,
            width=10cm,
            height=6cm
        ]

        % g_2(t)
        \addplot[red, thick, domain=-1.5:0] {0};
        \addplot[red, thick, domain=0:0.5] {2*x};
        \addplot[red, thick, domain=0.5:3] {1};

        \fill[red] (0.85,0.5) node[] {$H_\varepsilon(x)$};
        \fill[red] (0.5,0) circle(2pt) node[above right] {$\varepsilon$};
        \draw[red, dashed] (0.5,0) -- (0.5,1);

        \end{axis}
        \end{tikzpicture}
        \end{figure}


        \noindent
        Llamamos a estas funciones $H_\varepsilon(x)$, para $\varepsilon<<1$, con lo que el Teorema de la convergencia dominada nos dice que:
        \begin{equation*}
            \int_{-1}^{1} H(x)\varphi'(x)~dx  = \lim_{\varepsilon\to0}\int_{-1}^{1} H_\varepsilon(x)\varphi'(x)~dx  
        \end{equation*}
        Pero observemos que:
        \begin{equation*}
            \int_{-1}^{1} H_\varepsilon(x)\varphi'(x)~dx = -\int_{0}^{\varepsilon} \frac{1}{\varepsilon}\varphi(x)~dx  \stackrel{(\varepsilon\to0)}{\longrightarrow} -\varphi(0)
        \end{equation*}
        Tras un argumento intuitivo, podemos pensar en $\delta$ como en una masa unidad concentrada en el origen.

    Podemos pensar como una medida, dada por:
    \begin{equation*}
        \delta(E) = \left\{\begin{array}{ll}
            1 & \text{si\ } 0\in E \\
            0 & \text{si\ } 0\notin E
        \end{array}\right. , \qquad E\subset\mathbb{R}
    \end{equation*}
    En dicho caso, es habitual escribir:
    \begin{equation*}
        \int_{-1}^{1} \varphi(x)~d\delta(x) = \int_{-1}^{1} \delta(x)\varphi(x)~dx  
    \end{equation*}
    Para referirse a esta medida, se suele user $\delta_0$ o $\delta(x)$.

    Para una medida análoga pero cambiando $0$ por $x_0$ se escribe $\delta_{x_0}$ o $\delta(x-x_0)$. Se tiene así formalmente que:
    \begin{equation*}
        \int_{\mathbb{R}^d}\delta(x-x_0)\varphi(x)~dx = \varphi(x_0)
    \end{equation*}
    para $\varphi$ continua.
\end{itemize}

\begin{ejemplo}
    Todo lo anteriormente comentado se usa con mucha soluta en ingeniería, por ejemplo, para calcualr la derivada de: 
    \begin{equation*}
        f(x) = \left\{\begin{array}{ll}
            1 & \text{si\ } x<1 \\
            2 & \text{si\ } 1<x<3 \\
            0 & \text{si\ } x>3
        \end{array}\right. 
    \end{equation*}
    Algebraicamente se hace:
    \begin{equation*}
        f'(x) = \delta(x-1) -2\delta(x-3)
    \end{equation*}
    Aunque debemos tener en cuenta que esto es en el sentido de la teoría de distribuciones, no como derivada débil.
\end{ejemplo}

\section{Integración por partes en dos dimensiones}
\noindent
Aprendámos rápido y formalmente cómo se integra por partes en varias variables. Tendremos $u,v:[0,1]\times[0,1]\to \mathbb{R}$, escribiremos $u = u(x_1,x_2)$, $x = (x_1,x_2)$ y $\Omega = [0,1]\times [0,1]$. Se cumple que:
\begin{multline*}
    \int_\Omega u(x_1,x_2)\frac{\partial v}{\partial x_1}(x_1,x_2)~dx \AstIg \int_{0}^{1} \left(\int_{0}^{1} u(x_1,x_2)\frac{\partial v}{\partial x_1}(x_1,x_2)~dx_1\right) ~dx_2 \\
                                                                      = \int_{0}^{1} \left(-\int_{0}^{1} \frac{\partial u}{\partial x_1}(x_1,x_2)v(x_1,x_2)~dx_1 + [u(\cdot ,x_2)v(\cdot ,x_2)]_0^1\right)~dx_2  \\
                                                                      = -\int_\Omega \frac{\partial u}{\partial x_1}(x_1,x_2) v(x_1,x_2)~dx + \int_{0}^{1} u(1,x_2)v(1,x_2)~dx_2 -\int_{0}^{1}u(0,x_2)v(0,x_2)~dx_2 
\end{multline*}
donde en $(\ast)$ hemos aplicado el Teorema de Fubini. La fórmula general se conoce como \textbf{Fórmula de Green}: Dado $\Omega\subset \mathbb{R}^d$ con frontera regular, dadas $f,g$ regulares se tiene que:
\begin{equation*}
    \int_\Omega \frac{\partial f}{\partial x_i}g~dx = -\int_\Omega f\frac{\partial g}{\partial x_i}~dx + \int_{\partial\Omega} fgn\cdot e_i~d\sigma
\end{equation*}
donde $n$ es el vector normal exterior de módulo 1 a la curva $\partial \Omega$ y $e_i$ es el vector de la base canónica correspondiente a la coordenada $x_i$ de $x$. En el caso anterior, tenemos que $\Omega$ tiene una frontera que se separa en 4 segmentos, con normales $(0,1)$, $(1,0)$, $(0,-1)$ y $(-1,0)$. Como estamos haciendo parciales respecto $x_1$, solo nos quedamos con las primeras coordenadas de los vectores.\\

\noindent
En la práctica de esta asignatura, el segundo sumando será cero o bien será suficiente con estimar su tamaño, por lo que no será necesario calcularlo. Cuando $\Omega$ sea uan superficie tendremos que $\partial \Omega$ será una curva.\\

\noindent
Para las integrales sobre superficies, para no tener que tratar el concepto en toda su generalidad (lo cual es difícil), trabajaremos solo con superficies parametrizadas, es decir, superficies que podamos describir como la gráfica de una función de dos variables $S:O\to \mathbb{R}^3$, donde $O\subset \mathbb{R}^2$, $\Omega = Gr(S)$. Así, para $f:\Omega\to\mathbb{R}$ podemos considerar:
\begin{equation*}
    f\circ S:O\to \mathbb{R}
\end{equation*}
y sabemos integrar en $O$, puesto que es una integral doble. Además, la aplicación $S$ deforma las áreas, por lo que tenemos que considerar:
\begin{equation*}
    A(\Omega) = \int_O \left|\frac{\partial S}{\partial x_1}\land \frac{\partial S}{\partial x_2}\right|~dx_1dx_2
\end{equation*}
Así:
\begin{equation*}
    \int_\Omega f~d\sigma = \int_O (f\circ S)(x_1,x_2) \cdot \left|\frac{\partial S}{\partial x_1}\land \frac{\partial S}{\partial x_2}\right|~dx_1dx_2
\end{equation*}

\begin{teo}[de la divergencia]
    Dados $\Omega\subset\mathbb{R}^d$ con frontera regular y $F:\Omega\to\mathbb{R}^d$ regular con:
    \begin{equation*}
        \int_{\partial \Omega}F\cdot n~d\sigma = \int_\Omega div F~dx
    \end{equation*}
    siendo $n$ el normal exterior a $\partial \Omega$.
    \begin{proof}
        Tenemos que:
        \begin{equation*}
            \int_\Omega div F~dx = \int_\Omega 1\cdot \frac{\partial F}{\partial x_1} + \ldots + 1\cdot \frac{\partial F}{\partial x_d}~dx
        \end{equation*}
        Ahora, si denotamos por $n = (n_1,\ldots,n_d)$ al vector normal exterior unitario a $\Omega$:
        \begin{equation*}
            \int_\Omega 1\cdot \frac{\partial F_i}{\partial x_i}~dx =-\int_\Omega \cancelto{0}{\frac{\partial 1}{\partial x_1}}F_i~dx + \int_{\partial \Omega}F_in_i~d\sigma \qquad \forall i \in \{1,\ldots,d\}
        \end{equation*}
        Juntando todo esto:
        \begin{equation*}
            \int_\Omega div F~dx = \sum_{i=1}^{d}\int_{\partial \Omega}F_in_i~d\sigma = \int_{\partial \Omega}F\cdot n~d\sigma
        \end{equation*}
    \end{proof}
\end{teo}

\section{Derivada débil en varias variables}
\noindent
Si $f\in C^1(\Omega)$ para $\Omega\subset \mathbb{R}^d$ abierto, entonces se tiene la fórmula de Green:
\begin{equation*}
    \int_\Omega \frac{\partial f}{\partial x_i}\varphi~dx = -\int_\Omega f\frac{\partial \varphi}{\partial x_i}~dx \qquad \forall \varphi \in C^1_c(\Omega)
\end{equation*}
donde no tenemos términos de frontera por tener $\varphi \in C^1_c(\Omega)$, ya que $\varphi\big|_{\partial \Omega}\equiv 0$.\\

\noindent
Ahora, si $f\in C^3(\Omega)$, entonces tenemos que:
\begin{equation*}
    \int_\Omega \frac{\partial^3 f}{\partial x_1\partial x_2^2}\varphi~dx = -\int_\Omega f\frac{\partial^3\varphi}{\partial x_1\partial x_2^2}~dx\qquad \forall \varphi \in C_c^3(\Omega)
\end{equation*}
Donde tenemos un menos porque teníamos una cantidad impar de derivadas parciales.

\begin{notacion}
    Será usual denotar:
    \begin{equation*}
        D^{(1,2)} f = \frac{\partial^3 f}{\partial x_1\partial x_2^2}, \qquad D^{(2,0,1)}f = \frac{\partial^3 f}{\partial x_1^2\partial x_3}
    \end{equation*}
    Así como:
    \begin{equation*}
        |(a,b,c)| = a+b+c
    \end{equation*}
    Por lo que la $\alpha-$ésima derivada de $f$ para $\alpha = (\alpha_1,\ldots,\alpha_d)\in \mathbb{N}^d$ será:
    \begin{equation*}
        D^\alpha f =  \frac{\partial^{|\alpha|}f}{\partial x_1^{\alpha_1}\ldots \partial  x_d^{\alpha_d}}
    \end{equation*}
\end{notacion}

\noindent
Vista la fórmula de integración por partes para varias variables y de forma análoga a lo que hicimos para definir la derivada débil para una variable, definimos ahora:
\begin{definicion}
    Sean $\Omega\subset\mathbb{R}^d$ abierto, $f\in L^1_{\text{loc}}(\Omega)$ y $\alpha\in \mathbb{N}^d$, decimos que una aplicación $g\in L^1_{\text{loc}}(\Omega)$ es su $\alpha-$derivada débil si:
    \begin{equation*}
        \int_\Omega f(x)D^{\alpha}\varphi(x)~dx = {(-1)}^{|\alpha|}\int_\Omega g(x)\varphi(x)~dx\qquad \forall \varphi \in C_c^{|\alpha|}(\Omega)
    \end{equation*}
\end{definicion}

\begin{ejemplo}
    Veamos que $f(x)=|x|$ admite vector gradiente en sentido débil en $B(0,1)\subset \mathbb{R}^d$. Podríamos proceder a partir de la definición o bien dándonos cuenta de que el único ``problema'' (respecto a la definición usual de aplicación diferencial) es el origen. Elegimos la segunda, por lo que para $1>\varepsilon>0$ tenemos que $f\in C^\infty(B(0,1)\setminus B(0,\varepsilon))$, con gradiente:
    \begin{equation*}
        \nabla f(x) = \frac{x}{|x|}
    \end{equation*}
    Así, si hubiera gradiente en sentido débil, tendrá que coincidir con $\frac{x}{|x|}$, que vemos que claramente está en $L^1_{\text{loc}}(B(0,1))$. Fijado $i \in \{1,\ldots,d\}$, vemos que para $\varphi \in C^1_c(\Omega)$:
    \begin{equation*}
        \int_{B(0,1)}f(x)\frac{\partial \varphi}{\partial x_i}~dx \AstIg \lim_{\varepsilon\to0} \int_{B(0,1)\setminus B(0,\varepsilon)} f(x)\frac{\partial \varphi}{\partial x_i}~dx
    \end{equation*}
    donde en $(\ast)$ podemos usar el Teorema de la convergencia dominada\footnote{Este Teorema fija el dominio de integración y se mueve mediante una sucesión de funciones, pero podemos tomar $f_n$ como la función indicada multiplicada por $\chi_{B(0,1)\setminus B(0,\nicefrac{1}{n})}$ para poder mover el dominio.}, puesto que $f\frac{\partial \varphi}{\partial x_i}$ es acotado en $B(0,1)$. En este último conjunto podemos integrar por partes usando la fórmula de Green:
    \begin{align*}
        \int_{B(0,1)\setminus  B(0,\varepsilon)}f(x)\frac{\partial \varphi}{\partial x_i}~dx &= -\int_{B(0,1)\setminus B(0,\varepsilon)} \frac{\partial f}{\partial x_i}\varphi~dx + F(f,\varphi,B(0,1)\setminus B(0,\varepsilon)) \\
                                                                                             &= -\int_{B(0,1)\setminus B(0,\varepsilon)} \frac{x_i}{|x|}\varphi(x)~dx+ F(f,\varphi,B(0,1)\setminus B(0,\varepsilon)) 
    \end{align*}
    donde denotamos por $F(f,\varphi,B(0,1)\setminus B(0,\varepsilon))$ a los términos de frontera, que tiene una parte en la curva $S(0,1)$ y otra en $S(0,\varepsilon)$. Como $\varphi \in C_c^1(B(0,1))$, el término sobre $S(0,1)$ se anula, por lo que solo nos queda el término sobre $S(0,\varepsilon)$:
    \begin{equation*}
        F(f,\varphi,B(0,1)\setminus B(0,\varepsilon)) = \int_{S(0,\varepsilon)} f\varphi (n\cdot e_i) d\sigma = \int_{|x|=\varepsilon}|x|\varphi(x)(n\cdot e_i)~dx
    \end{equation*}
    donde $n$ es el vector normal exterior unitario en cada punto. Observemos que $n\cdot e_i$ es de módulo 1 por ser ambos de módulo 1. Como $\varphi$ es de clase 1, podemos escribir:
    \begin{equation*}
        \varphi(x) = \varphi(0) + \nabla\varphi(0)\cdot x + \ldots
    \end{equation*}
    Como $\varepsilon<<1$, podemos ``cambiar $\varphi(x)$ por $\varphi(0)$ sin equivocarnos demasiado'', con lo que:
    \begin{equation*}
        F(f,\varphi,B(0,1)\setminus B(0,\varepsilon)) \approx \int_{|x|=\varepsilon} \varepsilon\varphi(0)(n\cdot e_i)~dx
    \end{equation*}
    Ahora, usando que\footnote{Para $d=2$ tenemos que $\lm(S^1) = 2\pi r$, para $d=3$ tenemos $\lm(S^2) = 4\pi r^2$} $\lm(S(0,\varepsilon))\approx R^{d-1}$ siendo $d$ la dimensión, así como que también $|x|=\varepsilon \approx \varepsilon^{d-1}$, los términos de frontera son proporcionales a $\varepsilon^d\varphi(0)\to 0 \quad (\varepsilon\to0)$. En definitiva, tenemos que:
    \begin{align*}
        \int_{B(0,1)}|x|\frac{\partial \varphi}{\partial x_i}~dx &= \lim_{\varepsilon\to0}\int_{B(0,1)\setminus B(0,\varepsilon)}|x|\frac{\partial \varphi}{\partial x_i}~dx = \lim_{\varepsilon\to0} -\int_{B(0,1)\setminus B(0,\varepsilon)} \frac{x_i}{|x|}\varphi(x)~dx  \\ &= - \int_{B(0,1)}\frac{x_i}{|x|}\varphi(x)~dx
    \end{align*}
\end{ejemplo}

\begin{ejercicio}
    Si consideramos $f(x) = \frac{1}{|x|^\alpha}$ para $\alpha>0$ en $B(0,1)\subset\mathbb{R}^d$, entonces $f$ admite derivadas débiles de orden uno si $\alpha<d-1$ (la condición sobre $\alpha$ sale de exigir primero que $f\in L^1_{\text{loc}}(B(0,1)$ y luego que el candidato a derivada débil también lo sea). En cuyo caso:
    \begin{equation*}
        \nabla f(x) = -\alpha \frac{1}{|x|^{\alpha+1}} \frac{x}{|x|} = -\alpha \frac{x}{|x|^{\alpha+2}}
    \end{equation*}
\end{ejercicio}

\subsection{Motivación}
\noindent
El concepto de derivada débil lo hemos dado para introducirnos a las ecuaciones en derivadas parciales. Veamos un ejemplo:

\begin{ejemplo}
Consideremos la ecuación:
\begin{equation}\label{eq:ec_motivacion}
    \left\{\begin{array}{l}
        -u''(x) = f(x)  \\
        u(a) = 0 = u(b)
    \end{array}\right.
\end{equation}
en el dominio $I = \left]a,b\right[$, para $f:I\to \mathbb{R}$ dada.\\

\noindent
Si $u\in C^2(\left]a,b\right[)$ cumple la ecuación en todo punto, entonces:
\begin{equation*}
    -\int_{a}^{b} u''(x) \varphi(x)~dx  = \int_{a}^{b} f(x)\varphi(x)~dx  
\end{equation*}
para $\varphi(x)$ dada. La motivación de la derivada débil es que no sea necesario exigir que $u$ sea de clase $C^2(\left]a,b\right[)$ (salvo para escribir $u''$), por lo que buscamos otro concepto para escribir algo parecido a $u''$ que cumpla la ecuación integral sin ser necesariamente $u$ de clase $C^2(\left]a,b\right[)$.\\

\noindent
Forzando más esta situación, si $\varphi$ es suficientemente regular, por ejemplo $\varphi \in C^1(I)$, podemos integral formalmente por partes, por lo que solo aparecerá $u'$ en la expresión y ya solo bastará con que $u\in C^1(\left]a,b\right[)$. Así:
\begin{equation}\label{eq:ast_despreciar}
    \int_{a}^{b} u'(x)\varphi'(x)~dx  - \underbrace{[u'(x)\varphi(x)]_a^b}_{(\ast)} = \int_{a}^{b} f(x)\varphi(x)~dx 
\end{equation}
y queremos despreciar el término $(\ast)$.\\

\noindent
En resumen, si $u\in C^2(\left]a,b\right[)$ es solución del problema~\eqref{eq:ec_motivacion}, tendremos entonces que:
\begin{equation*}
    \int_{a}^{b} u'(x)\varphi'(x)~dx  = \int_{a}^{b} f(x)\varphi(x)~dx \qquad \forall \varphi \in C^1_0(\left]a,b\right[)
\end{equation*}
Así, para $u$ y $f$ con hipótesis suficientes para que estas integrales tengan sentido, un marco general para estudiar la ecuación integral superior es buscar funciones $u$ tales que su derivada débil cumpla la ecuación integral dada.\\

\noindent
Buscando que las parejas $u'$, $\varphi'$ y $f,\varphi$ estén en igualdad de condiciones (con esto se consigue el caso más fácil), trataremos de definir un producto escalar e interpretar la fórmula. COn exigir $u',\varphi'\in L^2$ bastaría para darle sentido\footnote{$L_2$ es más pequeño pero más cómodo de trabajar.}.
\end{ejemplo}

\section{Espacios de Sobolev}
\begin{definicion}
    Sea $I\subset \mathbb{R}$ abierto, consideramos\footnote{Donde la derivada que aparece es débil.}:
    \begin{equation*}
        H^1(I) = W^{1,2}(I) = \{f\in L^2(I):f'\in L^2(I)\}
    \end{equation*}
\end{definicion}

\begin{notacion}
    Lo notamos por $H^1$:
    \begin{itemize}
        \item La $H$ por ser un espacio de Hilbert.

            Si consideramos la norma:
            \begin{equation*}
                \|f\|_{H^1} = \|f\|_2 + \|f'\|_2
            \end{equation*}
            se tiene que es de Banach. Más aún, podemos considerar:
            \begin{equation*}
                \langle f,g \rangle_{H^1} = \langle f,g \rangle_{L^2} + \langle f',g' \rangle_{L^2}
            \end{equation*}
            Y obtenemos que $H^1$ es completo para dicha norma.
        \item El 1 porque hemos derivado 1 vez.
    \end{itemize}
    O también por $W^{2,1}$:
    \begin{itemize}
        \item La $W$ viene de \textit{weak}.
        \item El 1 porque hemos derivado 1 vez.
        \item El 2 porque exigimos $f,f'\in L^{\red{2}}(I)$.
    \end{itemize}
\end{notacion}

\begin{ejemplo}
    Tenemos que $|x|\in H^1(\left]-1,1\right[)$, pero $sgn(x),H(x)\notin H^1(\left]-1,1\right[)$. EN general, funcioens con discontinuidades de salto no están en espacios de Sobolev, pero cualquier función suave con picos en $\left]-1,1\right[$ sí que está en $H^1(\left]-1,1\right[)$.
\end{ejemplo}

\subsubsection{Varias variables}
\noindent
En varias variables, para $f:\Omega\to\mathbb{R}$ con $\Omega\subset\mathbb{R}^d$ abierto querremos exigir que:
\begin{equation*}
    f,\frac{\partial f}{\partial x_1},\ldots,\frac{\partial f}{\partial x_d}\in L^2(\Omega)
\end{equation*}
A partir de esto se define:

\begin{definicion}
    Sea $\Omega\subset\mathbb{R}^d$ un abierto, consideramos:
    \begin{equation*}
        H^1(\Omega) = \left\{f\in L^2(\Omega) : \nabla f \in (L^2(\Omega))^d\right\}
    \end{equation*}
\end{definicion}

\noindent
De forma análoga al caso para dimensión 1:
\begin{itemize}
    \item $H^1(\Omega)$ es un subespacio completo de $L^2(\Omega)$, para la norma:
        \begin{equation}\label{eq:norma_sobolev}
            \|f\|_{H^1} = \|f\|_2 + \sum_{i=1}^{d}\left\|\frac{\partial f}{\partial x_1}\right\|_2
        \end{equation}
    \item $H^1$ es un espacio de Hilbert, para el producto escalar:
        \begin{equation*}
            \langle f,g \rangle_{H^1} = \langle f,g \rangle_{L^2} + \sum_{i=1}^{d} \left\langle \frac{\partial f}{\partial x_i},\frac{\partial g}{\partial x_i} \right\rangle_{L^2}
        \end{equation*}
    \item Finalmente, escribiendo para abreviar $f_i = \frac{\partial f}{\partial x_i}$, como tenemos:
        \begin{equation*}
            (f,f_1,\ldots,f_d) \in {(L^2)}^{d+1}
        \end{equation*}
        Tenemos que combinar las magnitudes $\|f\|_2,\|f_1\|_2,\ldots,\|f_d\|_2$. En la fórmula~\eqref{eq:norma_sobolev} optamos por combinarlo mediante la suma, pero de forma análoga al caso de las normas $\|\cdot \|_p$ ($1\leq p\leq \infty$) en $\mathbb{R}^{d+1}$ (donde todas ellas son equivalentes), podríamos haber considerado:
        \begin{equation*}
            \|f\|_{H^1} = {\left(\int_\Omega f^2~dx + \sum_{i=1}^{d}\int_\Omega{\left(\frac{\partial f}{\partial x_i}\right)}^{2}\right)}^{\nicefrac{1}{2}}
        \end{equation*}
        Que es una norma equivalente\footnote{La demostración se basa en que $\|\cdot \|_1$ y $\|\cdot \|_2$ son equivalentes en $\mathbb{R}^{d+1}$.}.
\end{itemize}

\subsubsection{Derivadas sucesivas}
\noindent
En el caso de derivadas sucesivas, en dimensión $1$ nos gustaría definir:
\begin{equation*}
    H^2(I) = \{f\in L^2(I) : f',f''\in L^2(I)\}
\end{equation*}
Y trabajaremos en dicho espacio con la norma y producto escalar:
\begin{equation*}
    \|f\|_{H^2} = \|f\|_2 + \|f'\|_2 + \|f''\|_2, \qquad \langle f,g \rangle_{H^2} = \langle f,g \rangle_2 + \langle f',g' \rangle_2 + \langle f'',g'' \rangle_2
\end{equation*}

\begin{ejemplo}
    Observemos que $|x|\notin H^2(\left]-1,1\right[)$, puesto que $sgn(x)\notin H^1(\left]-1,1\right[)$.
\end{ejemplo}

\noindent
En varias variables para derivadas sucesivas es una definición análoga.\\

\subsubsection{Recordando la motivación de espacios de Sobolev}
\noindent
Para $I = \left]a,b\right[$ podemos introducir ahora un subespacio especial de $H^1(\left]a,b\right[)$:
\begin{equation*}
    H_0^1(\left]a,b\right[) = \overline{C^\infty_c(\left]a,b\right[)}^{\|\cdot \|_{H^1}}
\end{equation*}
¿Por qué nos interesa? Por construcción son ``las funciones $f\in H^1(I)$ que se anulan en la frontera del conjunto $I$''. La motivación para construir el conjunto es para despreciar el término $(\ast)$ de la ecuación~\eqref{eq:ast_despreciar}, teniendo en cuenta que estamos trabajando con clases de equivalencia y no podemos fijar el valor de la función en ningún punto.\\

\noindent
Así, se puede considerar la aplicación traza
\Func{Tr}{C_c^1(\bb{R}^d)}{L^2(\partial \Omega)}{f}{f\big|_{\partial \Omega}}
Puede verse que $Tr$ es lineal y que se puede extender a un operador lineal acotado
\Func{Tr}{H^1(\Omega)}{L^2(\partial \Omega)}{f}{f\big|_{\partial \Omega}}
Esto nos sirve para justificar la fórmula de Green en $H^1(\Omega)$, puesto que tenemos que interpretrar los valores de $fg$ en $\partial \Omega$.\\

\noindent
En vistas de que para:
\begin{equation}\label{eq:ec_motivacion2}
    \left\{\begin{array}{l}
        -u''(x) = f(x) \\
        u(a) = 0 = u(b)
    \end{array}\right.
\end{equation}
en $I$, si $u\in C^2(I)$ es solución y $\varphi \in C^1_c(I)$ , entonces:
\begin{equation*}
    \int_{a}^{b} u'(x)\varphi'(x)~dx = \int_{a}^{b} f(x)\varphi(x)~dx 
\end{equation*}~\\

\noindent
Consideramos ahora que si $u:I\to \mathbb{R}$ es una solución del problema~\eqref{eq:ec_motivacion2} en sentido débil si $u\in H^1_0(I)$ y además:
\begin{equation*}
    \int_{a}^{b} u'(x)\varphi'(x)~dx = \int_{a}^{b} f(x)\varphi(x)~dx \qquad \forall \varphi \in H^1_0(I)
\end{equation*}
donde vemos que $u',\varphi,\varphi'\in L^2(I)$ y si $f\in L^2(I)$ no hay problema.
