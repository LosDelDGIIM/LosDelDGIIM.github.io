    \begin{ejercicio}
        Tomemos $f=(x^4+2)(x^2-2)\in \mathbb{Q}[x]$ y $K$ el cuerpo de descomposición de $f$.
        \begin{enumerate}
            \item \textbf{[1 punto]} Demostrar que $K$ es también cuerpo de descomposición de $x^4-2\in \mathbb{Q}[x]$.
            \item \textbf{[1 punto]} Describir explícitamente los elementos del grupo $\Aut(K)$.
            \item \textbf{[1 punto]} Calcular, listando sus elementos, $\Aut_{\mathbb{Q}\left(i\sqrt{2}\right)}(K) \cap \Aut_{\mathbb{Q}\left(\sqrt{2}\right)}(K)$.
            \item \textbf{[1 punto]} Calcular el núero de subcuerpos de $K$ de grado 4 sobre $\mathbb{Q}$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Responder razonadamente a las siguientes cuestiones.
        \begin{enumerate}
            \item \textbf{[0.5 puntos]} Sea $\alpha$ una raíz real de $f=x^3-3x+1\in \mathbb{Q}[x]$. ¿Es $\mathbb{Q}(\alpha)$ un cuerpo de descomposición de $f$?
            \item \textbf{[0.5 puntos]} ¿Cuántos cuerpos finitos no isomorfos entre sí tienen cardinal menor que 25?
            \item \textbf{[0.5 puntos]} ¿Es cierto que si $K$ es un cuerpo de descomposición de un polinomio $f\in \mathbb{Q}[x]$ y $a\in K$, entonces $\Irr(a,\mathbb{Q})$ es un divisor de $f$?
            \item \textbf{[0.5 puntos]} ¿Satisface todo número constructible con regla y compás una ecuación resoluble por radicales?
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Sea $F$ un cuerpo de característica 2 y $a\in F$ tal que $F=\bb{F}_2(a)$ y $a^6 = a^3 + 1$.
        \begin{enumerate}
            \item \textbf{[1 punto]} Calcular el cardinal de $F$.
            \item \textbf{[1 punto]} Calcular $\Irr(a^{21},\bb{F}_2)$ y todos los homomorfismos de cuerpos de $\bb{F}_2(a^{21})$ a $F$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Sea $\omega \in \mathbb{C}$ una raíz décima primitiva de la unidad.
        \begin{enumerate}
            \item \textbf{[1 punto]} Calcular $\Irr(\omega, \mathbb{Q})$.
            \item \textbf{[1 punto]} Calcular, expresando sus generadores en términos de $\omega$, todos los subcuerpos de $\mathbb{Q}(\omega)$.
        \end{enumerate}
    \end{ejercicio}

    \noindent
    \textbf{Nota:} Se recuerda, por si es útil en algún ejercicio, que el discriminante de una cúbica reducida $f=x^3+px+q$ vale $-4p^3-27q^2$.
