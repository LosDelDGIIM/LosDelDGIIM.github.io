    \begin{ejercicio}
        Tomemos $f=x^3-2\in \mathbb{Q}\left(\sqrt{3}\right)[x]$ y $K$ el cuerpo de descommposición de $f$ sobre $\mathbb{Q}\left(\sqrt{3}\right)$.
        \begin{enumerate}[label=\alph*)]
            \item Calcular razonadamente $[K:\mathbb{Q}(\sqrt{3})]$.
            \item Describir los elementos del grup $\Aut_{(\mathbb{Q}(\sqrt{3}))}(K)$.
            \item Calcular todas las subextensiones de la extensión $\mathbb{Q}(\sqrt{3})\leq K$.
            \item Dar todas las subextensiones de $\mathbb{Q}(\sqrt{3})\leq K$ que contienen al número $\sqrt{3}+i$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Consideremos una raíz cúbica primitiva de la unidad $w\in \mathbb{C}$. Decidir razonadamente si $\mathbb{Q}(w) = \mathbb{Q}(\frac{1}{w+1})$.
    \end{ejercicio}

    \begin{ejercicio}
        Sea $g=x^3+x^2-1\in \bb{F}_3[x]$ y $F$ un cuerpo de descomposición sobre $\bb{F}_3$ de $g$.
        \begin{enumerate}[label=\alph*)]
            \item Describir los elementos del grupo $\Aut(F)$ y calcular todos los subcuerpos de $F$.
            \item Si $\alpha_1,\alpha_2,\alpha_3\in F$ son las raíces de $g$, decidir si $\alpha_1^2 + \alpha_2^2 + \alpha_3^2 \in \bb{F}_3$.
            \item Resolver la ecuación $x^2+1=0$ en $F$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Decidir razonadamente sobre la veracidad de las siguientes afirmaciones:
        \begin{enumerate}[label=\alph*)]
            \item $\sqrt[6]{32}-\sqrt[6]{16}\in \mathbb{Q}(\sqrt{2},\sqrt[3]{2})$.
            \item Si $F\leq K$ es una extensión de Galois y $\alpha\in K$ es tal que $\sigma(\alpha)\neq \alpha$ para todo $\sigma\in \Aut_F(K)$ distinto de la identidad, entonces $K= F(\alpha)$.
            \item Si $f\in \mathbb{Q}[x]$ es un polinomio de grado 3 y tiene una raíz construible, entonces $f$ es reducible.
            \item Sea $F$ un cuerpo de descomposición de $f=x^2+x+1\in \bb{F}_5[x]$ y $\alpha\in F$ una raíz de $f$. para $g=\Irr(\alpha+1,\bb{F}_5)$ se tiene que $g(\alpha) = 3\alpha$.
        \end{enumerate}
    \end{ejercicio}
