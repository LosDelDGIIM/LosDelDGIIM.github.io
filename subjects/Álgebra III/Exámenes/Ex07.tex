    \begin{ejercicio}[5 puntos]
        Sea $f= x^4 + 3x^2-2 \in \bb{Q}[x]$ y $K$ su cuerpo de descomposición. 
        \begin{enumerate}
            \item Demostrar que $f$ tiene una raíz real positiva $\alpha$ tal que $\alpha\notin\bb{Q}(\alpha^2)$.
            \item Demostrar que $i\sqrt{2} \in K$ y calcular $[K:\bb{Q}]$.
            \item Calcular $Aut(K)$ definiendo explícitamente todos sus elementos. 
            \item Calcular los subgrupos de $Aut(K)$ correspondientes por la conexión de Galois con los subcuerpos $\bb{Q}(\alpha^2)$ y $\bb{Q}(i\sqrt{2})$ de $K$, listando sus elementos. 
            \item Decidir razonadamente cuál es el grado del polinomio
                \begin{equation*}
                    g=Irr(\alpha+i\sqrt{2}, \bb{Q})
                \end{equation*}
                ¿Es g resoluble por radicales? ¿Son las raíces commplejas de f constructibles con regla y compás?
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[3 puntos]
        Sea $\eta \in \bb{C}$ una raíz decimocuarta primitiva de la unidad. 
        \begin{enumerate}
            \item Calcular el decimocuarto polinomio ciclotómico $\phi_{14}$.
            \item Calcular, definiendo explícigtamente todos sus elementos, $Aut(\bb{Q}(\eta))$ y todos sus subgrupos.
            \item Calcular, expresando sus generadores en función de $\eta$, todos los subcuerpos de $\bb{Q}(\eta)$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[2 puntos]
        Vimos en clase que $\bb{F}_{64}=\bb{F}_2(a)$ con a satisfaciendo la ecuación $a^6+a+1=0$, y que $a$ es un elemento primitivo del $\bb{F}_{64}$. Calcular, expresándolas en función de a, las raíces en $\bb{F}_{64}$ del polinomio $h=x^3+x+1 \in \bb{F}_2[x]$.
    \end{ejercicio}
