    \begin{ejercicio}
        Decidir razonadamente sobre la veracidad de las siguientes afirmaciones:
        \begin{enumerate}[label=\alph*)]
            \item El número real $\sum\limits_{n=1}^{8}\sqrt[n]{2}$ es algebraico sobre $\mathbb{Q}$.
            \item Si $K$ es un cuerpo de descomposición de un polinomio $f\in F[x]$ y $\alpha \in K$, entonces $\Irr(\alpha,F)$ es un divisor de $f$.
            \item Dada una torre de cuerpos $F\leq E\leq K$, si $F\leq E$ y $E\leq K$ son de Galois, entonces $F\leq K$ es de Galois.
            \item Si $z\in \mathbb{C}$ tiene grado 4 sobre $\mathbb{Q}$, entonces $z$ es un número construible.
        \end{enumerate}
    \end{ejercicio}

    \noindent
    \textbf{Solución.}

    \begin{ejercicio}
        Tomemos $f=x^3+3\in \mathbb{Q}[x]$ y $K$ el cuerpo de descomposición sobre $\mathbb{Q}$ de $f$.
        \begin{enumerate}[label=\alph*)]
            \item Decidir razonadamente si $\sqrt{3}\in K$.

                Las raíces de $f$ son las raíces cúbicas de $-3$, es decir, $-\sqrt[3]{3},-w\sqrt[3]{3}$ y $-w^2\sqrt[3]{3}$, donde $w$ es una raíz cúbica primitiva de la unidad, por ejemplo:
                \begin{equation*}
                    w = \frac{-1}{2}+i\frac{\sqrt{3}}{2}
                \end{equation*}
                Tenemos así que $K = \mathbb{Q}\left(\sqrt[3]{3},w\sqrt[3]{3},w^2\sqrt[3]{3}\right)$, pero como:
                \begin{equation*}
                    w = \frac{w\sqrt[3]{3}}{\sqrt[3]{3}} 
                \end{equation*}
                Observamos que $K =\mathbb{Q}(\sqrt[3]{3},w)$. Más aún, afirmamos que $K=\mathbb{Q}\left(\sqrt[3]{3},i\sqrt{3}\right)$, que es claro en vista de la expresión de $w$. Si calculamos $[K:\mathbb{Q}]$ vemos por el Lema de la Torre que:
                \begin{equation*}
                    [K:\mathbb{Q}] = \left[K:\mathbb{Q}\left(\sqrt[3]{3}\right)\right]\left[\mathbb{Q}\left(\sqrt[3]{3}\right):\mathbb{Q}\right] =2\cdot 3 = 6
                \end{equation*}
                ya que:
                \begin{itemize}
                    \item $x^3-3$ es irreducible en $\mathbb{Q}[x]$ por Eisenstein para $p=3$.
                    \item $x^2+3$ es irreducible en $\mathbb{Q}\left(\sqrt[3]{3}\right)[x]$ porque sus dos raíces son complejas.
                \end{itemize}
                Por reducción al absurdo, supuesto que $\sqrt{3}\in K$, tendríamos entonces que $\mathbb{Q}\left(\sqrt[3]{3},\sqrt{3},i\sqrt{3}\right) = K$, con $[K:\mathbb{Q}] = 6$, pero por el Lema de la Torre tenemos que:
                \begin{equation*}
                \left[\mathbb{Q}\left(\sqrt[3]{3},\sqrt{3},i\sqrt{3}\right):\mathbb{Q}\right] = \left[\mathbb{Q}\left(\sqrt[3]{3},\sqrt{3},i\sqrt{3}\right):\mathbb{Q}\left(\sqrt[3]{3},\sqrt{3}\right)\right]\left[\mathbb{Q}\left(\sqrt[3]{3},\sqrt{3}\right):\mathbb{Q}\right]
                \end{equation*}
                donde la primera es $2$ por ser $x^2+3$ irreducible e $\mathbb{Q}\left(\sqrt[3]{3},\sqrt{3}\right)$ ya que sus raíces son complejas y la segunda se ha visto en varias veces que es 6, puesto que podemos aplicar el Lema de la Torre para ver que es menor o igual que 6 y múltiplo de 2 y en el otro sentido para ver que también es múltiplo de 3. En definitiva, llegamos a que $6=12$, contradicción que viene de suponer que $\sqrt{3}\in K$.
            \item Describir los elementos del grupo $\Aut_\mathbb{Q}(K)$.

                Vemos que $K$ es cuerpo de descomposición de $f$ sobre $\mathbb{Q}$, por lo que $\mathbb{Q}\leq K$ es de Galois, con lo que $|\Aut_\mathbb{Q}(K)| = |\Aut(K)| = [K:F] = 6$. Para calcular sus elementos aplicaremos en reiteradas ocasiones la Proposición de extensión, calculando primero los homomorfismos $\eta:\mathbb{Q}\left(\sqrt[3]{3}\right)\to K$:
                \begin{figure}[H]
                    \centering
                    \shorthandoff{""}
                    \begin{tikzcd}
                        \mathbb{Q} \arrow[r, hook] \arrow[rd, hook] & K                         \\
                                                                    & {\mathbb{Q}(\sqrt[3]{3})}
                    \end{tikzcd}
                    \shorthandon{""}
                \end{figure}
                \noindent
                Como $x^3-3$ es irreducible en $\mathbb{Q}[x]$ encontramos 3 homomorfismos, $\eta_j$ para $j\in \{0,1,2\}$; cada uno determinado por una raíz distinta de este polinomio en $K$:
                \begin{equation*}
                    \sqrt[3]{3} \stackrel{\eta_j}{\longmapsto} w^j\sqrt[3]{3}
                \end{equation*}
                Cada uno de ellos se extiende a un automorfismo $K\to K$: 
                \begin{figure}[H]
                    \centering
                    \shorthandoff{""}
                    \begin{tikzcd}
                        {\mathbb{Q}(\sqrt[3]{3})} \arrow[r, "\eta_j"] \arrow[rd, hook] & K \\
                                                                                       & K
                    \end{tikzcd}
                    \shorthandon{""}
                \end{figure}
                \noindent
                Como $x^2+3\in \mathbb{Q}\left(\sqrt[3]{3}\right)[x]$ es irreducible tenemos que cada $\eta_j$ se extiende a dos automorfismos $\eta_{j,k}$ para $k \in \{0,1\}$, determinados por:
                \begin{align*}
                    \sqrt[3]{3} &\stackrel{\eta_{j,k}}{\longmapsto} w^j\sqrt[3]{3} \\
                    i\sqrt{3} &\stackrel{\eta_{j,k}}{\longmapsto} {(-1)}^{k}i\sqrt{3}
                \end{align*}
                Así, tenemos que:
                \begin{equation*}
                    \Aut(K) = \{\eta_{j,k}:j\in \{0,1,2\},k\in \{0,1\}\}
                \end{equation*}
            \item Calcular todos los subcuerpos de $K$. Señalar cuáles son extensiones de Galois de $\mathbb{Q}$.

                Como $\mathbb{Q}\leq K$ es de Galois, cada subcuerpo de $K$ está en correspondencia biunívoca con un único subgrupo de $\Aut(K)$. Por tanto, calculamos primero todos los subgrupos de $\Aut(K)$. Para ello, primero calculamos los órdenes de los automorfismos. Sabemos que $\Aut(K)$ es un subgrupo de $S_3$ y como $|\Aut(K)| = [K:\mathbb{Q}] = 6$ tiene que ser $\Aut(K)\cong S_3$, por lo que hay dos elementos de orden 3 y 3 de orden 2.
                \begin{itemize}
                    \item Claramente $\eta_{0,1}$ es de orden 2.
                    \item Para $\eta_{1,0}$:
                        \begin{equation*}
                            \sqrt[3]{3}\stackrel{\eta_{1,0}}{\longmapsto} w\sqrt[3]{3}\stackrel{\eta_{1,0}}{\longmapsto} w^2\sqrt[3]{3} \stackrel{\eta_{1,0}}{\longmapsto} w^3\sqrt[3]{3} = \sqrt[3]{3}
                        \end{equation*}
                        Por lo que tiene orden 3.
                    \item Para $\eta_{1,1}$:
                        \begin{equation*}
                            \sqrt[3]{3}\stackrel{\eta_{1,1}}{\longmapsto} w\sqrt[3]{3}\stackrel{\eta_{1,0}}{\longmapsto} w^2w\sqrt[3]{3} = \sqrt[3]{3}
                        \end{equation*}
                        Por lo que tiene orden 2.
                    \item Para $\eta_{2,0}$:
                        \begin{equation*}
                            \sqrt[3]{3}\stackrel{\eta_{2,0}}{\longmapsto} w^2\sqrt[3]{3}\stackrel{\eta_{2,0}}{\longmapsto} w^4\sqrt[3]{3} = w\sqrt[3]{3} \stackrel{\eta_{2,0}}{\longmapsto} w^3\sqrt[3]{3} = \sqrt[3]{3}
                        \end{equation*}
                        Por lo que tiene orden 3.
                    \item Tiene que ser por tanto $\eta_{2,1}$ de orden 2.
                \end{itemize}

                \begin{table}[H]
                \centering
                \begin{tabular}{cccccc}
                    $\eta_{0,0}$ & $\eta_{0,1}$ & $\eta_{1,0}$ & $\eta_{1,1}$ & $\eta_{2,0}$ & $\eta_{2,1}$ \\
                    \hline
                    1 & 2 & 3 & 2 & 3 & 2
                \end{tabular}
                \end{table}
                \noindent
                Así, los subgrupos de $\Aut(K)$ son:
                \begin{figure}[H]
                    \centering
                    \shorthandoff{""}
                    \begin{tikzcd}
                                                                        & Aut(K) \arrow[ld, no head] \arrow[dd, no head] \arrow[rrdd, no head] \arrow[rdd, no head] &                                                &                                                 \\
                        {\langle\eta_{1,0}\rangle} \arrow[rdd, no head] &                                                                                           &                                                &                                                 \\
                                                                        & {\langle\eta_{0,1}\rangle} \arrow[d, no head]                                             & {\langle\eta_{1,1}\rangle} \arrow[ld, no head] & {\langle\eta_{2,1}\rangle} \arrow[lld, no head] \\
                                                                                                                        & {\langle\eta_{0,0}\rangle}                                                                &                                                &                                                
                    \end{tikzcd}
                    \shorthandon{""}
                \end{figure}
                \noindent
                Buscamos ahora los subcuerpos asociados a cada subgrupo:
                \begin{itemize}
                    \item Para $K^{\langle \eta_{1,0} \rangle }$ buscamos un subcuerpo de grado 2 que quede fijo por $\langle \eta_{1,0} \rangle $. Observamos que $\mathbb{Q}\left(i\sqrt{3}\right)\leq K^{\langle \eta_{1,0} \rangle }$, con $\left[\mathbb{Q}\left(i\sqrt{3}\right):\mathbb{Q}\right] = 2$, por lo que este debe ser el subcuerpo.
                    \item Para $K^{\langle \eta_{0,1} \rangle }$ buscamos un subcuerpo de grado 3. Observamos que se cumple $\mathbb{Q}\left(\sqrt[3]{3}\right)\leq K^{\langle \eta_{0,1} \rangle }$, con $\left[\mathbb{Q}\left(\sqrt[3]{3}\right):\mathbb{Q}\right] = 3$, por lo que este debe ser el subcuerpo asociado.
                    \item Para $K^{\langle \eta_{1,1} \rangle }$ buscamos otro subcuerpos de grado 3. Observamos que:
                        \begin{equation*}
                            \sqrt[3]{3}\stackrel{\eta_{1,1}}{\longmapsto}w\sqrt[3]{3}, \qquad w\stackrel{\eta_{1,1}}{\longmapsto} w^2
                        \end{equation*}
                        Con lo que:
                        \begin{equation*}
                            w^2\sqrt[3]{3} \stackrel{\eta_{1,1}}{\longmapsto} w^5\sqrt[3]{3} = w^2\sqrt[3]{3}
                        \end{equation*}
                        de donde $\mathbb{Q}\left(w^2\sqrt[3]{3}\right)\leq K^{\langle \eta_{1,1} \rangle }$ y $\left[\mathbb{Q}\left(w^2\sqrt[3]{3}\right):\mathbb{Q}\right]= 3$ porque $x^3-3$ es irreducible.
                    \item Para $K^{\langle \eta_{2,1} \rangle }$ vemos de forma análoga al caso anterior que $K^{\langle \eta_{2,1} \rangle } = \mathbb{Q}\left(w\sqrt[3]{3}\right)$.
                \end{itemize}
            \item Calcular el cardinal del grupo $\Aut_\mathbb{Q}(K(i))$.
        \end{enumerate}
    \end{ejercicio}
