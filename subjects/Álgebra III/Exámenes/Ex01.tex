    \begin{ejercicio}
        Calcular $\phi_9\in \mathbb{Q}[x]$.\\

        \noindent
        Aplicando la fórmula recursiva:
        \begin{equation*}
            \phi_9 = \frac{x^9-1}{\phi_1\phi_3}
        \end{equation*}
        Y tenemos que:
        \begin{align*}
            \phi_1 &= x-1 \\
            \phi_3 &= \frac{x^3-1}{x-1} = x^2+x+1
        \end{align*}
        Por lo que:
        \begin{equation*}
            \frac{x^9-1}{\phi_1} = \frac{x^9-1}{x-1} = x^8+x^7+ \ldots + 1
        \end{equation*}
        Y si dividimos ahora este polinomio entre $x^2+x+1$, obtenemos $\phi_9 = x^6+x^3+1$.
    \end{ejercicio}

    \begin{ejercicio}
        Sea $\zeta\in \mathbb{C}$ una raíz primitiva novena de la unidad, calcular todos los subcuerpos de $\mathbb{Q}(\zeta)$.\\

        \noindent
        Tenemos que $\mathbb{Q}\leq\mathbb{Q}(\zeta)$ es de Galois y en teoría se ha visto que $\mathbb{Q}(\zeta)\cong \cc{U}(\mathbb{Z}_9)$, que tiene cardinal 6. Por tanto, tenemos que:
        \begin{equation*}
            \Aut(\mathbb{Q}(\zeta)) = \{\tau_1,\tau_2, \tau_4, \tau_5,\tau_7, \tau_8\}
        \end{equation*}
        donde $\tau_j$ viene determinada por $\zeta\longmapsto \zeta^j$. Tenemos que $\Aut(\mathbb{Q}(\zeta))\cong \cc{U}(\mathbb{Z}_9)$, que es abeliano y de 6 elementos, luego tiene que ser isomorfo a $\mathbb{Z}_6$, por lo que tenemos 2 subgrupos no triviales de $\mathbb{Z}_6$, con lo que estamos buscando dos subcuerpos de $\mathbb{Q}(\zeta)$ no triviales.

        \noindent
        Buscamos ahora un generador de $\Aut(\mathbb{Q}(\zeta))$, sabiendo que $\{1,\zeta,\zeta^2,\zeta^3,\zeta^4,\zeta^5\}$ son $\mathbb{Q}-$linealmente independientes.
        \begin{align*}
            \zeta &\stackrel{\tau_2}{\longmapsto} \zeta^2 \stackrel{\tau_2}{\longmapsto} \eta^4 \stackrel{\tau_2}{\longmapsto} \zeta^8 
        \end{align*}
        Todos ellos distintos de $\zeta$, por lo que $\tau_2$ no es de orden 1, 2, o 3, luego ha de ser de orden $6$ (por el Teorema de Lagrange), luego un generador de $G:=\Aut(\mathbb{Q}(\zeta))$. Además, sabemos que $\tau_2^2 = \tau_4$, así como que $\tau_2^3=\tau_8$. De esta forma, los subgruposde $G$ son $G,\{id_{\mathbb{Q}(\zeta)}\}, \langle \tau_2^3 \rangle, \langle \tau_2^2 \rangle  $; con $\langle \tau_2^3 \rangle $ de 2 elementos y $\langle \tau_2^2 \rangle $ de 3 elementos. Por la conexión de Galois, tenemos que calcular los subcuerpos:
        \begin{equation*}
            \mathbb{Q}(\zeta)^{\langle \tau_8 \rangle }, \qquad \mathbb{Q}(\zeta)^{\langle \tau_4 \rangle }
        \end{equation*}
        de los que conocemos su grado:
        \begin{equation*}
            [\mathbb{Q}^{\langle \tau_8 \rangle }: \mathbb{Q}] = (G:\langle \tau_8 \rangle ) = 3, \qquad [\mathbb{Q}^{\langle \tau_4 \rangle }: \mathbb{Q} ] = 2
        \end{equation*}
        \begin{itemize}
            \item Para $\mathbb{Q}^{\langle \tau_4 \rangle }$, sabemos que:
                \begin{equation*}
                    \zeta^6 + \zeta^3 + 1 = 0
                \end{equation*}
                Por lo que $\zeta^3$ es raíz de $x^2+x+1\in \mathbb{Q}[x]$, polinomio irreducible (por ser polinomio ciclotómico por ejemplo), con lo que $\zeta^3$ tiene grado 2 sobre $\mathbb{Q}$, por lo que seguro tendremos:
                \begin{equation*}
                    \mathbb{Q}(\zeta^3) = \mathbb{Q}^{\langle \tau_4 \rangle }
                \end{equation*}
            \item Para $\mathbb{Q}^{\langle \tau_8 \rangle }$, observemos que:
                \begin{equation*}
                    \zeta^8 = \zeta^{-1} = \overline{\zeta}
                \end{equation*}
                con lo que:
                \begin{align*}
                    \zeta &\stackrel{\tau_8}{\longmapsto} \zeta^8 = \overline{\zeta} \\
                    \zeta^8 &\stackrel{\tau_8}{\longmapsto} \zeta^{64} = \zeta
                \end{align*}
                Por lo que $\tau_8$ mantedrá fijo el elemento $\zeta + \zeta^8$, de donde $\zeta^8+\zeta \in \mathbb{Q}^{\langle \tau_8 \rangle }$, por lo que $\mathbb{Q}(\zeta^8+\zeta)\leq \mathbb{Q}^{\langle \tau_8 \rangle }$. Finalmente, veamos el grado de $\zeta^8+\zeta$ y comprobemos que es 3. Supuesto que $\zeta+\zeta^8\in \mathbb{Q}$ y usando que $\{1,\zeta, \ldots, \zeta^5\}$ son $\mathbb{Q}-$linealmente dependientes, vemos que:
                \begin{equation*}
                    \zeta^8 = \zeta^2 \zeta^6 = \zeta^2(-\zeta^3-1) = -\zeta^5 - \zeta^2
                \end{equation*}
                Por lo que:
                \begin{equation*}
                    \zeta + \zeta^8 = \zeta - \zeta^5 -\zeta^2
                \end{equation*}
                que no puede estar en $\mathbb{Q}$, porque esto contradeciría que estos fueran $\mathbb{Q}-$linealmente independientes
        \end{itemize}
    \end{ejercicio}
