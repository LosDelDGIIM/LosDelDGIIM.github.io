    \begin{ejercicio}
        Enuncia:
        \begin{enumerate}[label=\alph*)]
            \item Definición de \textit{Conjunto mayorado}.

                Un conjunto $A\subset \mathbb{R}$ decimos que está mayorado si existe $M\in \mathbb{R}$ de forma que $a\leq M \quad \forall a\in A$.
            \item Definición de \textit{Sucesión convergente}.

                Una sucesión $\{x_n\}$ de números reales decimos que es convergente si:
                \begin{equation*}
                    \forall \varepsilon>0~\exists m\in \mathbb{N} : n\geq m \Longrightarrow |x_n - x| < \varepsilon
                \end{equation*}
            \item Definición de \textit{Sucesión de Cauchy}.

                Una sucesión $\{x_n\}$ de números reales decimos que es de Cauchy si:
                \begin{equation*}
                    \forall \varepsilon>0~\exists m\in \mathbb{N} : p,q\geq m \Longrightarrow |x_p - x_q| < \varepsilon
                \end{equation*}
            \item Enuncia el \textit{Teorema de Bolzano-Weierstrass}.

                Toda sucesión de numeros reales acotada admite una sucesión parcial convergente.
            \item Enuncia el \textit{Teorema de complitud de $\mathbb{R}$}.

                En $\mathbb{R}$ toda sucesión de Cauchy es convergente.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Decide \textbf{razonadamente} si las siguientes afirmaciones son verdaderas o falsas:
        \begin{enumerate}[label=\alph*)]
            \item $a,b\in \mathbb{R}$, $|ab|<|a| \Longrightarrow b<1$.

                Es \textbf{verdadera}, puesto que si $a,b\in \mathbb{R}$ con $|ab|<|a|$, tenemos entonces que:
                \begin{equation*}
                    |a||b| = |ab|  < |a| \Longleftrightarrow |b| < 1 \Longleftrightarrow b \in \left]-1,1\right[
                \end{equation*}
                En particular, tenemos que $b<1$.
            \item Todo conjunto no vacío de números naturales tiene mínimo.

                Es \textbf{verdadera}, y se conoce como el principio de buena ordenación de los naturales.
            \item Todo conjunto no vacío de números positivos tiene mínimo.

                Es \textbf{falsa}, pues $\left]0,1\right[$ es un conjuno no vacío de números positivos y no tiene mínimo.
            \item Toda sucesión monótona y mayorada es convergente.

                Es \textbf{falsa}, pues $\{-n\}$ es una sucesión monótona (es decreciente) y mayorada por $4$ pero no es convergente.
            \item Toda sucesión que admita una parcial de Cauchy, es acotada.

                Es \textbf{falsa}, pues la sucesión $\{x_n\}$ dada por:
                \begin{equation*}
                    x_{2n} = 0, \qquad x_{2n+1} = n \qquad \forall n\in \mathbb{N}
                \end{equation*}
                Admite una parcial de Cauchy, $\{x_{2n}\} = \{0\}$, pero no es acotada.
        \end{enumerate}
    \end{ejercicio}
