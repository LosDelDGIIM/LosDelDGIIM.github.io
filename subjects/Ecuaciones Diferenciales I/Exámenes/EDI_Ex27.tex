    \begin{ejercicio}
        Demuestra que la ecuación
        \begin{equation*}
            \arctan(x) + 2t + x = 1
        \end{equation*}
        define una única función implícita $x = x(t)$, $t \in \mathbb{R}$. ¿Es derivable esta función?
        En caso afirmativo encuentra una ecuacion diferencial de la que sea solución.
    \end{ejercicio}

    \begin{ejercicio}
        Encuentra la solución del problema de valores iniciales
        \begin{equation*}
            x' = \frac{t}{\cos(x)}, \quad x(2) = 0.
        \end{equation*}
        Se especificará el intervalo de definición de esta solución.
    \end{ejercicio}

    \begin{ejercicio}
        Encuentra la solución del problema de valores iniciales
        \begin{equation*}
            x' = \frac{2tx}{t^2 + x^2}, \quad x(-3) = -3.
        \end{equation*}
        ¿En qué intervalo está definida?
    \end{ejercicio}

    \begin{ejercicio}
        Se supone que $\varphi : \mathbb{R}^2 \to \mathbb{R}^2, (t,x) \mapsto (s,y)$ es un $C^1$ difeomorfismo con inversa
        $\psi : \mathbb{R}^2 \to \mathbb{R}^2, (s,y) \mapsto (t,x)$. Expresa $\frac{\partial \psi_2}{\partial s}$ en términos de $\varphi_1, \varphi_2$ y sus derivadas parciales.
    \end{ejercicio}

    \begin{ejercicio}
        Un subconjunto $C$ del plano $\mathbb{R}^2$ es una curva explícita si es posible encontrar un intervalo abierto $I$
        y una función $f : I \to \mathbb{R}$ de clase $C^1$ para los que se cumple
        \begin{equation*}
            C = \{(x,y): x \in I, y = f(x)\}.
        \end{equation*}   

        Supongamos que $C$ es una curva explícita y $\varphi : \mathbb{R}^2 \to \mathbb{R}^2, \varphi = (\varphi_1, \varphi_2)$ es un
        difeomorfismo de clase $C^1$ que cumple 
        \begin{equation*}
            \frac{\partial \varphi_1}{\partial x}(x, f(x)) + \frac{\partial \varphi_1}{\partial y}(x, f(x)) f'(x) \neq 0. \quad \forall x \in I.
        \end{equation*}

        Demuestra que $\varphi(C)$ es una curva explícita.
    \end{ejercicio}
