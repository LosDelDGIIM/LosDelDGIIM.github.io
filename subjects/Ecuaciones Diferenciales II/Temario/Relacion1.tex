\begin{ejercicio} % // TODO: Ejercicio de Olga
    Tratamos de resolver el sistema:
    \begin{equation*}
        \left\{\begin{array}{l}
                \dot{x} = y \\
                \dot{y} = -x \\
                \dot{z} = 0
        \end{array}\right.
    \end{equation*}
    de donde obtenemos:
    \begin{align*}
        x(t) &= c_1\cos t + c_2\sen t \\
        y(t) &= -c_1\sen t +c_2\cos t \\
        z(t) &= c_3 \in \mathbb{R}
    \end{align*}
    para ciertas constantes $c_1,c_2,c_3\in \mathbb{R}$. El ejercicio consiste en demostrar que la curva $\alpha = (x,y,z)$ tiene como traza una circunferencia.\\

    \noindent
    Para ello, vemos por una parte que la curva $\alpha=(x,y,z)$ está contenida en el plano $P(z=c_3)$ y que esta misma cumple:
    \begin{equation*}
        x^2(t) + y^2(t) = c_1^2 + c_2^2 \qquad \forall t\in \mathbb{R}
    \end{equation*}
    Queda claro que $\tr \alpha$ queda contenida en la circunferencia de centro $0$ y radio $\sqrt{c_1^2+c_2^2}$. Veamos ahora que $\alpha$ recubre toda la circunferencia. Para ello, recordamos que todos los puntos de la circunferencia son de la forma:
    \begin{align*}
        x(t) &= r\cos(t+\theta) \\
        y(t) &= r\sen(t+\theta)
    \end{align*}
    para ciertos $r>0$ y $\theta\in \mathbb{R}$. Buscamos relacionar $r$ y $\theta$ con $c_1$ y $c_2$:
    \begin{align*}
        c_1\cos t + c_2 \sen t &= r\cos(t+\theta) \\
                               &= r(\cos t\cos\theta - \sen t\sen \theta) \\
                               &= r\cos t\cos\theta - r\sen t\sen\theta
    \end{align*}
    de donde:
    \begin{equation*}
        \left\{\begin{array}{l}
            c_1 = r\cos\theta \\
            c_2 = -r\sen\theta \\
            c_1^2 + c_2^2 = r^2(\cos^2\theta + \sen^2\theta) = r^2 \quad\Longrightarrow\quad r = \sqrt{c_1^2 + c_2^2}
        \end{array}\right. 
    \end{equation*}
    Despejando y suponiendo que $(c_1,c_2)\neq (0,0)$:
    \begin{equation}\label{eq:ej1}
       \cos\theta = \frac{c_1}{\sqrt{c_1^2+c_2^2}} , \qquad \sen\theta = \frac{-c_2}{\sqrt{c_1^2+c_2^2}}
    \end{equation}
    Vemos que el vector:
    \begin{equation*}
        \left(\frac{c_1}{\sqrt{c_1^2+c_2^2}}, \frac{-c_2}{\sqrt{c_1^2+c_2^2}}\right)
    \end{equation*}
    Tiene módulo 1, por lo que podemos encontrar $\theta\in \mathbb{R}$ verificando~\refeq{eq:ej1}.\\

    \noindent
    \textbf{Otra forma:} Buscamos ver que:
    \begin{equation*}
        \left\{\begin{array}{l}
                c_1\cos t + c_2\sen t = r\cos(t+\theta) \\
                -c_1\sen t + c_2\cos t = r\sen(t+\theta)
        \end{array}\right.
    \end{equation*}
    Tomando $r = \sqrt{c_1^2 + c_2^2}$ y $\theta\in \mathbb{R}$ verificando:
    \begin{equation*}
        \cos\theta = \frac{c_1}{\sqrt{c_1^2+c_2^2}}, \qquad \sen\theta = \frac{-c_2}{\sqrt{c_1^2+c_2^2}}
    \end{equation*}
    Vemos que se da la igualdad.
\end{ejercicio}

\begin{ejercicio}
    Sean $f_n:[0,1]\to \mathbb{R}$ dadas por:
    \begin{equation*}
        f_n(t) = \sqrt{t+\frac{1}{n}} \qquad \forall t\in [0,1], \quad \forall n\in \mathbb{N}
    \end{equation*}
    Probar que $\{f_n\}$ es equicontinua.\\

    \noindent
    Para ello, si miramos a la definición, buscamos acotar la diferencia:
    \begin{equation*}
        f_n(t) - f_n(s) = \sqrt{t+\frac{1}{n}} - \sqrt{s+\frac{1}{n}}
    \end{equation*}
    Por $\varepsilon$ sabiendo acotar $t-s$ por $\delta$. Si multiplicamos y dividimos por el conjugado:
    \begin{equation*}
        |f_n(t) - f_n(s)| = \frac{|t-s|}{\sqrt{t+\frac{1}{n}}+\sqrt{s+\frac{1}{n}}}
    \end{equation*}
    Ahora:
    \begin{equation*}
        |f_n(t) - f_n(s)| = \frac{|t-s|^{\nicefrac{1}{2}}}{\sqrt{t+\frac{1}{n}}+\sqrt{s+\frac{1}{n}}}|t-s|^{\nicefrac{1}{2}}
    \end{equation*}
    Tenemos que:
    \begin{equation*}
        |t-s| \leq t+s \quad\Longrightarrow\quad|t-s|^{\nicefrac{1}{2}} \leq \sqrt{t} + \sqrt{s}
    \end{equation*}
    De donde podemos acotar el primer término del producto anterior por 1, de donde:
    \begin{equation*}
        |f_n(t) - f_n(s)| \leq |t-s|^{\nicefrac{1}{2}} \qquad \forall t,s\in [0,1], \quad \forall n\geq 0
    \end{equation*}
    Ahora, para $\varepsilon>0$, tomando $\delta = \varepsilon^2$ tenemos que si $|t-s|<\delta$ entonces $|t-s|^{\nicefrac{1}{2}}<\varepsilon$, de donde:
    \begin{equation*}
        |f_n(t) - f_n(s)| \leq |t-s|^{\nicefrac{1}{2}} < \varepsilon \qquad \forall n\geq 0
    \end{equation*}
\end{ejercicio}

\begin{ejercicio}
    Probar que la sucesión anterior converge uniformemente.\\

    \noindent
    Tenemos que probar que:
    \begin{equation*}
        \forall \varepsilon>0~~\exists N\in \mathbb{N}~:~n\geq N \quad\Longrightarrow\quad |f_n(t) - f(t)| < \varepsilon \qquad \forall t\in [0,1]
    \end{equation*}
    Para cierta función $f:[0,1]\to \mathbb{R}$. En primer lugar, observamos que fijado $t\in [0,1]$ y usando que la raíz es continua obtenemos que:
    \begin{equation*}
        \{f_n(t)\} = \left\{\sqrt{t+\frac{1}{n}}\right\} \to \sqrt{t}
    \end{equation*}
    Por lo que si definimos $f:[0,1]\to \mathbb{R}$ como $f(t) = \sqrt{t}$ para todo $t\in [0,1]$ tenemos entonces que $\{f_n\}\to f$ puntualmente. Como la convergencia uniforme implica la puntual si la sucesión $\{f_n\}$ converge uniformemente, esta deberá hacerlo a $f$.\\

    \noindent
    Una vez tenemos claro cuál es la función $f$, tratamos de estudiar la siguiente diferencia con el fin de acotarla para $n\in \mathbb{N}$ suficientemente grande:
    \begin{align*}
        |f_n(t) - f(t)| &= \left|\sqrt{t+\frac{1}{n}}-\sqrt{t}\right| = \frac{\nicefrac{1}{n}}{\sqrt{t+\frac{1}{n}} + \sqrt{t}} \leq \frac{\nicefrac{1}{n}}{\nicefrac{1}{\sqrt{n}}} = \frac{1}{\sqrt{n}} \qquad \forall n\in \mathbb{N}
    \end{align*}
    Así, fijado $\varepsilon>0$, tenemos que existe $N\in \mathbb{N}$ de forma que $\nicefrac{1}{\sqrt{N}}< \varepsilon$, y si tomamos $n\in \mathbb{N}$ con $n\geq N$ obtendremos que:
    \begin{equation*}
        |f_n(t) - f(t)| \leq \frac{1}{\sqrt{n}} \leq \frac{1}{\sqrt{N}} < \varepsilon \qquad \forall t\in [0,1]
    \end{equation*}
    Acabamos de probar que $\{f_n\}\to f$ uniformemente.\\

    \noindent
    \textbf{Otra forma.} (ejercicio 2 de 2022) y tenía 3 apartados.
    \begin{enumerate}[label=\alph*)]
        \item Probar $\sqrt{a+b}\leq \sqrt{a}+\sqrt{b}\quad \forall a,b\geq 0$, como $x^2$ es creciente en $\mathbb{R}^+$ tenemos la desigualdad si y solo si:
            \begin{equation*}
                a+b \leq a+b+2\sqrt{ab} \quad\Longleftrightarrow\quad 0 \leq \sqrt{ab}
            \end{equation*}
        \item Hacer el ejercicio anterior, para el que podemos usar el apartado a).
            \begin{align*}
                |f_n(t) - f(t)| &= \sqrt{t+\frac{1}{n}} - \sqrt{t} \leq \sqrt{\frac{1}{n}}
            \end{align*}
        \item Se pregunta sobre la convergencia uniforme de la sucesión de las derivadas.

            Tenemos que:
            \begin{equation*}
                f_n'(t) = \frac{1}{2\sqrt{t+\frac{1}{n}}} \qquad \forall t\in [0,1]
            \end{equation*}
            Y para $t=0$ vemos que:
            \begin{equation*}
                f_n'(0) = \frac{\sqrt{n}}{2} \to \infty
            \end{equation*}
            Por lo que $\{f_n'(0)\}$ ni siquiera converge puntualmente, con lo que no puede converger uniformemente.
    \end{enumerate}
\end{ejercicio}

\begin{ejercicio}
    Tenemos una sucesión de funciones $\{f_n\}_{n\geq 1}$ que vienen definidas como $f_n:[0,1]\to \mathbb{R}$ de modo que:
    \begin{equation*}
        |f_n(0)| \leq 7 \qquad \forall n\in \mathbb{N}
    \end{equation*}
    Con $\{f_n\}$ equicontinua, entonces existe una parcial $\{f_{\sigma(n)}\}$ que converge uniformemente.\\

    \noindent
    Buscamos aplicar el Teorema de Ascoli, tenemos que ver que:
    \begin{equation*}
        |f_n(t)| \leq M \qquad \forall n\in \mathbb{N}, \quad \forall t\in [0,1]
    \end{equation*}
    Recordamos que $\{f_n\}$ es equicontinau si:
    \begin{equation*}
        \forall \varepsilon>0~~\exists \delta>0~:~ |f_n(t) - f_n(s)| < \varepsilon, \quad |t-s| < \delta\qquad \forall n\in \mathbb{N}
    \end{equation*}
    La equicontinuidad dice que la función ``crece por escaleras''. Vamos a probar la cota por inducción.\\

    \noindent
    Tomamos $\varepsilon = 1$ y la equicontinuidad nos da $\delta>0$, tomamos $m = \frac{\delta}{2}$, $k\geq 1$ y si $t\in [(k-1)m, km]\cap [0,1]\neq \emptyset $, vamos a probar que:
    \begin{equation*}
        |f_n(t)| \leq 7 + k \qquad \forall n\in \mathbb{N}
    \end{equation*}
    \begin{itemize}
        \item Para $k = 1$ tenemos $t\in [0,m]$, de donde:
            \begin{equation*}
                |f_n(t)| \leq |f_n(t)-f_n(0)| + |f_n(0)| \leq 1 + 7 = 8
            \end{equation*}
        \item Suponiendo ahora que $k > 1$, tomando $t\in [km, (k+1)m]$, tenemos que:
            \begin{equation*}
                |f_n(t)| \leq |f_n(t) - f_n(km)| + |f_n(km)| \leq 1 + 7 + k = 7 + (k+1)
            \end{equation*}
    \end{itemize}
    Como tenemos cota uniforme, aplicando el Teorema de Arcoli conseguimos el ejercicio.
\end{ejercicio}

\begin{ejercicio}
    % // TODO: Copiar el enunciado del ejercicio
    Tenemos $F\in C^1(\mathbb{R}^2)$ con la condición $F(x_0,y_0)=0$ y tenemos:
    \begin{equation*}
        \frac{\partial F}{\partial y}(x_0,y_0) \neq 0
    \end{equation*}
    % NO es problema de valores iniciales, es una ecuación funcional
    \begin{enumerate}[label=\alph*)]
        \item Tenemos que $\mathbb{R}^2$ es un abierto, que $F\in C^1(\mathbb{R}^2)$ y que:
            \begin{equation*}
                \frac{\partial F}{\partial y}(x_0,y_0) \neq 0
            \end{equation*}
            Aplicando el Teorema de la función implícita obtenemos directamente la existencia de la función.
        \item Queremos ahora llegar a la misma solución utilizando Cauchy-Peano. Para ello, hacemos derivación implícita, suponemos que tenemos una función $y$ que depende de $x$ y suponemos que tenemos una solución $F$, en cuyo caso se tendrá la ecuación:
            \begin{equation*}
                \frac{d}{dx}F(x,y) = \frac{\partial F}{\partial x}(x,y) + \frac{\partial F}{\partial y}(x,y)y' = 0
            \end{equation*}
            ¿Esta ecuación equivale a encontrar la $F$? tendríamos que $F(x,y(x)) = c$, de donde tendríamos la ecuación inicial que nos han dado. La idea es que dada una ecuación funcional se puede obtener una ecuación diferencial, pero en este caso podemos invertir este procedimiento. Aquí comienza la prueba, que va a consistir en aplicar el Teorema de Cauchy-Peano a la ecuación diferencial:

            \begin{equation*}
                y' = \frac{-\frac{\partial F(x,y)}{\partial y}}{\frac{\partial F}{\partial y}(x,y)}, \qquad y(x_0) = y_0
            \end{equation*}
            Como $F$ es de clase $C^1$ tenemos que es continua y como su parcial es distinta de cero, podemos exigir que existe un entorno $\Omega$ abierto y conexo de $(x_0,y_0)$ donde tengamos el denominador no nulo. Vemos que el campo es continuo como cociente de continuos, que $(x_0,y_0) \in \Omega^\circ$ y aplicando el Teorema de Cauchy-Peano conseguimos una solución $y:I\to \mathbb{R}$  de forma que $Gr(y)\subset \Omega$ y cumpliendo la ecuación diferencial. Ahora se continuaría hacia detrás.
        \item Damos un contraejemplo:
            \begin{equation*}
                F(x,y) = x^2+y = 0, \qquad (x_0,y_0) = (-1,1)
            \end{equation*}
            Tenemos que:
            \begin{equation*}
                \frac{\partial F}{\partial y}(x,y) = 2y, \qquad \frac{\partial F}{\partial y}(-1,1) = 2 \neq 0
            \end{equation*}
            Y despejando de $F(x,y) = 0$ vemos que:
            \begin{equation*}
                y = \pm \sqrt{-x}, \qquad x\in \mathbb{R}^-
            \end{equation*}
            Derivando $y^2+x=0$ respecto $x$ para buscar la ecuación diferencial (y por tanto su solución):
            \begin{equation*}
                2yy'+1 = 0 \quad\Longrightarrow\quad y' = -\frac{1}{2y}
            \end{equation*}
    \end{enumerate}
\end{ejercicio}

\begin{ejercicio}
    Consideramos el problema $\dot{x} = X(t,x)$, donde $X:D\subset \mathbb{R}\times\mathbb{R}^d\to \mathbb{R}^d$ con $X\in C(D,\mathbb{R}^d)$. Tenemos además una solución $x:I\to \mathbb{R}^d$ de la misma. Se pide:
    \begin{enumerate}
        \item Probar que $x\in C^1(I,\mathbb{R}^d)$.

            Como $x$ es solución, tenemos entonces que es derivable, luego en particular continua. Por ser solución, será:
            \begin{equation*}
                \dot{x}(t) = X(t,x(t)), \qquad \forall t\in I
            \end{equation*}
            Tenemos entonces que:
            \begin{equation*}
                t\longmapsto (t,x(t)) \stackrel{X}{\longmapsto} \dot{x}(t)
            \end{equation*}
            Con $x$ y $X$ son continuas, tenemos entonces que $\dot{x}$ es continua, luego $x\in C^1(I,\mathbb{R}^d)$.
        \item Supuesto que $X\in C^k(D,\mathbb{R}^d)$ para $k\geq 1$, probar que $x\in C^{k+1}(I,\mathbb{R}^d)$. % Fácil
        \item ¿Existe $\dot{x} = X(t,x)$ con $X\in C(D,\mathbb{R}^d)$ no de clase $C^1$ pero que $x\in C^2(I,\mathbb{R}^d)$?

            Sea $X:\mathbb{R}^2\to \mathbb{R}$, tomamos:
            \begin{equation*}
                X(t,x) = 3x^{\nicefrac{2}{3}}
            \end{equation*}
            Y una solución es $x(t) = t^3$, que es de hecho de clase $C^\infty(\mathbb{R})$.
    \end{enumerate}
\end{ejercicio}

\begin{ejercicio}
    Dar una sución $f_n:\left]0,1\right[\to \mathbb{R}$ continuas y que convergen uniformemente pero que no son equicontinuas.\\

    \noindent
    Tomamos:
    \begin{equation*}
        f_n(t) = \frac{1}{t} + \frac{1}{n}
    \end{equation*}
    Son continuas, tenemos para todo $t$ que:
    \begin{equation*}
        \left|f_n(t)-\frac{1}{t}\right| = \left|\frac{1}{n}\right|\to 0
    \end{equation*}
    Por lo que convergen uniformemente. Si fuera la sucesión equicontinua, tenemos que para $\varepsilon=\frac{1}{3}$ tenemos que existe $\delta>0$ de forma que si $|t-s|<\delta$ entonces:
    \begin{equation*}
        |f_n(t)-f_n(s)| < \varepsilon\qquad \forall n\in \mathbb{N}, \quad \forall t,s\in \left]0,1\right[
    \end{equation*}
    Sin embargo, si tomamos $N>\frac{1}{\sqrt{\delta}}$ y $n>N$ entonces:
    \begin{equation*}
        \left|\frac{1}{n}-\frac{1}{n+1}\right| = \left|\frac{1}{n^2+n}\right| < \left|\frac{1}{n^2}\right| < \delta
    \end{equation*}
    Y al evaluar vemos que:
    \begin{equation*}
        \left|f_n\left(\frac{1}{n}\right) - f_n\left(\frac{1}{n+1}\right)\right| = 1 > \frac{1}{3}
    \end{equation*}
    Podríamos haberlo hecho también viendo que un término cualquiera de la sucesión no es uniformemente continua, puesto que en dicho caso no es posible que la sucesión sea equicontinua.
\end{ejercicio}

\begin{ejercicio}
    Dadas dos funciones $X_1,X_2\in C^1(\mathbb{R}\times\mathbb{R},\mathbb{R})$ verificando:
    \begin{equation*}
        X_1(t,x) = X_2(t,x) \qquad \forall t\in \mathbb{R}, |x|\leq 2
    \end{equation*}
    Nos dan dos soluciones $x_i:I_i\to \mathbb{R}$, $i \in \{1,2\}$ de los problemas:
    \begin{equation*}
        \dot{x} = X_i(t,x), \qquad x(0) = 0
    \end{equation*}
    ¿Hay algún entorno de $0$ donde coinciden?\\

    \noindent
    Queremos responder que sí aplicando el Teorema de unicidad. Sin embargo, tenemos dos problemas de valores iniciales, por lo que tenemos que tratar de restringir el problema para conseguir un único problema. Sea por tanto $D=\mathbb{R}\times \left]-2,2\right[$ y sea:
    \begin{equation*}
        \overline{X} = X_1\big|_D = X_2\big|_D
    \end{equation*}
    Vemos claramente que $\overline{X}\in C^1(D)$. Planteamos ahora el problema de valores iniciales:
    \begin{equation}\label{eq:nuevo_problema}
        \dot{x} = \overline{X}(t,x), \qquad x(0) = 0
    \end{equation}
    Para cada $i \in \{1,2\}$, como $x_i$ era solución de cada problema de valores iniciales, tenemos entonces que cada una es continua y que $I_i^\circ \ni 0$. Por la continuidad de cada función, podemos tomar $I_i^\ast\subset I$ con $0\in (I_i^\ast)^\circ$ de forma que:
    \begin{equation*}
        |x_i(t)| < 2 \qquad \forall t\in I_i^\ast, \quad i \in \{1,2\}
    \end{equation*}
    Así, $x_1$ y $x_2$ son solución del problema~\eqref{eq:nuevo_problema}. Como $\overline{X}\in C^1(D)$, podemos ahora sí aplicar el Teorema de unicidad, obteniendo que:
    \begin{equation*}
        x_1(t) = x_2(t) \qquad \forall t\in I_1^\ast\cap I_2^\ast
    \end{equation*}
\end{ejercicio}

\begin{ejercicio}
    Consideramos un sistema de valores iniciales:
    \begin{equation*}
        \dot{x} = X(t,x), \qquad x(t_0) = x_0
    \end{equation*}
    con $X:D\subset \mathbb{R}\times\mathbb{R}^d\to \mathbb{R}^d$ continuo, $(t_0,x_0)\in D$. Vamos a probar que existe una solución maximal para este problema de valores iniciales.\\

    \noindent
    La demostración es análoga a la demostración del Teorema de Hahn-Banach versión analítica.\\

    \noindent
    El Teorema de Cauchy-Peano nos dice que existe una solución $x:I\to \mathbb{R}^d$ del problema de valores iniciales con $t_0\in I^\circ$. Supondremos que $I$ es un intervalo abierto, pues en otro cacso podemos seguir extendiendo la solución, hasta alcanzar todo $\mathbb{R}$, en cuyo caso sería maximal, o bien un intervalo abierto.\\

    \noindent
    Sea ahora:
    \begin{equation*}
        \Lambda = \left\{\tilde{x}:\tilde{I}\to \mathbb{R}^d :
            \begin{array}{l}
                \tilde{x} \text{\ es solución del problema de valores iniciales} \\
                I\subset \tilde{I} \text{\ intervalo abierto} \\
                \tilde{x}(t) = x(t)
            \end{array}
        \right\}
    \end{equation*}
    Vemos que $\Lambda\neq \emptyset $, pues contiene a la solución $x$. Dados $\tilde{x}_1,\tilde{x}_2\in \Lambda$, diremos que:
    \begin{equation*}
        \tilde{x} \leq \tilde{x} \quad\Longleftrightarrow\quad I_1\subset I_2, \quad \tilde{x}_2(t) = \tilde{x}_1(t) \quad \forall t\in I_1
    \end{equation*}
    Veamos que esto define una relación de orden:
    \begin{itemize}
        \item $\tilde{x}\leq \tilde{x}$ es inmediato.
        \item Dadas $\tilde{x}_1,\tilde{x}_2\in \Lambda$ con $\tilde{x}_1\leq \tilde{x}_2$ y $\tilde{x}_2\leq \tilde{x}_1$, tenemos entonces por doble inclusión que $I_1 = I_2$ y ahora la segunda condición nos dice que $\tilde{x}_1 = \tilde{x}_2$, por lo que es antisimétrica la relación.
        \item Si tenemos $\tilde{x}_1,\tilde{x}_2,\tilde{x}_3\in \Lambda$ con $\tilde{x}_1\leq \tilde{x}_2$ y $\tilde{x}_2\leq \tilde{x}_3$, tenemos entonces que
            \begin{equation*}
                I_1\subset I_2 \subset I_3 \quad\Longrightarrow\quad I_1\subset I_3
            \end{equation*}
            Por otra parte, como $\tilde{x}_1(t) = \tilde{x}_2(t)$ en $I_1$ y $\tilde{x}_2(t)= \tilde{x}_3(t)$ en $I_2\supset I_1$, llegamos a que $\tilde{x}_1(t) = \tilde{x}_3(t)\quad \forall t\in I_1$, de donde $\tilde{x}_1\leq \tilde{x}_3$.
    \end{itemize}
    Comprobemos las hipótesis del Lema de Zorn. Sea:
    \begin{equation*}
        \Omega = \{x_\gamma : I_\gamma\to \mathbb{R}^d, \gamma\in \Gamma\} \subset \Lambda
    \end{equation*}
    un conjunto totalmente ordenado, definimos:
    \begin{equation*}
        I_\ast = \bigcup_{\gamma \in  \Gamma} I_\gamma
    \end{equation*}
    Que es abierto como unión arbitraria de abiertos y además tenemos que es un intervalo como unión de intervalos que se cortan en un punto. Definimos ahora $x_\ast:I_\ast\to \mathbb{R}^d$ dada por:
    \begin{equation*}
        x_\ast(t) = x_\gamma(t), \qquad \text{si\ } t\in I_\gamma
    \end{equation*}
    Vemos que $x_\ast$ está bien definida, pues si tomamos $t\in I_\ast$ vemos que existe $\gamma\in \Gamma$ de forma que $t\in I_\gamma$, podría existir $\tilde{\gamma}\in \Gamma$ y de forma que $t\in I_{\tilde{\gamma}}$. Como $\Omega$ es totalmente ordenado, suponemos sin pérdida de generalidad que $x_{\tilde{\gamma}} \leq x_\gamma$, de donde $I_{\tilde{\gamma}}\subset I_\gamma$ y además:
    \begin{equation*}
        x_\gamma(t) = x_{\tilde{\gamma}}(t) \qquad \forall t\in I_{\tilde{\gamma}}
    \end{equation*}
    Falta ver ahora que $x_\ast \in \Lambda$ y que $x_\ast \geq x_\gamma$ para todo $\gamma\in \Gamma$, siendo esto último claro por la definición de $x_\ast$.\\

    \noindent
    Sea $t_1\in I_\ast$, existe $\gamma\in \Gamma$ de forma que $t_1\in I_\gamma$, tenemos que $x_\ast(t) = x_\gamma(t)$ para todo $t\in I_\gamma$. Si restringimos $x_\ast$ a $I_\gamma$ tenemos:
    \begin{equation*}
        (x_\ast)\big|_{I_\gamma} = x_\gamma
    \end{equation*}
    Como $I_{\gamma}$ es abierto y $x_\gamma$ es derivable, tenemos que $(x_\ast)\big|_{I_\gamma}$ es también derivable, con lo que:
    \begin{equation*}
        (\dot{x}_\ast)\big|_{I_\gamma}(t) = \dot{x}_\gamma(t) = X(t,x_\gamma(t)) \qquad \forall t\in I_\gamma
    \end{equation*}
    Aplicando el carácter local de la derivabilidad, vemos que $x_\ast$ es derivable en $t_1$ y:
    \begin{equation*}
        \dot{x}_\ast(t_1) = X(t,x_\gamma(t_1)) = X(t,x_\ast(t_1))
    \end{equation*}
    Y es claro que $x_\ast(t_0) = x_0$ por definición de $x_\ast$.\\

    \noindent
    Así, todo conjunto totalmente ordenado tiene una cota superior, lo que nos permite aplicar el Lema de Zorn, obteniendo que $\Lambda$ tiene un elemento maximal, existe $\overline{x}:J\to \mathbb{R}^d$ elemento maximal de $\Lambda$, con lo que $\overline{x}$ es solución del problema de valores iniciales, y veamos que tiene que ser maximal (en el sentido de solución maximal de ecuaciones diferenciales).\\

    \noindent
    Por reducción al absurdo, supongamos que $\overline{x}$ no es solución maximal del problema de valores iniciales, por lo que podemos extenderla a un intervalo abierto (ya que si no sería prolongable y podemos tomar su prolongación hasta obtener una prolongable) que contenga estrictamente a $J$, es decir, existe $\tilde{x}:\tilde{J}\to \mathbb{R}^d$ solución del problema de valores iniciales con $J\subsetneq \tilde{J}$ y $\tilde{x}(t) = \overline{x}(t) \quad \forall t\in J$.\\

    \noindent
    Así, vemos que $\tilde{x}\in \Lambda$, de donde $\tilde{x}>\overline{x}$, contradicción con que $\overline{x}$ era maximal para la relación de orden enunciada.
\end{ejercicio}

    \end{equation*}

    \end{equation*}

    \end{align*}

\begin{ejercicio}
    Se supone que $\{x_n\}_{n\geq 1}$ es una sucesión de funciones en $C^1[-1,1]$ que cumplen, para cada $t\in [-1,1]$,
    \begin{equation*}
        |\dot{x}_n(t)| \leq 6, \qquad \left|x_n(t) - 7 - 3 \int_{0}^{t} \frac{x_n(s)}{1+x_n(s)^2}~ds \right| \leq \frac{1}{n}
    \end{equation*}
    \begin{enumerate}
        \item[$i)$] $\exists \{x_{\sigma(n)}\}\to x$ convergente uniformemente.

            La primera condición nos dice que la sucesión es equicontinua, puesto que por el Teorema del Valor Medio:
            \begin{equation*}
                |x_n(t) - x_n(s)| = |\dot{x}_n(t-s)| \leq 6 |t-s|
            \end{equation*}
            Ahora, para ver que es uniformemente acotada vemos que:
            \begin{equation*}
                |x_n(0) - 7| \leq \frac{1}{n} \leq 1
            \end{equation*}
            de donde $|x_n(0)|\leq 8$. Ahora:
            \begin{equation*}
                |x_n(t) - x_n(0)| \leq 6 |t| \leq 6 \quad\Longrightarrow\quad|x_n(t)| \leq |x_n(0)| + |x_n(t) - x_n(0)| \leq 8 + 6 = 14
            \end{equation*}
            El Teorema de Ascoli-Arzelá nos da el primer apartado.
        \item[$ii)$] PVI para $x(t)$.

            Repitiendo la demostración del Teorema de existencia, tenemos que $\{x_{\sigma(n)}\}\to x$ para cierto $x$, con $x$ continua y la condición:
            \begin{equation*}
                \left|x_{\sigma(n)}(t) - 7 - 3 \int_{0}^{t} \frac{x_{\sigma(n)}(s)}{1+x_{\sigma(n)}(s)^2}~ds \right| \leq \frac{1}{\sigma(n)}
            \end{equation*}
            Es claro que $\{x_{\sigma(n)}(t) - 7\} \to x(t) - 7$. Tenemos que estudiar:
            \begin{equation*}
                \int_{0}^{t} \frac{x_{\sigma(n)}(s)}{1+x_{\sigma(n)}(s)^2}~ds 
            \end{equation*}
            Vemos que la función $g:\mathbb{R}\to \mathbb{R}$
            \begin{equation*}
                g(\xi) = \frac{\xi}{1+\xi^2}
            \end{equation*}
            es continua en $\mathbb{R}$, por lo que $\{g(x_{\sigma(n)}(t))\}\to g(x(t))$
            \begin{equation*}
                g(\xi) \leq \frac{1}{2}\qquad \forall \xi \in \mathbb{R}
            \end{equation*}
            Por lo que podemos aplicar el Teorema de la convergencia dominada, obteniendo así que:
            \begin{equation*}
                \int_{0}^{t} \frac{x_{\sigma(n)}(s)}{1+x_{\sigma(n)}(s)^2}~ds \to \int_{0}^{t} \frac{x(s)}{1+x(s)^2}~ds  \qquad \forall t\in [-1,1]
            \end{equation*}
            Así, obtenemos que:
            \begin{equation*}
                \left|x(t) - 7 - 3 \int_{0}^{t} \frac{x(s)}{1+x(s)^2}~ds \right| \leq 0
            \end{equation*}
            Por lo que $x(t)$ cumple la ecuación integral de Volterra:
            \begin{equation*}
                x(t) = 7+3\int_{0}^{t} \frac{x(s)}{1+x(s)^2}~ds  \qquad \forall t\in [-1,1]
            \end{equation*}
            Sabemos además que $x$ es continua por ser límite uniforme de funciones continuas, de donde $x$ es solución del PVI:
            \begin{equation*}
                \dot{x} = \frac{3x}{1+x^2}, \qquad x(0) = 7
            \end{equation*}
        \item[$iii)$] $\{x_n\}$ converge uniformemente.

            Tenemos un PVI con $D=\mathbb{R}\times \mathbb{R}$ y $X(t,x) = \frac{3x}{1+x^2}\in C^1[-1,1]$, con lo que hay una única solución maximal. Ahora, vemos que:
            \begin{equation*}
                |X(t,x)| \leq \frac{3}{2}, \qquad \forall (t,x)\in D
            \end{equation*}
            Aplicando el Teorema de crecimiento lineal, tenemos que esta solución única está definida en todo $[-1,1]$ (aunque con lo anterior ya teníamos que $x$ estaba definida en $[-1,1]$). El resultado viene de un argumento clásico: unicidad + Arcoli nos da la convergencia. La sucesión converge uniformemente porque tiene un único punto límite, $x$ es un punto límite y debe ser el único, pues el PVI tiene solución única. Podemos verlo por reducción al absurdo, si $\{x_n\}$ no converge uniformemente a $x$, existirá $\delta>0$  y $\tau:\mathbb{N}\to \mathbb{N}$ de forma que:
            \begin{equation*}
                \|x_{\tau(n)} - x \|_\infty \geq \delta
            \end{equation*}
            Pero tenemos que existe una parcial $\hat{\tau}$ de forma que $x_{\hat{\tau(n)}}\to x$ converge uniformemente, contradicción.
    \end{enumerate}
\end{ejercicio}

\begin{ejercicio}
    Tenemos un problema $\dot{x} = X(t,x)$ con condición inicial $x(0) = x_0$ donde $X:\mathbb{R}\times \left]0,\infty\right[\to \mathbb{R}$ continuo y tal que existen $m,n:\mathbb{R}\to \mathbb{R}$ continuas tales que:
    \begin{equation*}
        |X(t,x)| \leq m(t) |x| + n(t) \qquad \forall (t,x)\in \mathbb{R}\times\mathbb{R}^+
    \end{equation*}
    Pero no podemos llegar a las mismas tesis que el otro Teorema, pues ya no solo podría no explotar (controlado por el crecimiento lineal) sino acercarse a la frontera. Para evitar esto segundo, tendríamos que suponer además la hipótesis:
    \begin{equation*}
        \exists \varepsilon > 0 ~:~ X(t,x) >0 \quad \text{si}\quad x\in \left]0,\varepsilon\right], \quad t>0
    \end{equation*}
    Para así obtener $\omega = +\infty$.
    \begin{proof}
        Si $\omega < \infty$, el Teorema de continuación nos dice que existe $t_n\to \omega$ de forma que $x(t_n)\to \xi$ con $(\omega,\xi)\in \delta D$ o bien $\lim_{t\to\omega^-}x(t) = \infty$.
        \begin{itemize}
            \item Para la primera, observamos que $\delta D = \mathbb{R}\times \{0\}$, por lo que $\xi = 0$, como $x_0 > 0$ el Teorema de Bolzano extendido nos dice que tiene que existir $t_\ast \in \left]0,\omega\right[$ tal que $x(t_\ast) = \varepsilon$ y de forma que $x(t)>\varepsilon \quad t \in \left[0,t_\ast\right[$. Por esta propiedad debemos tener:
                \begin{equation*}
                    0 \geq \dot{x}(t_\ast) = X(t_\ast,x(t_\ast)) = X(t_\ast,\varepsilon) > 0
                \end{equation*}
                Contradicción.
            \item Suponiendo lo segundo, la solución debe cumplir la ecuación integral de Volterra:
                \begin{equation*}
                    x(t) = x_0 + \int_{0}^{t} X(s,x(s))~ds 
                \end{equation*}
                Tomamos:
                \begin{equation*}
                    M = \max_{[0,\omega]}|m(t)|, \qquad N = \max_{[0,\omega]}|n(t)|
                \end{equation*}
                de donde:
                \begin{align*}
                    |x(t)| &\leq |x_0| + \int_{0}^{t} [M(s)|x(s)| + n(s)]~ds  \leq |x_0| + \int_{0}^{t} [M|x(s)| + N]~ds  \\
                           &\leq |x_0| + N\omega + M\int_{0}^{t} |x(s)|~ds
                \end{align*}
                Por lo que aplicando el Lema de Gronwall:
                \begin{equation*}
                    |x(t)| \leq (|x_0| + N\omega) e^{Mt} \leq (|x_0| + N\omega) e^{N\omega} \qquad \forall t\in \left[0,\omega\right[
                \end{equation*}
        \end{itemize}
    \end{proof}
\end{ejercicio}
