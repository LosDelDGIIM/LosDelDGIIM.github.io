    \begin{ejercicio}
        Para cada $n \geq 1$ la solución del problema de valores iniciales
        \[
        \ddot{\theta} + 7 \sen \theta = 0, \quad \theta(0) = \pi - \frac{1}{n}, \quad \dot{\theta}(0) = \frac{1}{n}
        \]
        se denota por $\theta_n(t)$. Calcula, si existe, $\lim\limits_{n \to \infty} \dot{\theta}_n(3\pi)$.
    \end{ejercicio}

    \begin{ejercicio}
        Se considera la ecuación diferencial
        \[
        x' = |x|x.
        \]
        ¿Hay unicidad del problema de Cauchy para toda condición inicial $(t_0, x_0) \in \bb{R}^2$? En caso afirmativo encuentra la solución general.
    \end{ejercicio}

    \begin{ejercicio}
        La solución del problema
        \[
        x' = x + y \sen x, \quad y' = -7y + x \sen y, \quad x(0) = \veps, \quad y(0) = 3\veps
        \]
        se denota por $(x(t; \veps), y(t; \veps))$. Calcula, si existe, el siguiente límite,
        \[
        \lim\limits_{\veps \to 0} \frac{1}{\veps} \left[y(t; \veps) - y(t; 0)\right].
        \]
    \end{ejercicio}

    \begin{ejercicio}
        Se supone que $f : \left[0, 1\right] \to \bb{R}$ es una función continua para la que la ecuación
        \[
        f(x) = 0
        \]
        tiene una única solución $x_* \in \left[0, 1\right]$. Nos dan dos sucesiones de números en el intervalo $\left[0, 1\right]$, $\{x_n\}$ y $\{y_n\}$, que cumplen $f(x_n) = y_n$. Demuestra que si $\{y_n\} \to 0$ entonces $\{x_n\} \to x_*$.
    \end{ejercicio}

    \begin{ejercicio}
        Dada una función $U \in C^1(\bb{R}^2)$ se considera el sistema
        \[
        \ddot{x} = \del{U}{x}(x, y), \quad \ddot{y} = \del{U}{y}(x, y).
        \]
        \begin{enumerate}[label=\alph*)]
            \item Dada una solución $(x(t), y(t))$ definida en un intervalo $I$, demuestra que existen números $a, b, c \in \bb{R} \setminus \{0\}$ tales que la función
            \[
            a\dot{x}(t)^2 + b\dot{y}(t)^2 + cU(x(t), y(t))
            \]
            es constante en $I$.
            \item Se supone que $U(x, y) \leq -3$ para cada $(x, y) \in \bb{R}^2$. Prueba que todas las soluciones maximales están definidas en $\left]-\infty, +\infty\right[$.
        \end{enumerate}
    \end{ejercicio}
