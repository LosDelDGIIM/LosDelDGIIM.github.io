    \begin{ejercicio}
        Demuestra que la solución maximal del problema de valores iniciales $x'' + x + x^3 = 0$, $x(0) = 1$, $x'(0) = 0$ está definida en $\left]-\infty, +\infty\right[$.
    \end{ejercicio}

    \begin{ejercicio}
        Resuelve el problema de valores iniciales
        \[
        x' = \frac{1}{x}, \quad x(t_0) = x_0,
        \]
        con $x_0 \neq 0$. Determina en cada caso el intervalo maximal.
    \end{ejercicio}

    \begin{ejercicio}
        Se considera el problema de valores iniciales
        \[
        x'' + e^x - 1 = 0, \quad x(0) = x_0, \quad x'(0) = 0.
        \]
        Demuestra que si $|x_0|$ es suficientemente pequeño entonces hay una solución bien definida en el intervalo $\left[-10, 10\right]$.
    \end{ejercicio}

    \begin{ejercicio}
        Se considera el problema de valores iniciales
        \[
        x'_1 = t|x_1|x_1 + x_2, \quad x'_2 = x_1 + x_2, \quad x_1(0) = 0, \quad x_2(0) = 0.
        \]
        ¿Hay unicidad?
    \end{ejercicio}

    \begin{ejercicio}
        Para cada $n = 1, 2, \dots$ se considera la sucesión de funciones dada por $x_n : \left[0, 1\right] \to \bb{R}$, que cumplen
        \[
        x_n(t) = 0, \ t \in \left[0, \frac{1}{n}\right], \quad x_n(t) = \int_0^{t-\frac{1}{n}} \cos x_n(s) \, ds, \ t \in \left[\frac{1}{n}, 1\right].
        \]
        \begin{enumerate}[label=\alph*)]
            \item Demuestra que cada $x_n$ está bien definida y es continua.
            \item Demuestra que $\{x_n\}$ es uniformemente acotada y equicontinua.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        La solución general del sistema
        \[
        x'_1 = tx_1^2 + 9x_2, \quad x'_2 = x_1
        \]
        se denota por $\chi(t; t_0, \chi_0)$ donde $\chi$ y $\chi_0$ son los vectores columna $\chi = (x_1, x_2)^t$, $\chi_0 = (x_{01}, x_{02})^t$. Calcula la diferencial de la aplicación $\chi_0 \in \bb{R}^2 \mapsto \chi(t; t_0, \chi_0) \in \bb{R}^2$ en $t = 1$, $t_0 = 0$, $\chi_0 = (0,0)^t$.
    \end{ejercicio}
