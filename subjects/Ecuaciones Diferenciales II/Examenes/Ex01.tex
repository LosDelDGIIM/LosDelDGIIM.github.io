    \begin{ejercicio}
        Se considera el problema $\dot{x} = g(x)$, $x(0) = 1$ donde
        \[
        g : \bb{R} \longrightarrow \bb{R}, \quad g(x) = \begin{cases} 
            \dfrac{x^2}{2} & \text{si } x \geq 1 \\[2ex]
            \dfrac{1}{2} & \text{si } x < 1.
        \end{cases}
        \]
        Calcula su solución maximal ¿es única?
    \end{ejercicio}

    \begin{ejercicio}
        Resuelve cada uno de los siguientes problemas:
        \begin{enumerate}[label=\alph*)]
            \item $\dot{x} = x^{\nicefrac{1}{5}}$, $x(0) = 0$
            \item $x(t) = 2 + 4 \displaystyle\int_3^t x(s) \, ds$
            \item $\dot{x} = tx$, $x(0) = 1$, $x(2) = 0$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Se considera una sucesión de funciones $\{f_n\}$ donde cada $f_n \in C^1([0, 1], \bb{R}^d)$ cumple
        \[
        \|f'_n(t)\| \leq 7 \quad \text{para cada } t \in [0, 1].
        \]
        Demuestra que la sucesión es equicontinua. ¿Es siempre uniformemente acotada?
    \end{ejercicio}

    \begin{ejercicio}
        Se considera el problema de valores iniciales
        \[
        (PC_1) \quad \dot{x} = x + f(t, x), \quad x(0) = x_0
        \]
        con $f : \left]-1, 1\right[ \times \bb{R}^d \longrightarrow \bb{R}^d$ continua y acotada. Dado $\veps \in \left]0, 1\right[$ se define la función
        \[
        x_{\veps}(t) = \begin{cases} 
            e^t x_0 & \text{si } t \in [-\veps, \veps] \\[1ex]
            e^t x_0 + \displaystyle\int_{\veps}^t e^{t-s} f(s, x_{\veps}(s)) \, ds & \text{si } t \in \left]\veps, 1\right[ \\[2ex]
            e^t x_0 + \displaystyle\int_{-\veps}^t e^{t-s} f(s, x_{\veps}(s)) \, ds & \text{si } t \in \left]-1, -\veps\right[.
        \end{cases}
        \]
        Demuestra que existe una sucesión $\{\veps_n\}$ de manera que $x_{\veps_n}$ converge uniformemente en $]-1, 1[$ a una solución de $(PC_1)$.
    \end{ejercicio}

    \begin{ejercicio}
        Sea $f_n : \bb{R}^2 \longrightarrow \bb{R}$, $(x, y) \mapsto f_n(x, y)$ una sucesión de funciones que cumplen
        \[
        |f_n(x, y)| \leq 2 \quad \text{para cada } n \in \bb{N}, \, (x, y) \in \bb{R}^2.
        \]
        Demuestra que existe un subconjunto denso $D$ de $\bb{R}^2$, una función $f : D \longrightarrow \bb{R}$ y una sucesión parcial $\{f_{\sigma(n)}\}$ tales que $f_{\sigma(n)}$ converge a $f$ puntualmente en el conjunto $D$.
    \end{ejercicio}
