    \begin{ejercicio}[3 puntos]
        Sea $X = C([0,1],\mathbb{R})$ el espacio vectorial de las funciones continuas de $[0,1]$ a $\mathbb{R}$. Se define
        \begin{equation*}
            \|f\|_1 := \int_{0}^{1} |f(t)|~dt , \qquad \|f\|_\infty := \sup\{|f(t)| : t\in [0,1]\}\qquad \forall f\in X
        \end{equation*}
        y se escribe $X_1 = (X,\|\cdot \|_1)$, $X_{\infty} = (X,\|\cdot \|_\infty)$. Definimos el funcional lineal $\delta:X\to \mathbb{R}$ por
        \begin{equation*}
            \delta(f) = f(0) \qquad \forall f\in X
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[0.75 puntos]} ¿Son equivalentes $\|\cdot \|_1$ y $\|\cdot \|_\infty$ en $X$? Justifica tu respuesta.
            \item \textbf{[0.75 puntos]} ¿Es $X_1$ completo? Justifica tu respuesta.
            \item \textbf{[0.75 puntos]} Demuestra que $\delta\in X_\infty^\ast$ y calcula su norma.
            \item \textbf{[0.75 puntos]} ¿Es $\delta$ continuo respecto a $\|\cdot \|_1$? Justifica tu respuesta.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[3 puntos]
        Sea $X = C([0,1],\mathbb{R})$ el espacio vectorial de las funciones continuas de $[0,1]$ a $\mathbb{R}$. Se define
        \begin{equation*}
            \|f\|_1 := \int_{0}^{1} |f(t)|~dt , \qquad \|f\|_\infty := \sup\{|f(t)| : t\in [0,1]\}\qquad \forall f\in X
        \end{equation*}
        y se escribe $X_1 = (X,\|\cdot \|_1)$, $X_{\infty} = (X,\|\cdot \|_\infty)$. Definimos el funcional lineal $\delta:X\to \mathbb{R}$ por
        \begin{equation*}
            \delta(f) = f(0) \qquad \forall f\in X
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[0.75 puntos]} ¿Son equivalentes $\|\cdot \|_1$ y $\|\cdot \|_\infty$ en $X$? Justifica tu respuesta.

            No pueden ser equivalentes: si fueran equivalentes tendríamos que las topologías que da cada norma serían iguales, pero sin embargo tenemos que $\delta$ es continua para $\|\cdot \|_\infty$ y no es continua para $\|\cdot \|_1$, por lo que sus topologías no pueden contener los mismos abiertos (recordamos que $\delta$ es continua si y solo si la preimagen de todo abierto de $\mathbb{R}$ es un abierto en la topología que consideramos en $X$), por lo que las dos normas no pueden ser equivalentes.
            
            \item \textbf{[0.75 puntos]} ¿Es $X_1$ completo? Justifica tu respuesta.

            No es completo, si consideramos la sucesión $\{f_n\}$ de funciones de $X$ donde cada $f_n$ es la función que en el intervalo $[0,\nicefrac{1}{n}]$ une el punto $(0,1)$ con el $(\nicefrac{1}{n},0)$ y en el intervalo $[\nicefrac{1}{n},1]$ es constantemente igual a 0, tendremos que la gráfica de cada función es:

                    axis lines=middle,
                    xmin=0, xmax=1.1,
                    ymin=0, ymax=3.2,
                    xtick={0,1/3+0.11,1},

                % Parte constante [1/3,1]
                \addplot[blue,domain=1/3+0.11:1]{0};
                \end{axis}
                \end{tikzpicture}
                \end{figure}
        
            \item \textbf{[0.75 puntos]} Demuestra que $\delta\in X_\infty^\ast$ y calcula su norma.
                Veamos que $\delta\in X_\infty^\ast$:
                \begin{itemize}
                    \item $\delta$ es lineal, pues si $f,g\in X$ y $\lm\in \mathbb{R}$ tenemos que:
                        \begin{equation*}
                            \delta(\lm f + g) = (\lm f + g)(0) = \lm f(0) + g(0) = \lm\delta(f) + \delta(g)
                        \end{equation*}
                    \item $\delta$ es continuo, pues si $f\in X$ tenemos:
                        \begin{equation*}
                            |\delta(f)| = |f(0)| \leq \sup_{t\in [0,1]}|f(t)| = \|f\|_\infty
                        \end{equation*}
                \end{itemize}
                Por lo que $\delta\in X_\infty^\ast$. De hecho, en el último apartado hemos probado además que $\|\delta\|\leq 1$. Veamos que $\|\delta\| = 1$, puesto que si consideramos cualquier función $f\in X$ de forma que $\max\limits_{t\in [0,1]}|f(t)| = |f(0)|$ tenemos entonces que:
                \begin{equation*}
                    |\delta(f)| = |f(0)| = \max_{t\in [0,1]}|f(t)| = \|f\|_\infty
                \end{equation*}
            
            \item \textbf{[0.75 puntos]} ¿Es $\delta$ continuo respecto a $\|\cdot \|_1$? Justifica tu respuesta.

            Si $\delta$ fuera continuo respecto a $\|\cdot \|_1$, en particular será continua en la función constantemente igual a $0$, por lo que:
            \begin{equation*}
                \forall \varepsilon>0~\exists \eta>0 : \|f\|_1 < \eta \Longrightarrow |\delta(f)|<\varepsilon
            \end{equation*}
            Es decir, si $\delta$ fuera continua podríamos acotar el valor de cualquier función $f\in X$ en $0$ sabiendo acotar el valor de su integral. No parece que esto sea posible, por lo que tratamos de probar que $\delta$ no es continua. Para ello, buscamos probar que:
            \begin{equation*}
                \exists \varepsilon>0 : \forall \eta>0~\exists f\in X \quad \text{con}\quad \|f\|_1<\eta \quad \text{y}\quad |\delta(f)|>\varepsilon
            \end{equation*}
            Si consideramos $\varepsilon=\nicefrac{1}{2}$ y nos dan $\eta>0$, si consideramos la función cuya gráfica es:
            \begin{figure}[H]
                \centering
            \begin{tikzpicture}
            \begin{axis}[
                axis lines=middle,
                xmin=0, xmax=1.1,
                ymin=0, ymax=3.2,
                xtick={0,1/3+0.11,1},
                ytick={3},
                xticklabels={$0$,$\eta$,$1$},
                yticklabels={$1$},
                width=10cm,
                height=6cm,
                samples=100,
                domain=0:1,
                thick
            ]

            % Parte lineal [0,1/3]
            \addplot[blue,domain=0:1/3+0.11]{3 - (3 - 3/4)/(1/3)*x};

            % Parte constante [1/3,1]
            \addplot[blue,domain=1/3+0.11:1]{0};
            \end{axis}
            \end{tikzpicture}
            \end{figure}
            \noindent
            Es decir, que en el intervalo $[0,\eta]$ es la recta que une el punto $(0,1)$ con el $(\eta,0)$ y en el intervalo $[\eta,1]$ vale $0$, tenemos que $f\in X$, así como que:
            \begin{equation*}
                \|f\|_1 = \int_{0}^{1} |f(t)|~dt  = \int_{0}^{1} f(t)~dt  = \int_{0}^{\eta}f(t) ~dt  + \int_{\eta}^{1} f(t)~dt  = \frac{\eta}{2} < \eta
            \end{equation*}
            (ya que el área del triángulo es base por altura entre 2) y tenemos que:
            \begin{equation*}
                |\delta(f)| = |f(0)| = 1 > \frac{1}{2}
            \end{equation*}
            Acabamos de probar que $\delta$ no es continua para $\|\cdot \|_1$.

        \end{enumerate}        
    \end{ejercicio}

    \begin{ejercicio}[4 puntos]
        Sean $1<p<\infty$ ($p'=\nicefrac{p}{p-1}$), $\emptyset\neq\Omega\subset\mathbb{R}^N$ un conjunto medible y
        $T:L^{p'}(\Omega)\to L^p(\Omega)^\ast$ definido por
        \begin{equation*}
        \langle Tu,f\rangle := \int_\Omega u f,
        \qquad
        \forall u\in L^{p'}(\Omega),\ \forall f\in L^p(\Omega).
        \end{equation*}
        
        \begin{enumerate}[label=\alph*)]
        
        \item \textbf{Prueba que $\|Tu\|_{L^p(\Omega)^\ast}=\|u\|_{L^{p'}(\Omega)}$ y que $T$ es inyectivo.}
        
        Sea $u\in L^{p'}(\Omega)$. Por definición de la norma en el dual,
        \begin{equation*}
        \|Tu\|_{(L^p(\Omega))^\ast}
        =
        \sup_{\|f\|_{L^p(\Omega)}\le 1}
        \left|\int_\Omega u f\right|.
        \end{equation*}
        Por la desigualdad de Hölder,
        \begin{equation*}
        \left|\int_\Omega u f\right|
        \le
        \|u\|_{L^{p'}(\Omega)}\,\|f\|_{L^p(\Omega)},
        \end{equation*}
        y por tanto
        \begin{equation*}
        \|Tu\|_{(L^p(\Omega))^\ast}\le \|u\|_{L^{p'}(\Omega)}.
        \end{equation*}
        
        Veamos ahora la desigualdad contraria. Si $u=0$ es trivial. Supongamos $u\neq 0$ y definimos
        \begin{equation*}
        f_0:=u\,|u|^{p'-2}=\operatorname{sgn}(u)\,|u|^{p'-1}.
        \end{equation*}
        Primero probamos que $f_0\in L^p(\Omega)$. En efecto,
        \begin{equation*}
        |f_0|^p = |u|^{p(p'-1)}.
        \end{equation*}
        Como $p'=\frac{p}{p-1}$, se tiene $p'-1=\frac{1}{p-1}$ y por tanto
        \begin{equation*}
        p(p'-1)=\frac{p}{p-1}=p'.
        \end{equation*}
        Luego
        \begin{equation*}
        \int_\Omega |f_0|^p=\int_\Omega |u|^{p'}<\infty,
        \end{equation*}
        y así $f_0\in L^p(\Omega)$. Además,
        \begin{equation*}
        \|f_0\|_{L^p(\Omega)}^p=\int_\Omega |u|^{p'}=\|u\|_{L^{p'}(\Omega)}^{p'}
        \quad\Longrightarrow\quad
        \|f_0\|_{L^p(\Omega)}=\|u\|_{L^{p'}(\Omega)}^{\frac{p'}{p}}
        =\|u\|_{L^{p'}(\Omega)}^{p'-1}.
        \end{equation*}
        
        Ahora evaluamos el funcional en $f_0$:
        \begin{equation*}
        \int_\Omega u f_0=\int_\Omega u\,u\,|u|^{p'-2}=\int_\Omega |u|^{p'}
        =\|u\|_{L^{p'}(\Omega)}^{p'}.
        \end{equation*}
        Consideramos entonces $\displaystyle f:=\frac{f_0}{\|f_0\|_{L^p(\Omega)}}$, que satisface $\|f\|_{L^p(\Omega)}=1$. Se tiene:
        \begin{multline*}
        \|Tu\|_{(L^p(\Omega))^\ast}
        =
        \sup_{\|g\|_{L^p(\Omega)}\le 1}
        \left|\langle Tu,g\rangle\right|
        =
        \sup_{\|g\|_{L^p(\Omega)}\le 1}
        \left|\int_\Omega u g\right| \\
        \ge
        \left|\int_\Omega u\, f\right|
        =
        \frac{\int_\Omega |u|^{p'}}{\|f_0\|_{L^p(\Omega)}}
        =
        \frac{\|u\|_{L^{p'}(\Omega)}^{p'}}{\|u\|_{L^{p'}(\Omega)}^{p'-1}}
        =
        \|u\|_{L^{p'}(\Omega)}.
        \end{multline*}
        
        Junto con la desigualdad anterior concluimos
        \begin{equation*}
        \|Tu\|_{(L^p(\Omega))^\ast}=\|u\|_{L^{p'}(\Omega)}.
        \end{equation*}
        
        Finalmente, $T$ es una isometría lineal, y sabemos que toda isometría lineal es inyectiva: si $Tu=0$ porque
        \begin{equation*}
        0=\|Tu\|_{(L^p(\Omega))^\ast}=\|u\|_{L^{p'}(\Omega)}
        \quad\Longrightarrow\quad
        u=0 \ \text{a.e.}
        \end{equation*}
        %-------------------------------------------------
        \item \textbf{Usando que $L^p(\Omega)$ es reflexivo, prueba que $T(L^{p'}(\Omega))=L^p(\Omega)^\ast$.}
        
        Por el apartado anterior, $T$ es una isometría lineal. Como $L^{p'}(\Omega)$ es Banach,
        se sigue que $T(L^{p'}(\Omega))$ es Banach y por tanto un subespacio cerrado de $L^p(\Omega)^\ast$.\\
        Para ver que es todo el dual, basta probar que $T(L^{p'}(\Omega))$ es denso en $L^p(\Omega)^\ast$.
        
        Usamos el último corolario del teorema de Hahn--Banach: un subespacio $M$ de un espacio normado $E$ es denso si y solo si el único funcional continuo de $E^\ast$ que se anula en $M$ es el funcional nulo.


        Sea
        $\Phi\in L^p(\Omega)^{\ast\ast}$ tal que
        \begin{equation*}
        \langle \Phi, Tu\rangle = 0
        \qquad \forall u\in L^{p'}(\Omega).
        \end{equation*}
        Como $L^p(\Omega)$ es reflexivo, existe $f\in L^p(\Omega)$ tal que
        $\Phi = J(f)$, donde $J$ es la inmersión canónica. Entonces
        \begin{equation*}
        0
        =
        \langle J(f),Tu\rangle
        =
        \langle Tu,f\rangle
        =
        \int_\Omega u f
        \qquad \forall u\in L^{p'}(\Omega).
        \end{equation*}
        Esto implica que $f=0$ $a.e.$ , luego $\Phi=0$.
        Por el teorema de Hahn--Banach, $T(L^{p'}(\Omega))$ es denso pero además sabíamos que era cerrado,
        por lo que
        \begin{equation*}
        T(L^{p'}(\Omega))=L^p(\Omega)^\ast.
        \end{equation*}
        
        \end{enumerate}
        
        \vspace{2mm}
        
        Ahora supongamos $\Omega=(0,1)$ y definamos
        \begin{equation*}
        g_n(x)=n^{1/p}e^{-nx}, \qquad x\in(0,1).
        \end{equation*}
        
        \begin{enumerate}[label=\alph*),resume]
        
        %-------------------------------------------------
        \item \textbf{Prueba que $g_n\to 0$ a.e. y que $\{g_n\}$ es acotada en $L^p(\Omega)$.}
        
        Para todo $x\in(0,1)$,
        \begin{equation*}
        g_n(x)=n^{1/p}e^{-nx}\xrightarrow[n\to\infty]{}0,
        \end{equation*}
        ya que la exponencial domina a cualquier potencia. Luego $g_n\to 0$ a.e.
        
        Además,
        \begin{equation*}
        \|g_n\|_{L^p(\Omega)}^p
        =
        \int_0^1 n e^{-npx}\,dx
        =
        \frac{1}{p}\left(1-e^{-np}\right)
        \le
        \frac{1}{p}.
        \end{equation*}
        Por tanto, $\{g_n\}$ es acotada en $L^p(\Omega)$.
        
        %-------------------------------------------------
        \item \textbf{Demuestra que $g_n\not\to 0$ en $L^p(\Omega)$.}
        
        Se tiene
        \begin{equation*}
        \|g_n\|_{L^p(\Omega)}
        =
        \left(\frac{1}{p}(1-e^{-np})\right)^{1/p}
        \xrightarrow[n\to\infty]{}
        \frac{1}{p^{1/p}}\neq 0.
        \end{equation*}
        Luego $\|g_n\|_{L^p(\Omega)}$ no converge a $0$, por tanto .
        
        %-------------------------------------------------
        \item \textbf{Usando que $C_c(0,1)$ es denso en $L^{p'}(\Omega)$, prueba que $g_n\to 0$
        en la topología débil $\sigma(L^p(\Omega),L^{p'}(\Omega))$.}

        Recordemos que $g_n \rightharpoonup 0$ en $\sigma(L^p(\Omega),L^{p'}(\Omega))$ significa que
        \[
        \int_0^1 g_n(x)\,u(x)\,dx \longrightarrow 0
        \qquad \forall\,u\in L^{p'}(0,1),
        \]
        es decir, que toda forma lineal continua sobre $L^p$ (identificada con un $u\in L^{p'}$) aplicada a $g_n$
        converge a $0$.
        
        Sea $u\in L^{p'}(\Omega)$ y $\varepsilon>0$. Como $C_c(0,1)$ es denso en $L^{p'}(\Omega)$,
        existe $\varphi\in C_c(0,1)$ tal que
        \begin{equation*}
        \|u-\varphi\|_{L^{p'}(\Omega)}<\varepsilon.
        \end{equation*}
        Entonces,
        \begin{equation*}
        \left|\int_0^1 g_n u\right|
        \le
        \left|\int_0^1 g_n (u-\varphi)\right|
        +
        \left|\int_0^1 g_n \varphi\right|.
        \end{equation*}
        Para el primer término, por Hölder,
        \begin{equation*}
        \left|\int_0^1 g_n (u-\varphi)\right|
        \le
        \|g_n\|_{L^p(\Omega)}\|u-\varphi\|_{L^{p'}(\Omega)}
        \le
        \frac{1}{p^{1/p}}\varepsilon \qquad \forall \varepsilon >0, \forall u \in L^{p'}(\Omega)
        \end{equation*}
        Para el segundo término, como $\varphi$ es continua con soporte compacto y $g_n\to 0$ a.e.,
        por convergencia dominada,
        \begin{equation*}
        \int_0^1 g_n \varphi \xrightarrow[n\to\infty]{} 0.
        \end{equation*}
        Concluimos que
        \begin{equation*}
        \lim_{n\to\infty}\int_0^1 g_n u = 0
        \qquad \forall u\in L^{p'}(\Omega),
        \end{equation*}        
        es decir,
        \begin{equation*}
        g_n \rightharpoonup 0
        \quad \text{en } \sigma(L^p(\Omega),L^{p'}(\Omega)).
        \end{equation*}
        \end{enumerate}
    \end{ejercicio}
