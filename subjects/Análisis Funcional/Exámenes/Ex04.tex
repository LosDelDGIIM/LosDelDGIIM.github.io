    \begin{ejercicio}[3 puntos]
        Sea $X = C([0,1],\mathbb{R})$ el espacio vectorial de las funciones continuas de $[0,1]$ a $\mathbb{R}$. Se define
        \begin{equation*}
            \|f\|_1 := \int_{0}^{1} |f(t)|~dt , \qquad \|f\|_\infty := \sup\{|f(t)| : t\in [0,1]\}\qquad \forall f\in X
        \end{equation*}
        y se escribe $X_1 = (X,\|\cdot \|_1)$, $X_{\infty} = (X,\|\cdot \|_\infty)$. Definimos el funcional lineal $\delta:X\to \mathbb{R}$ por
        \begin{equation*}
            \delta(f) = f(0) \qquad \forall f\in X
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[0.75 puntos]} ¿Son equivalentes $\|\cdot \|_1$ y $\|\cdot \|_\infty$ en $X$? Justifica tu respuesta.
            \item \textbf{[0.75 puntos]} ¿Es $X_1$ completo? Justifica tu respuesta.
            \item \textbf{[0.75 puntos]} Demuestra que $\delta\in X_\infty^\ast$ y calcula su norma.
            \item \textbf{[0.75 puntos]} ¿Es $\delta$ continuo respecto a $\|\cdot \|_1$? Justifica tu respuesta.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[3 puntos]
        Sean $S:c_0\to c_0$ y $T:l_1\to l_1$ operadores lineales y supongamos que
        \begin{equation*}
            \sum_{k=1}^{\infty}[Sx](k)y(k) = \sum_{k=1}^{\infty}x(k)[Ty](k) \qquad \forall x\in c_0, \quad \forall y\in l_1
        \end{equation*}
        Demuestra que $S$ y $T$ son continuos.
    \end{ejercicio}

    \begin{ejercicio}[4 puntos]
        Sean $1<p<\infty$ ($p' = \nicefrac{p}{p-1}$), $\emptyset \neq \Omega\subset \mathbb{R}^N$ un subconjunto medible y $T:L^{p'}(\Omega)\to L^p(\Omega)^\ast$ definida mediante
        \begin{equation*}
            \langle Tu,f \rangle  = \int_\Omega uf \qquad \forall u\in L^{p'}(\Omega), \quad \forall f\in L^p(\Omega)
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[1 punto]} Prueba que $\|Tu\|_{L^p(\Omega)^\ast} = \|u\|_{L^{p'}(\Omega)}$ y que $T$ es inyectivo.
            \item \textbf{[1 punto]} Usando que $L^p(\Omega)$ es reflexivo, prueba que $T(L^{p'}(\Omega)) = L^p(\Omega)^\ast$.
        \end{enumerate}
        Si para $\Omega = \left]0,1\right[$ consideramos la sucesión de funciones $\{g_n\}$ definidas mediante $g_n(x) = n^{\nicefrac{1}{p}}e^{-nx}$ para todo $x\in \left]0,1\right[$:
        \begin{enumerate}[label=\alph*)]
            \item[c)] \textbf{[0.5 puntos]} prueba que $\{g_n\}\to 0$ a.e. y que $\{g_n\}$ es acotada en $L^p(\Omega)$.
            \item[d)] \textbf{[0.5 puntos]} Demuestra que $\{g_n\}\not\to 0$ en $L^p(\Omega)$.
            \item[e)] \textbf{[1 punto]} Usando que $C_c(0,1)$ es denso en $L^{p'}(\Omega)$, prueba que $\{g_n\}\to 0$ en la topología débil $\sigma(L^p(\Omega),L^{p'}(\Omega))$.
        \end{enumerate}
    \end{ejercicio}
