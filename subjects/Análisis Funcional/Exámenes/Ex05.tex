    \begin{ejercicio}[1.5 puntos]
        Sea $X$ el espacio normado de las funciones continuas en $[0,1]$ que se anulan en cero con la norma del máximo:
        \begin{equation*}
            X = \{f\in C[0,1] : f(0) = 0\}, \qquad \|f\|_\infty = \max\{|f(t)| : 0\leq t \leq 1\}
        \end{equation*}
        En $X$ se considera el funcional lineal $\varphi:X\to \mathbb{R}$ dado por:
        \begin{equation*}
            \varphi(f) = \int_{0}^{1} tf(t)~dt \qquad \forall f\in X
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[1 punto]} Calcula $\|\varphi\|$.
            \item \textbf{[0.5 puntos]} Prueba que $\varphi$ no alcanza su norma.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[2 puntos]
        Sea $T:c_0\to l_2$ el operador lineal definido para todo $x\in c_0$ por
        \begin{equation*}
            [Tx](n) = \frac{x(n)}{n}\qquad \forall n\in \mathbb{N}
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[1 punto]} Calcula su norma y prueba que $T$ no la alcanza.
            \item \textbf{[0.5 punos]} Prueba que $Y=T(c_0)$ es un subespacio denso en $l_2$ y que $T$ es una biyección lineal de $c_0$ sobre $Y$.
            \item \textbf{[0.5 puntos]} ¿Es continua la aplicación $T^{-1}:Y\to c_0$? ¿Es $Y$ cerrado en $l_2$?
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[1.5 puntos]
        Sean $X$ un espacio de Banach, $Y$ un espacio normado y $\cc{F}$ un subconjunto de $L(X,Y)$. Prueba que las siguientes afirmaciones son equivalentes:
        \begin{enumerate}[label=\alph*)]
            \item $\cc{F}$ está acotado.
            \item Para cada $x\in X$ y cada $g\in Y^\ast$ el conjunto $\{g(T(x)) : T\in \cc{F}\}$ está acotado.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[2 puntos]
        Responde a las sigueintes cuestiones indicando en cada caso, según proceda, un resultado de teoría que justifique tu respuesta o un contraejemplo, o bien dando una prueba.
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[0.5 puntos]} ¿Puede definirse alguna norma completa en el espacio vectorial $c_0$ de las sucesiones casi nulas?
            \item \textbf{[0.5 puntos]} Si $H$ es un espacio de Hilbert y $M$ es un subespacio de $H$ verificando que $M^\perp = \{0\}$, entonces $M = H$.
            \item \textbf{[0.5 puntos]} Sea $\{f_n\}$ una sucesión puntualmente acotada de funciones lineales continuos en un espacio de Banach $X$, definiendo $T(x) = \{f_n(x)\}$ se obtiene un operador lineal continuo de $X$ en $t_\infty$.
            \item \textbf{[0.5 puntos]} Sean $\|\cdot \|$ y $|\|\cdot |\|$ dos normas completas no equivalentes en un espacio vectorial $X$. ENtonces la aplicación identidad $I_X:(X,\|\cdot \|)\to (X,|\|\cdot |\|)$ no es continua.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[3 puntos]
        Escribe sobre alguno de los siguientes temas:
        \begin{enumerate}[label=\alph*)]
            \item Ortogonalidad. Teorema de la proyección ortogonal. Teorema de Riesz-Fréchet.
            \item Separación de conjuntos convexos en espacios normados.
            \item Lema de categoría de Baire. Teorema de Banach-Steinhaus.
        \end{enumerate}
    \end{ejercicio}
