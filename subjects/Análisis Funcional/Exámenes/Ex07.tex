    \begin{ejercicio}[3 puntos]
        \begin{enumerate}[label=\alph*)]
            \item Enuncia el teorema de la proyección ortogonal.
            \item Enuncia el Teorema de Riesz-Fréchet.
            \item Enuncia la caracterización de las bases ortonormales en un espacio de Hilbert.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[3.5 puntos]
        Sea $X=L_2([0,1])$:
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[0.5 puntos]} Prueba que $M = \left\{x\in X : \int_{0}^{\nicefrac{1}{2}} x(t)~dt = 0 \right\}$ es un subespacio vectorial cerrado de $X$.
            \item \textbf{[1 punto]} Describe $M^\perp$.
            \item \textbf{[1.5 puntos]} Calcula $P_M$ y $P_M^\perp$.
            \item \textbf{[0.5 puntos]} Calcula la mínima distancia de la función $x_0(t) = t$ al subespacio $M$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[3. puntos]
        Sea
        \begin{equation*}
            M = \left\{x\in l_2 : x(1) = 0 \qquad \text{y}\qquad \sum_{n=1}^{\infty}\frac{x(2n)}{n} = 0\right\}
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item\textbf{[0.5 puntos]} Prueba que $M$ es un subespacio cerrado de $l_2$.
            \item\textbf{[1 punto]} Describe $M^\perp$.
            \item \textbf{[1.5 puntos]} Calcula $P_M$ y $P_M^\perp$.
            \item \textbf{[0.5 puntos]} Calcula la mínima distancia en $l_2$ de $e_1+e_3$ a $M$.
        \end{enumerate}
    \end{ejercicio}
