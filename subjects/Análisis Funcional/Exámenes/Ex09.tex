    \textbf{Parte Estándar} (hasta 8 puntos)
    \begin{ejercicio}[5 puntos]
        En el espacio de Banach $C[0,1]$ se considera el funcional lineal $\varphi$ definido por
        \begin{equation*}
            \varphi(f) = \int_{0}^{\nicefrac{1}{2}} f(t)~dt - \int_{\nicefrac{1}{2}}^{1} f(t)~dt \qquad \forall f\in C[0,1]
        \end{equation*}
        Probar que $\varphi$ es continuo y calcular su norma.
    \end{ejercicio}

    \begin{ejercicio}[5 puntos]
        Probar el Teorema de Hausdorff de 1932 (junto con su lema previo): \textit{toda biyección lineal entre espacios normados de dimensión finita es un isomorfismo.}
    \end{ejercicio}

    \noindent
    \textbf{Parte Extra} (hasta 2 puntos)
    \begin{ejercicio}
        Justificar si el funcional $\varphi$ del Ejercicio 1 alcanza o no su norma.
    \end{ejercicio}

    \begin{ejercicio}
        Sea $X$ un espacio normado y $T:X\to Y$ un operador lineal y continuo. Demostrar que si $dim(T(X)) < \infty$ y $\ker T$ es proximal en $X$, entonces $T$ alcanza su norma.
    \end{ejercicio}
