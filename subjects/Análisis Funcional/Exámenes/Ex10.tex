    \begin{ejercicio}[7 puntos]
        Se consiera el operador lineal $T:c_0\to c_0$ definido por
        \begin{equation*}
            T(x)(k) = \frac{k}{k+1}x(k) \qquad \forall k\in \mathbb{N}, \quad \forall x\in c_0
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[3 puntos]} Demuestra que $T$ es continuo y $\|T\|\leq 1$.
            \item \textbf{[3 puntos]} Demuestra que $\|T\| = 1$.
            \item \textbf{[1 punto]} Demuestra que $T$ no alcanza su norma.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[3 puntos]
        Sea $H$ un espacio de Hilbert, sea $u\in H$ con $u\neq 0$ y sea
        \begin{equation*}
            M = \{x\in H : \langle x,u \rangle =0\}
        \end{equation*}
        para cada $x\in H$:
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[2 puntos]} Demuestra que
                \begin{equation*}
                    dist(x,M) = \frac{|\langle x,y \rangle |}{\|u\|}
                \end{equation*}
            \item \textbf{[1 punto]} Demuestra que existe un único punto $y\in M$ tal que $dist(x,M) = \|x-y\|$ y calcula ese punto.
        \end{enumerate}
    \end{ejercicio}
