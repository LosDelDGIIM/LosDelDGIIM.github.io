        \begin{equation*}
            T(x)(k) = \frac{k}{k+1}x(k) \qquad \forall k\in \mathbb{N}, \quad \forall x\in c_0
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[3 puntos]} Demuestra que $T$ es continuo y $\|T\|\leq 1$.
            \item \textbf{[3 puntos]} Demuestra que $\|T\| = 1$.
            \item \textbf{[1 punto]} Demuestra que $T$ no alcanza su norma.
        \end{enumerate}
    \end{ejercicio}

        \begin{enumerate}[label=\alph*)]

        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[7 puntos]
        Se considera el operador lineal $T:c_0\to c_0$ definido por
        \begin{equation*}
            T(x)(k) = \frac{k}{k+1}x(k) \qquad \forall k\in \mathbb{N}, \quad \forall x\in c_0
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[3 puntos]} Demuestra que $T$ es continuo y $\|T\|\leq 1$.

                Recordamos que teníamos la norma:
                \begin{equation*}
                    \|x\| = \sup_{n\in \mathbb{N}}|x(n)|
                \end{equation*}
                Si $x\in c_0$ y $k\in \mathbb{N}$ tenemos que:
                \begin{equation*}
                    |T(x)(k)| = \frac{k}{k+1}|x(k)| \leq |x(k)| \leq \|x\|
                \end{equation*}
                Por lo que:
                \begin{equation*}
                    \|T(x)\| = \sup_{k\in \mathbb{N}}|T(x)(k)| \leq \|x\|
                \end{equation*}
                de donde $\|T\|\leq 1$.
            \item \textbf{[3 puntos]} Demuestra que $\|T\| = 1$.

                Si consideramos la sucesión de sucesiones:
                \begin{equation*}
                    x_n(k) = \left\{\begin{array}{ll}
                        1 & \text{si\ } k\leq n \\
                        0 & \text{si\ } k > n
                    \end{array}\right. 
                \end{equation*}
                Tenemos entonces que $\|x_n\|\leq 1 \quad \forall n\in \mathbb{N}$, así como que:
                \begin{equation*}
                    \|T(x_n)\| = \sup_{k\in \mathbb{N}}\left|\frac{k}{k+1}x_n(k)\right| = \max_{k \in \{1,\ldots,n\}} \left|\frac{k}{k+1}\right| = \frac{n}{n+1} \qquad \forall n\in \mathbb{N}
                \end{equation*}
                Y vemos que $\{\|T(x_n)\|\} = \{\nicefrac{n}{n+1}\}\to 1$, por lo que ha de ser $\|T\| = 1$.
            \item \textbf{[1 punto]} Demuestra que $T$ no alcanza su norma.

                Supuesto que existe $x\in c_0$ de forma que $\|x\| \leq 1$ y $\|Tx\| = 1$, tenemos:
                \begin{equation*}
                    1 = \|Tx\| = \sup_{k\in \mathbb{N}}\left|\frac{k}{k+1}x(k)\right| = \sup_{k \in \mathbb{N}}\left(\frac{k}{k+1}|x(k)|\right)
                \end{equation*}
                Y como $x\to 0$, para $\varepsilon=\nicefrac{1}{2}$ existe $m\in \mathbb{N}$ de forma que:
                \begin{equation*}
                    |x(k)| < \frac{1}{2} \qquad \forall k\geq m
                \end{equation*}
                De donde:
                \begin{equation*}
                    1 = \sup_{k \in \mathbb{N}}\left(\frac{k}{k+1}|x(k)|\right) = \max\left\{\max_{k \in \{1,\ldots,m-1\}}\left(\frac{k}{k+1}|x(k)|\right), \sup_{k\geq m}\left(\frac{k}{k+1}|x(k)|\right)\right\}
                \end{equation*}
                Pero:
                \begin{itemize}
                    \item $\sup_{k\geq m}\left(\frac{k}{k+1}|x(k)|\right) \leq \nicefrac{1}{2}$.
                    \item y tenemos que $\nicefrac{k|x(k)|}{k+1}<1$ para $k \in \{1,\ldots,m-1\}$.
                \end{itemize}
                Por lo que hemos llegado a una contradicción.
        \end{enumerate}
    \end{ejercicio}
