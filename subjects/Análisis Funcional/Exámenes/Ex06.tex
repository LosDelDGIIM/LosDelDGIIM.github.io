    \begin{ejercicio}[3 puntos]
        \begin{enumerate}[label=\alph*)]
            \item Enuncia la caracterización de las aplicaciones lineales y continuas entre espacios normados.
            \item Definición y caracterización de los isomorfismos topológicos entre espacios normados.
            \item Enuncia el Teorema de Riesz.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}[4 puntos]
        Sea $X=C([0,1],\mathbb{R})$. Se dfine $T:X\to X$ mediante
        \begin{equation*}
            T(f)(x) = \int_{0}^{x^3} f(t)~dt  \qquad \forall x\in [0,1]
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item Prueba que $T$ es lineal, continua e inyectiva.
            \item Calcula la norma de $T$.
            \item Prueba que $T$ no es un isomorfismo topológico de $X$ sobre $T(X)$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Sea $\{P_n\}_{n\in \mathbb{N}}$ una sucesión de polinomios de grado menor o igual que 20 tal que, para cada $k \in \{0,1,\ldots,20\}$ la sucesión $\{P_n^{k)}(0)\}_{n\in \mathbb{N}}$ es convergente. Prueba que $\{P_n\}_{n\in \mathbb{N}}$ converge uniformemente en $[-1,1]$.
    \end{ejercicio}
