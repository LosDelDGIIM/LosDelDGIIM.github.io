    \begin{ejercicio}[5 puntos]
        Sean $E$ un espacio de Banach y $\{f_n\}$ una sucesión en el espacio dual $E^\ast$ y $f\in E^\ast$. Probad que
        \begin{equation*}
            \{f_n\}\stackrel{\ast}{\rightharpoonup} f \Longleftrightarrow \{\langle f_n,x \rangle \}\to \langle f,x \rangle  \quad \forall x\in E
        \end{equation*}
    \end{ejercicio}

    \begin{ejercicio}[5 puntos]
        Sean $C\neq \emptyset $ un subconjnto convexo y cerrado de un espacio de Hilbert $H$ y $T:C\to C$ una aplicación verificando
        \begin{equation*}
            \|Tu -Tv\| \leq \|u-v\| \qquad \forall u,v\in C
        \end{equation*}
        \begin{enumerate}[label=\alph*)]
            \item Probad que si existe una sucesión $\{u_n\}\subset C$ verificando
                \begin{equation*}
                    \{u_n\}\rightharpoonup u \qquad \text{y}\qquad \{u_n-Tu_n\}\rightarrow f
                \end{equation*}
                entonces $u-Tu = f$.\newline
                (\textbf{Pista:} Empieza por el caso $C=H$ y usa la desigualdad 
                \begin{equation*}
                    \langle (u-Tu)-(v-Tv),u-v \rangle \geq 0 \qquad \forall u,v)
                \end{equation*}
            \item Deduce que si $C$ es además acotado, entonces $T$ tiene un punto fijo.\newline
                (\textbf{Pista:} Considera $T_\varepsilon u=(1-\varepsilon)Tu + \varepsilon a$ siendo $a\in C$ fijo y $\varepsilon>0$, con $\varepsilon\rightarrow0$.)
        \end{enumerate}
    \end{ejercicio}
