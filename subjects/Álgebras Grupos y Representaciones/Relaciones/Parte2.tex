\begin{ejerciciostarstar}\label{ej25}
    Supongamos $T:V\to V$ un endomorfismo $K$-lineal, donde $V$ es un espacio vectorial de dimensión finita que consideramos, como de costumbre, como un $K[x]$-módulo. Supongamos que el polinomio mínimo $m(x)$ de $T$ es irreducible en $K[x]$. Demostrar que existen $K[x]$-submódulos simples $V_1\ldots, V_t$ tales que $V = V_1\oplus \cdots\oplus V_t$ como $K[x]$-módulo.
\end{ejerciciostarstar}

\begin{ejerciciostarstar}
    En las condiciones del Ejercicio~\ref{ej25}, demostrar que el polinomio característico de $T$ es ${(m(x))}^{t}$. 
\end{ejerciciostarstar}

\begin{ejercicio}
    Supongamos que $M$ es un módulo sobre un anillo $A$. Entonces $End_A(M)$ es un subanillo de $End(M)$, llamado \textit{anillo de endomorfismos del} $A$\textit{-módulo M} (y es una subálgebra de $End_K(M)$ si $A$ es una $K$-álgebra). Si $M = M_1 \oplus \cdots \oplus M_n$, denotemos, para cada $i=1\ldots n$, por $\iota_i:M_i \to M$ la inclusión canónica, y por $\pi_i:M \to M_i $ la aplicación que asigna $\pi_i(m) = m_i $ a cada $m = m_1 + \cdots + m_n \in M$, con $m_j \in M_j$ para $j = 1\ldots n$. Si denotamos por $e_i = \iota_i\pi_i $ para $i=1\ldots n$, no es difícil ver que $\{e_1, \ldots, e_n\}$ es un CCIO para el anillo $End_A(M)$. Comprobar todas las afirmaciones.
\end{ejercicio}

\begin{ejercicio}
    Sea $M$ un $A$-módulo, y pongamos $R=End_A(M)$. Supongamos que $M=M_1 \oplus \cdots \oplus M_n$ para submódulos $M_i $ tales que existe un $A$-módulo $N$ y un isomorfismo de $A$-módulos $M_i \cong N$ para cada $i=1\ldots n$. Si $\Delta = End_A(N)$, entonces exxiste un isomorfismo de anillos $R\cong M_n(\Delta)$ que, si $A$ es una $K$-álgebra, es de $K$-álgebras. Cubrir los detalles respecto a los apuntes. 
\end{ejercicio}

\begin{ejercicio}
    Sea $\{e_1,\ldots,e_n\}$ un CCIO para $R$. Demostrar que los idempotentes $e_1,\ldots,e_n$ son centrales si, y solo si, $e_iRe_j = \{0\}$ para todo $i\neq j$. 
\end{ejercicio}

\begin{ejercicio}\label{ej33}
    Demostrar que, si $M$ un $R$-módulo y $e\in R$ un idempotente central, entonces $eM = \{em : m\in M\}$ es un submódulo de $M$. Si ahora $\{e_1,\ldots,e_n\}$ es un CCIO centrales de $R$, demostrar que $M = e_1M \oplus \cdots \oplus e_nM$.
\end{ejercicio}

\begin{ejercicio}
    Con la notación del Ejercicio~\ref{ej33}, demostrar que $M$ es cíclico si, y solo si, $e_iM$ es cíclico para todo $i=1\ldots n$. 
\end{ejercicio}

\begin{ejerciciostarstar}
    Sea $V$ un $K$-espacio vectorial de dimensión finita $n$ y $T:V \to V$ una aplicación lineal. Diremos que un vector $v \in V$ es cíclico para $T$ si $\{v, T(v), \ldots, T^{n-1}(v)\}$ es una base de $V$ como $K$-espacio vectorial. Demostrar que $V$ admite un vector cíclico si, y solo si, el polinomio mínimo de $T$ tiene grado $n$. ¿Cuál es entonces la longitud de $V$ en tanto que $K[x]$-módulo?
\end{ejerciciostarstar}

\begin{ejercicio}
    Sea $A = K[x]/\langle x^2 \rangle $, donde $K$ es un cuerpo, y $K[x]$ es el anillo de polinomios. Demostrar que, visto como $A$-modulo, $A$ no tiene sumandos directos no triviales.
\end{ejercicio}

\begin{ejercicio}
    Sean $M$ y $N$ módulos semisimples con descomposiciones como suma directa de submódulos simples $M = S_1\oplus \ldots\oplus S_t$ y $N=T_1\oplus \ldots\oplus T_s$. Supongamos que $S_i$ no es isomorfo a $T_j$ para todo $i = 1,\ldots,t,j=1,\ldots,s$. Demostrar que todo homomorfismo de módulos de $M$ a $N$ es cero.
\end{ejercicio}

\begin{ejercicio}\label{ej:39}
    Sea $M$ un módulo semisimple de dimensión finita con estructura $(\Sigma_1,n_1), \ldots, (\Sigma_t,n_t)$. Si $N$ es un submódulo de $M$, demostrar que su estructura es $(\Sigma_1,m_1), \ldots, (\Sigma_t,m_t)$ para ciertos $m_j\leq n_j$ (admitimos que $m_j=0$ significa que $\Sigma_j$ no aparece en la estructura de $N$).
\end{ejercicio}

\begin{ejercicio}

\end{ejercicio}

\begin{ejercicio}
    Sea $\varphi:R\to S$ un isomorfismo de $K-$álgebras e $I,J$ ideales a izquierda de $R$. Demsotrar que $\varphi(I),\varphi(J)$ son ideales a izquierda de $S$ y que dado cualquier homomorfismo de $R-$módulos $f:I\ŧo J$, la aplicación $\hat{f}:\varphi(I)\to \varphi(J)$ definida por $\hat{f}(y) = \varphi f \varphi^{-1}(y)$ para $y\in \varphi(I)$ es un homomorfismo de $S-$módulos.
\end{ejercicio}
