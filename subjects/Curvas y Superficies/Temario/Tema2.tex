\chapter{Superficies en el espacio}
\section{Definición de superficie}

\begin{definicion}
    Una superficie en $\mathbb{R}^3$ es un subconjunto $S\subset\mathbb{R}^3$ no vacío tal que $\forall p\in S$ existen:
    \begin{enumerate}[label=\alph*)]
        \item $U\subset\mathbb{R}^2$ abierto.
        \item Un entorno abierto $V$ de $p$ en $S$.
        \item Una aplicación $X:U\to \mathbb{R}^3$ diferenciable.
    \end{enumerate}
    de forma que:
    \begin{enumerate}
        \item $X(U) = V$.
        \item $X:U\to V$ es un homeomorfismo\footnote{Nos gustaría exigir que sea un difeomorfismo, pero $V$ no es necesariamente un espacio normado, con lo que no tenemos definido qué es ser una aplicación diferenciable $X:U\to V$.}.
        \item $(dX)_q:\mathbb{R}^2\to \mathbb{R}^3$ es inyectiva $\forall q\in U$.
    \end{enumerate}
\end{definicion}

\begin{observacion}
    Vemos que definimos anteriormente las curvas como aplicaciones, pero ahora las superficies las hemos dado como un subconjunto de $\mathbb{R}^3$, de forma que para cada punto $p\in S$ obtenemos una aplicación diferenciable $X:U\to \mathbb{R}^3$.\\

    \noindent
    La definición hipotética de que una superficie sea la imagen de una aplicación diferenciable $X:U\to \mathbb{R}^3$ no nos gusta porque entonces la esfera no sería una superficice, sino que habría que cortar y pegar trozos, algo que no queremos hacer en geometría.\\

    \noindent
    Así, una superficie se obtiene juntando varios trozos de $\mathbb{R}^2$ y pegándolos juntos.
\end{observacion}

\begin{notacion}
    Usualmente notaremos a la aplicación $X:U\to \mathbb{R}^3$ que obtenemos en cada punto $p\in S$ por:
    \begin{equation*}
        X(u,v) = (x(u,v), y(u,v), z(u,v))
    \end{equation*}
    donde $x,y,z:U\to \mathbb{R}$ son aplicaciones diferenciables. Notaremos también:
    \begin{equation*}
        q = (u,v), \quad p = (x,y,z)
    \end{equation*}
\end{notacion}

\begin{observacion}
    Como $X:U\to V$ ha de ser un homeomorfismo, las superficies que estudiamos no pueden tener autointersecciones.
\end{observacion}

\begin{observacion}
    Otra forma de decir que cada aplcación $(dX)_q$ es inyectiva es decir que:
    \begin{equation*}
        \frac{\partial X}{\partial u}(q), \frac{\partial X}{\partial v}(q)
    \end{equation*}
    son linealmente independientes, que es tanto como el análogo a decir que tenemos una curva regular. Otra forma para ver esto es tanto como decir que no son linealmente dependientes, es decir, que:
    \begin{equation*}
        \frac{\partial X}{\partial u}(q) \land \frac{\partial X}{\partial v}(q) \neq 0 \quad \forall q\in U
    \end{equation*}
    Cuando estudiemos el plano tangente a una superficie $S$ entenderemos el por qué de esta condición, pues es necesario exigirla para poder obtener en cada punto $p\in S$ un plano tangente a la superficie que pase por el punto $p$, algo que parece razonable de esperar de una superficie.
\end{observacion}

\begin{notacion}
    Si tenemos una superficie $S$, fijado $p\in S$ podemos obtener la $X$ que nos dice su definición, por lo que podemos escribir:
    \begin{equation*}
        X(u,v) = (x,y,z), \qquad X(q) = p
    \end{equation*}
    Llamaremos:
    \begin{itemize}
        \item A $(u,v)$ coordenadas de $p$.
        \item A $X$ parametrización (local), o a veces mapa o carta (de navegación).
        \item A $V$ entorno parametrizado.
        \item A $U$ dominio de la parametrización.
    \end{itemize}
\end{notacion}

\subsection{Ejemplos de superficies}
\begin{ejemplo}
    Ejemplos de superficies:
    \begin{enumerate}
        \item Los planos de $\mathbb{R}^3$ son superficies.

            Para verlo, sea $P\subset\mathbb{R}^3$ un plano:
            \begin{equation*}
                P = \{(x,y,z)\in \mathbb{R}^3 : ax + by + cz + d= 0\} \qquad (a,b,c)\neq (0,0,0), d\in \mathbb{R}
            \end{equation*}
            Como $(a,b,c)\neq (0,0,0)$ tendremos que al menos uno de ellos será no nulo. Supongamos que $c\neq 0$, con lo que:
            \begin{equation*}
                P = \{(x,y,z)\in \mathbb{R}^3 : z = Ax + By + C\} \qquad A,B,C\in \mathbb{R}
            \end{equation*}
            Sea $p\in P$, elegimos en vista de la definición:
            \begin{equation*}
                U = \mathbb{R}^2, \quad V = P, \quad X:U\to \mathbb{R}^3
            \end{equation*}
            dada por:
            \begin{equation*}
                X(u,v) = (u, v, Au + Bv + C)
            \end{equation*}
            observamos que es diferenciable. Para revisar las condiciones:
            \begin{itemize}
                \item $X(U) = X(\mathbb{R}^2) = \{(u,v,Au+Bv+C) \in \mathbb{R}^3 : (u,v)\in \mathbb{R}^2\} = P = V$.
                \item Vemos que $X:U=\mathbb{R}^2\to V=P$ es un homeomorfismo\footnote{No tenemos que demostrar ni la continuidad de $X$ ni su sobreyectividad, puesto que sabemos ya que $X(U) = V$ y que es un difeomorfismo.}, puesto que $X^{-1}:P\to \mathbb{R}^3$ dada por $(x,y,z)\mapsto (x,y)$ es su aplicación inversa, que es continua.
                \item $X_u(u,v) = (1,0,A)\quad \forall (u,v)\in U$ y que $X_v(u,v) = (0,1,B)\quad \forall (u,v)\in U$, que son linealmente independientes.
            \end{itemize}
        \item En realidad hemos probado algo más fuerte que lo de que un plano sea superficie, porque observamos que lo único que hemos usado de que $P$ sea un plano es que podíamos haber despejado $z$ en función de $x$ e $y$. Así, demostramos a continuación que el grafo de cualquier función real de dos variables diferenciable es una superficie:

            Sea $O\subset \mathbb{R}^2$ un abierto y $f:O\to \mathbb{R}$ una aplicación diferenciable, tenemos que:
            \begin{equation*}
                Gr f = \{(x,y,z)\in \mathbb{R}^3 : (x,y)\in O, z=f(x,y)\}
            \end{equation*}
            Fijado $p\in Gr f$, elegimos:
            \begin{equation*}
                U = O, \quad V = Gr f, \quad X:U\to \mathbb{R}^3
            \end{equation*}
            dada por:
            \begin{equation*}
                X(u,v) = (u,v,f(u,v))
            \end{equation*}
            que es una aplicación diferenciable. Revisamos las condiciones:
            \begin{itemize}
                \item $X(U) = X(O) = Gr f = V$.
                \item Veamos que $X:U\to V$ es un homeomorfismo, algo que se ha probado ya en anteriores cursos de Topología.
                \item Para ver que la diferencial sea inyectiva miramos las derivadas parciales:
                    \begin{equation*}
                        X_u(u,v) = (1,0,f_u(u,v)), \qquad X_v(u,v) = (0, 1, f_v(u,v))
                    \end{equation*}
                    Obtenemos vectores linealmente independientes.
            \end{itemize}
            Tanto el plano como el grafo tienen en común que $U,V,X$ no dependen del punto $p\in S$ escogido, por lo que estos dos objetos cumplirían aquella definición hipotética de una aplicación diferenciable de $U$ en $\mathbb{R}^3$.
        \item Sea $S$ una superficie y $O\subset S$ un abierto suyo, entonces $O$ es una superficie.

            Para verlo, sea $p\in O$, tenemos que $p\in S$ superficie, con lo que existen $U\subset \mathbb{R}^2$ abierto, $V$ entorno abierto de $p$ en $S$ y $X:U\to \mathbb{R}^3$ diferenciable cumpliendo 1, 2 y 3. Tomamos: $V_O = U\cap O$, que es un entorno abierto de $p$ en $V$, $U_O = X^{-1}(V\cap O)$, que por la condición 2 y ser $V\cap O$ un abierto de $V$ es un abierto de $\mathbb{R}^2$. Finalmente, $X_O = X\big|_{U_O}:U_O\to \mathbb{R}^3$ que sigue siendo diferenciable por el carácter local.

            Vemos ahora que: 
            \begin{equation*}
                X_O (U_O) = X\big|_{U_O}(X^{-1}(V\cap O)) = V\cap O = V_O
            \end{equation*}
            Así como que $X_O:U_O\to V_O$ es un homeomorfismo, puesto que es igual a:
            \begin{equation*}
                X\big|_{U_O}:X^{-1}(V\cap O) \to V\cap O
            \end{equation*}
            Finalmente, para $q\in U_O = X^{-1}(V\cap O)$ tenemos que:
            \begin{equation*}
                (dX_O)_q = (dX\big|_{X^{-1}(V\cap O)})_q = (dX)_q
            \end{equation*}
            Por lo que $(dX_O)_q:\mathbb{R}^2\to \mathbb{R}^3$ es inyectiva.

            Por tanto, acabamos de probar que $O$ es una superficie.
        \item Sea $S\subset\mathbb{R}^3$ no vacío, $S_i\subset S$ abierto de $S$ y superficie para cada $i \in I$ de forma que $S=\bigcup\limits_{i \in I}S_i$, entonces $S$ es una superficie.

            Para verlo, sea $p\in S=\bigcup\limits_{i \in I}S_i$, existe por tanto $i_0\in I$ de manera que $p\in S_{i_0}$ superficie, con lo que existen $U_0\subset \mathbb{R}^2$ abierto, $V_0$ entorno abierto de $p$ en $S_{i_0}$ y $X:U_0\to \mathbb{R}^3$ diferenciable de forma que verifican 1, 2 y 3.

            Elegimos $V=V_0\subset S$ entorno abierto de $p$ en $S$ por ser $V_0$ un entorno abierto de $p$ en $S_{i_0}\subset S$ siendo este último abierto; $U=U_0$ abierto de $\mathbb{R}^2$ y $X=X_0$ con lo que $X:U = U_0 \to \mathbb{R}^3$ es diferenciable. Falta ver que $U,V$ y $X$ verifican 1, 2 y 3. % // TODO: HACER

            % // TODO: Explicar que a partir de aqui se demustra que la union de superficies es superficie. Union de dos planos paralelos si pero union de dos planos que se cortan no por ser dos abiertos relativos.
        \item Sea $S\subset \mathbb{R}^3$ una superficie y $\phi:O\to O'$ un difeomorfismo con $O$ y $O'$ abiertos de $\mathbb{R}^3$ de forma que $S\subset O$, entonces $S' = \phi(S)$ es otra superficie en $\mathbb{R}^3$.

            Para ello, sea $p'\in S'$, existe un único $p\in S$ de manera que $\phi(p) = p'$. Como $S$ es superficie, asociados a $p$ tenemos $U,V$ y $X$ dados por la definición de superficie. Tomamos $U' = U$, $V' = \phi(V)$ y $X' = \phi\circ X$, tenemos que:
            \begin{itemize}
                \item $X'(U') = (\phi\circ X)(U) = \phi(V) = V'$.
                \item $X':U'\to V'$, es igual a $\phi\circ X:U\to \phi(V)$, que es composición de los homeomorfismos $X:U\to V$ y $\phi:V\to \phi(V)$, por lo que $X'$ es un homeomorfismo.
                \item Sea ahora $q'\in U'=U$, tenemos que:
                    \begin{equation*}
                        (dX')_{q'} = d(\phi\circ X)_{q'} = (d\phi)_{X(q')}\circ (dX)_{q'}
                    \end{equation*}
                    Siendo la última inyectiva por hipótesis y la primera biyectiva por ser una diferencial de un difeomorfismo, por lo que $(dX')_{q'}$ será también inyectiva.
            \end{itemize}
        \item Las esferas son superficies: $S^2(c,R)$ es una superficie, para todo $c\in \mathbb{R}^3$ y $R>0$.

            Para verlo, suponemos que $c = 0$ y que $R = 1$, por lo que trabajamos con $S^2 = S^2(0,1)$. Tomamos $N = (0,0,1)\in S^2$ el polo norte de la esfera. Si consideramos:
            \begin{equation*}
                H_N^+ = \{(x,y,z)\in S^2 : z > 0\}
            \end{equation*}
            es decir, el hemisferio abierto (sin el ecuador, denotado por el $+$) de centro $N$ en $S^2$. Es fácil ver que si tomamos $g:D(0,1)\to \mathbb{R}$ dada por:
            \begin{equation*}
                g(x,y) = \sqrt{1-x^2-y^2}
            \end{equation*}
            tenemos que $g$ es diferenciable y que $H_N^+$ es el grafo de $g$, por lo que $H_N^+$ es una superficie.

            Ahora, para todo $p\in S^2$ sabemos que existe un giro $\phi_p:\mathbb{R}^3\to\mathbb{R}^3$ que cumple $\phi_p(N) \to p$ y es un difeomorfismo, con lo que tendremos:
            \begin{equation*}
                \phi_p(H_N^+) = H_p^+
            \end{equation*}
            Y tenemos por tanto que $H_p^+$ es una superficie, puesto que es un grafo. Finalmente, podemos ver que\footnote{Aunque realmente solo tendríamos que haber usado 6 hemisferios.}:
            \begin{equation*}
                S^2 = \bigcup_{p\in S^2} H_p^+
            \end{equation*}
            Así, obtenemos que $S^2$ es una superficie.

            Ahora, para una esfera cualquiera $S=S^2(c,R)$ tenemos que existen una traslación $T$ y una homotecia $H$ de $\mathbb{R}^3$ tales que $(H\circ T)(S^2) = S^2(c,R)$, siendo $H\circ T$ un difeomorfismo de $\mathbb{R}^3$, por lo que el ejemplo 5 nos dice que $S^2(c,R)$ es una superficie.

            Finalmente, observemos que los elipsoides también son superficies, puesto que podemos pasar de un elipsoide a una esfera por una aplicación afín.

        \item Superficies en forma implícita (gran herramienta para obtener superficies). Sea $O\subset \bb{R}^3$ abierto $f:O\to \bb{R}$ diferenciable. Tenemos que $a\in \bb{R}$ se llama \textbf{valor regular} de $f$ cuando
        \begin{gather*}
            (df)_p = (\nabla f)_p \neq 0 \ \ \ \ \forall p\in f^{-1}(\{a\})
        \end{gather*}
        Consideramos $S=f^{-1}(\{a\})\subset O \subset \bb{R}^3$ con $a$ un valor regular y queremos ver si $S$ es o no una superficie. Recordamos que el Teorema de la Función implícita dice que para cada punto $p_0 = (x_0, y_0, z_0)\in S$ si se tiene que $(\nabla f)_{p_0}\neq 0$, entonces suponemos sin pérdida de generalidad que $f_x(p_0)\neq 0$. Tenemos que existe un entorno abierto $U$ de $(x_0,y_0)$ en $\bb{R}^2$, un entorno abierto $V$ de $z_0$ en $\bb{R}$ y una aplicación $g:U\to V\subset \bb{R}$ diferenciable tales que 
        \begin{gather*}
            S \cap (U\times V) = Grafo(g)
        \end{gather*}
        Por tanto $S$ es unión de grafos abiertos en $S$.

        \item Sea $A\subset M_4(\bb{R})$ tal que $A=A^t$. Sea
        \begin{gather*}
            S = \left\{p\in \bb{R}^3 : (1,p^t)A \begin{pmatrix} 1\\p\end{pmatrix} = 0\right\} 
        \end{gather*}
        que será la cuádrica en $\bb{R}^3$ determinada por la matriz $A$. Queremos ver ahora entonces si las cuádricas de $\bb{R}^3$ son superficies.\\

        Si definimos $f:O=\bb{R}^3 \to \bb{R}$ dada por 
        \begin{gather*}
            f(p) =  (1,p^t)A \begin{pmatrix} 1\\p\end{pmatrix}
        \end{gather*}
        tendremos que es diferenciable y además con esta definición se tiene que $S=f^{-1}(\{0\})$. Nos preguntamos ahora si $0$ es un valor regular de $f$ (ya que en dicho caso podremos aplicar el apartado anterior). Aplicando la definición de valor regular tomamos $p\in f^{-1}(\{0\})=S$ y consideramos $(df)_p:\bb{R}^3 \to \bb{R}$. Tomamos $v\in \bb{R}^3$ y se tiene 
        \begin{gather*}
            (df)_p(v) = (0,v^t)A\begin{pmatrix} 1\\p\end{pmatrix} + (1,p^t)A\begin{pmatrix} 0\\v\end{pmatrix} = 2(1,p^t)A\begin{pmatrix} 0\\v\end{pmatrix}
        \end{gather*}
        Supongamos que $0$ no es valor regular. En ese caso existirá un $p\in f^{-1}(\{0\})$ tal que $(df)_p = 0$ lo que nos diría que 
        \begin{gather*}
            (df)_p(v) = 0 \ \ \ \ \forall v \in \bb{R}^3
        \end{gather*}
        o equivalentemente
        \begin{gather*}
            0 = (df)_p(v) = 2(1,p^t)A \begin{pmatrix} 0\\v\end{pmatrix} \ \ \ \ \forall v \in \bb{R}^3 \Rightarrow (1,p^t)A = (\lambda, 0)
        \end{gather*}
        Si lo unimos con lo anterior tenemos que 
        \begin{gather*}
            0 = f(p) = (1,p^t)A \begin{pmatrix} 0\\v\end{pmatrix} = (\lambda, 0)\begin{pmatrix} 1\\p\end{pmatrix} = \lambda
        \end{gather*}
        Si $0$ no es valor regular, entonces existe un $p\in S$ tal que $(1,p^t)A = 0$. El contrarrecíproco de esta afirmación sería que si para todo $p\in S$ se tiene que $(1,p^t)A\neq 0$, entonces se tiene que $0$ es regular para $f$. En este caso tendríamos que o bien $S$ es vacío o bien $S$ es una superficie.\\
        
        Si la cuádrica $S$ no tiene punto singulares (los tiene en el infinito) y no es vacía, entonces es una superficie de $\bb{R}^3$. Esto nos dice que los elipsoides, hiperboloides, paraboloides, cilindros y planos son superficies. Nos quedarían fuera los conos y los planos secantes, que en principio no podremos concluir nada con el ejemplo anterior.

        \item Consideramos $S^1((0,a,0), r)$ con $0<r<a$. Definimos el siguiente conjunto:
        \begin{gather*}
            S = \{(x,y,z)\in \bb{R}^3 : (x,y,z) \text{ se obtiene de algún punto de }S^1((0,a,0), r)\\
            \text{ rotando alrededor del eje }z\}
        \end{gather*}
        que llamaremos \textbf{toro de revolución}\footnote{revolución no porque sea muy rebelde o porque quiera derrocar al gobierno.}. Tomamos $(x,y,z)\in S$ por lo que se habrá obtenido de un punto de $S^1((0,a,0), r)\owns (0,a+r\cos(u), r\sen(u))$. Las rotaciones alrededor del eje $z$ serán producto de multiplicar por la siguiente matriz:
        \begin{gather*}
            \begin{pmatrix}
                \cos(v) & \sen(v) & 0\\
                -\sen(v) & \cos(v) & 0\\
                0 & 0 & 1
            \end{pmatrix}
        \end{gather*}
        Por tanto, al aplicar el giro sobre el eje $z$ tendremos:
        \begin{align*}
            (x,y,z) &= (0,a+r\cos(u), r\sen(u)) \begin{pmatrix}
                \cos(v) & \sen(v) & 0\\
                -\sen(v) & \cos(v) & 0\\
                0 & 0 & 1
            \end{pmatrix}
            =\\ &=((a+r\cos(u))\sen(v), (a+r\cos(u))\cos(v), r\sen(u)) \ \ \ \ \forall u,v\in \bb{R}
        \end{align*}
        Esto sería una construcción discreta de la parametrización $X(u,v)$. Si estudiamos la condición que nos ha salido antes nos quedará
        \begin{gather*}
            x^2 + y^2 = (a+r\cos(u))^2 \Rightarrow \sqrt{x^2 + y^2} = a+r\cos(u) \Rightarrow \sqrt{x^2 + y^2} -a = r\cos(u)
        \end{gather*}
        Sabemos que $z=r\sen(u)$ por lo que juntando estas ecuaciones nos queda
        \begin{gather*}
            (\sqrt{x^2 + y^2}-a)^2 + z^2 = r^2
        \end{gather*}
        Entonces $S\subset \{(x,y,z)\in \bb{R}^3 : (\sqrt{x^2 + y^2}-a)^2 + z^2 = r^2\}$ y queremos buscar la igualdad. Para ello tomamos $(x,y,z)$ de este conjunto, y tendremos
        \begin{gather*}
            \frac{(\sqrt{x^2 + y^2}-a)^2 + z^2}{r^2} = 1
        \end{gather*}
        por lo que para cierto $u\in \bb{R}$ se tendrá que
        \begin{gather*}
            \left\{
            \begin{array}{l}
                \sqrt{x^2 + y^2} -a = r\cos(u) \Rightarrow \sqrt{x^2 + y^2} = a + r\cos(u) \Rightarrow x^2 + y^2 = (a+r\cos(u))^2\\
                z = r\sen(u) 
            \end{array}
            \right.
        \end{gather*}
        de la primera ecuación llegamos a que 
        \begin{gather*}
            \left\{
            \begin{array}{l}
                x = (a+r\cos(u))\cos(v)\\
                y = (a+r\cos(u))\sen(v)
            \end{array}
            \right. \ \ \ \ v\in \bb{R}
        \end{gather*}
        Definimos $f(x,y,z) = (\sqrt{x^2 + y^2} -a)^2 + z^2$ que será diferenciable en $O=\bb{R}^3 \setminus\{\text{eje }z\}$. Nos preguntamos ahora si $r^2$ es un valor regular de $f$. Para ello tomamos $(x,y,z)\in f^{-1}(\{r^2\})=S$ y estudiamos su gradiente:
        \begin{gather*}
            (\nabla f)_{(x,y,z)} = \left(\frac{2(\sqrt{x^2 + y^2} -a)x}{\sqrt{x^2 + y^2}}, \frac{2(\sqrt{x^2 + y^2} -a)y}{\sqrt{x^2 + y^2}}, 2z \right) 
        \end{gather*}
        Si $r^2$ no es regular para $f$, entonces existe un $(x,y,z)\in S$ para el cual $(\nabla f)_{(x,y,z)} = (0,0,0)$. Si escribimos esto tendremos
        \begin{gather*}
            \left\{
            \begin{array}{l}
                2(\sqrt{x^2 + y^2} -a)x = 0\\
                2(\sqrt{x^2 + y^2} -a)y = 0\\
                2z = 0 \Rightarrow z=0
            \end{array}
            \right.
        \end{gather*} 
        Si distinguimos casos tenemos 
        \begin{itemize}
            \item Si $\sqrt{x^2 + y^2}-a \neq 0$, entonces $x=y=z=0$ pero entonces $(x,y,z)\notin O$.
            \item Si $\sqrt{x^2 + y^2}-a = 0$, entonces $f(x,y,z) = 0 \neq r^2$
        \end{itemize}
        En cualquier caso llegamos a una clara contradicción por lo que $r^2$ es un valor regular. Por el ejemplo 7 sabemos que $S$ o bien es vacío o bien es una superficie. Como la circunferencia inicial estaba en $S$ sabemos que no es vacío por lo que concluimos finalmente que el toro de revolución es una superficie.
    \end{enumerate}
\end{ejemplo}

\begin{ejercicio}
    El cono $\cc{C}$ dado por 
    \begin{gather*}
        \cc{C} = \{(x,y,z)\in \bb{R}^3 : x^2 + y^2 = z^2\}
    \end{gather*}
    no es una superficie.\\

    \noindent
    Si $\cc{C}$ fuera una superficie entonces existirían $V$ entorno del $(0,0,0)$, $U\subset \mathbb{R}^2$ abierto y conexo y $X:U\to V$ homeomorfismo entre ellos. Sin embargo, cualquier entorno abierto $V$ del $(0,0,0)$ verificará que $V\setminus\{(0,0,0)\}$ tendrá 2 componentes conexas. Por otro lado, cualquier abierto $U\subset \bb{R}^2$ al quitarle un punto $p\in U$ tendrá una única componente conexa por lo que no podrá existir ningún homeomorfismo entre $U$ y $V$.\\

    \noindent
    Para ver que $U\setminus\{p\}$ solo tiene una componente conexa tenemos que ver que al ser $U$ abierto existirá un abierto básico $A_p$ para todo $p\in U$ tal que $p\in A_p\subset U$. Además $A_p=B(p,\veps)$ con $\veps>0$ por lo que $A_p\setminus\{p\}$ seguirá siendo conexo. En consecuencia $U\setminus\{p\}$ será conexo.
\end{ejercicio}

Sean $S,S'\subset \mathbb{R}^3$ superficies y $O\subset \mathbb{R}^n$ un abierto, consideraremos 3 tipos de aplicaciones en la siguiente definición:
\begin{enumerate}[label=\alph*)]
    \item Aplicaciones $f:S\to \mathbb{R}^m$ para $m\geq 1$.
    \item Aplicaciones $f:O\to S$.
    \item Aplicaciones $f:S\to S'$.
\end{enumerate}

\begin{definicion}[Aplicación diferenciable de tipos $a)$, $b)$ y $c)$]
    Damos la definición de una aplicación diferenciable, distinguiendo en qué caso estamos:
    \begin{enumerate}[label=\alph*)]
        \item Diremos que $f:S\to \mathbb{R}^m$ es diferenciable si y solamente si $f\circ X:U\to \mathbb{R}^m$ es diferenciable (en el sentido del análisis) para toda parametrización $X:U\to S$.
        \item Diremos que $f:O\to S$ es diferenciable si y solamente si $i\circ f:O\to \mathbb{R}^3$ es diferenciable (en el sentido del análisis), donde $i:S\hookrightarrow \mathbb{R}^3$ es la aplicación inclusión.
        \item Diremos que $f:S\to S'$ es diferenciable si y solamente si $i'\circ f:S\to \mathbb{R}^3$ es diferenciable (en el sentido $a)$), donde $i':S'\to \mathbb{R}^3$ es la aplicación inclusión.
    \end{enumerate}
\end{definicion}

\begin{observacion}
    Notemos que hemos definido el nuevo concepto de diferenciabilidad globalmente (para todo el dominio de la aplicación $f$), aunque en análisis primero definíamos qué era ser diferenciable en un punto.

    Sin embargo, es fácil llegar a la definición de lo que debería ser la diferenciabilidad en un punto de una aplicación que esté relacionada con las superficies.
\end{observacion}

\noindent
Vemos propiedades topológicas de las apliaciones diferenciables:

\begin{prop}
    Si $f$ es diferenciable en sentido $a)$, $b)$ o $c)$, entonces $f$ es continua.
    \begin{proof}
        Distinguimos casos:
        \begin{enumerate}[label=\alph*)]
            \item Tenemos $f:S\to \mathbb{R}^m$ diferenciable, es decir, que $f\circ X:U\to \mathbb{R}^m$ es diferenciable (en el sentido del análisis) para toda parametrización $X$, por lo que ya sabemos que $f\circ X$ es continua para toda parametrización $X:U\to S$. Si consideramos ahora:
                \begin{equation*}
                    X(U) \stackrel{X^{-1}}{\longrightarrow} U \stackrel{f\circ X}{\longrightarrow} \mathbb{R}^m
                \end{equation*}
                con $X^{-1}:X(U)\to U$ homeomorfismo y $f\circ X$ continua, por lo que la composición $f = (f\circ X)\circ X^{-1}$ es continua en $X(U)$ para toda parametrización $X:U\to S$.

                Ahora, como los $X(U)$ son abiertos que cubren toda la superficie $S$, deducimos que $f$ es continua en $S$.
            \item Si $f:O\to S$ es diferenciable tenemos entonces que la aplicación $f:O\to\mathbb{R}^3$ es diferenciable (en el sentido del análisis), luego $f:O\to \mathbb{R}^3$ es continua, de donde $f:O\to S$ es continua.
            \item Si $f:S\to S'$ es diferenciable, tenemos entonces que $i\circ f:S\to\mathbb{R}^3$  es diferenciable, por lo que por el apartado $a)$ tenemos que es continua, por lo que $f$ será continua.
        \end{enumerate}
    \end{proof}
\end{prop}

\noindent
Destacamos a continuación una propiedad algebraica de las aplicaciones diferenciables:
\begin{prop}
    Sean $f,g:S\to \mathbb{R}^m$ y $\lm:S\to \mathbb{R}$ aplicaciones diferenciables en sentido $a)$, entonces las aplicaciones $f+g,\lm f:S\to \mathbb{R}^m$ son diferenciables.

    \noindent
    Más aún:
    \begin{itemize}
        \item $\langle f,g \rangle :S\to \mathbb{R}$ es diferenciable.
        \item Si $m=3$, $f\land g:S\to \mathbb{R}^3$ es diferenciable.
    \end{itemize}
    \begin{proof}
        % // TODO: Terminar
        Para el caso $\lm f:S\to \mathbb{R}$ tenemos que comprobar la definición del tipo $a)$. Por tanto, sea $X:U\subset\mathbb{R}^2\to S$ una parametrización, vemos que:
        \begin{multline*}
            [(\lm f)\circ X](u,v) = (\lm f)(X(u,v)) = \lm(X(u,v)) f(X(u,v)) \\ = (\lm\circ X)(u,v) (f\circ X)(u,v) = [(\lm\circ X)(f\circ X)](u,v)
        \end{multline*}
        Pero $\lm\circ X:U\to \mathbb{R}$ y $f\circ X:U\to \mathbb{R}^m$ son diferenciables (en el sentido del análisis), luego su producto (que coincide con $(\lm f)\circ X$) también es diferenciable.
    \end{proof}
\end{prop}

\begin{ejemplo}
    Veamos varios ejemplos de aplicaciones diferenciables en el nuevo sentido:
    \begin{enumerate}
        \item Las restricciones son diferenciables, es decir, si tenemos una superficie $S\subset O$ donde $O\subset \mathbb{R}^3$ es un abierto y tenemos $F:O\to \mathbb{R}^m$ diferenciable, definimos $f=F\big|_S:S\to \mathbb{R}^m$. 

            Sea $X:U\subset\mathbb{R}^2\to S$ una parametrización de la superficie, tenemos que: 
            \begin{equation*}
                f\circ X = F\big|_S \circ X \stackrel{(X(U)\subset S)}{=} F\circ X
            \end{equation*}
            con $X:U\to O\subset \mathbb{R}^3$ diferenciable en el sentido del análisis y $F:O\to \mathbb{R}^m$ diferenciable en el sentido del análisis, por lo que sabemos ya que $F\circ X$ es diferenciable para toda parametrización $X$, es decir, $f$ es diferenciable en el sentido $a)$.

            Este ejemplo nos proporciona una buena herramienta para obtener muchas aplicaciones $f:S\to \mathbb{R}^m$ diferenciables, pues ya conocemos muchas aplicaciones $F:O\to \mathbb{R}^m$ diferenciables.
        \item[$a_1)$] Si tomamos $O = \mathbb{R}^3$, $F:O\to \mathbb{R}^m$ una aplicación constante, una superficie $S\subset O$ y $f=F\big|_S$, tenemos entonces que $f$ es diferenciable.

            En conclusión, las aplicaciones $f:S\to \mathbb{R}^m$ constantes son diferenciables.
        \item[$a_2)$] Si tomamos $F = 1_{\mathbb{R}^3}:\mathbb{R}^3\to \mathbb{R}^3$ la aplicación identidad, tenemos que $F$ es diferenciable. Así, si $S$ es una superficie y tomamos $f = F\big|_S:S\to \mathbb{R}^3$, tenemos que $f = i:S\to \mathbb{R}^3$ (es decir, la aplicación inclusión) es diferenciable.

            Más aún, $i(S) = S$, con lo que si restringimos en codominio $i$ a $S$ tenemos que $1_S:S\to S $ es diferenciable en el sentido $c)$.
        \item[$a_3)$] Sea $p_0\in \mathbb{R}^3$, $F:\mathbb{R}^3\to \mathbb{R}$ dada por:
            \begin{equation*}
                F(p) = |p-p_0|^2
            \end{equation*}
            es decir, $F$ es la distancia al cuadrado al punto $p_0$. $F$ es diferenciable en $\mathbb{R}^3$ por ser un polinomio de segundo grado. Sea $S$ una superficie, tendremos que $f=F\big|_S:S\to \mathbb{R}$ es diferenciable, donde $f$ es:
            \begin{equation*}
                f(p) = |p-p_0|^2 \qquad \forall p\in S
            \end{equation*}

            Ahora, si consideramos $F(p) = |p-p_0|$, tenemos que $F$ es diferenciable en $O=\mathbb{R}^3\setminus \{p_0\}$. Sea $S\subset O$ una superficie, tenemos entonces que $p_0\notin S$, por lo que $f=F\big|_S$ es diferenciable.

            Así, si queremos mirar la distancia de cualquier punto $p\in S$ a otro $p_0\in \mathbb{R}^3$, si $p_0\in S$ tenemos que considerar la distancia al cuadrado para obtener una aplicación diferenciable.
        \item[$a_4)$] Sea $R\subset\mathbb{R}^3$ una recta, $p_0\in R$ y $v\in \mathbb{R}^3\setminus \{0\}$ unitario tal que $R = p + \langle v \rangle $. Sea ahora $p\in \mathbb{R}^3$, ¿cómo podemos calcular la distancia de $p$ a $R$?
            \begin{equation*}
                dist(p,R)^2 = |p-p_0|^2 - \langle p-p_0,v \rangle^2
            \end{equation*}
            Con lo que la aplicación $p\in S\mapsto |p-p_0|^2 - \langle p-p_0,v \rangle^2$ es diferenciable para todo $p_0$. Ahora, si consideramos la aplicación $dist(\cdot ,R):S\to \mathbb{R}$ dada por:
            \begin{equation*}
                dist(p,R) = \sqrt{|p-p_0|^2 - \langle p-p_0,v \rangle^2 }
            \end{equation*}
            Tenemos que esta es diferenciable siempre que $R\cap S = \emptyset $.
        \item[$a_5)$] Sea $P\subset \mathbb{R}^3$ un plano que pasa por $p_0\in \mathbb{R}^3$ y un vector unitario normal $v\in \mathbb{R}^3$.

            Sea $F:\mathbb{R}^3\to \mathbb{R}$ dada por:
            \begin{equation*}
                F(p) = \langle p-p_0,v \rangle 
            \end{equation*}
            Tenemos que $F$ es la función distancia orientada a $P$, y $F$ es diferenciable en el sentido del análisis en $\mathbb{R}^3$. Sea $S$ una superficie, $f=F\big|_S$ es diferenciable, sin importar si $p_0 \in S$ o no.
    \end{enumerate}
\end{ejemplo}

\begin{observacion}
    A pesar de la facilidad en estos ejemplos, ¿comprobar la condición de diferenciabilidad en el sentido $a)$ es fácil o difícil?\\

    \noindent
    Será generalmente difícil, puesto que debemos comprobar la definición para \textbf{todas} las parametrizaciones de la superficie.
\end{observacion}

Enunciaremos a continuación un Lema técnico (que no demostraremos, puesto que presenta una demostración larga, tediosa y poco didáctica), que nos será de gran utilidad para ver que no es necesario comprobar la definición $a)$ con todas las parametrizaciones, sino solo con un recubrimiento de la superficie.

\begin{lema}\label{lema:tecnico}
    Sea $S$ una superficie, $X_i:U_i\to S$ con $i \in \{1,2\}$ dos parametrizaciones\footnote{Podemos pensar intuitivamente en que tenemos una superficie y dos mapas distintos en los que orientarnos por la superficie con sistemas de coordenadas distintos.} de $S$ tales que $X_1(U_1)\cap X_2(U_2) \neq \emptyset $, tenemos entonces que:
    \begin{equation*}
        X_1^{-1}(X_1(U_1)\cap X_2(U_2)) \subset U_1\subset \mathbb{R}^2, \qquad 
        X_2^{-1}(X_1(U_1)\cap X_2(U_2)) \subset U_2\subset \mathbb{R}^2
    \end{equation*}
    son dos abiertos del plano tales que pasamos del primero al segundo por $X_2^{-1}\circ X_1$, que es un homeomorfismo entre tales abiertos de $\mathbb{R}^2$. Este homeomorfismo se llama ``cambio de parámetros (o de coordenadas)'', del que afirmamos que es un difeomorfismo entre tales abiertos.

\end{lema}

\begin{coro}\label{coro:recubrimiento}
    Sea $S$ una superficie y $f:S\to \mathbb{R}^m$, suponemos que $\forall p\in S$ existe una parametrización $X_p:U_p\to S$ tal que $p\in X_p(U_p)$ y $f\circ X_p$ es diferenciable (en el sentido del análisis), entonces $f$ es diferenciable (en el sentido $a)$).
    \begin{proof}
        Sea $X:U\to S$ una parametrización arbitraria, tenemos que ver que $f\circ X:U\to \mathbb{R}^m$ es diferenciable (en el sentido del análisis). Para ello, veámoslo punto a punto tomando $q\in U$ y obtenemos que $p := X(q)\in S$. Para dicho $p$ existe por hipótesis $X_p:U_p\to S$ con $p \in X_p(U_p)$ y de forma que $X_p\circ f:U_p\to \mathbb{R}^m$ es diferenciable (en el sentido del análisis).\\

        \noindent
        En vista del Lema~\ref{lema:tecnico}, estudiamos el cambio de cartas entre $X$ y $X_p$, que sabemos que existe puesto que $p \in O := X(U)\cap X_p(U_p)$. Nos interesa el cambio de cartas\newline $X_p^{-1}\circ X:X^{-1}(O)\to X_p^{-1}(O)$, que es una aplicación diferenciable y sabemos que $q\in X^{-1}(O)\subset U$. Ahora, vemos que:
        \begin{equation*}
            (f\circ X)\big|_{X^{-1}(O)} = f\big|_O \circ X\big|_{X^{-1}(O)} = (f\big|_O \circ X_p)\circ (X_p^{-1}\circ X\big|_{X^{-1}(O)})
        \end{equation*}
        Del segundo sabemos que es un difeomorfismo (en el sentido del análisis) por el Lema~\ref{lema:tecnico}, por ser un cambio de cartas. Por otra parte, del primero sabemos también que es diferenciable en el sentido del análisis. En definitiva, tenemos que $(f\circ X)\big|_{X^{-1}(O)}$ es diferenciable, con $X^{-1}(O)$ un abierto que contiene a $q$, por lo que $f\circ X$ es diferenciable en $q$ por el carácter local de la diferenciabilidad.\\

        \noindent
        Como esto se cumple para todo $q\in U$, concluimos que $X\circ f$ es diferenciable (en el sentido del análisis), para toda parametrización $X$, por lo que $f$ es diferenciable (en el sentido $a)$).
    \end{proof}
\end{coro}

\noindent
Así, la definición de diferenciable ya no nos parece tan difícil, puesto que no tenemos que comprobar la diferenciabilidad de $f$ para cualquier parametrización $X$ suya, nos basta con comprobar la condición de la definición con un conjunto de parametrizaciones que recubren a $X$.

\begin{ejemplo}
    Sea $S$ una superficie y $X:U\to S$ una parametrización: 
    \begin{itemize}
        \item ¿es $X:U\to X(U)$ diferenciable? 
        \item ¿es $X^{-1}:X(U)\to U$ diferenciable?
    \end{itemize}
    Repondamos a las preguntas:
    \begin{itemize}
        \item En la primera observemos que $X(U)$ es un abierto de $S$, por lo que $X(U)$ es una superficie y nos están preguntando si es diferenciable en en sentido $b)$, por lo que $X:U\to X(U)$ es diferenciable si y solo si al componer con la inclusión, $(i\circ X):U\to \mathbb{R}^3 = X:U\to \mathbb{R}^3$ es diferenciable en el sentido del análisis, que sí lo es por definición de parametrización.
        \item Ahora, para $X^{-1}:X(U)\to U$ sale de una superficie y $U\subset \mathbb{R}^2$ abierto, por lo que tenemos que comprobar diferenciabilidad en el sentido $a)$, para toda parametrización necesitamos comprobar si al componerla con $X^{-1}$ es diferenciable en el sentido del análisis.

            Ahora, el Corolario~\ref{coro:recubrimiento} nos dice que basta verlo para un recubrimiento por parametrizaciones, y solamente la parametrización $X$ nos sirve como parametrización de $X^{-1}$ $\forall p\in S$ . Así, tenemos que $X^{-1}:X(U)\to U$ es diferenciable en el sentido $a)$ si y solo si lo es:
            \begin{equation*}
                f\circ X_p = X^{-1}\circ X = 1_U
            \end{equation*}
            Y $1_U:U\to U$ es diferenciable en el sentido del análisis, por lo que $f\circ X_p$ es diferenciable en el sentido del análisis y por tanto $X^{-1}:X(U)\to U$ es diferenciable en el sentido $a)$.
    \end{itemize}
\end{ejemplo}

\begin{ejercicio}
    Varios ejemplos (ejercicios) que está bien hacer por uno mismo:
    \begin{enumerate}
        \item Una curva diferenciable en una superficie $S$ es una curva $\alpha:I\to \mathbb{R}^3$ (diferenciable) tal que $\alpha(I)\subset S$.

            \begin{equation*}
                S=\bigcup_{i \in I}S_i
            \end{equation*}
            con cada $S_i\subset S$ abierto y $f\big|_{S_i}:S_i\to S'$ es diferenciable, entonces $f:S\to S'$ también es diferenciable.

\begin{definicion}
    Si $S,S'$ son superficies y tenemos $f:S\to S'$ diferenciable, biyectiva y su inversa $f^{-1}:S'\to S$ también es diferenciable, diremos que $f$ es un \textbf{difeomorfismo}.
\end{definicion}

\begin{ejercicio}
    Demostrar que las superficies:
    \begin{align*}
        C &= \{(x,y,z)\in \mathbb{R}^3 : x^2+y^2 = 1\} \\
        H &= \{(x,y,z)\in \mathbb{R}^3 : x^2+y^2-z^2 = 1\}
    \end{align*}
    Son difeomorfas.\\

    \noindent
    Interpretemos estas superficies:
    \begin{itemize}
        \item Para $C$, vemos que para cualquier plano $P(z=c)$ obtenemos la circunferencia de radio 1.
        \item Para $H$, vemos que aumentando $z$ o disminuyendo $z$\ldots (hiperboloide)
    \end{itemize}

    \noindent
    En primer lugar, ¿son superficies? Pues sí, ya que son cuádricas y no tienen puntos singulares.\\

    \noindent
    Para buscar el difeomorfismo seguimos una estrategia similar a la que hacíamos en topología con los homeomorfismos. Dado $(x,y,z)\in C$ tenemos así que $x^2+y^2 = 1$. Así, tomamos (queremos para cada altura $z$ obtener una circunferencia de radio $1+z^2$.):
    \begin{equation*}
        f(x,y,z) = \left(x\sqrt{1+z^2}, y\sqrt{1+z^2}, z\right)
    \end{equation*}
    Así, si tenemos $(X,Y,Z)=f(x,y,z)$ tenemos que:
    \begin{equation*}
        X^2 + Y^2 - Z^2 = x^2(1+z^2) + y^2(1+z^2) - z^2 = (1+z^2)\underbrace{(x^2 + y^2)}_1 - z^2 = 1
    \end{equation*}
    Por lo que $f(x,y,z)\in H$ para todo $(x,y,z)\in C$.\\

    \noindent
    Para ver que $f:C\to H$ es un difeomorfismo, comprobamos en primer lugar que es diferenciable. Tenemos que comprobar en el sentido $c)$, por lo que estudiamos si $f:C\to \mathbb{R}^3$ dada por:
    \begin{equation*}
        f(x,y,z) = \left(x\sqrt{1+z^2}, y\sqrt{1+z^2},z\right)
    \end{equation*}
    es diferenciable. Se tiene que $f = F\big|_C$ para $F:\mathbb{R}^3\to \mathbb{R}^3$ dada por:
    \begin{equation*}
        F(x,y,z) = \left(x\sqrt{1+z^2}, y\sqrt{1+z^2},z\right)
    \end{equation*}
    con $F$ diferenciable en sentido del análisis, por lo que $f:C\to \mathbb{R}^3$ también será diferenciable en el sentido $a)$, luego $f:C\to H$ también (en el sentido $c)$).\\

    \noindent
    Vemos además que $f$ es biyectiva, con inversa:
    \Func{g}{H}{C}{(x,y,z)}{\left(\frac{x}{\sqrt{1+z^2}}, \frac{y}{\sqrt{1+z^2}},z\right)}
    Que está bien definida (es decir, que $g(H)\subset C$), que también está bien definida en $\mathbb{R}^3$, razonar que es diferenciable usando el mismo argumento que para $f$ y ver que $g$ es inversa de $f$.\\

    \noindent
    Notemos que hemos seguido la misma estrategia para buscar un difeomorfismo entre dos superficies: la misma estrategia que hacíamos en topología para buscar homeomorfismos entre espacios topológicos.
\end{ejercicio}

\subsection{Diferencial de una aplicación diferenciable}
\noindent
En análisis, para una aplicación $f:U\to \mathbb{R}^m$, fijado $p\in U$ definíamos para cada $v\in U$:
\begin{equation*}
    (Df)(p,v) = \frac{d}{dt}\big|_{t=0}\ f(p+tv)
\end{equation*}
Pero esta definición no podemos copiarla para una aplicación $f:S\to \mathbb{R}^m$, debido a la elección que tenemos que hacer de $v$:
\begin{itemize}
    \item No podemos tomar un $v$ arbitrario, puesto que puede ser que la recta $p+tv$ ``atraviese a la superficie'', y la aplicación $f$ la tenemos definida en la propia superficie, por lo que no podemos evaluarla en $p+tv$.
    \item Podríamos pensar ahora en calcular solo el diferencial para vectores $v$ en el plano tangente a la superficie en el punto $p$, concepto que no hemos definido hasta ahora, y con el que nos encontramos la dificultad de que ya no es tan fácil como lo hacíamos con las curvas, puesto que antes sencillamente calculábamos $\alpha'(t_0)$ cuando queríamos obtener el vector director de la recta tangente a $\tr\alpha$ en el punto $\alpha(t_0)$; ahora tenemos $X:U\subset\mathbb{R}^2\to \mathbb{R}^3$.
    \item Lo que sí podemos hacer es tomar una curva en $S$ que pase por el punto $p$.
\end{itemize}

\begin{definicion}[Vector tangente a una superficie en un punto]~\\
    \noindent
    Para una superficie $S$ y $p\in S$, decimos que $v\in \mathbb{R}^3$ ``\underline{es tangente a $S$ en $p$}'' si existe una curva diferenciable $\alpha:\left]-\varepsilon,\varepsilon\right[\to S$ (para $\varepsilon>0$) tal que: 
    \begin{equation*}
        \alpha(0) = p \quad \text{y}\quad  \alpha'(0) = v
    \end{equation*}

    \noindent
    Al \underline{conjunto} de vectores tangentes a $S$ en $p$ lo representaremos por $T_pS\subset \mathbb{R}^3$.
\end{definicion}~\\

% // TODO: Quizas meter dibujos
\noindent
Sea $S$ una superficice, $p\in S$ y $X:U\to S$ una parametrización arbitraria de $S$ tal que $p\in X(U)$, tenemos entonces que $X^{-1}(p)\in U\subset \mathbb{R}^2$.\\

\noindent
Para $w\in \mathbb{R}^2$ arbitrario, consideramos la aplicción $\beta:R\to \mathbb{R}^2$ dada por:
\begin{equation*}
    \beta(t) = X^{-1}(p) + tw
\end{equation*}
Que verifica $\beta(0) = X^{-1}(p)\in U$ y $\beta'(0) = w$.\\

\noindent
Observemos que $\tr\beta$ no tiene por qué estar contenida en $U$, pero por continuidad de $\beta$ y por ser $U$ abierto existirá $\varepsilon>0$ de manera que $\beta\left(\left]-\varepsilon,\varepsilon\right[\right)\subset U$. Podemos considerar $\alpha=X\circ \beta:\left]-\varepsilon,\varepsilon\right[\to S$, que es una curva diferenciable (como composición de aplicaciones diferenciables), que verifica:
\begin{align*}
    \alpha(0) &= (X\circ \beta)(0) = X(\beta(0)) = X(X^{-1}(p)) = p \\
    T_pS \stackrel{(\ast)}{\ni} \alpha'(0) &= (X\circ \beta)'(0) \stackrel{(\ast\ast)}{=} (dX)_{\beta(0)}(\beta'(0)) = (dX)_{X^{-1}(p)}(w)
\end{align*}
donde tenemos $(\ast)$ por su propia definición y en $(\ast\ast)$ hemos usado la regla de la cadena, puesto que tenemos $\beta:\left]-\varepsilon,\varepsilon\right[\to U$ y $X:U\to \mathbb{R}^3$, ambas aplicaciones diferenciables en el sentido del análisis\footnote{Observemos que todavía no tenemos la regla de la cadena para las nuevas definiciones de diferenciabilidad.}.\\

\noindent
En definitiva, hemos descubierto que $\forall w\in \mathbb{R}^2$, $(dX)_{X^{-1}(p)}(w)\in T_pS$, es decir:
\begin{equation*}
    \boxed{(dX)_{X^{-1}(p)}(\mathbb{R}^2) \subset T_pS}
\end{equation*}
Si ahora tomamos $v\in T_pS$, tenemos entonces que existe $\varepsilon>0$ y $\alpha:\left]-\varepsilon,\varepsilon\right[\to S$ diferenciable con:
\begin{equation*}
    \alpha(0) = p, \qquad \alpha'(0) = v
\end{equation*}
Sea $X:U\to S$ una parametrización con $p\in X(U)$, observemos que $\alpha\left(\left]-\varepsilon,\varepsilon\right[\right)$ no tiene por qué estar dentro de $U$, pero por ser $\alpha$ continua y $\alpha(0)=p\in X(U)$ abierto, ha de existir $0<\delta< \varepsilon$ de manera que $\alpha:\left]-\delta,\delta\right[\to X(U)\subset S$. Tomando ahora:
\begin{equation*}
    \beta=X^{-1}\circ \alpha:\left]-\delta,\delta\right[ \to U\subset \mathbb{R}^2
\end{equation*}
Tenemos entonces que:
\begin{equation*}
    \beta(0) = X^{-1}(\alpha(0)) = X^{-1}(p)
\end{equation*}
Observamos que $\alpha = X\circ \beta$, con lo que aplicando la regla de la cadena (ambas aplicaciones son diferenciables en el sentido del análisis):
\begin{equation*}
    v = \alpha'(0) = (dX)_{\beta(0)}(\beta'(0)) = (dX)_{X^{-1}(p)}(\beta'(0))
\end{equation*}
Observemos que $\beta'(0)\in \mathbb{R}^2$, con lo que hemos demostrado que:
\begin{equation*}
    v \in (dX)_{X^{-1}(p)}(\mathbb{R}^2) \quad\Longrightarrow\quad \boxed{T_pS \subset (dX)_{X^{-1}(p)}(\mathbb{R}^2)}
\end{equation*}~\\

\noindent
Así, acabamos de probar que si $S$ es una superficie, $p\in S$ y $X:U\to S$ es una parametrización con $p\in X(U)$, entonces:
\begin{equation*}
    T_pS = (dX)_{X^{-1}(p)}(\mathbb{R}^2)
\end{equation*}
Como consecuencias:
\begin{enumerate}
    \item Como $(dX)_{X^{-1}(p)}:\mathbb{R}^2\to \mathbb{R}^3$ es una aplicación lineal\footnote{Por ser la diferencial de una aplicación en un punto.}, tenemos entonces que $(dX)_{X^{-1}(p)}(\mathbb{R}^2)\subset \mathbb{R}^3$ es un subespacio vectorial. Más aún, como $dim(\mathbb{R}^2) = 2$ y $(dX)_{X^{-1}(p)}$ es inyectiva\footnote{Se imponía en la propia definición de una superficie, toda parametrización debía verificar para todo punto $q\in U$ que $(dX)_q:\mathbb{R}^2\to\mathbb{R}^3$ fuera inyectiva.}, tenemos que $T_pS=(dX)_{X^{-1}(p)}(\mathbb{R}^2)$ es un \underline{plano vectorial}.

        Podemos ya llamar a $T_pS$ \underline{plano tangente} a $S$ en $p$.

        Acabamos de ver que $T_pS$ es un plano vectorial, aunque siempre se dibuja el plano afín $p+\langle T_pS \rangle $.
    \item $(dX)_{X^{-1}(p)}(\mathbb{R}^2)$ no depende de $X$, pues $T_pS$ no depende de $X$ por su definición.
    \item Como $T_pS = (dX)_{X^{-1}(p)}(\mathbb{R}^2)$, la imagen de cualquier base de $\mathbb{R}^2$ nos dará una base de $T_pS$.

        En particular, $\{(dX)_{X^{-1}(p)}(1,0),(dX)_{X^{-1}(p)}(0,1)\}$ es una base de $T_pS$. Más aún, observemos que (si escribimos $X=X(u,v)$), $\{X_u(X^{-1}(p)),X_v(X^{-1}(p))\}$ es una base de $T_pS$ para cualquier parametrización $X:U\to S$ con $p\in X(U)$.

        Recordemos que la definición de superficie nos garantizaba que: 
        \begin{equation*}
            X_u = \frac{\partial X}{\partial u}, \qquad X_v = \frac{\partial X}{\partial v}
        \end{equation*}
        son linealmente independientes en todo punto $x\in \mathbb{R}^2$.
\end{enumerate}

    Veamos ejemplos de planos tangentes a superficies:
    \begin{enumerate}
        \item Sea $O\subset \mathbb{R}^3$ un abierto y $f:O\to\mathbb{R}$ diferenciable, si $a\in \mathbb{R}$ es un valor regular de $f$ tal que $\emptyset  \neq S = f^{-1}(\{a\})$. Fijamos $p\in S$.

            Tomamos ahora una curva $\alpha:\left]-\varepsilon,\varepsilon\right[\to S$ con:
            \begin{equation*}
                \alpha(0) = p, \qquad \alpha'(0) = v \in T_pS
            \end{equation*}
            Tenemos entonces que $\tr\alpha \subset S = f^{-1}(a)$, es decir, que:
            \begin{equation*}
                f(\alpha(t)) = a \qquad \forall |t|<\varepsilon
            \end{equation*}
            Derivando en $0$ respecto $t$ y usando la regla de la cadena:
            \begin{equation*}
                (df)_{\alpha(0)}(\alpha'(0)) = (d f)_p(v) = 0 \quad\Longrightarrow\quad v\in \ker (d f)_p
            \end{equation*}
            Acabamos de probar que $T_pS\subset \ker (d f)_p$. Además, como $dim T_p = 2$ y $\ker(d f)_p = 2$ (ya que $dim (d f)_p(\mathbb{R}^3) = 1$ por ser no nula, valor regular), tenemos entonces que:
            \begin{equation*}
                T_pS = \ker (d f)_p = \langle \nabla f_p \rangle^\perp
            \end{equation*}
        \item[1a.] Sea $P$ un plano que pasa por $p_0\in \mathbb{R}^3$ y $a$ un vector normal unitario a $P$, tenemos:
            \begin{equation*}
                P = \{p\in \mathbb{R}^3  : \langle p-p_0,a \rangle = 0\} = f^{-1}(\{0\})
            \end{equation*}
            para $f:\mathbb{R}^3\to \mathbb{R}$ dada por $f(p) = \langle p-p_0,a \rangle $, diferenciable y $0$ es valor regular de $f$ (compruébese), por lo que $P$, al no ser vacio, es una superficie (algo que ya sabíamos, pero ahora nos será de utilidad por el apartado anterior). Para $p\in P$ hemos visto en el apartado anterior que:
            \begin{equation*}
                T_pS = \langle (\nabla f)_p \rangle^\perp = \ker(d f)_p
            \end{equation*}

            \begin{equation*}
                (d f)_p(v) = \frac{d}{dt}\big|_{t=0}\ f(p+tv) = \frac{d}{dt}\big|_{t=0}\ \langle p+tv-p_0,a \rangle  = \langle v,a \rangle 
            \end{equation*}
            Así:
            \begin{equation*}
                (\nabla f)_p = a

            De donde:
            \begin{equation*} 
                T_pS = \langle a \rangle^\perp = \vec{P} \qquad \forall p\in S
            \end{equation*}

\begin{definicion}
    Sea $S\subset \mathbb{R}^3$ una superficie y $p\in S$, llamamos ``recta normal a $S$ en $p$'' a la recta afín que pasa por $p$ y es perpendicular a $T_pS$.
\end{definicion}

            \end{equation*}

            Por lo que tenemos que $v\in T_{p'}S$, de donde $T_{p'}S'\subset T_{p'}S$, de donde por ser ambos espacios vectoriales de dimensión 2 tenemos que $T_{p'}S' = T_{p'}S$.

\noindent
Sea $S$ una superficie y $f:S\to \mathbb{R}^m$ diferenciable, no sabemos todavía lo que significa $(d f)_p$, puesto que solo tenemos definida la diferencial para aplicaciones diferenciables en el sentido del análisis.
\begin{definicion}
    Para $S$ una superficie y $f:S\to \mathbb{R}^m$ diferenciable, definimos la diferencial de $f$ en el punto $p\in S$ por una aplicación $(d f)_p:T_pS\to \mathbb{R}^m$, que para cada $v\in T_pS$ tendremos que existirá $\alpha:\left]-\varepsilon,\varepsilon\right[\to S$ con:
    \begin{equation*}
        \alpha(0) = p, \qquad \alpha'(0) = v
    \end{equation*}
    y definiremos la diferencial por:
    \begin{equation*}
        (d f)_p(v) = (f\circ \alpha)'(0) = \frac{d}{dt}\big|_{t=0}\ f(\alpha(t))
    \end{equation*}
\end{definicion}

\noindent
A partir de la definición anterior, parece que $(d f)_p(v)$ depende de la curva $\alpha$ escogida para $v\in T_pS$, pero veremos que realmente $(f\circ \alpha)'(0)$ no depende de $\alpha$, sino solo de $\alpha'(0) = v$, para que $(d f)_p(v)$ esté bien definido. Antes de ello:

\begin{observacion}
    Notemos que el dominio de definición de $(d f)_p$ debe ser $T_pS$, pues necesitamos encontrar una curva $\alpha:\left]-\varepsilon,\varepsilon\right[\to \mathbb{R}^3$ con $\tr\alpha\subset S$, ya que necesitamos realizar la composición $f\circ \alpha$.
\end{observacion}

\noindent
Veamos ahora que la definición de $(d f)_p(v)$ para $v\in T_pS$ no depende de la curva $\alpha$ escogida, sino solo de $v$. Para ello, tomamos una parametrización $X:U\to S$ que tape a $p$, es decir, que $p\in X(U)$. Por continuidad de $\alpha$, podemos considerar sin pérdida de generalidad que $\tr \alpha \subset X(U)$ (ya que si no existirá $\delta>0$ de forma que $\alpha(\left]-\delta,\delta\right[)\subset X(U)$). Podemos ahora escribir:
\begin{equation*}
    f\circ \alpha = (f \circ X) \circ (X^{-1} \circ \alpha)
\end{equation*}
donde $X^{-1}\circ \alpha$ y $f\circ X$ son diferenciables en el sentido del análisis. Derivamos ahora en el cero, usando la regla de la cadena (que por ahora solo la tenemos para aplicaciones del análisis):
\begin{equation*}
    (f\circ \alpha)'(0) = d(f\circ X)_{X^{-1}(p)}\circ (X^{-1}\circ \alpha)'(0)
\end{equation*}
Ahora, tratando de quitar la dependencia de $\alpha$ a la derecha:
\begin{equation*}
    X\circ X^{-1}\circ \alpha = \alpha
\end{equation*}
Derivando en cero:
\begin{equation*}
    v = \alpha'(0) = (dX)_{X^{-1}(p)}\circ (X^{-1}\circ \alpha)'(0)
\end{equation*}
Ahora, como $(dX)_{X^{-1}(p)}$ es inyectiva tenemos que es biyectiva sobre su imagen, con lo que:
\begin{equation*}
    (X^{-1}\circ \alpha)'(0) = [(dX)_{X^{-1}(p)}]^{-1}(v)
\end{equation*}
Así, hemos conseguido:
\begin{equation*}
    (d f)_p(v) = (f\circ \alpha)'(0) = d(f\circ X)_{X^{-1}(p)}\circ (X^{-1}\circ \alpha)'(0) = d(f\circ X)_{X^{-1}(p)}\circ [(dX)_{X^{-1}(p)}]^{-1}(v)
\end{equation*}
Por lo que $(d f)_p(v)$ no depende de $\alpha$, y además vemos que como $d(f\circ X)_{X^{-1}(p)}$ y $[(dX)_{X^{-1}(p)}]^{-1}$ son lineales, $(d f)_p(v)$ depende linealmente de $v$. Además, vemos que $(d f)_p(v)$ no depende de $X$ por su definición, por lo que la fórmula de la derecha tampoco depende de $X$.\\

\noindent
Así, hemos probado además que $(d f)_p:T_pS\to \mathbb{R}^m$ es una aplicación lineal.\\

\begin{definicion}
    Si $S,S'$ son superficies y $O\subset \mathbb{R}^n$ es abierto:
    \begin{itemize}
        \item Si $f:O\to S$ es diferenciable en el sentido $b)$, para $x\in O$ definimos la diferencial de $f$ en el punto $x$ como la aplicación $(d f)_x:\mathbb{R}^n\to T_{f(x)}S\subset \mathbb{R}^3$.
        \item Si $f:S\to S'$ es diferenciable en el sentido $c)$, para $p\in S$ definimos la diferencial de $f$ en el punto $p$ como la aplicación $(d f)_p:T_pS\to T_{f(p)}S'$.
    \end{itemize}
\end{definicion}
\noindent
Pensar sobre estas aplicaciones, sobre cómo han de estar definidas y sobre cómo probar que son lineales.

\begin{ejemplo}
    Veamos ejemplos de cálculo de diferenciales:
    \begin{enumerate}
        \item \textbf{Restricciones.} Sea $O\subset \mathbb{R}^3$ un abierto, si tenemos $F:O\to \mathbb{R}^m$ diferenciable (en el sentido del análisis) y tenemos una superficie $S\subset O$, si tomamos ahora $f=F\big|_S:S\to \mathbb{R}^m$, vimos ya que $f$ es diferenciable, por lo que tiene derecho a una aplicación diferencial.

            Para $p\in S$ podemos considerar $(d f)_p:T_pS\to \mathbb{R}^m$, que para $v\in T_pS$ tenemos que existe $\alpha:\left]-\varepsilon,\varepsilon\right[\to S$ con:
            \begin{equation*}
                \alpha(0) = p, \qquad \alpha'(0) = v
            \end{equation*}
            por lo que:
            \begin{equation*}
                (d f)_p(v) = (f\circ \alpha)'(0) = (F\circ \alpha)'(0) = (dF)_p(v) \qquad \forall v\in T_pS
            \end{equation*}
            Sin embargo, recordemos que $(d F)_p$ está definida en $\mathbb{R}^3$, por lo que no tenemos una igualdad, sino una restricción:
            \begin{equation*}
                (d F\big|_S)_p = (dF)_p\big|_{T_pS}
            \end{equation*}
            En los siguientes ejemplos usaremos sin mencionarlo que:
            \begin{equation*}
                (dF)_p(v) = \frac{d}{dt}\big|_{t=0} F(p+tv)
            \end{equation*}
        \item[1a.] Si consideramos $F:\mathbb{R}^3\to \mathbb{R}$ dada por:
            \begin{equation*}
                F(p) = \langle p-p_0,u \rangle 
            \end{equation*}
            para cierto $u\in \mathbb{R}^3$ con $|u|^2 = 1$ y $p_0\in \mathbb{R}^3$, que nos da la distancia orientada de $p$ al plano que pasa por $p_0$ y que es ortogonal a $u$.

            Consideramos ahora para una superficie $S$ la aplicación $h=F\big|_S:S\to \mathbb{R}$, que es diferenciable y queremos calcular ahora para $p\in S$ la diferencial $(dh)_p:T_pS\to \mathbb{R}$, que para cada $v\in T_pS$ vale:
            \begin{equation*}
                (dh)_p(v) = (dF)_p(v) = \langle v,u \rangle 
            \end{equation*}
        \item[1b.] Consideramos ahora $F:\mathbb{R}^3\to\mathbb{R}$ dada por:
            \begin{equation*}
                F(p) = |p-p_0|^2
            \end{equation*}
            para cierto $p_0\in \mathbb{R}^3$, que nos da la distancia de un punto $p$ a $p_0$ al cuadrado. Consideramos ahora para una superficie $S$, $f=F\big|_S$. Para $p\in S$ calculamos $(d f)_p:T_pS\to \mathbb{R}$:
            \begin{equation*}
                (df)_p(v) = (dF)_p(v) = 2\langle p-p_0,v \rangle 
            \end{equation*}
        \item[1c.] Tomamos $F:\mathbb{R}^3\setminus \{p_0\}\to\mathbb{R}$ para $p_0\in \mathbb{R}^3$, dada por:
            \begin{equation*}
                F(p) = |p-p_0|
            \end{equation*}
            Para una superficie $S$ tal que $p_0\notin S$ tomamos $f=F\big|_S$ y para $p\in S$ podemos considerar $(d f)_p:T_pS\to \mathbb{R}$, dada por:
            \begin{equation*}
                (d f)_p(v) = (dF)_p(v) = \frac{\langle p-p_0,v \rangle }{|p-p_0|}
            \end{equation*}
        \item[1d.] Sea $R$ una recta que pasa por $p_0\in \mathbb{R}^3$ y lleva dirección $u$ con $|u\vDash1$, consideramos $F:\mathbb{R}^3\to \mathbb{R}$ dada por:
            \begin{equation*}
                F(p) = dist(p,R)^2 = |p-p_0|^2 - \langle p-p_0,u \rangle^2
            \end{equation*}
            Para una superficie $S$ consideramos $f=F\big|_S$, que para cada $p\in S$ y $v\in T_pS$ tenemos:
            \begin{equation*}
                (d f)_p(v) = 2\langle p-p_0,v \rangle  - 2\langle p-p_0,u \rangle \langle v,u \rangle 
            \end{equation*}
        \item[2.] \textbf{Inclusiones.} Sea $S$ una superficie, consideramos la aplicación $i:S\hookrightarrow \mathbb{R}^3$, que es diferenciable por ser restricción de la aplicación identidad. Para cada $p\in S$ calculamos su diferencial $(di)_p:T_pS\to \mathbb{R}^3$, dada por:
            \begin{equation*}
                (di)_p = \left(dI_3\big|_S\right)_p = \left((dI_3)_p\right)\big|_{T_pS} = I_3\big|_{T_pS} = i_{T_pS}
            \end{equation*}
            donde esta última es la aplicación inclusión $i_{T_pS}:T_pS\to \mathbb{R}^3$.
        \item[3.] \textbf{Constantes.} Sea $S$ una superficie, consideramos $f:S\to S'$ donde $S'$ es una superficie o un abierto de $\mathbb{R}^m$. Siendo $f$ constantes, es decir, existe $a\in S'$ tal que $f(p) = a\quad \forall p\in S$. Fijado $p\in S$, calculamos $(df)_p:T_pS\to T_aS'$ (pensando que $T_aS'=\mathbb{R}^m$ en el caso de que $S'$ sea un abierto de $\mathbb{R}^m$), que para $v\in T_pS$ calculamos directamente a partir de la definición, existe $\alpha:\left]-\varepsilon,\varepsilon\right[\to S$ con:
            \begin{equation*}
                \alpha(0) = p, \qquad \alpha'(0) = v
            \end{equation*}
            por lo que ($f\circ \alpha$ es constante):
            \begin{equation*}
                (d f)_p(v) = (f\circ \alpha)'(0) = 0
            \end{equation*}
            Así, $(d f)_p\equiv 0\quad \forall p\in S$.
    \end{enumerate}
\end{ejemplo}

\subsection{Propiedades importantes}
\begin{enumerate}
    \item \textbf{Regla de la cadena.} Sean $f:S\to S'$ y $g:S'\to S''$ aplicaciones diferenciables entre superficies y/o abiertos de un euclídeo. Entonces $\forall p\in S$ se tiene que:
        \begin{equation*}
            d(g\circ f)_p = (dg)_{f(p)}\circ (d f)_p
        \end{equation*}
        \begin{proof}
            Queremos ver que el siguiente diagrama es conmutativo:
            \begin{figure}[H]
                \centering
                \shorthandoff{""}
                \begin{tikzcd}
                    d(g\circ f)_p:T_pS \arrow[rr] \arrow[rd, "(df)_p"'] &                                       & T_{g(f(p))}S'' \\
                                                                      & T_{f(p)}S' \arrow[ru, "(dg)_{f(p)}"'] &               
                \end{tikzcd}
                \shorthandon{""}
            \end{figure}
            \noindent
            Para $v\in T_pS$ tenemos que $\exists \alpha:\left]-\varepsilon,\varepsilon\right[\to S$ con $\alpha(0) = p$ y $\alpha'(0) = v$. Observamos que así tenemos $f\circ\alpha:\left]-\varepsilon,\varepsilon\right[\to S'$ que cumple:
            \begin{equation*}
                (f\circ \alpha)(0) = f(p), \qquad (f\circ \alpha)'(0) = (d f)_p(v)
            \end{equation*}
            Para esta mismo $v$ pero aplicando la definición de $d(g\circ f)_p$:
            \begin{equation*}
                d(g\circ f)_p(v) = ((g\circ f)\circ \alpha)'(0) = (g\circ (f\circ \alpha))'(0) \AstIg (d g)_{f(p)}[(d f)_p(v)] 
            \end{equation*}
            donde en $(\ast)$ hemos usado la definición de $dg_{f(p)}:T_{f(p)}S'\to T_{g(f(p))}S''$.
        \end{proof}
    \item \textbf{Extremos locales de una función.} Sea $S$ una superficie y $f:S\to \mathbb{R}$ una aplicación diferenciable, suponemos que $f$ tiene un extremo local en $p_0\in S$, entonces $(d f)_{p_0}\equiv 0$.
        \begin{proof}
            Para $(d f)_{p_0}:T_pS\to \mathbb{R}$ tenemos que para $v\in T_pS$ por definición existe $\alpha:\left]-\varepsilon,\varepsilon\right[\to S$ con $\alpha(0) = p_0$ y $\alpha'(0) = v$. Por ser $p_0$ un extremo local de $f$ (supondremo un máximo), existe $V\subset S$ entorno abierto de $p_0$ de forma que $f(p)\leq f(p_0)\quad \forall p\in V$. Como $\alpha(0) = p_0$ y $V$ es abierto, podemos suponer sin pérdida de generalidad que $\tr\alpha\subset V$, con lo que $0$ es un máximo relativo de $(f\circ \alpha)$, luego $(f\circ \alpha)'(0) = 0$ y recordemos que:
            \begin{equation*}
                (d f)_{p_0}(v) = (f\circ \alpha)'(0) = 0
            \end{equation*}
            Así, obtenemos que $(d f)_{p_0}\equiv 0$.
        \end{proof}
    \item \textbf{Diferencial nula en todo punto.} Sea $S$ una superficie conexa y $f:S\to S'$ una aplicación diferenciable donde $S'$ es una superficie o un abierto de un euclídeo, si $(d f)_p\equiv 0\quad \forall p\in S$, entonces $f$ es constante.
        \begin{proof}
            Sea $a\in f(S)$ consideramos:
            \begin{equation*}
                A = \{p\in S : f(p) = a\} = f^{-1}(\{a\})
            \end{equation*}
            que es no vacío por hipótesis, cerrado en $S$ y bastará ver que $A$ es abierto en $S$ para demostrar que $f$ es constante. Para ello, si tomamos $p\in A$, sabemos que tiene que existir $X:U\subset \mathbb{R}^2\to S$ parametrización con $p\in X(U)$. Podemos escribir así:
            \begin{equation*}
                f\big|_{X(U)} = (f\circ X) \circ X^{-1}
            \end{equation*}
            Usando la regla de la cadena (la propiedad 1), para $q\in X(U)$ tenemos:
            \begin{equation*}
                0\equiv (d f)_q = d(f\circ X)_{X^{-1}(q)}\circ (dX^{-1})_q
            \end{equation*}
            Y como $X^{-1}$ es un difeomorfismo tenemos que $(dX^{-1})_q$ es un isomorfismo, por lo que debemos tener $d(f\circ X)_{X^{-1}(q)}\equiv 0$, para $f\circ X:U\to S'$ una aplicación del análisis (si $S'$ es uan superficie basta componer con la inclusión) que tiene diferencial cero en todo punto, por lo que $f\circ X$ es constante, luego $f\big|_{X(U)}$ es constante, e igual a $a$. Hemos probado que $p\in X(U)\subset A$, por lo que $A$ es abierto en $S$. La arbitrariedad de $p$ nos dice que $A = S$, con lo que $f$ es constante.
        \end{proof}
    \item \textbf{Teorema de la función inversa.} Sea $f:S\to S'$ una aplicación diferenciable donde $S,S'$ son superficies o bien abiertos de $\mathbb{R}^2$. Sea $p_0\in S$ donde tenemos que $(d f)_{p_0}:T_{p_0}S\to T_{f(p_0)}S'$ es regular (isomorfismo\footnote{En este contexto también es equivalente tener rango máximo, ser inyectiva, \ldots}). Entonces existe un entorno abierto $V$ de $p_0$ en $S$ y un entorno abierto $W$ de $f(p_0)$ en $S'$ tales que $f(V)=W$ y con $f\big|_V:V\to W$ un difeomorfismo.
    \item[4.1.] Si $(d f)_p$ es regular $\forall p\in S$, entonces $f$ es un difeomorfismo local. En particular, $f$ es un homeomorfismo local, por lo que $f$ es abierta, de donde $f(S)$ es abierto en $S'$.
\end{enumerate}

\begin{ejemplo}
    Sean $S,S'$ dos superficies con $S'\subset S$, si tomamos $i:S'\hookrightarrow S$ para $p'\in S'$ anteriormente vimos que:
    \begin{equation*}
        (d i)_{p'} = i_{T_{p'}S'}
    \end{equation*}
    pero ahora observamos que:
    \begin{equation*}
        (d i)_{p'}(T_{p'}S')\subset T_{p'}S
    \end{equation*}
    con $(d i)_{p'}$ lineal y ambos de la misma dimensión, con lo que $(d i)_{p'}(T_{p'}S')= T_{p'}S$. En particular, $(di)_{p'}$ es regular $\forall p'\in S'$. Por la propiedad 4.1. deducimos que:
    \begin{equation*}
        i(S') = S'\subset S
    \end{equation*}
    es un abierto. Esta es una primera propiedad geométrica que obtenemos del estudio de la diferenciabilidad. Sabemos ya que la única forma de tener una superficie dentro de otra es que la pequeña sea un abierto en la grande.
\end{ejemplo}
