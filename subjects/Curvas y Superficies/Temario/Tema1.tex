\chapter{Curvas en el plano y en el espacio}
\section{Definición. Curva regular. Longitud de arco}
\begin{definicion}
    Una \textbf{curva} en el espacio es una aplicación diferenciable\footnote{En esta asignatura cada vez que se mencione ``diferenciable'' entenderemos que será derivable infinitas veces. Hacemos esto por aligerar su estudio, pero recomendamos al lector que trate de averiguar qué hipótesis hace falta imponer sobra la regularidad de cada curva en cada caso.} $\alpha:I\to \mathbb{R}^3$ donde $I\subseteq \mathbb{R}$ es un intervalo abierto.\\

    \noindent
    Normalmente, si queremos señalar explícitamente las componentes de $\alpha$, escribiremos:
    \begin{equation*}
        \alpha(t) = (x(t), y(t), z(t))
    \end{equation*}
    En realidad, a un geómetra solo le interesa la llamada ``traza de la curva'':
    \begin{equation*}
        \tr \alpha = \im \alpha = \alpha(I)
    \end{equation*}
\end{definicion}

\begin{ejemplo}
    Sean $a\in \mathbb{R}^3$, $v\in \mathbb{R}^3$, definimos $\alpha:\mathbb{R}\to \mathbb{R}^3$ dada por:
    \begin{equation*}
        \alpha(t) = a + tv 
    \end{equation*}
    En dicho caso:
    \begin{equation*}
        \tr \alpha = \left\{\begin{array}{ll}
            \{a\} & \text{si\ } v=0 \\
             R_{a,v} & \text{si\ } v\neq 0
        \end{array}\right. 
    \end{equation*}
    donde denotamos por $R_{a,v}$ a la única recta que pasar por $a$ y con dirección $v$:
    \begin{equation*}
        R_{a,v} = a + \langle v \rangle 
    \end{equation*}

    \begin{figure}[H]
        \centering
        \begin{tikzpicture}
        \draw (-2,0) -- (2,0);
        \draw[-Stealth] (0,0) -- (1,0);
        \fill (0,0) circle(2pt) node[below] {$a$};
        \fill (0.5, 0) node[above] {$v$};
        \fill (-1, 0) node[above] {$R_{a,v}$};
        \end{tikzpicture}
    \end{figure}
\end{ejemplo}

\begin{definicion}
    Una curva $\alpha:I\to \mathbb{R}^3$ se llama ``\textbf{plana}'' si existe un plano $P\subset \mathbb{R}^3$ tal que $\tr \alpha \subset P$.\\

    \noindent
    Como $P$ y $P(z=0)$ son equivalentes salvo un movimiento rígido\footnote{Es decir, una aplicación afín que conserve longitudes y ángulos.} de $\mathbb{R}^3$, podemos considerar que la curva $\alpha$ está definida como $\alpha:I\to P(z=0)\subset \mathbb{R}^3$, cuyas componentes son:
    \begin{equation*}
        \alpha(t) = (x(t), y(t), 0) \equiv (x(t), y(t))
    \end{equation*}
    De esta forma, podemos abstraernos y pensar que una curva plana es una aplicación diferenciable $\alpha:I\to \mathbb{R}^2$.
\end{definicion}~\\

\noindent
En los libros es común llamar a estar curvas ``parametrizadas'' y ``diferenciables''. En nuestra definición imponemos que una curva ha de ser diferenciable, y el adjetivo parametrizada se debe a la dependencia en cada coordenada de la variable independiente.

\begin{ejemplo}
    Varios ejemplos de curvas:
    \begin{enumerate}
        \item Extrapolando el ejemplo anterior:
            \begin{equation*}
                \alpha(t) = a + tv
            \end{equation*}
            para $a\in \mathbb{R}^2, v\in \mathbb{R}^2$ y $\alpha'(t) = v$. Se trata de un Movimiento Rectilíneo Uniforme. Ahora, vemos que $\alpha''(t) = 0$, no hay aceleración ninguna.
        \item Tomando $a\in \mathbb{R}^2$ y $v\in \mathbb{R}^2\setminus \{0\}$, consideramos $\beta:\mathbb{R}\to \mathbb{R}^2$ dada por:
            \begin{equation*}
                \beta(t) = a+t^2v
            \end{equation*}
            Observamos ahora que tenemos $\tr \beta\subset \tr \alpha$, y la traza de esta curva es la semirrecta de extremo $a$ y dirección $v$:
            \begin{equation*}
                \tr \beta = R_0^+(a,v) 
            \end{equation*}
            \begin{figure}[H]
                \centering
                \begin{tikzpicture}
                \draw (0,0) -- (3,0);
                \draw[-Stealth] (0,0) -- (1,0);
                \fill (0,0) circle(2pt) node[below] {$a$};
                \fill (0.5,0) node[above] {$v$};
                \fill (2,0) node[above] {$\tr \beta$};
                \end{tikzpicture}
            \end{figure}

            Desde el punto de vista físico, muy en el pasado (límite en $-\infty$) estábamos muy alejados del punto $a$. A medida que nos vamos acercando a tiempo $0$ nos vamos acercando al punto $a$ con una velocidad de módulo decreciente que se hace cero cuando alcanzamos el instante $t=0$ y que luego aumenta posteriormente mientras el móvil se aleja del punto $a$ en la dirección en la que vino.

            Vemos que tenemos una velocidad $\beta'(t) = 2tv$, así como una aceleración $\beta''(t) = 2v$, que es constante, por lo que estamos ante un Movimiento Rectilíneo Uniformemente Acelerado.

            El punto $a$ (que se alcanza en $t=0$) es un punto extraño, es el único punto de $\tr \beta$ en su frontera. Además, podemos observar que en dicho punto tenemos $\beta'(0) = 0$.
    \end{enumerate}
\end{ejemplo}

\begin{ejercicio} 
    Estudiar $\gamma(t) = a+t^3 v$, con $a\in \mathbb{R}^2$ y $v\in \mathbb{R}^2\setminus \{0\}$.

    \noindent
    Es claro que $\tr \gamma = R_{a,v} = \tr \alpha$. En este caso tenemos:
    \begin{equation*}
        \gamma'(t) = 3t^2v, \qquad \gamma''(t) = 6tv
    \end{equation*}
    Vemos que la velocidad es siempre positiva y en este caso la aceleración no es constante: es decreciente cuando $t<0$ y es creciente cuando $t>0$, simula la situación en la que un móvil se acerca al punto $a$ frenando cada vez más fuerte y a medida que pasa el punto $a$ comienza a acelerar cada vez más rápido.
\end{ejercicio}

\begin{ejemplo}
    Siguiendo con más ejemplos:
    \begin{enumerate}
        \setcounter{enumi}{2}
        \item Consideramos ahora $\delta:\mathbb{R}\to \mathbb{R}^2$ dada por:
            \begin{equation*}
                \delta(t) = (t^3-4t, t^2-4)
            \end{equation*}
            Para pensar la curva primero analizamos dónde esta corta el eje $x$:
            \begin{equation*}
                t^2-4 = 0 \Longleftrightarrow t = \pm 2
            \end{equation*}
            En dichas abscisas, la curva toma la ordenada:
            \begin{equation*}
                \delta(-2) = 0 = \delta(2)
            \end{equation*}
            Por lo que en ambos instantes de tiempo la curva pasa por el origen. Si estudiamos el corte con el eje $y$ vemos que tiene 3 puntos de corte, dos de ellos ya los conocemos y el que falta es en el instante $t=0$; donde alcanza una ordenada de $-4$. Teniendo en cuenta también los límites en $-\infty$ y en $+\infty$ podemos finalmente deducir que la curva será algo del estilo:
            \ifshowfigures
            \begin{figure}[H]
                \centering
                \begin{tikzpicture}[scale=0.8]
                    % Ejes
                    \draw[-Stealth] (-5,0) -- (5,0) node[right] {$x$};
                    \draw[-Stealth] (0,-4.5) -- (0,3) node[above] {$y$};

                    % Curva parametrizada
                    \draw[domain=-2.5:2.5, samples=200, smooth, thick, blue] 
                        plot ({\x*\x*\x - 4*\x}, {\x*\x - 4});
                \end{tikzpicture}
            \end{figure}
            \fi

            \noindent
            Vemos que esta curva tiene autointersecciones, por lo que las curvas no tienen por qué ser inyectivas.

        \item Si consideramos $\varepsilon:\mathbb{R}\to \mathbb{R}^2$ dada por $\varepsilon(t) = (t^3,t^2)$.

            Su velocidad es $\varepsilon'(t) = (3t^2, 2t)$, que se anula en el origen. Observamos que en el dibujo de la traza vemos un pico.

            Aunque $\varepsilon$ sea diferenciable, $\tr \varepsilon$ tiene ``picos''. Observamos además que $\im \varepsilon$ es la gráfica de la aplicación\footnote{Lo hemos obtenido igualando $x=t^3$, $y=t^2$, despejando $t$ de la primera e igualando en la segunda.} $y=x^{\nicefrac{2}{3}}$. Esta función no es derivable en el origen, a pesar de que la curva sí lo sea.

                    \draw[domain=-1.5:1.5, samples=20, smooth, thick, blue] 
                        plot ({\x*\x*\x}, {\x*\x});
                \end{tikzpicture}
            \end{figure}
            \fi

        \item Sea $\zeta:\mathbb{R}\to \mathbb{R}^3$ dada por $\zeta(t) = a+r\left(\cos\left(\frac{t}{r}\right)e_1 + \sen\left(\frac{t}{r}\right)e_2\right)$ donde $a\in \mathbb{R}^3$, $e_1,e_2$ con $|e_1| = 1 = |e_2|$ con $\langle e_1,e_2 \rangle =0$ y $r>0$.

            Observamos que tenemos siempre:
            \begin{equation*}
                \zeta(t) \in P = a + \langle \{e_1,e_2\} \rangle \qquad \forall t\in \mathbb{R}
            \end{equation*}
            Por lo que:
            \begin{equation*}
                \im \zeta = \tr \zeta \subset P
            \end{equation*}
            Es decir, $\zeta$ es una curva plana. Además, vemos que:
            \begin{equation*}
                |\zeta(t) - a|^2 = r^2 \quad\Longrightarrow\quad |\zeta(t)-a| = r \qquad \forall t\in \mathbb{R}
            \end{equation*}
            Como $\tr \zeta$ no se puede salir del plano y equidista una cantidad $r$ de $a$, tenemos que $\tr \zeta \subset C(a,r)\subset P$.

            A esta curva la llamaremos \textbf{la\footnote{A esta la parametrizaremos siempre de la misma forma, aunque distintas parametrizaciones den la misma traza.}} circunferencia de centro $a$ y radio $r>0$ en $P$.

            Observamos que:
            \begin{equation*}
                \zeta'(t) = -\sen\left(\frac{t}{r}\right)e_1 + \cos\left(\frac{t}{r}\right)e_2 \qquad \forall t\in \mathbb{R}
            \end{equation*}
            No es constante, por lo que no es un MRU. Además vemos que $\zeta''(t)$ no es constante, por lo que tampoco es un MRUA. Sin embargo, apreciamos que $|\zeta''(t)|$ sí que es constante, así como que $|\zeta'(t)| = 1$.

            Se trata de la traza de un Movimiento Circular Uniforme.
        \item Sea $\eta:I\to \mathbb{R}^2$ con $I = \left]-1,+\infty\right[$ dada por:
            \begin{equation*}
                \eta(t) = \left(\frac{3t}{1+t^3}, \frac{3t^2}{1+t^3}\right)
            \end{equation*}
            Si pensamos en su representación, tras un poco de análisis (límite en más y menos infinito, puntos de corte con los ejes, observar que si $t>0$ siempre se encuentra en el primer cuadrante, \ldots) podemos llegar a deducir que su forma ha de ser similar a algo como:
            \ifshowfigures
            \begin{figure}[H]
            \centering
            \begin{tikzpicture}
                % Ejes
                \draw[-Stealth] (-2,0) -- (2,0) node[right] {$x$};
                \draw[-Stealth] (0,-0.5) -- (0,2) node[above] {$y$};

                % Curva parametrizada
                \draw[domain=-0.6:20, samples=100, smooth, thick, blue] 
                    plot ({3*\x/(1 + \x*\x*\x)}, {3*\x*\x/(1 + \x*\x*\x)});

                \draw[thick, blue] (0.01,0) -- (0.01,0.2);
            \end{tikzpicture}
            \end{figure}
            \fi

            \noindent
            Vemos que $\eta(0) = (0,0)$ y que $\lim_{t\to+\infty}\eta(t) = (0,0)$, pero a pesar de ello la curva no se autointerseca, ya que podemos probar que es inyectiva:

            \noindent
            Sean $u,t\in \mathbb{R}$ con $u,t>-1$ tales que $\eta(t) = \eta(u)$ tenemos entonces que:
            \begin{equation*}
                \left.\begin{array}{l}
                        \dfrac{3t}{1+t^3} = \dfrac{3u}{1+u^3} \\
                        \\
                        \dfrac{3t^2}{1+t^3} = \dfrac{3u^2}{1+u^3} 
                \end{array}\right\} \quad\Longrightarrow\quad \dfrac{3u^2}{1+u^3} = t\cdot \dfrac{3t}{1+t^3} = \dfrac{3ut}{1+u^3} 
            \end{equation*}
            Si $u\neq 0$ podemos dividir entre $u$ y concluir que $u=t$ y si $u=0$ ha de ser $t=0 =u$ para obtener $\eta(t) = \eta(u)$. Hemos probado que $\eta$ es inyectiva.

            Vemos otro hecho que hemos de tener en cuenta, y es que en este ejemplo $I$ no es homeomorfo a $\tr \eta$, puesto que si hubiera un homeomorfismo, podemos tomar cualquier entorno de $\eta(0)$ este ha de contener una bola abierta de centro $\eta(0)$ y radio $r>0$ lo suficientemente pequeña para que su intersección con $\tr \eta$ menos $\eta(0)$ tenga 3 componentes conexas y la correspondiente imagen de este conjunto mediante el homeomorfismo tendría 2, lo que llevaría a una contradicción.

            Si a esta curva le añadimos otra para que sea simétrica respecto al eje $y=x$ obtenemos el \textit{folium de Descartes}.
    \end{enumerate}
\end{ejemplo}

\begin{definicion}
    Sea $\alpha:I\to \mathbb{R}^3$ una curva, definimos la \textbf{recta tangente} a la curva $\alpha$ en el instante $t\in I$ como la recta afín $\alpha(t) + \langle \alpha'(t) \rangle $, que denotaremos por $R_{\alpha(t),\alpha'(t)} \equiv R_t$.\\

    \noindent
    Observemos que si $\alpha'(t) = 0$ para cierto $t\in I$, $\alpha(t) + \langle 0 \rangle $ no es una recta, es decir, en los puntos donde $\alpha'(t) = 0$ no hay\footnote{No diremos que la recta tangente es un punto.} recta tangente.\\

    \noindent
    Los puntos $\alpha(t)$ de $\tr \alpha$ tales que $\alpha'(t) \neq 0$ se llaman \textbf{regulares}. En otro caso, los llamaremos \textbf{singulares}. La curva $\alpha$ se dice \textbf{regular} si $\alpha'(t)\neq 0 \quad \forall t\in I$.
\end{definicion}

\begin{ejercicio}
    Determinar las curvas regulares que han aparecido en los ejemplos anteriores.
    \begin{itemize}
        \item $\alpha(t) = a+tv$, teníamos $\alpha'(t) = v$, por lo que $\alpha$ es regular si y solo si $v\neq 0$.
        \item $\beta(t) = a+t^2v$, tenemos $\beta'(t) = 2tv$, por lo que la curva no es regular, ya que el punto $\beta(0)$ es singular.
        \item $\gamma(t) = a+t^3v$, tenemos $\gamma'(t) = 3t^2v$, por lo que la curva no es regular, puesto que $\gamma(0)$ es un punto singular.
        \item $\delta(t) = (t^3-4t, t^2-4)$, tenemos $\delta'(t) = (3t^2-4, 2t)\neq 0 \quad \forall t\in \mathbb{R}$ (la segunda componente solo se anula si $t=0$ y a la primera no le sucede esto), por lo que es una curva regular.
        \item $\varepsilon(t) = (t^3,t^2)$, $\varepsilon'(t) = (3t^2, 2t)$ no es regular, el punto $\varepsilon(0)$ es singular.
        \item $\zeta(t) = a+r(\cos(\nicefrac{t}{r})e_1 + \sen(\nicefrac{t}{r})e_2)$, teníamos que $|\zeta'(t)| = 1$ para todo $t\in \mathbb{R}$, por lo que se trata de una curva regular.
        \item $\eta:\left]-1,+\infty\right[\to \mathbb{R}^2$, dada por:
            \begin{equation*}
                \eta(t) = \left(\frac{3t}{1+t^3}, \frac{3t^2}{1+t^3}\right)
            \end{equation*}
            Tenemos que:
            \begin{equation*}
                \eta'(t) = \left(\frac{3(1-2t^3)}{{(1+t^3)}^{2}}, \frac{3t(2-t^3)}{{(1+t^3)}^{2}}\right)
            \end{equation*}
            Veamos si se anula la derivada en algún punto:
            \begin{equation*}
                \left\{\begin{array}{rcc}
                        3-6t^3 = 0 & \Longleftrightarrow& t = \sqrt[3]{\frac{1}{2}} \\
                        6t-3t^3 = 0 &\Longleftrightarrow& t = 0 \text{\ ó\ } t = \pm \sqrt{2}
                \end{array}\right.
            \end{equation*}
            Como ambos sucesos son incompatibles tenemos que $\eta'(t) \neq 0\quad \forall t\in \left]-1,+\infty\right[$.
    \end{itemize}
\end{ejercicio}

\subsection{Longitud de una curva}
\begin{definicion}
    Sea $\alpha:I\to \mathbb{R}^3$ una curva y $a,b\in \mathbb{R}$ con $[a,b]\subset I$, definimos la \textbf{longitud} de $\alpha$ entre\footnote{Algunos libros mencionan en su lugar ``entre $\alpha(a)$ y $\alpha(b)$''.} $a$ y $b$ correspondiente a la partición 
    \begin{equation*}
        P = \{a = t_0 < t_1  < \ldots < t_n = b\}
    \end{equation*}

    como:
    \begin{equation*}
        L_a^b(\alpha, P) = \sum_{i=1}^{n} |\alpha(t_i) - \alpha(t_{i-1})|
    \end{equation*}
\end{definicion}

\noindent
Observemos que si aplicamos la desigualdad triangular un número finito de veces obtenemos que:
\begin{equation*}
    L_a^b(\alpha, P) \geq |\alpha(t_n) - \alpha(t_0)| = |\alpha(b) - \alpha(a)|
\end{equation*}
Y observemos que esta desigualdad no depende de la partición $P$ escogida.

\noindent
Además, la igualdad se dará si en todas las desigualdades triangulares se daba la igualdad, es decir, si toda la partición se podía contener en un segmento. Así, el camino más corto entre dos puntos parece ser un segmento. Vemos que esta cota inferior es una mala aproximación del valor que estamos buscando, que es la longitud de la curva $\alpha$ entre $a$ y $b$.\\

\noindent
Ahora, podemos ver que:
\begin{align*}
    L^b_a(\alpha,P) &= |\alpha(t_1) - \alpha(t_0)| + \ldots + |\alpha(t_n) - \alpha(t_{n-1})| \\
                    &= \left|\int_{t_0}^{t_1} \alpha'(t)~dt \right| + \ldots + \left|\int_{t_{n-1}}^{t_n} \alpha'(t)~dt \right| \\
                    &\leq \int_{t_0}^{t_1} |\alpha'(t)|~dt  + \ldots + \int_{t_{n-1}}^{t_n} |\alpha'(t)|~dt = \int_{t_0}^{t_n} |\alpha'(t)|~dt  = \int_{a}^{b} |\alpha'(t)|~dt 
\end{align*}
Y esta cantidad tampoco depende de la partición $P$ escogida. Así:
\begin{equation*}
    |\alpha(b) - \alpha(a)| \leq L_a^b(\alpha,P) \leq \int_{a}^{b} |\alpha'(t)|~dt  \qquad \forall P \text{\ partición de\ } [a,b]
\end{equation*}
Y la última desigualdad tiene una gran interpretación física, pues si recordamos la fórmula $e=vt$ de la física, vemos que $|\alpha'(t)|$ es la velocidad y la multiplicamos por una diferencia de tiempo pequeña $dt$, obteniendo así la ``suma'' (integral) de las diferencias de espacio. \\

\noindent
Supuesto ahora que $a\neq b$, definimos $f:[a,b]\times [a,b]\times [a,b] \to \mathbb{R}$ como:
\begin{equation*}
    f(\beta,\gamma,\eta) = +\sqrt{x'(\beta)^2 + y'(\gamma)^2 + z'(\eta)^2} \qquad \forall \beta,\gamma,\eta\in [a,b]
\end{equation*}
para $x,y,z$ las componentes de $\alpha$: $\alpha = (x,y,z)$. La aplicación $f$ es continua sobre el compacto $[a,b]^3$, por lo que es uniformemente continua, es decir, dado $\varepsilon>0$ existe $\delta>0$ tal que si $t_1,t_2,t_3,t_1',t_2',t_3'\in [a,b]$, con $|t_1-t_1'|, |t_2-t_2'|, |t_3-t_3'| < \delta$ se tiene que:
\begin{equation*}
    |f(t_1,t_2,t_3) - f(t_1',t_2',t_3')| < \frac{\varepsilon}{b-a}
\end{equation*}
Sea $P$ una partición de $[a,b]$ con $|P|< \delta$ (es decir, que $|t_i - t_{i-1}| < \delta$), vemos que:
\begin{align*}
    \left|L_a^b(\alpha,P) - \int_{a}^{b} |\alpha'(t)|~dt \right| &= \left|\sum_{i=1}^{n}\left(|\alpha(t_i)-\alpha(t_{i-1})| - \int_{t_{i-1}}^{t_i} |\alpha'(t)|~dt\right) \right|  \\
                                                          &\leq \sum_{i=1}^{n} \left|\left(|\alpha(t_i)-\alpha(t_{i-1})|- \int_{t_{i-1}}^{t_i} |\alpha'(t)|~dt \right)\right| \\
                                                          &\AstIg \sum_{i=1}^{n}\left|f(\beta_i,\gamma_i,\eta_i)(t_i-t_{i-1}) - f(\varepsilon_i,\varepsilon_i,\varepsilon_i)(t_{i}-t_{i-1})\right| \\
                                                          &= \sum_{i=1}^{n} \left|f(\beta_i,\gamma_i,\eta_i)-f(\varepsilon_i,\varepsilon_i,\varepsilon_i)\right|(t_i - t_{i-1}) \\
                                                          &< \frac{\varepsilon}{b-a}\sum_{i=1}^{n} (t_{i-1}-t_i) = \frac{\varepsilon}{b-a}\cdot (b-a) = \varepsilon
\end{align*}
donde en $(\ast)$ hemos aplicado el Teorema del valor Intermedio, teniendo en cuenta que $\alpha=(x,y,z)$ y que este Teorema solo aplica para funciones reales de variable real, por lo que en cada intervalo $[t_{i-1},t_i]$ encontramos un punto distinto. Hemos aplicado también el Teorema del Valor Medio Integral, a la función real de variable real:
\begin{equation*}
    t\longmapsto |\alpha'(t)|
\end{equation*}
Posteriormente hemos usado que como $|P|<\delta$ tendremos que $|t_{i}-t_{i-1}|<\delta$, con $\beta_i,\gamma_i,\eta_i,\varepsilon_i \in [t_{i-1}-t_i]$, de donde:
\begin{equation*}
    |f(\beta_i,\gamma_i,\eta_i)-f(\varepsilon_i,\varepsilon_i,\varepsilon_i)| < \frac{\varepsilon}{b-a}
\end{equation*}
Acabamos de ver que $\forall \varepsilon>0~~\exists \delta>0$ tal que
\begin{equation*}
    |P| < \delta \quad\Longrightarrow\quad \left|L_a^b(\alpha,P)-\int_{a}^{b} |\alpha'(t)|~dt \right| < \varepsilon
\end{equation*}

es decir:
\begin{equation}\label{eq:lim}
    \lim_{|P|\to0} L_a^b(\alpha,P) =  \int_{a}^{b} |\alpha'(t)|~dt 
\end{equation}
Todo esto motiva la siguiente definición:

\begin{definicion}
    Sea $\alpha:I\to \mathbb{R}^3$ Una curva, para $a,b\in I$ con $[a,b]\subset I$ definiremos la longitud de $\alpha$ entre $a$ y $b$ como:
    \begin{equation*}
        L_a^b(\alpha) = \int_{a}^{b} |\alpha'(t)|~dt 
    \end{equation*}
\end{definicion}

\begin{ejercicio} 
    Probar que $L_a^b(\alpha) = \sup_{P}L_a^b(\alpha,P)$.\\

    \noindent
    Hemos visto anteriormente que:
    \begin{equation*}
        L_a^b(\alpha,P) \leq L_a^b(\alpha) \qquad \forall P \text{\ partición de\ } [a,b]
    \end{equation*}
    Por lo que $L_a^b(\alpha)$ es un mayorante del conjunto del que buscamos supremo. Además, la condición~\eqref{eq:lim} nos permite argumentar que si tomamos una sucesión de particiones $\{P_n\}$ de $[a,b]$ con $\{|P_n|\}\to 0$ (nos sirve cualquiera) tenemos entonces que:
    \begin{equation*}
        \{L_a^b(\alpha,P_n)\} \to L_a^b(\alpha)
    \end{equation*}
    Por lo que tenemos un mayorante que es límite una sucesión de elementos del conjunto, dicho mayorante debe ser el supremo.
\end{ejercicio}

\begin{ejemplo}
    Apliquemos esta nueva definición a varias curvas:
    \begin{enumerate}
        \item Para $p\in \mathbb{R}^3$ y $v\in \mathbb{R}^3$, consideramos $\alpha(t) = p+tv$, $t\in \mathbb{R}$.

            Fijados $a,b\in \mathbb{R}$ con $a\leq b$, tenemos:
            \begin{equation*}
                L_a^b(\alpha) = \int_{a}^{b} |\alpha'(t)|~dt  = \int_{a}^{b} |v|~dt  = |v|(b-a)
            \end{equation*}
            Así:
            \begin{itemize}
                \item Si $v=0$, $L_a^b(\alpha) = 0$.
                \item Si $v\neq 0$, recuperamos la fórmula $e = |v|t$ del MRU.
            \end{itemize}
        \item Consideramos ahora $\beta:\mathbb{R}\to \mathbb{R}^2$ que queremos que sea:
            \begin{equation*}
                \tr \beta = C(a,r), \qquad a\in \mathbb{R}^2,\quad  r>0
            \end{equation*}
            Así:
            \begin{equation*}
                \beta(t) = (a_1, a_2) + r\left(\cos\left(\frac{t}{r}\right), \sen\left(\frac{t}{r}\right)\right)
            \end{equation*}
            donde hemos dividido entre $r$ para obtener velocidad de 1. Calculamos la longitud tras dar una vuelta:
            \begin{equation*}
                L_0^{2\pi r}(\beta) = \int_{0}^{2\pi r} |\beta'(t)|~dt = \int_{0}^{2\pi r}~dt = 2\pi r
            \end{equation*}
            Observamos que el centro no tiene nada que ver con la longitud de la curva.
        \item Sea $\gamma:\mathbb{R}\to \mathbb{R}^2$ dada por:
            \begin{equation*}
                \gamma(t) = (e^{-t}\cos t, e^{-t}\sen t) = e^{-t}(\cos t, \sen t)
            \end{equation*}
            Sean $a,b\in \mathbb{R}$ con $a\leq b$, se pide calcular $L_a^b(\alpha)$.

            Es un buen ejercicio tratar de determinar la forma de la curva.

            Se trata de una espiral logarítmica. Calculamos ahora la longitud entre dos intantes de tiempo. Para ello, derivaremos la expresión:
            \begin{equation*}
                \gamma'(t) = -e^{-t}(\cos t, \sen t) + e^{-t}(-\sen t, \cos t) 
            \end{equation*}
            Y estamos interesados en el módulo de la misma:
            \begin{equation*}
                |\gamma'(t)|^2 = e^{-2t} + e^{-2t} + \cancelto{0}{2\langle -e^{-t}(\cos t,\sen t),e^{-t}(-\sen t,\cos t) \rangle}  = 2e^{-2t}
            \end{equation*}
            Y además vemos que $|\gamma'(t)|^2>0$ para todo $t\in \mathbb{R}$. Seguimos calculando:
            \begin{equation*}
                |\gamma'(t)| = \sqrt{2}e^{-t}
            \end{equation*}

            de donde:
            \begin{equation*}
                L_a^b(\gamma) = \int_{a}^{b} \sqrt{2}e^{-t}~dt  = \sqrt{2}\left[-e^{-t}\right]_a^b = \sqrt{2}\left(e^{-a}-e^{-b}\right)
            \end{equation*}
            Y como $a\leq b$ tenemos que $L_a^b(\gamma)\geq 0$. Además, vemos que:
            \begin{equation*}
                L_a^\infty(\gamma) := \lim_{b\to\infty}L_a^b(\gamma) = \sqrt{2}e^{-a}
            \end{equation*}
            \underline{Es una ``cantidad infinita'' de curva de longitud finita}.
    \end{enumerate}
\end{ejemplo}

\section{Curvas a partir de curvas}
\subsection{Reparametrizaciones}
\noindent
Dada una curva $\alpha:I\to \mathbb{R}^3$, queremos poder componer $\alpha$ con una aplicación $h:J\to I$. ¿Qué condiciones tiene que cumplir $h$ para que $\alpha\circ h$ siga siendo una curva? $h$ debe ser diferenciable y $J$ debe ser un intervalo abierto de $\mathbb{R}$.\\

\noindent
La nueva curva $\beta = \alpha\circ h$ es una parametrizacaión de $\alpha$ por medio de $h$. Veamos cuál es su nueva velocidad:
\begin{equation*}
    \beta'(t) = (\alpha\circ h)'(t) = h'(t)\alpha'(h(t))  \qquad \forall t\in J
\end{equation*}
Supongamos ahora que $\alpha$ es regular y queremos que $\beta$ también lo sea. Para ello, en vista de la expresión superior, tendremos que suponer que $h'(t)\neq 0 \quad \forall t\in J$, es decir, que bien $h'(t)>0 \quad \forall t\in J$ o bien que $h'(t)<0\quad \forall t\in J$. En vista de estas condiciones, tendremos que $h:J\to I$ es un \textbf{difeomorfismo\footnote{Pues será inyectiva por ser estrictamente creciente y al tener derivada distinta de cero el Teorema de la función inversa nos dice que la aplicación admite una inversa, que también será diferenciable. Recordamos que difeomorfismo es una aplicación diferenciable de inversa diferenciable.}} (bien creciente o decreciente).

\noindent
Tenemos además que:
\begin{equation*}
    \tr \beta = \tr (\beta\circ h)(J) = \alpha(h(J)) \subset \tr \alpha
\end{equation*}
Por lo que para obtener $\tr \beta = \tr \alpha$ debemos imponer que $h(J) = I$, es decir, que $h$ sea \textbf{sobreyectiva}.\\

\noindent
Así, si estamos en la situación:
\begin{equation*}
    J\stackrel{h}{\to}I\stackrel{\alpha}{\to}\mathbb{R}^3
\end{equation*}
Tomamos $[a,b]\subset J$ y queremos calcular la longitud de $\beta = \alpha\circ h$ entre $a$ y $b$:
\begin{equation*}
    L_a^b(\beta) = \int_{a}^{b} |\beta'(t)|~dt  = \int_{a}^{b} |h'(t)||\alpha'(h(t))|~dt 
\end{equation*}
Hemos llegado a la fórmula del Teorema de cambio de variable, tomando $u = h(t)$:
\begin{equation*}
    \boxed{
    L_a^b(\beta) = \int_{a}^{b} |h'(t)||\alpha'(h(t))|~dt \AstIg \int_{h(a)}^{h(b)} |\alpha'(u)|~du  = L_{h(a)}^{h(b)}(\alpha)}
\end{equation*}
Así, \underline{la longitud es invariante por reparametrizaciones}.

\subsection{Deformaciones}
\noindent
Dada una curva $\alpha:I\to \mathbb{R}^3$, queremos poder componer $\alpha$ con una aplicación $\Phi:\mathbb{R}^3\to \mathbb{R}^3$. ¿Qué condiciones tiene que cumplir $\Phi$ para que $\beta = \Phi\circ \alpha$ siga siendo una curva? $\Phi$ ha de ser diferenciable.\\

\noindent
¿Qué le pasa ahora a la traza de la curva al aplicarle $\Phi$?
\begin{equation*}
    \tr \beta = (\Phi\circ \alpha)(I) = \Phi(\alpha(I)) = \Phi(\tr\alpha)
\end{equation*}
Como vemos, la traza de la curva no se mantiene. Mucho menos lo hará su longitud.\\

\noindent
Ahora, si $\alpha$ es regular, queremos que $\beta$ también sea regular:
\begin{equation*}
    \beta'(t) = (\Phi\circ \alpha)'(t) = (d\Phi)_{\alpha(t)} (\alpha'(t))
\end{equation*}

\begin{notacion}
    En Geometría se usa la notación:
    \begin{equation*}
        (d\Phi)_{\alpha(t)}(\alpha'(t)) := (D\Phi)(\alpha(t), \alpha'(t))
    \end{equation*}
\end{notacion}
\noindent
Tenemos que $\alpha'(t)\neq 0$, por lo que para obtener $\beta'(t)\neq 0$ necesitamos que la aplicación lineal $(d\Phi)_{\alpha(t)}$ lleve vectores distintos de cero en vectores distintos de cero, es decir, que $(d\Phi)_{\alpha(t)}$ sea biyectiva.\\

\noindent
Así, necesitamos que $\Phi$ sea un \textbf{difeomorfismo local}. Usaremos habitualmente difeomorfismos.

\subsubsection{Movimientos rígidos}
\noindent
Hay difeomorfismos $\Phi:\mathbb{R}^3\to \mathbb{R}^3$ que conservan la ``geometría'', los movimientos rígidos $M:\mathbb{R}^3\to\mathbb{R}^3$ dada por:
\begin{equation*}
    Mx = Ax + b 
\end{equation*}
Con\footnote{Una matriz $3\times 3$ es $O(3)$ (ortogonal) si su inversa coincide con su traspuesta, condición que buscamos si queremos que las matrices mantengan invariante el producto escalar.} $A\in O(3)$ y $b\in \mathbb{R}^3$. En este caso las trazas son geométricamente equivalentes:
\begin{equation*}
    \tr\beta = M(\tr \alpha)\equiv \tr\alpha
\end{equation*}
Veamos qué ocurre con las longitudes, tomamos $[a,b]\subset I$ y calculamos:
\begin{equation*}
    L_a^b(\beta) = \int_{a}^{b} |\beta'(t)|~dt  = \int_{a}^{b} |A\alpha'(t)|~dt  = \int_{a}^{b} |\alpha'(t)|~dt  = L_a^b(\alpha)
\end{equation*}
\underline{Vemos que los movimientos rígidos conservan la longitud}.\\

\subsection{Curvas parametrizadas por el arco}
\begin{definicion}
    Sea $\alpha:I\to \mathbb{R}^3$ una curva y $a\in I$, definimos\footnote{Donde $S$ viene de \textit{space}.} $S_a:I\to \mathbb{R}$ dada por:
    \begin{equation*}
        S_a(t) = \int_{a}^{t} |\alpha'(u)|~du 
    \end{equation*}
    Observamos que:
    \begin{equation*}
        S_a(t) = \left\{\begin{array}{rl}
            L_a^t(\alpha) & \text{si\ } a\leq t \\
            -L_t^a(\alpha) & \text{si\ } a \geq t
        \end{array}\right. 
    \end{equation*}
    Podemos llamar a esta aplicación ``longitud orientada de $\alpha$ respecto del instante $a\in I$''.
\end{definicion}

\begin{observacion}
    Vemos que $S_a\in C^1(I)$, con derivada $S_a'(t) = |\alpha'(t)|$. En caso de que $\alpha$ sea regular observamos además que $S_a'$ es una aplicación diferenciable, y lo mismo le sucede a $S_a:I\to S_a(I)=J$, por lo que será un difeomorfismo. En otro caso $S_a'$ será solo continua.

    \noindent
    Este es otro motivo por el que preferir las curvas regulares. Así, a continuación se considerarán siempre curvas regulares, salvo que se especifique lo contrario.
\end{observacion}~\\

\noindent
Suponiendo así que $\alpha$ es regular, sea $h = {(S_a)}^{-1}:J\to I$, podemos componer esta aplicación con $\alpha:I\to \mathbb{R}^3$, obteniendo $\beta = \alpha\circ h:J\to \mathbb{R}^3$, la reparametrización de $\alpha$ por medio del difeomorfismo $h = {(S_a)}^{-1}$.

Observemos que para $s\in J$ tenemos $S_a(h(s)) = s\quad \forall s\in J$, y derivando:
\begin{equation*}
    1 = h'(s)S_a'(h(s)) = h'(s)|\alpha'(h(s))| \qquad \forall s\in J
\end{equation*}

de donde:
\begin{equation*}
    h'(s) = \frac{1}{|\alpha'(h(s))|} \qquad \forall s\in J
\end{equation*}
Vemos así que:
\begin{equation*}
    \beta'(s) = h'(s)\alpha'(h(s)) = \frac{\alpha'(h(s))}{|\alpha'(h(s))|} \quad \forall s\in J \quad\Longrightarrow\quad |\beta'(s)| = 1 \quad \forall s\in J
\end{equation*}
Así, toda curva regular admite una reparametrización con celeridad constantemente igual a 1. A todas estas curvas les pasa que:
\begin{equation*}
    L_a^b(\beta) = \int_{a}^{b} |\beta'(t)|~dt  = \int_{a}^{b}dt  = b-a
\end{equation*}

\begin{definicion}
    Sea $\alpha:I\to \mathbb{R}^3$ una curva con celeridad constantemente igual a 1:
    \begin{equation*}
        |\alpha'(t)| = 1 \qquad \forall t\in I
    \end{equation*}
    Entonces decimos que $\alpha$ es una ``curva parametrizada por el arco'' (abreviado como p.p.a.).
\end{definicion}

\noindent
Las curvas parametrizadas por el arco son de gran importancia cuando queremos hacer un estudio geométrico de las curvas, pues al recorrerse con celeridad constante toda aceleración debe ser ortogonal a la dirección de velocidad de la curva en cada instante, ya que si no la aceleración aumentaría o disminuiría la celeridad del móvil. Así, el estudio de la segunda derivada de la curva (aceleración) queda restringida a un plano de estudio geométrico, que será el que nos interese más adelante cuando estudiemos el concepto de curvatura de una curva.\\

\noindent
Siguiendo por donde íbamos, acabmos de ver que toda curva regular admite una parametrización por el arco.

\begin{ejercicio}
    Si $\alpha:I\to \mathbb{R}^3$ es una curva parametrizada por el arco, ¿cuántas reparametrizaciones por el arco admite?\\

    \noindent
    Sea $h:J\to I$ una reparametrización de $\alpha$ (es decir, un difeomorfismo), si $\alpha$ era una curva p.p.a. y si queremos que $\beta = \alpha\circ h$ también lo sea debemos exigir que:
    \begin{equation*}
        1 = |\beta'(t)| = |h'(t)\alpha'(h(t))| = |h'(t)||\alpha'(h(t))| = |h'(t)| \qquad \forall t\in I
    \end{equation*}
    En este caso, tendremos que:
    \begin{equation*}
        h'(t) \in \{-1,1\}\qquad \forall t\in I
    \end{equation*}
    Sin embargo, como $h$ es un difeomorfismo, $h'$ debe ser una función continua, por lo que tendremos entonces que $h'(t) = 1\quad \forall t\in I$ o bien $h'(t) = -1\quad \forall t\in I$. En el primer caso tendríamos que $h(t) = t+c$ para cierta constante $c\in \mathbb{R}$ y en el segundo que $h(t) = -t+c$, para otra cierta constante $c\in \mathbb{R}$. En definitiva, tenemos que $h$ es de la forma:
    \begin{equation*}
        h(t) = \pm t + c, \qquad c\in \mathbb{R}
    \end{equation*}
\end{ejercicio}

\begin{ejercicio}
    Si $\alpha:I\to \mathbb{R}^2$ es una curva regular plana, llamamos recta normal a la curva $\alpha$ en el instante $t\in I$ a la recta que pasa por $\alpha(t)$ y es perpendicular a la recta tangente en dicho instante.\\

    \noindent
    Se pide ver cómo son las rectas normales de una recta y de una circunferencia. ¿Hay otras curvas que satisfagan algunas de esas dos propiedades?\\

    \noindent
    Si todas las rectas normales a una curva son paralelas tendremos entonces que todas las rectas tangentes son paralelas, es decir, si $s\in I$, tenemos que la recta tangente a la curva es:
    \begin{equation*}
        \alpha(s) + \langle \alpha'(s) \rangle 
    \end{equation*}
    Como las rectas tangentes son todas paralelas entre sí debemos tener (y como $\alpha$ es regular):
    \begin{equation*}
        \alpha'(s) = \lm(s)v, \qquad v\in \mathbb{R}^2\setminus \{0\}, \quad \lm(s) \neq 0\quad \forall s\in I
    \end{equation*}
    Puede probarse que $\lm$ es diferenciable a partir de esta expresión, pues multiplicando escalarmente por $v$ podemos despejar $\lm(s)$:
    \begin{equation*}
        \lm(s) = \frac{\langle \alpha'(s),v \rangle }{|v|^2}
    \end{equation*}
    con lo que bien tenemos $\lm(s)> 0 $ o $\lm(s)<0$ para todo $s\in I$. Ahora para $s_0\in I$ tenemos:
    \begin{equation*}
        \int_{s_0}^{s} \alpha'(u)~du = \left(\int_{s_0}^{s} \lm(u)~du \right)v\qquad \forall s\in I
    \end{equation*}
    De donde:
    \begin{equation*}
        \alpha(s) - \alpha(s_0) = f(s)v \quad\Longrightarrow\quad \alpha(s) = \alpha(s_0) + f(s)v \in R_{\alpha(s_0),v} \qquad \forall s\in I
    \end{equation*}
    Por lo que:
    \begin{equation*}
        \tr\alpha\subset R_{\alpha(s_0),v}
    \end{equation*}
    De donde $\tr\alpha$ es un segmento abierto (pensar por qué) de recta.\\

    \noindent
    Si ahora todas las rectas normales a la curvas pasan por el mismo punto (o bien que todas sus rectas normales son concurrentes), sea $t\in I$, la recta normal a $\alpha$ en el instante $t$ es la recta que pasa por $\alpha(t)$ y lleva dirección $J\alpha'(t)$:
    \begin{equation*}
        \alpha(t) + \lm J\alpha'(t), \qquad \lm \in \mathbb{R}
    \end{equation*}
    Sabemos que existe un punto $c\in \mathbb{R}^2$ tal que $c$ pertenece a todas estas rectas:
    \begin{equation*}
        \forall t\in I~~\exists \lm(t)~:~ \alpha(t) + \lm(t)J\alpha'(t) = c
    \end{equation*}
    Tenemos así una aplicación $\lm:I\to \mathbb{R}$ que es diferenciables puesto que:
    \begin{equation*}
        \lm(t) J\alpha'(t) = c - \alpha(t) \qquad \forall t\in I
    \end{equation*}

    de donde:
    \begin{equation*}
        \lm(t) |\alpha'(t)|^2 = \langle \lm(t)J\alpha'(t),J\alpha'(t) \rangle  = \langle c-\alpha(t),J\alpha'(t) \rangle   \qquad \forall t\in I
    \end{equation*}
    Por lo que:
    \begin{equation*}
        \lm(t) = \frac{\langle c-\alpha(t),J\alpha'(t) \rangle }{|\alpha'(t)|^2} \qquad \forall t\in \mathbb{R}
    \end{equation*}
    Si ahora vemos que:
    \begin{equation*}
        \alpha(t)-c = -\lm(t)J\alpha'(t) \qquad \forall t\in I
    \end{equation*}
    Multiplicamos escalarmente por $\alpha'(t)$:
    \begin{equation*}
        \langle \alpha(t)-c,\alpha'(t) \rangle = 0 \qquad \forall t\in I
    \end{equation*}
    Y podemos observar ahora que:
    \begin{equation*}
        \frac{1}{2}(|\alpha(t)-c|^2)' = 0 \quad \forall t\in I \quad\Longrightarrow\quad|\alpha(t)-c|^2 = r> 0\quad \forall t\in I
    \end{equation*}
    De donde $\alpha(t)\in C(c,r)\quad \forall t\in I\quad\Longrightarrow\quad\tr\alpha\subset C(c,r)$. Es decir, $\tr\alpha$ es un arco de circunferencia abierto.
\end{ejercicio}

\section{Teoría local de curvas planas. Diedro de Frenet. Curvatura}
\noindent
Sea $\alpha:I\to \mathbb{R}^2$ una curva plana regular que consideramos parametrizada por el arco. Representaremos el vector tangente $\alpha'(s)$ de $\alpha$ en el instante $s\in I$ por $T(s)\in \mathbb{R}^2$, del que sabemos que $|T(s)| = 1 \quad \forall s\in I$.

\begin{definicion}
    Dada una curva $\alpha:I\to \mathbb{R}^2$ plana y regular, llamamos vector normal de $\alpha$ en el instante $s\in I$ a $N(s) = JT(s)$, donde $J:\mathbb{R}^2\to \mathbb{R}^2$ viene dada por:
    \begin{equation*}
        J(x,y) = (-y,x)
    \end{equation*}
    Vemos que $J$ es una rotación de $90º$ en sentido antihorario. Tendremos así:
    \begin{equation*}
        |N(s)| = |JT(s)| = |T(s)| = 1 \qquad \forall s\in I
    \end{equation*}
    Y además:
    \begin{equation*}
        \langle T(s),N(s) \rangle = \langle T(s),JT(s) \rangle  = 0 \qquad \forall s\in I
    \end{equation*}
\end{definicion}

\begin{observacion}
    Vemos que para cada $s\in I$ tenemos que $\{T(s),N(s)\}$ es una base ortonormal del plano positivamente orientada, que depende diferenciablemente de $s$. Llamaremos a esta base ``diedro de Frenet-(Senret)'' de $\alpha$ en el instante $s$.
\end{observacion}

\noindent
Si queremos estudiar la curvatura de una curva plana parametrizada por el arco tenemos que ver cómo cambia el diedro de Frenet a lo largo de la curva, es decir, estudiar su derivada. Vemos que:
\begin{equation*}
    \langle T(s),T(s) \rangle = 1 \qquad \forall s\in I
\end{equation*}
Derivando a ambos lados:
\begin{equation*}
    \langle T'(s),T(s) \rangle = 0 \qquad \forall s\in I
\end{equation*}
Así, tenemos para cada $s\in I$ que existe $k(s)\in \mathbb{R}$ de manera que $T'(s) = k(s) N(s)$, donde recordamos que $\alpha''(s) = T'(s)$. El escalar $k(s)$ recibirá el nombre ``\textbf{curvatura} de $\alpha$ en el instante $s$'', y se define como aquel escalar $k(s)$ que verifica:
\begin{equation*}
    \boxed{T'(s) = k(s)N(s)}
\end{equation*}
Como queremos estudiar cómo cambia todo el diedro de Frenet, realizamos el mismo estudio con $N(s)$:
\begin{equation*}
    \langle N(s),N(s) \rangle = 1 \qquad \forall s\in I
\end{equation*}

derivando llegamos a que:
\begin{equation*}
    \langle N'(s),N(s) \rangle  = 0 \qquad \forall s\in I
\end{equation*}
De donde:
\begin{equation*}
    N'(s) = \nu(s)T(s) \qquad \forall s\in I
\end{equation*}
Usando ahora que $\langle T(s),N(s) \rangle = 0 \quad \forall s\in I$ y derivando:
\begin{equation*}
    \langle T'(s),N(s) \rangle  + \langle T(s),N'(s) \rangle  = 0 \qquad \forall s\in I
\end{equation*}

de donde:
\begin{equation*}
    \langle T'(s),N(s) \rangle  + \langle T(s),N'(s) \rangle  = k(s)\langle N(s),N(s) \rangle + \nu(s) \langle T(s),T(s) \rangle  = 0 
\end{equation*}
De aquí deducimos que $\nu(s) = -k(s) \qquad \forall s\in I$, por lo que:
\begin{equation*}
    \boxed{N'(s) = -k(s)T(s)}
\end{equation*}
Estas dos ecuaciones que hemos destacado reciben el nombre de ``ecuaciones de Frenet'', y son las que nos permiten definir el concepto de curvatura, junto con su uso e interpretación. 

\begin{observacion}
    Un par de observaciones:
    \begin{itemize}
        \item En primer lugar, observemos que hemos definido el concepto de curvatura solo para curvas parametrizadas por el arco. 

            Parece lógico pensar que puede extenderse la definición a cualquier curva regular, pues de cada curva regular podemos extraer de forma canónica una curva p.p.a. Sin embargo, nos esperaremos a hacerlo más adelante, pues obtendremos una fórmula que nos será de mayor utilidad que la que podemos desarrollar en este momento.

        \item Lo mismo sucede con el diedro de Frenet y los vectores $T(s)$ y $N(s)$.

        \item Conviene pensar en la diferencia que hay entre la curvatura de una curva que ``gira a izquierda'' con la curvatura de una curva que ``gira a derecha''. 
            \begin{figure}[H]
                \centering
                \begin{tikzpicture}

                % Curva suave tipo spline
                \draw plot[smooth] coordinates {
                        (0,0) (1,0.5) (2,1.2) (3,1.5) (4,1.3) (5,0.8)
                };

                % Punto sobre la curva (aproximadamente en el tramo central)
                \fill (3,1.5) circle (1.5pt);

                % Vector tangente (aproximado)
                \draw[-Stealth] (3,1.5) -- (3.95,1.5);
                % Vector normal (perpendicular aproximado)
                \draw[-Stealth] (3,1.5) -- (3,2.4);

                \fill (2.3,2) node[] {$N(s)$};
                \fill (4.6,1.5) node[] {$T(s)$};
                \fill (3,1.5) node[below] {$\alpha(s)$};
                \end{tikzpicture}
            \end{figure}

            \noindent
            Si imaginamos que vamos en un coche recorriendo la carretera descrita por la curva ilustrada, vemos que en un entorno del punto $s$ que estamos considerando tenemos que ``girar el volante a la derecha'' para seguir la trayectoria de la curva. Este concepto se entiende formalmente como que la aceleración centrífuga o centrípeta $\alpha''(s) = T'(s)$ lleva la misma dirección pero sentido opuesto al vector normal $N(s)$, por lo que tenemos en dicho punto una curvatura negativa, $k(s)<0$.

            Si por el contrario estamos ``girando el volante a la izquierda'', vemos que tenemos entonces que la aceleración centrípeta sí que lleva la misma dirección y sentido que el vector normal $N(s)$, por lo que en dicho caso tenemos una curvatura positiva, $k(s)>0$. 

            Así, el caso $k(s) = 0$ corresponde con una aceleración igual a cero, por lo que la curva no presenta variación de la velocidad en dicho punto, algo que podemos interpretar como que la curva no presenta curvatura en dicho instante, pues la traza simplemente sigue la velocidad del vector velocidad, que en dicho instante presenta derivada nula.
        \item En vista de la fórmula que nos define el concepto de curvatura:
            \begin{equation*}
                \alpha''(s) = T'(s) = k(s)N(s) 
            \end{equation*}
            Vemos que $|N(s)| = 1$, por lo que podríamos haber tomado módulo a ambos lados, obteniendo que:
            \begin{equation*}
                |\alpha''(s)| = |k(s)N(s)| = |k(s)|
            \end{equation*}
            E interpretar así lo que es la curvatura, pues su módulo es el módulo de la aceleración de la curva $\alpha$ en cada instante. Sin embargo, al cosniderar módulo perdemos el signo de la curvatura, que nos interesará para saber si estamos ``girando a izquierdas o a derechas'', como ya hemos comentado. Esta idea será buena tenerla en mente para cuando exploremos la curvatura de curvas en el espacio.
    \end{itemize}
\end{observacion}

\begin{ejemplo}
    Veamos varios ejemplos de calcular la curvatura de una curva:
    \begin{enumerate}
        \item Sea $\alpha:\mathbb{R}\to \mathbb{R}^2$ dada por $\alpha(s) = p+v$ con $v\in \mathbb{R}^2\setminus \{0\}$, vemos que $T(s) = v$, con lo que tenemos que exigir $|v|=1$ para que $\alpha$ esté p.p.a. Así, obtenemos que:
            \begin{equation*}
                T'(s) = 0 \qquad \forall s\in I \quad\Longrightarrow\quad k(s) =0 \qquad \forall s\in I
            \end{equation*}
        \item Sea $\beta:\mathbb{R}\to \mathbb{R}^2$ dada por:
            \begin{equation*}
                \beta(s) = \left(a_1 + r\cos\left(\frac{s}{r}\right), a_2 + r\sen\left(\frac{s}{r}\right)\right)
            \end{equation*}
            Vemos que:
            \begin{equation*}
                T(s) = \beta'(s) = \left(-\sen\left(\frac{s}{r}\right),\cos\left(\frac{s}{r}\right)\right) \qquad \forall s\in I
            \end{equation*}
            por lo que obtenemos $|\beta'(s)| = 1 \quad \forall s\in I$. Ahora:
            \begin{equation*}
                N(s) = JT(s) = \left(-\cos\left(\frac{s}{r}\right),-\sen\left(\frac{s}{r}\right)\right) \qquad \forall s\in I
            \end{equation*}
            Y si calculamos la derivada del vector tangente:
            \begin{equation*}
                T'(s) = \frac{1}{r}\left(-\cos\left(\frac{s}{r}\right), -\sen \left(\frac{s}{r}\right)\right) = \frac{1}{r}N(s) \qquad \forall s\in I
            \end{equation*}
            Hemos obtenido que $k(s) = \frac{1}{r} \quad \forall s\in I$. Vemos así que:
            \begin{itemize}
                \item La curvatura de una circunferencia es constante en cada instante.
                \item La curvatura de una circunferencia disminuye conforme el radio de la misma aumenta.
            \end{itemize}
    \end{enumerate}
\end{ejemplo}

\begin{ejercicio}
    Estudiar las curvas de $\mathbb{R}^2$ p.p.a. con curvatura constante.\\

    \noindent
    Tenemos $\alpha:I\to \mathbb{R}^2$ p.p.a. con:
    \begin{equation*}
        k(s) = k_0\qquad \forall s\in I
    \end{equation*}
    Distinguimos casos:
    \begin{itemize}
        \item Si $k_0 = 0$, tenemos entonces que:
            \begin{equation*}
                \alpha''(s) = T'(s) = k_0N(s) = 0 \qquad \forall s\in I
            \end{equation*}
            Por lo que $\alpha'(s) = v\in \mathbb{R}^2\quad \forall s\in I$, $|v|=1$. Fijado ahora $s_0\in I$ tenemos que:
            \begin{equation*}
                \alpha(s) - \alpha(s_0) = \int_{s_0}^{s} \alpha'(u)~du = \left(\int_{s_0}^{s} ~du \right) v = (s-s_0)v \qquad \forall s\in I
            \end{equation*}
            de donde:
            \begin{equation*}
                \alpha(s) = \alpha(s_0) + (s-s_0)v \qquad \forall s\in I
            \end{equation*}
            Es decir, $\alpha$ es un segmento de recta p.p.a.
        \item Si $k_0\neq 0$, defino $f:I\to \mathbb{R}^2$ dada por:
            \begin{equation*}
                f(s) = \alpha(s) + \frac{1}{k_0}N(s)
            \end{equation*}
            que es una función diferenciable. Si calculamos:
            \begin{equation*}
                f'(s) = T(s) + \frac{1}{k_0}(-k_0T(s)) = 0 \qquad \forall s\in I
            \end{equation*}
            De donde:
            \begin{equation*}
                \alpha8s) + \frac{1}{k_0}N(s) = c \qquad \forall s\in I
            \end{equation*}
            para cierto $c\in \mathbb{R}^2$, con lo que:
            \begin{equation*}
                \alpha(s) - c = \frac{-1}{k_0}N(s) \quad\Longrightarrow\quad |\alpha(s)-c|^2 = \frac{1}{k_0^2}
            \end{equation*}
            De donde $\tr\alpha\subset C\left(c,\frac{1}{|k_0|}\right)$ y $\alpha$ es un arco de circunferencia.
    \end{itemize}
    Hemos demostrado que las únicas curvas de curvatura constante p.p.a. son la recta y la circunferencia.
\end{ejercicio}

\begin{definicion}
    Sea $\alpha:I\to \mathbb{R}^2$ una curva regular, tomamos una reparametrización de $\alpha$ por el arco $\beta = \alpha\circ S_a^{-1}$ vista anteriormente (que es estrictamente creciente), definimos la curvatura de $\alpha$ en un instante $t\in I$ como:
    \begin{equation*}
        k_\alpha(t) = k_{\beta}(S_a(t))
    \end{equation*}
\end{definicion}

\noindent
Veamos que $k_{\alpha}(t)$ no depende del punto $a\in I$ escogido. Para ello, observamos que:
\begin{equation*}
    (\beta\circ S_a)(t) = \alpha(t) \qquad \forall t\in I
\end{equation*}
Derivando:
\begin{equation}\label{eq:der_cur}
    S_a'(t)T_\beta(S_a(t)) = \alpha'(t) \qquad \forall t\in I
\end{equation}
Aplicando $J$:
\begin{equation*}
    S_a'(t)N_\beta(S_a(t)) = J\alpha'(t) \qquad \forall t\in I
\end{equation*}
Derivando la expresión~\eqref{eq:der_cur}:
\begin{equation*}
    S_a''(t)T_\beta(S_a(t)) + S_a'(t)^2 k_\beta(S_a(t))N(S_a(t)) = \alpha''(t) \qquad \forall t\in I
\end{equation*}
Multiplicamos escalarmente estas dos últimas ecuaciones:
\begin{equation*}
    \langle J\alpha'(t),\alpha''(t) \rangle = S_a'(t)^3 k_\beta(S_a(t)) = |\alpha'(t)|^3 k_\alpha(t)
\end{equation*}
Así:
\begin{equation*}
    \boxed{k_\alpha(t) = \frac{\langle J\alpha'(t),\alpha''(t) \rangle }{|\alpha'(t)|^3} = \frac{\det(\alpha'(t), \alpha''(t))}{|\alpha'(t)|^3}}
\end{equation*}
Donde hemos usado que si $\alpha'(t) = (a_1, a_2)$, $\alpha''(t) = (b_1, b_2)$ entonces:
\begin{equation*}
    \langle J\alpha'(t), \alpha''(t) \rangle  = \langle (-a_2, a_1), (b_1,b_2) \rangle = a_1b_2 - a_2b_1 = \left|\begin{array}{cc}
         a_1 & a_2 \\
         b_1 & b_2
    \end{array}\right| = \det(\alpha'(t), \alpha''(t))
\end{equation*}

\noindent
Hemos llegado a que la curvatura no depende de las reparametrizaciones crecientes.\\

\begin{observacion}
    Vemos que la fórmula que hemos obtenido para la curvatura de las curvas regulares no puede extenderse a las curvas no regulares, por el denominador de la misma.\\

    \noindent
    Además, vemos que el signo de la curvatura depende del determinante indicado, que será positivo cuando los vectores del determinante estén positivamente orientados, es decir, cuando ``giramos nuestro volante a la izquierda'', por lo que no hemos perdido la interpretación que teníamos del signo de la curvatura.
\end{observacion}

\begin{ejemplo}
    Si tenemos $\alpha:\left]-a,a\right[\to \mathbb{R}^2$ regular, tomamos ahora $\beta:\left]-a,a\right[\to \mathbb{R}^2$ dada por $\beta(t) = \alpha(-t)$. Vemos que:
    \begin{equation*}
        \beta'(t) = -\alpha'(-t), \qquad \beta''(t) = \alpha''(-t)
    \end{equation*}
    Por lo que:
    \begin{equation*}
        k_\beta(t) = \frac{\det(\beta'(t), \beta''(t))}{|\beta'(t)|^3} = \frac{\det(-\alpha'(-t), \alpha''(-t))}{|\alpha'(-t)|^3} = -\frac{\det(\alpha'(-t), \alpha''(-t))}{|\alpha'(-t)|^3} = -k_\alpha(-t)
    \end{equation*}
    Así, la curvatura cambia de signo por reparametrizaciones decrecientes.
\end{ejemplo}

\begin{ejercicio}
    Calcular la curvatura de la espiral logarítmica. Obtener conclusiones.\\

    \noindent
    Tenemos $\alpha:\mathbb{R}\to \mathbb{R}^2$ dada por:
    \begin{equation*}
        \alpha(t) = e^{-t}(\cos t, \sen t)
    \end{equation*}
    Calculamos $\alpha'$ y $\alpha''$ con vistas a calcular la curvatura de $\alpha$:
    \begin{align*}
        \alpha'(t) &= -e^{-t}(\cos t, \sen t) + e^{-t}(-\sen t, \cos t) = (e^{-t}(-\cos t-\sen t), e^{-t}(\cos t - \sen t)) \\
        \alpha''(t) &= e^{-t}(\cos t, \sen t) - e^{-t}(-\sen t, \cos t) - e^{-t}(-\sen t, \cos t) + e^{-t}(-\cos t, -\sen t) \\
                    &= -2e^{-t}(-\sen t, \cos t) = (2e^{-t}\sen t, -2e^{-t}\cos t) \qquad \forall t\in \mathbb{R}
    \end{align*}
    Por lo que:
    \begin{align*}
        \det(\alpha'(t),\alpha''(t)) &= \left|\begin{array}{cc}
            e^{-t}(-\cos t-\sen t) & e^{-t}(\cos t-\sen t) \\
            2e^{-t}\sen t & -2e^{-t}\cos t  
        \end{array}\right| \\
        &= -2e^{-2t}(-\cos^2t - \sen t\cos t) - 2e^{-2t}(\sen t\cos t - \sen^2t) \\
        &= -2e^{-2t}(-\cos^2t - \sen^2 t) = 2e^{-2t}  \\
            |\alpha'(t)|^2 &= e^{-2t}(\cos^2 t + \sen^2 t + 2\cos t\sen t) + e^{-2t}(\cos^2 t + \sen^2 t - 2\cos t\sen t) \\
                           &= 2e^{-2t} \quad\Longrightarrow\quad |\alpha'(t)| = \sqrt{2}e^{-t}\qquad \forall t\in \mathbb{R} 
    \end{align*}
    Tenemos así que:
    \begin{equation*}
        k_\alpha(t) = \frac{\det(\alpha'(t), \alpha''(t)}{|\alpha'(t)|^3} =  \frac{2e^{-2t}}{{(\sqrt{2}e^{-t})}^{3}} = \frac{1}{\sqrt{2}e^{-t}} = \frac{e^t}{\sqrt{2}} \qquad \forall t\in \mathbb{R}
    \end{equation*}
    Vemos que la curvatura es siempre positiva, por lo que la curva debe ser levógira. Además, vemos que cuanto más grande es $t$ más grande es $k_\alpha(t)$, por lo que cuanto más nos acercamos a $\lim\limits_{t\to\infty} \alpha(t)= (0,0)$ más curvatura tiene la curva.
\end{ejercicio}

\begin{ejercicio} 
    Estudiar la longitud y curvatura para las gráficas de funciones derivables $f:I\to \mathbb{R}$.\\

    \noindent
    Hay una curva parametrizada correspondiente a $f$, $\alpha_f$, tal que  $\tr \alpha_f = \Gr f$. Esta era $\alpha_f:I\to \mathbb{R}^2$ dada por $\alpha_f(t) = (t,f(t))$. Vemos que:
    \begin{equation*}
        \alpha_f'(t) = (1, f'(t)) \neq 0 \qquad \forall t\in I
    \end{equation*}
    Por lo que $\alpha_f$ es una curva regular. Sea ahora $[a,b]\subset I$, tenemos: 
    \begin{equation*}
        L_a^b(\alpha_f) = \int_{a}^{b} \sqrt{1+f'(t)^2}~dt 
    \end{equation*}
    Si queremos calcular ahora la curvatura de $\alpha_f$ ($\alpha_f''(t) = (0, f''(t))$):
    \begin{equation*}
        k_{\alpha_f}(t) = \frac{\left|\begin{array}{cc}
            1 & f'(t) \\
            0 & f''(t)
        \end{array}\right|}{\sqrt{1+f'(t)^2}^3} = \frac{f''(t)}{\sqrt{1+f'(t)^2}^3}
    \end{equation*}
    Así, los conceptos ``cóncava'' y ``convexa'' se vuelven en puntos con curvatura positiva y negativa, y los puntos de inflexión en puntos con curvatura cero.
\end{ejercicio}

\subsection{Curvatura y movimientos rígidos}
\noindent
Sea $M:\mathbb{R}^2\to \mathbb{R}^2$ un movimiento rígido y $x\in \mathbb{R}^2$, tenemos:
\begin{equation*}
    Mx = Ax + b
\end{equation*}
con $A\in O(2)$ y $b\in \mathbb{R}^2$. Consideramos $\beta = M\circ \alpha$ y veamos quién es su curvatura:
\begin{equation*}
    k_{\beta}(t) = \frac{\det(\beta'(t),\beta''(t))}{|\beta'(t)|^3} 
\end{equation*}
Para ello:
\begin{align*}

\end{align*}
de donde:

\begin{teo}[Fundamental de la Teoría Local de Curvas Planas]
    Sea $I\subset \mathbb{R}$ un intervalo abierto y $k_0:I\to \mathbb{R}$ una función diferenciable, entonces existe una curva p.p.a. $\alpha:I\to \mathbb{R}^2$ tal que
    \begin{equation*}
        k_{\alpha}(s) = k_0(s) \qquad \forall s\in I
    \end{equation*}
    Además, $\alpha$ es única salvo movimientos rígidos directos.
    \begin{proof}
        Tenemos dos partes en la demostración:
        \begin{description}
            \item [Existencia.] Fijado $s_0 \in I$, definimos $\theta:I\to \mathbb{R}$ dada por:
                \begin{equation*}
                    \theta(s) = \int_{s_0}^{s} k_0(t)~dt 
                \end{equation*}
                que es una aplicación diferenciable. Como $k(t)$ podemos entenderlo como el cambio infinitesimal que realiza el vector tangente en cada instante, podemos ver que $\theta(s)$ es la acumulación de todos esos cambios desde el instante inicial $s_0$, es decir, el ángulo que varía el vector tangente a la curva desde el instante $s_0$ hasta $s$.

                Fijados $s_1,s_2\in I$, definimos $\alpha:I\to \mathbb{R}^2$ como
                \begin{equation*}
                    \alpha(s) = \left(\int_{s_1}^{s} \cos\theta(t)~dt, \int_{s_2}^{s} \sen\theta(t)~dt  \right)
                \end{equation*}
                que también es una aplicación diferenciable, por lo que $\alpha$ es una curva. Calculamos:
                \begin{equation*}
                    \alpha'(s) = (\cos\theta(s), \sen\theta(s)), \qquad |\alpha'(s)| = 1 \quad \forall s\in I
                \end{equation*}
                Por lo que $\alpha$ es una curva p.p.a. Como queremos ver ahora su curvatura, vemos que:
                \begin{align*}
                    T_\alpha(s) &= \alpha'(s) = (\cos\theta(s), \sen\theta(s)) \\
                    N_\alpha(s) &= JT_\alpha(s) = (-\sen\theta(s), \cos\theta(s)) \\
                    \\
                    T_\alpha'(s) &= \theta'(s) (-\sen\theta(s), \cos\theta(s)) \qquad \forall s\in I
                \end{align*}
                De donde:
                \begin{equation*}
                    k_\alpha(s) = \theta'(s) = k_0(s) \qquad \forall s\in I
                \end{equation*}
            \item [Unicidad.] Supuesto ahora que tenemos $\alpha,\beta:I\to \mathbb{R}^2$ dos curvas p.p.a. verificando
                \begin{equation*}
                    k_\alpha(s) = k_\beta(s) = k_0(s) \qquad \forall s\in I
                \end{equation*}
                Fijado $s_0\in I$, tomamos $A\in O(2)$ con $\det A = 1$ tal que:
                \begin{equation*}
                    AT_\beta(s_0) = T_\alpha(s_0), \qquad AN_\beta(s_0) = N_\alpha(s_0)
                \end{equation*}
                Dicha matriz existe, pues estamos pasando una base ortonormal positivamente orientada en otra base ortonormal positivamente orientada. Si consideramos ahora:
                \begin{equation*}
                    b = \alpha(s_0) - A\beta(s_0)\in \mathbb{R}^2
                \end{equation*}
                Defino $M:\mathbb{R}^2\to \mathbb{R}^2$ por $Mx = Ax+b$ y consideramos $\gamma = M\circ \beta$. Tenemos en primer lugar que:
                \begin{equation*}
                    k_\gamma(s) = k_\beta(s) = k_\alpha(s) = k_0(s) \qquad \forall s\in I
                \end{equation*}
                Y además que:
                \begin{align*}
                    \gamma(s_0) &= A\beta(s_0) + b = \alpha(s_0) \\
                    T_\gamma(s_0) &= AT_\beta(s_0) = T_\alpha(s_0)
                \end{align*}
                De la segunda condición obtenemos además: 
                \begin{equation*}
                    N_\gamma(s_0) = JT_\gamma(s_0) = JT_\alpha(s_0) = N_\alpha(s_0)
                \end{equation*}
                Defino $f:I\to \mathbb{R}$ dada por:
                \begin{equation*}
                    f(s) = \frac{1}{2}\left(|T_{\alpha}(s) - T_\gamma(s)|^2 + |N_\alpha(s) - N_\gamma(s)|^2\right)
                \end{equation*}
                que es diferenciable, con:
                \begin{align*}
                    f'(s) &= \langle T_\alpha'(s) - T_\gamma'(s), T_\alpha(s)-T_\gamma(s) \rangle + \langle N_\alpha'(s) - N_\gamma'(s), N_\alpha(s)-N_\gamma(s) \rangle  \\
                          &\AstIg k_0(s)\langle N_\alpha(s)-N_\gamma(s), T_\alpha(s)-T_\gamma(s) \rangle \\
                          &\quad - k_0(s) \langle T_\alpha(s)-T_\gamma(s),N_\alpha(s)-N_\gamma(s) \rangle  = 0 \qquad \forall s\in I
                \end{align*}
                Por lo que $f(s) = c\quad \forall s\in I$, para $c\in \mathbb{R}$. Sin embargo, observando la definición de $f$ en $s_0$ obtenemos que:
                \begin{equation*}
                    c = f(s_0) = \frac{1}{2}\left(|T_\alpha(s_0)-T_\gamma(s_0)|^2 + |N_\alpha(s_0)-N_\gamma(s_0)|^2\right) = 0 
                \end{equation*}
                De donde obtenemos: 
                \begin{equation*}
                    \alpha'(s) = T_\alpha(s) = T_\gamma(s) = \gamma'(s) \qquad \forall s\in I
                \end{equation*}
                Por lo que $\alpha(s) = \gamma(s) + d\quad \forall s\in I$, para $d\in \mathbb{R}^2$, pero nuevamente en $s_0$ obtenemos:
                \begin{equation*}
                    \gamma(s_0) = \alpha(s_0) = \gamma(s_0) + d \quad\Longrightarrow\quad d = 0
                \end{equation*}
                En definitiva, obtenemos que $\alpha(s) = \gamma(s)\quad \forall s\in I$. Como habíamos definido $\gamma = M\circ \beta$, vemos que $\alpha$ es igual a $\beta$ salvo movimientos rígidos directos.
        \end{description}
    \end{proof}
\end{teo}

\section{Teoría local de las curvas en el espacio}
\noindent
Sea $\alpha:I\to \mathbb{R}^3$ una curva p.p.a. al igual que en curvas planas escribimos:
\begin{equation*}
    T_\alpha(s) := \alpha'(s)
\end{equation*}
Definimos la \textbf{curvatura de $\alpha$ en el instante $s\in I$} como:
\begin{equation*}
    k_\alpha(s) = |T_\alpha'(s)| = |\alpha''(s)| \qquad \forall s\in I
\end{equation*}
Estamos usando la idea que teníamos en las curvas planas:
\begin{equation*}
    T_\alpha'(s) = k_\alpha(s)N_\alpha(s) \quad\Longrightarrow\quad |T_\alpha'(s)| = |k_\alpha(s)||N_\alpha(s)| = |k_\alpha(s)|
\end{equation*}
Observamos así que tenemos por definición:
\begin{equation*}
    k_\alpha(s) = |T_\alpha'(s)| \geq 0 \qquad \forall s\in I
\end{equation*}
Siguiendo con lo que teníamos en las curvas planas:
\begin{equation*}
    T_\alpha'(s) = k_\alpha(s) N_\alpha(s)
\end{equation*}
Buscando definir el \textbf{vector normal} a $\alpha$ en el instante $s$, suponiendo que para $s\in I$ tenemos $k_\alpha(s) > 0$, obtenemos:
\begin{equation*}
    N_\alpha(s) = \frac{T_\alpha'(s)}{k_\alpha(s)} = \frac{T_\alpha'(s)}{|T_\alpha'(s)|}
\end{equation*}
Así, para tener el vector normal de una curva en cierto instante $s\in I$ necesitamos ahora que la curva sea p.p.a. y que $k_\alpha(s) > 0$.\\

\noindent
Supondremos por tanto que\footnote{Puede entenderse como una ``regularidad de orden 2''.} $k_\alpha(s) > 0 \quad \forall s\in I$ a partir de ahora, para tener siempre definido el vector normal a una curva en cada instante $s\in I$.\\

\noindent
Tenemos ahora la aplicación $s\in I\longmapsto \{T_\alpha(s), N_\alpha(s)\}$, y observamos que:
\begin{equation*}
    |N_\alpha(s)| = \frac{|T_\alpha'(s)|}{|T_\alpha'(s)|} = 1 \qquad \forall s\in I
\end{equation*}
Además, usando ahora que:
\begin{equation*}
    |T_\alpha(s)|^2 = 1 \qquad \forall s\in I
\end{equation*}
Y derivando obtenemos que:
\begin{equation*}
    2|T_\alpha'(s)| \langle T_\alpha(s), N_\alpha(s) \rangle = 2\langle T_\alpha(s),T_\alpha'(s) \rangle  = 0 \qquad \forall s\in I
\end{equation*}
Por lo que los vectores $T_\alpha(s)$ y $N_\alpha(s)$ son ortogonales.\\

\noindent
Al igual que en el caso plano buscamos poder asignar a cada instante $s\in I$ una base del espacio. Como ahora tenemos dos vectores ortogonales, buscamos ampliarlo para obtener una base del espacio con un vector normal a ambos y de módulo 1 que haga la base positivamente orientada. Sea $B_\alpha:I\to \mathbb{R}^3$ la función vectorial dada por\footnote{Donde denotamos por $\land$ al producto vectorial.}:
\begin{equation*}
    B_\alpha(s) = T_\alpha(s)\land N_\alpha(s) 
\end{equation*}
Hemos construido para cada $s\in I$ un \textbf{triedro de Frenet}:
\begin{equation*}
    s\in I\longmapsto \{T_\alpha(s), N_\alpha(s), B_\alpha(s)\}
\end{equation*}
Al vector $B_\alpha(s)$ lo llamaremos \textbf{vector binormal de $\alpha$ en el instante $s$}.

\noindent
Tratamos de ver ahora cómo varía el triedro de Frenet. En primer lugar, a partir de la definición de $N_\alpha(s)$ vemos que:
\begin{equation*}
    \boxed{T_{\alpha}'(s) = k_\alpha(s) N_\alpha(s)}\qquad \forall s\in I
\end{equation*}
Para el vector binormal, vemos que: % // TODO: Comprobar que así se deriva \land
\begin{align*}
    B_\alpha'(s) &= (T_\alpha(s)\land N_\alpha(s))' = T_\alpha'(s)\land N_\alpha(s) + T_\alpha(s)\land N_\alpha'(s) \\
                 &= k_\alpha(s)\cancelto{0}{(N_\alpha(s)\land N_\alpha(s))} + T_\alpha(s)\land N_\alpha'(s)
\end{align*}
Además, sabíamos que $|B_\alpha(s)|^2 = 1\quad \forall s\in I$. Derivando:
\begin{equation*}
    \langle B_\alpha'(s),B_\alpha(s) \rangle = 0 
\end{equation*}
Por lo que $B_\alpha'(s)$ no tiene componente en $B_\alpha(s)$. Si hacemos ahora producto escalar en la fórmula obtenida anteriormente vemos que:
\begin{equation*} % // TODO: Interpretar <u x w, v> = det(u, v, w)
    \langle B_\alpha'(s),T_\alpha(s) \rangle  = \langle T_\alpha(s)\land N_\alpha'(s), T_\alpha(s) \rangle  = \det(T_\alpha(s), N_\alpha'(s), T_\alpha(s)) = 0 \qquad \forall s\in I
\end{equation*}
Por lo que $B_\alpha'(s)$ no tiene componente en $T_\alpha(s)$. Por tanto, debemos tener:
\begin{equation*}
    \boxed{B_\alpha'(s) = \tau_\alpha(s)N_\alpha(s)} \qquad s\in I
\end{equation*}
Para cierta constante $\tau_\alpha(s)$, a la que llamaremos \textbf{torsión de $\alpha$ en $s\in I$}.\\ 

\noindent
Para el vector normal sabemos que:
\begin{equation*}
    \langle N_\alpha(s),T_\alpha(s) \rangle  = 0 \qquad \forall s \in I
\end{equation*}
Derivando:
\begin{equation*}
    \langle N_\alpha'(s),T_\alpha(s) \rangle  = - \langle N_\alpha(s),T_\alpha'(s) \rangle = -k_\alpha(s) \qquad \forall s\in I
\end{equation*}
Obtenemos la componente de $N_\alpha'(s)$ en $T_\alpha(s)$. Para calcular ahora su componente en $B_\alpha(s)$:
\begin{equation*}
    \langle N_\alpha(s),B_\alpha(s) \rangle = 0 \qquad \forall s \in I
\end{equation*}
Derivando al igual que antes:
\begin{equation*}
    \langle N_\alpha'(s),B_\alpha(s) \rangle  = -\langle N_\alpha(s),B_\alpha'(s) \rangle = -\tau_\alpha(s) \qquad \forall s\in I
\end{equation*}
Finalmente, la componente de $N_\alpha'(s)$ en $N_\alpha(s)$ la obtenemos derivando:
\begin{equation*}
    |N_\alpha(s)|^2 = 1 \qquad \forall s\in I
\end{equation*}
Obtenemos:
\begin{equation*}
    \langle N_\alpha'(s),N_\alpha(s) \rangle  = 0 \qquad \forall s\in I
\end{equation*}
Así, el vector $N_\alpha'(s)$ se expresa en el tiedro de Frenet con las coordenadas:
\begin{equation*}
    \boxed{N_\alpha'(s) = (-k_\alpha(s), -\tau_\alpha(s), 0)} \qquad \forall s\in I
\end{equation*}
En definitiva, las \textbf{ecuaciones de Frenet} de la curva $\alpha$ p.p.a. quedan:
\begin{align*}
    T_\alpha'(s) &= k_\alpha(s) N_\alpha(s) \\
    N_\alpha'(s) &= -k_\alpha(s)T_\alpha(s) - \tau_\alpha(s)B_\alpha(s) \\
    B_\alpha'(s) &= \tau_\alpha(s)N_\alpha(s)
\end{align*}
Recordamos que para poder hablar de torsión (o si quiera de los vectores normales y binormal) necesitamos que la curva tenga normal, es decir, que $k_\alpha(s)>0\quad \forall s\in I$.

\begin{ejercicio}
    Sea $\alpha:I\to \mathbb{R}^3$ una curva plana ($\tr\alpha\subset P$) p.p.a. Entonces:
    \begin{equation*}
        k_\alpha^{\mathbb{R}^3}(s) = |k_\alpha^{P}(s)|
    \end{equation*}
    Para verlo, si tomamos las ecuaciones de Frenet como curva plana:
    \begin{equation*}
        T_\alpha'(s) = k_\alpha^P(s) N_\alpha^P(s) \qquad \forall s\in I
    \end{equation*}
    Tomando módulo:
    \begin{equation*}
        k_\alpha^{\mathbb{R}^3}(s) = |T_\alpha'(s)| = |k_\alpha^P(s)| |N_\alpha^P(s)| = |k_\alpha^P(s)| \qquad \forall s\in I
    \end{equation*}
    Como consecuencias:
    \begin{enumerate}
        \item Si $\alpha:I\to \mathbb{R}^3$ es una recta (un segmento de recta), entonces:
            \begin{equation*}
                k_\alpha^{\mathbb{R}^3}(s) = 0 \qquad \forall s\in I
            \end{equation*}
            Y no se puede hablar del diedro de Frenet ni de torsión, porque no están definidos los vecores normal y binormal.
        \item Si $\beta:I\to \mathbb{R}^3$ es una circunferencia (un arco) de radio $r$, entonces:
            \begin{equation*}
                k_\beta^{\mathbb{R}^3}(s) = \frac{1}{r} \qquad \forall s\in I
            \end{equation*}
            La circunferencia $\beta$ es una curva plana, por lo que existe un (en este caso, único) plano $P\subset \mathbb{R}^3$ con $\tr\beta\subset P$. Supuesto que la ecuación implícita de $P$ es:
            \begin{equation*}
                \langle x,a \rangle  = b \qquad a\in \mathbb{R}^3\setminus\{0\}, b\in \mathbb{R}
            \end{equation*}
            Obtenemos así que:
            \begin{equation*}
                \langle \beta(s),a \rangle  = b \qquad \forall s\in I
            \end{equation*}
            Derivando:
            \begin{equation*}
                \langle T_\beta(s),a \rangle = 0 \qquad \forall s\in I
            \end{equation*}
            Por lo que $T_\beta(s)$ es ortogonal con $a$. Derivando otra vez:
            \begin{equation*}
                \frac{1}{r}\beta(s)\langle N_\beta(s),a \rangle = 0
            \end{equation*}
            Por lo que $N_\beta(s)$ es ortogonal con $a$. Derivando nuevamente:
            \begin{equation*}
                \langle N_\beta'(s),a \rangle = \tau_\alpha(s) \langle B_\alpha(s),a  \rangle  = 0
            \end{equation*}
            Deducimos por tanto que:
            \begin{equation*}
                0\neq B_\beta(s) \in \{\pm a\} 
            \end{equation*}
            Por lo que $\tau_\beta(s) = 0\quad \forall s\in I$.
    \end{enumerate}
\end{ejercicio}

            \end{equation*}

            \begin{equation*}
                0 = B'(s) = \tau N(s) \quad\Longrightarrow\quad \tau(s) = 0 \qquad \forall s\in I
            \end{equation*}
        \item[3'.] Sea $\alpha:I\to \mathbb{R}^3$ una curva p.p.a. y con curvatura positiva, supongamos que $\tau(s) = 0 \quad \forall s\in I$. La última ecuación de Frenet nos dice:
            \begin{equation*}
                B'(s) = \tau(s)N(s) = 0 \qquad \forall s\in I
            \end{equation*}
            Luego $B(s)$ es una función constante. Supongamos que $B(s) = v\in \mathbb{R}^3 \quad \forall s\in I$, con $|v| = 1$, como producto vectorial $T(s)\land N(s)$. Observemos además que $T(s) \perp B(s) = v$, con lo que:
            \begin{equation*}
                \langle \alpha'(s),v \rangle = \langle T(s), v \rangle  = 0 \qquad \forall s\in I
            \end{equation*}
            Observemos que:
            \begin{equation*}
                \langle \alpha(s),v \rangle ' = \langle \alpha'(s),v \rangle  = 0 \qquad \forall s\in I
            \end{equation*}
            Por lo que esta cantidad es constante, $\langle \alpha(s),v \rangle = c\in \mathbb{R} \quad \forall s\in I$. Tenemos así que:
            \begin{equation*}
                v_1 x(s) + v_2 y(s) + v_3 z(s) = c \qquad \forall s\in I
            \end{equation*}
            con $(v_1,v_2,v_3) = (0, 0,0)$, de donde $\tr \alpha\subset P$ donde $P$ es el plano normal a $v$ y que tiene valor en origen $c$.
    \end{enumerate}
\end{ejemplo}

Acabamos de probar que para cualquier curva $\alpha:I\to \mathbb{R}^3$ p.p.a. y de curvatura estrictamente positiva se tiene que:
\begin{equation*}
    \alpha \text{\ es una curva plana\ } \quad\Longleftrightarrow\quad\tau(s) = 0 \quad \forall s\in I
\end{equation*}

\noindent
La intuición es que la torsión mide las ganas que tiene la curva de salirse del plano formado por $T$ y $N$. La torsión debe ser una función con signo, para indicarnos si se sale por arriba o por abajo. Así, la torsión mide la ``planitud'' de la curva, si la torsión es muy grande, la curva tratará de escapar del plano con mucha violencia.

\begin{notacion}
    A partir de ahora, omitiremos la notación $(s)$ en las funciones tangente, \ldots, al igual que hacemos en ecuaciones diferenciales.
\end{notacion}

\begin{ejemplo}
    \item[4.] Recuperamos la hélice circular recta, $\alpha:\mathbb{R}\to \mathbb{R}^3$ dada por:
        \begin{equation*}
            \alpha(t) = \left(a\cos \frac{s}{\sqrt{a^2+b^2}}, a\sen \frac{s}{\sqrt{a^2+b^2}}, \frac{bs}{\sqrt{a^2+b^2}}\right)
        \end{equation*}
        donde $ab\neq 0$, ya que si no obtendríamos una curva plana. Calculemos:
        \begin{equation*}
            \boxed{\alpha'(s) = \frac{1}{\sqrt{a^2+b^2}}\left(-a\sen\frac{s}{\sqrt{a^2+b^2}}, a\cos \frac{s}{\sqrt{a^2+b^2}}, \frac{b}{\sqrt{a^2+b^2}}\right)}
        \end{equation*}
        Y calculemos:
        \begin{equation*}
            |\alpha'(s)|^2 = \frac{1}{a^2+b^2}(a^2 + b^2) = 1 \qquad \forall s\in I
        \end{equation*}
        Así, tenemos que $\alpha$ es una curva p.p.a. con lo que podemos notar $T(s) = \alpha'(s)$. Ahora, al igual que siempre hacemos en el espacio, calculamos:
        \begin{equation*}
            T'(s) = \frac{1}{a^2+b^2} \left(-a\cos \frac{s}{\sqrt{a^2+b^2}}, -a\sen \frac{s}{\sqrt{a^2+b^2}},0\right)
        \end{equation*}
        de donde (como $ab\neq 0$):
        \begin{equation*}
            k(s) = |T'(s)| = \frac{|a|}{a^2 + b^2} > 0 \qquad \forall s\in I
        \end{equation*}
        Observemos además que $k(s)$ es constante. Como $k(s)>0$, podemos considerar:
        \begin{equation*}
            \boxed{N(s) = \frac{T'(s)}{k(s)} = \frac{-a}{|a|} \left(\cos\frac{s}{\sqrt{a^2+b^2}}, \sen\frac{s}{\sqrt{a^2+b^2}},0\right)}
        \end{equation*}
        Buscamos ahora calcular el binormal (donde omitimos lo que hay dentro de las funciones $\sen$ y $\cos$, que siempre es lo mismo):
        \begin{align*}
            B(s) &= T(s)\land N(s) = \left|\begin{array}{ccc}
                 i & j & k \\
                 -a\sen & a\cos & b \\
                 \cos & \sen & 0
            \end{array}\right| \frac{1}{\sqrt{a^2+b^2}} \frac{-a}{|a|} \\
                 &=\boxed{ \frac{-\sgn(a)}{\sqrt{a^2+b^2}}\left(-b\sen \frac{s}{\sqrt{a^2+b^2}}, b\cos\frac{s}{\sqrt{a^2+b^2}}, -a\right)}
        \end{align*}
        Buscamos calcular ahora la torsión, que obtenemos derivando $B(s)$ y viendo el factor de colinealidad con $N(s)$:
        \begin{equation*}
            \boxed{B'(s) = \frac{\sgn(a)b}{a^2+b^2}\left(\cos \frac{s}{\sqrt{a^2+b^2}}, \sen\frac{s}{\sqrt{a^2+b^2}},0\right)}
        \end{equation*}
        Vemos que es colineal con $N(s)$, de donde ($b \neq 0$):
        \begin{equation*}
            \boxed{\tau(s) = \frac{-b}{a^2+b^2}} \neq 0 \qquad \forall s\in I
        \end{equation*}
        Observamos que es \underline{constante y no nula}. Este signo nos dirá si la hélice es levógira o dextrógira.\\

        \noindent
        Geométricamente, $|a|$ es el radio del cilindro en el que está contenida la hélice, pues es la curvatura de la curva. Además, $b$ nos dice ``las ganas de la curva por salirse del plano formado por el tangente y el normal'', por lo que nos da el ``paso del tornillo'' (\textit{peech}).
\end{ejemplo}

\begin{ejercicio}
    Sea $\alpha:I\to \mathbb{R}^3$ una curva p.p.a. y con curvatura estrictamente positiva. Supongamos que $k$ y $\tau$ son constantes. ¿Podemos concluir que $\tr\alpha$ está contenida en una hélice circular recta?\\

    \noindent
    Sospechamos que la respuesta es afirmativa, por lo que a partir de la curvatura y torsión de una hélice circular recta, defino $a,b\in \mathbb{R}$ dados por la ecuación:
    \begin{equation*}
        k_0 = \frac{|a|}{a^2+b^2}, \qquad \tau_0 = \frac{-b}{a^2+b^2}
    \end{equation*}
    ¿Tiene solución? Observemos que:
    \begin{equation*}
        k_0^2 = \frac{a^2}{a^2+b^2}, \qquad \tau_0^2 = \frac{b^2}{a^2+b^2}
    \end{equation*}
    de donde ($k_0 > 0$):
    \begin{equation*}
        k_0^2 + \tau_0^2 = \frac{1}{a^2+b^2} \quad\Longrightarrow\quad a^2+b^2 = \frac{1}{k_0^2 + \tau_0^2}
    \end{equation*}
    Despejando ahora de las ecuaciones:
    \begin{align*}
        |a| &= k_0 (a^2 + b^2) = \frac{k_0}{k_0^2+\tau_0^2} \\
        b &= -\tau_0 (a^2+b^2) = \frac{-\tau_0}{k_0^2+\tau_0^2}
    \end{align*}
    Observamos que $a$ y $b$ están bien definidos, salvo el signo de $a$. Parece que van a salir dos curvas, pero realmente sale una curva y un giro de la misma, que debería ser geométricamente a la anterior.\\

    \noindent
    Falta tomar una hélice circulear 
\end{ejercicio}
