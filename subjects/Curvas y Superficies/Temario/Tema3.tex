\chapter{Segunda forma fundamental. Curvaturas}
\noindent
Sea $S$ una superficie y $p\in S$, ¿cuántos vectores $n\in \mathbb{R}^3$ cumplen $n\perp T_pS$ con $|n| = 1$? Hay exactamente dos. Si elegimos uno, orientamos $T_pS$, porque dividimos las bases $\{u,v\}$ de $T_pS$ en dos clases:
\begin{itemize}
    \item Aquellas para las que $\det(u,v,n)>0$.
    \item Aquellas para las que $\det(u,v,n)<0$.
\end{itemize}
Además, un cambio de $n$ a $-n$ invierte la orientación en $T_pS$.

\begin{definicion}
    Diremos que la superficie $S$ es \underline{orientable} si se pueden orientar todos sus planos tangentes mediante una aplicación diferenciable, es decir, si existe $N:S\to \mathbb{R}^3$ diferenciable tal que
    \begin{equation*}
        \forall p\in S \quad \text{se tiene que} \quad |N(p)| = 1 \quad \text{y}\quad   N(p)\perp T_pS
    \end{equation*}
    Si $S$ es orientable, diremos que está orientada si se ha elegido una aplicación $N$ con las características anteriores.
\end{definicion}

\begin{notacion}
    Escribiremos usualmente $N(p) = N_p$ para la aplicación $N$ anteriormente enunciada, a la que es costumbre referirnos por ``orientación'' para $S$, o también campo normal unitario o aplicación de Gau\ss.
\end{notacion}

\subsubsection{Propiedades}
\begin{enumerate}
    \item Sea $S$ una superficie conexa y orientable, entonces existen exactamente dos orientaciones sobre $S$.

        Por ser $S$ orientable, debe existir $N:S\to \mathbb{R}^3$ verificando:
        \begin{equation*}
            N_p\perp T_pS \quad \text{y}\quad |N_p| = 1 \qquad \forall q\in S
        \end{equation*}
        Fijamos dicho $N$ y supongamos que $N':S\to\mathbb{R}^3$  es otra orientación de $S$. Si hacemos el producto escalar de ambas obtenemos $\langle N,N' \rangle :S\to \mathbb{R}$, que:
        \begin{equation*}
            \langle N,N' \rangle (p) = \langle N_p,N_p' \rangle  = \pm 1
        \end{equation*}
        Como $\langle N,N' \rangle $ es continua (es diferenciable de hecho), toma valores en un discreto y $S$ es conexo tenemos que $\langle N,N' \rangle $ tiene que ser constante, con lo que hay dos posibilidades:
        \begin{itemize}
            \item Bien $\langle N,N' \rangle (p) = 1\quad \forall p\in S$, de donde $N_p = N_p'\quad \forall p\in S$.
            \item Bien $\langle N,N' \rangle (p) = -1\quad \forall p\in S$, de donde $N_p = -N_p'\quad \forall p\in S$.
        \end{itemize}
    \item Toda superficie $S$ es localmente orientable, es decir, cada punto $p\in S$ admite un entorno $V$ en $S$ de manera que $V$ es orientable\footnote{Observemos que $V$ es una superficie por ser un abierto de una superficie, por lo que tiene sentido hablar de orientablidad de $V$.}.

        Sea $X:U\to\mathbb{R}^3$ una parametrización de $S$ con $p\in X(U) = V$. Fijado $(u,v)\in U$, podemos considerar $T_{X(u,v)}S$, plano para el que $X$ proporciona una base:
        \begin{equation*}
            \{X_u(u,v), X_v(u,v)\}
        \end{equation*}
        Así, $\{X_u,X_v\}$ forman una base de $T_XS$. Definimos ahora $N^X:U\to\mathbb{R}^3$ dado por:
        \begin{equation*}
            N^X = \frac{X_u\land X_v}{|X_u\land X_v|}
        \end{equation*}
        que tiene módulo 1 y es perpendicular a $X_u$ y a $X_v$, luego a $T_XS$. Para orientar $V$ falta considerar finalmente $N:V\to\mathbb{R}^3$ como la composición $N = N^X\circ X^{-1}$, que es diferenciable. Hemos conseguido orientar $V$.

        Observemos que una vez fijemos una parametrización estamos así fijando una orientación.
\end{enumerate}

\begin{ejemplo}
    Veamos ejemplos de superficies orientables:
    \begin{enumerate}
        \item Para un plano afín $P\subset \mathbb{R}^3$  que pasa por un punto $p_0\in \mathbb{R}^3$ y tiene por vector normal $a\in \mathbb{R}^3$:
            \begin{equation*}
                P = \{p\in \mathbb{R}^3 : \langle p-p_0,a \rangle =0\}
            \end{equation*}
            Tenemos que $T_pP = \vec{P} = \langle a \rangle^\perp$. Así, vemos que $a\perp T_pP$ con $|a| =1$. Si tomamos $N:P\to \mathbb{R}^3$ dada por $N_p=a$, hemos orientado $P$.
        \item Para una esfera $S^2(p_0,R)$ con $p_0\in \mathbb{R}^3$ y $R>0$. Para $p\in S^2(p_0,R)$ teníamos:
            \begin{equation*}
                T_pS^2(p_0,R) = \{v\in \mathbb{R}^3 : v\perp p-p_0\} = \langle p-p_0 \rangle^\perp
            \end{equation*}
            definimos una aplicación de Gau\ss\ $N:S^2(p_0,R)\to \mathbb{R}^3$ dada por:
            \begin{equation*}
                N_p = \frac{p-p_0}{|p-p_0|} = \frac{1}{R}(p-p_0)
            \end{equation*}
        \item Todos los grafos son orientables, porque existe una parametrización que los cubre por completo.

            Tenemos una aplicación diferenciable $h:U\subset\mathbb{R}^2\to \mathbb{R}^3$ que define una superficie: 
            \begin{equation*}
                \Gr g = S = \{(x,y,z)\in \mathbb{R}^3 : (x,y)\in U, \ z = h(x,y)\}
            \end{equation*}
            Podemos tomar la parametrización $X:U\to \Gr h$ dada por:
            \begin{equation*}
                X(u,v) = (u,v,h(u,v))
            \end{equation*}
            que cumple $X(U)=S$. Ahora, tenemos:
            \begin{equation*}
                X_u = (1, 0, h_u), \qquad X_v = (0, 1, h_v)
            \end{equation*}
            Sea $N^X:U\to \mathbb{R}^3$ dada por
            \begin{equation*}
                N^x = \frac{(-h_u,-h_v,1)}{\sqrt{1+h_u^2+h_v^2}} = \frac{1}{\sqrt{1+|\nabla h|^2}}(-h_u,-h_v,1)
            \end{equation*}
            Con este $N^X$ definimos la correspondiente aplicación de Gau\ss, $N=N^X\circ X^{-1}$.
        \item Imágenes inversas de un valor regular. Tenemos $f:O\subset\mathbb{R}^3\to \mathbb{R}$ y $a\in \mathbb{R}$ un valor regular de $f$, con $f^{-1}(\{a\})\neq \emptyset $. Si tomamos:
            \begin{equation*}
                S = \{(x,y,z)\in \mathbb{R}^3 : f(x,y,z)=a\}
            \end{equation*}
            es una superficie. Para $(x,y,z)\in S$, tenemos que:
            \begin{equation*}
                T_{(x,y,z)}S = \left\langle \nabla f_{(x,y,z)} \right\rangle^\perp
            \end{equation*}
            Definimos así una aplicación de Gau\ss\ $N:S\to \mathbb{R}^3$ por:
            \begin{equation*}
                N(x,y,z) = \frac{\nabla f}{|\nabla f|}(x,y,z) = \frac{1}{\sqrt{f_x^2 + f_y^2 + f_z^2}} (f_x,f_y,f_z)
            \end{equation*}
        \item \textbf{Teorema de Brower-Samelson.} Toda superficie $S\subset\mathbb{R}^3$ compacta (basta con cerrada) es orientable\footnote{Notemos que en Topología II vimos ejemplos de superficies compactas no orientables, pero en dicha asignatura nuestra definición de superficie era distinta, pues admitía autointersecciones, no como en esta asignatura.}.
    \end{enumerate}
\end{ejemplo}

\begin{ejercicio}
    Un par de ejercicios:
    \begin{enumerate}
        \item Supongamos que $S = S_1\cup S_2$ para $S_i$ una superficie orientable para cada $i=1,2$. Si $S_1\cap S_2$ es conexa entonces $S$ es orientable.
        \item Supongamos que $S=S_1\cup S_2$, para $S_i$ una superficie orientable y conexa para cada $i=1,2$. Si $S_1\cap S_2$ tiene exactamente dos componentes conexas $A$ y $B$. Si $S_1$ y $S_2$ se pueden orientar de forma que las orientaciones inducidas en $A$ y en $B$ son opuestas, entonces $S$ no es orientable. Aplicarlo a la cinta de Möbius, $M = \im X$ para $X:\mathbb{R}^2\to \mathbb{R}^3$ dada por:
            \begin{equation*}
                X(u,v) = \left(\left(2-v\sen \frac{u}{2}\right)\sen u,\left(2-v\sen\frac{u}{2}\right)\cos u,v\cos\frac{u}{2}\right)
            \end{equation*}
    \end{enumerate}
\end{ejercicio}

\section{Curvatura}
\noindent
Sea $S$ una superficie orientable, consideramos una aplicación de Gau\ss\ $N:S\to\mathbb{R}^3$ que:
\begin{equation*}
    N_p\perp T_pS, \quad |N_p| = 1 \qquad \forall p\in S
\end{equation*}
Podemos considerarla también como $N:S\to S^2$ diferenciable que verifica:
\begin{equation*}
    N_p\perp T_pS\qquad \forall p\in S
\end{equation*}
Para $p\in S$, consideramos $(dN)_p:T_pS\to T_{N_p}S^2 = \langle N_p \rangle^\perp = T_pS$.\\

\noindent
Así, vemos que $(dN)_p$ es un endomorfismo de $T_pS$. Esto nos interesa, pues los endomorfismos presentan invariantes geométricos, las raíces de su polinomio característico. En el caso de dimensión dos, estos se corresponden con el determinante y la traza de la matriz que representa la aplicación lineal. Así, notaremos:
\begin{equation*}
    K(p) = \det(dN)_p, \qquad H(p)=-\frac{1}{2}\tr(dN)_p
\end{equation*}
Llamaremos a $K(p)$ \textbf{curvatura de Gau\ss} y a $H(p)$ \textbf{curvatura media}\footnote{La $H$ proviene de \textit{half}.}.\\

\noindent
Observemos que tanto $K$ como $H$ son aplicaciones $K,H:S\to\mathbb{R}$, que a priori no sabemos si son continuas.\\

\noindent
Por ser $(dN)_p$ un endomorfismo, este presenta una forma bilineal asociada, que definiremos por $\sigma_p:T_pS\times T_pS\to\mathbb{R}$ dada por:
\begin{equation*}
    \sigma_p(v,w) = -\langle (dN)_p(v),w \rangle 
\end{equation*}
Llamaremos a esta aplicación por \textbf{La segunda forma fundamental de $S$ en $p$} (para la orientación $N$). Finalmente, es costumbre considerar también al trabajar con endomorfismos la forma cuadrática asociada a la forma bilineal, $\sigma_p:T_pS\to\mathbb{R}$ dada por:
\begin{equation*}
    \sigma_p(v) = \sigma_p(v,v)
\end{equation*}

\begin{observacion}
    Si $S$ es una superficie conexa, solo presentará dos orientaciones, y si cambiamos la orientación $N$ a $-N$, la segunda forma fundamental y la curvatura media cambian de signo, pero la curvatura de Gau\ss\ no, pues el determinante de una matriz $2\times 2$ se conserva al cambiar la matriz por su opuesta, por tener dimensión $2$.
\end{observacion}

\noindent
Además, la \textbf{primera forma fundamental $g_p$ de $S$ en $p$} no es más que el producto escalar de $\mathbb{R}^3$ restringido a $T_pS$, $g_P:T_pS\times T_pS\to\mathbb{R}$ dada por:
\begin{equation*}
    g_p(v,w) = \langle v,w \rangle 
\end{equation*}
Finalmente, es costumbre trabajar con $-(dN)_p$ en lugar que con la propia diferencial en el punto $p$. Así, escribiremos normalmente:
\begin{equation*}
    A_p = -(dN)_p
\end{equation*}
y lo llamaremos \textbf{endomorfismo de Weingarten}. Así, podemos expresar la curvatura de Gau\ss, la curvatura media y la segunda forma fundamental en función de este nuevo endomorfismo:
\begin{gather*}
    K(p) = \det A_p, \qquad H(p) = \frac{1}{2}\tr(A_p) \\
    \sigma_p(v,w) = \langle A_p(v),w \rangle 
\end{gather*}~\\

\subsection{Intuición geométrica}
\noindent
Sea $S$ una superficie orientada por $N:S\to S^2$ una aplicación de Gau\ss, consideramos $p\in S$ y tomamos $v\in T_pS$ con $|v|=1$. Podemos considerar:
\begin{equation*}
    P_v = p + \langle v,N_p \rangle 
\end{equation*}
Así, el conjunto $\{P_v : v\in T_pS\}$ es un haz de planos con eje $N_p$.\\

\noindent
Para cada $v\in T_pS$ parece intuitivo\footnote{Habría que demostrarlo, pero la demostración es larga y tediosa.} ver que $P_v\cap S$ es la traza de una curva, es decir, que $\exists \alpha:\left]-\varepsilon,\varepsilon\right[\to S\cap P_v$ con:
\begin{equation*}
    \alpha(0) = p, \qquad \alpha'(0) = v
\end{equation*}
Así, $P_v\cap S=\tr \alpha$ con $\alpha$ una curva regular\footnote{Es al menos regular en $0$, luego en un entorno de $0$ y podemos restringirnos al intervalo en el que la curva sea regular.} (por lo que podemos suponer que es p.p.a. ya que si no la reparametrizamos). Observemos que $\alpha$ es una curva plana, pues está contenida en el plano $P_v$.\\

\noindent
Por ser $\alpha$ una curva plana, tiene curvatura. Consideramos:
\begin{equation*}
    \langle N_{\alpha(s)},\alpha'(s) \rangle = 0 \qquad \forall s\in \left]-\varepsilon,\varepsilon\right[
\end{equation*}
derivando en el cero:
\begin{equation*}
    \langle (dN)_p(v),v \rangle + \langle N_p, \alpha''(0) \rangle = 0
\end{equation*}
Despejando:
\begin{equation*}
    \sigma_p(v,v) = \langle \alpha''(0),N_p \rangle = k_\alpha(p) \cancelto{1}{\langle N_p,N_p \rangle} 
\end{equation*}
A estas curvas $\alpha$ que han aparecido por cada $v\in T_pS$ son secciones normales a la superfice $S$ en $p$.\\

\noindent
Así, hemos visto que $\sigma_p$ mide la curvatura de las secciones normales a $S$ en $p$, vistas como curvas planas.

\subsection{Propiedades}
\noindent
Sea $N:S\to S^2$ la aplicación de Gau\ss\ de una superficie, sea $X:U\to S$ una parametrización, para $(u,v)\in U$ consideramos $N_{X(u,v)} = (N\circ X)(u,v)\perp T_{X(u,v)}S$, pero sabemos que $X$ nos da una base de este plano, $\{X_u(u,v),X_v(u,v)\}$. Así:
\begin{equation*}
    \langle (N\circ X)(u,v),X_u(u,v) \rangle  = 0 = \langle (N\circ X)(u,v),X_v(u,v) \rangle  
\end{equation*}
Como esto vale para todo $(u,v)\in U$, seguimos escribiendo a continuación lo mismo pero sin indicar las variables en las que se evalúan las funciones:
\begin{equation*}
    \langle N\circ X, X_u \rangle  = 0 = \langle N\circ X, X_v \rangle 
\end{equation*}
Derivando la primemra respecto $v$ y la segunda respecto $u$:
\begin{align*}
    \langle (N\circ X)_v,X_u \rangle  + \langle N\circ X,X_{uv} \rangle  &= 0 \\
    \langle (N\circ X)_u,X_v \rangle + \langle N\circ X, X_{vu} \rangle  &= 0
\end{align*}
Pero la igualdad de Schwartz nos dice que los dos sumandos son idénticos, por lo que deducimos que los dos primeros son iguales también:
\begin{align*}
    \sigma_X(X_v,X_u) &= \langle (dN)_XXv,X_u \rangle = \langle (N\circ X)_v,X_u \rangle  = \langle (N\circ X)_u,X_v \rangle \\ &= \langle (dN)_XX_u, X_v \rangle = \sigma_X(X_u,X_v)
\end{align*}
Pero $\{X_u,X_v\}$ es una base de $T_XS$, por lo que deducimos que $\sigma_{X(u,v)}$ es simétrico $\forall (u,v)\in U$. Para cada $p\in S$ podemos encontrar una parametrización $X:U\to S$ con $p\in X(U)$, por lo que deducimos que $\sigma_p$ es simétrica $\forall p\in S$.\\

\noindent
En vista de la definición de la segunda forma fundamental deducimos que $(dN)_p$ y $A_p$ son autoadjuntos respecto de la primera forma fundamental:
\begin{equation*}
    \langle A_pv,w \rangle = \langle v,A_pw \rangle  \qquad \forall v,w\in T_pS
\end{equation*}
Tenemos así que tanto $A_p$ como $(dN)_p$ \textbf{son diagonalizables}, por lo que existe una base ortonormal $\{e_1,e_2\}$ de $T_pS$ de forma que:
\begin{equation*}
    A_pe_i = k_i(p)e_i, \quad i \in \{1,2\}
\end{equation*}

\begin{notacion}
    A los valores propios de $A_p$, $k_1(p)\leq k_2(p)$ se les llama \textbf{curvaturas principales} de $S$ en $p$. Son aplicaciones $k_1,k_2:S\to\mathbb{R}$ que verifican:
    \begin{equation*}
        K = k_1\cdot k_2, \qquad H = \frac{1}{2}(k_1+k_2)
    \end{equation*}
    Notemos que por el trabajo que realizamos en la sección anterior, $k_1(p)$ y $k_2(p)$ son los valores máximo y mínimo de la sgunda forma fundamental:
    \begin{equation*}
        k_1(p) = \min_{\substack{v\in T_pS\\|v| =1}} \sigma_p(v,v), \qquad k_2(p) = \max_{\substack{v\in T_pS\\|v| =1}}\sigma_p(v,v)
    \end{equation*}
    A $e_1$ y $e_2$ se les llama direcciones principales de $S$ en $p$.\\

    \noindent
    El polinomio característico de $A_p$ es:
    \begin{equation*}
        \lm^2 - 2H\lm + K 
    \end{equation*}
    Que sabemos que tiene raíces (por ser simétrica), luego:
    \begin{equation*}
        \Delta = 4H^2 - 4K \geq 0 \quad\Longleftrightarrow\quad K\leq H^2
    \end{equation*}
    Y la igualdad se alcanza si y solamente si $k_1(p) = k_2(p)$, por lo que se alcanza si y solo si $\sigma_p$ es constante.\\

    \noindent
    Si $p\in P$ verifica $k_1(p) = k_2(p) \quad\Longleftrightarrow\quad K(p) = H^2(p)$, entonces $p$ se dice un punto \textbf{umbilical} o \textbf{umbílico}.
\end{notacion}

\begin{observacion}
    Notemos que si nos encontramos en una superficie, la dirección de máxima curvatura es ortogonal a la dirección de mínima curvatura.
\end{observacion}

\begin{ejemplo}
    Veamos ejemplos:
    \begin{enumerate}
        \item Un plano:
            \begin{equation*}
                P = \{p\in \mathbb{R}^3 : \langle p-p_0,u \rangle = 0\}, \qquad p_0,u\in \mathbb{R}^3, \quad |u| = 1
            \end{equation*}
            Tenemos que $N_p = u\quad \forall p\in P$, con lo que $(dN)_p = 0 = A_p$, de donde:
            \begin{equation*}
                k_1(p) = 0 = k_2(p) \qquad \forall p\in P
            \end{equation*}
            O bien $K(p) = 0 = H(p)\quad \forall p\in P$.

            Así, $P$ cumple que todos sus puntos son umbilicales, se suele decir que $P$ es totalmente umbilical.
        \item Para una esfera $S=S^2(p_0,R)$ con $p_0\in \mathbb{R}^3$ y $R>0$, sea la aplicación de Gau\ss\ $N_p:S^2(p_0,R)\to S^2$ dada por:
            \begin{equation*}
                N_p = \frac{1}{R}(p-p_0)
            \end{equation*}
            Cuya diferencial es para $v\in T_pS^2(p_0,R)$:
            \begin{equation*}
                (dN)_p(v) = \frac{1}{R}v
            \end{equation*}
            Así, el endomorfismo de Weingarten es:
            \begin{equation*}
                A_p = -\frac{1}{R}I_{T_pS^2(p_0,R)}
            \end{equation*}
            Por lo que podemos calcular ya nuestras curvaturas:
            \begin{equation*}
                k_1(p) = -\frac{1}{R} = k_2(p)
            \end{equation*}
            Así, $S^2(p_0,R)$ también es totalmente umbilical. Podemos también calcular:
            \begin{equation*}
                K(p) = \frac{1}{R^2}, \qquad H(p) = -\frac{1}{R}
            \end{equation*}
            La segunda forma fundamental es:
            \begin{equation*}
                \sigma_p(v,v) = -\frac{1}{R}|v|^2 
            \end{equation*}
            Vemos que es proporcional a la primera.
        \item Para un cilindro circular recto de radio $R>0$, trabajaremos ahora con componentes:
            \begin{equation*}
                C = \{(x,y,z)\in \mathbb{R}^3 : x^2+y^2 = R^2\}
            \end{equation*}
            Para $(x,y,z)\in C$ tenemos que (es una superficie dada en forma implícita, de las que ya sabemos calcular su plano tangente a partir de la ecuación implícita):
            \begin{equation*}
                T_{(x,y,z)}C = \{(v_1,v_2,v_3)\in \mathbb{R}^3 : xv_1 + yv_2 = 0\}
            \end{equation*}
            Una base de $T_{(x,y,z)}C$ es (sacada a ojo) $\{(0,0,1), (y,-x,0)\}$. Así:
            \begin{equation*}
                N_{(x,y,z)} = \frac{1}{R}(x,y,0)
            \end{equation*}
            diferenciando\footnote{Sustituimos mentalmente $x,y,z$ por una curva $x(t),y(t),z(t)$, derivamos y evaluamos en 0.} conseguimos $(dN)_{(x,y,z)}:T_{(x,y,z)}C\to T_{(x,y,z)}C$:
            \begin{equation*}
                (dN)_{(x,y,z)}(v_1,v_2,v_3) = \frac{1}{R}(v_1,v_2,0)
            \end{equation*}
            Así, vemos que $(dN)_{(x,y,z)}(0,0,1) = 0 = 0\cdot (0,0,1)$, por lo que $(0,0,1)$ es un vector propio con valor $0$. Observamos ahora que:
            \begin{equation*}
                -A_{(x,y,z)} = (dN)_{\{(x,y,z)\}(y,-x,0)} = \frac{1}{R}(y,-x,0)
            \end{equation*}
            Hemos obtenido así que:
            \begin{equation*}
                k_1(x,y,z) = -\frac{1}{R}, \quad k_2(x,y,z) = 0\qquad \forall (x,y,z)\in C
            \end{equation*}
            El cilindro no tiene ningún punto umbilical. Calculamos ahora:
            \begin{equation*}
                K(x,y,z) = 0,\quad H(x,y,z) = -\frac{1}{2R}
            \end{equation*}
    \end{enumerate}
\end{ejemplo}

    \end{equation*}
    Se trata de un paraboloide hiperbólico. Como mínimo, estudiarlas en el punto $(0,0,0)\in S$.
\end{ejercicio}

\noindent
Resolvemos ahora un ejercicio que se nos ha quedado pendiente a partir de los ejemplos anteriores:
\begin{ejercicio}
    Sea $S\subset\mathbb{R}^3$ una superficie conexa, totalmente umbilical y orientada, estudiar qué superficie es $S$.\\

    \noindent
    Por ser $S$ totalmente umbilical, tenemos entonces que:
    \begin{equation*}
        k_1(p) = \lm(p) = k_2(p) \qquad \forall p\in S
    \end{equation*}
    Así, tenemos que $A_p = \lm(p) I_{T_pS}\quad \forall p\in S$, con lo que tenemos una aplicación con dominios $\lm:S\to \mathbb{R}$, ¿es esta aplicación diferenciable?\\

    \noindent
    Es una aplicación de tipo $a)$, por lo que tomamos $X:U\to S$ una parametrización de $S$ y estudiamos la aplicación $\lm\circ X$. Para $(u,v)\in U$ tenemos que $X(u,v)\in S$, con lo que podemos considerar $(dN)_{X(u,v)}$, que aplicamos a la base que nos da la parametrización $X$ por sus parciales:
    \begin{equation*}
        (dN)_{X(u,v)}(X_u(u,v)) = -\lm(X(u,v))X_u(u,v)
    \end{equation*}
    Escrito sin variables:
    \begin{equation*}
        (dN)_X(X_u) = -(\lm \circ X)X_u
    \end{equation*}
    Y para $X_v$ es un caso análogo:
    \begin{equation*}
        (dN)_X(X_v) = -(\lm \circ X)X_v
    \end{equation*}
    Estas dos ecuaciones son equivalentes a decir que $S$ es totalmente umbilical. Obsevamos ahora la igualdad siguiente, que nos simplifica la parte de la izquierda:
    \begin{equation*}
        (N\circ X)_u = (dN)_X(X_u), \qquad (N\circ X)_v = (dN)_X(X_v)
    \end{equation*}
    Así, tenemos que:
    \begin{equation}\label{eq:ecs_ejm}
        \left\{\begin{array}{l}
                (N\circ X)_u = -(\lm \circ X)X_u \\
                (N\circ X)_v = -(\lm \circ X)X_v 
        \end{array}\right.
    \end{equation}
    Tratando de despejar $\lm \circ X$:
    \begin{equation*}
        \langle (N\circ X)_u,X_u \rangle = -(\lm\circ X)|X_u|^2 \quad\Longrightarrow\quad \lm\circ X = -\frac{\langle (N\circ X)_u,X_u \rangle }{|X_u|^2}
    \end{equation*}
    Vemos que la aplicación de la derecha es diferenciable, por lo que $\lm\circ X$ también es diferenciable, luego $\lm$ también. Derivamos ahora~\eqref{eq:ecs_ejm} de forma cruzada:
    \begin{equation*}
        \left\{\begin{array}{l}
            (N\circ X)_{uv} = -(\lm\circ X)_vX_u - (\lm\circ X)X_{uv} \\
            (N\circ X)_{vu} = -(\lm\circ X)_uX_v - (\lm\circ X)X_{vu}
        \end{array}\right.
    \end{equation*}
    La igualdad de Schwartz nos dice que los términos de la izquierda y que los segundos sumandos de la derecha son iguales, con lo que:
    \begin{equation*}
        (\lm\circ X)_v X_u = (\lm\circ X)_u X_v
    \end{equation*}
    Como $\{X_u,X_v\}$ es una base, deducimos que:
    \begin{equation*}
        (\lm\circ X)_u = 0 = (\lm\circ X)_v 
    \end{equation*}
    en $U$, que podemos tomarla conexa, con lo que la aplicación $\lm\circ X:U\to\mathbb{R}$ es constante, luego $\lm\big|_{X(U)}$ es constante para toda parametrización $X$. Acabamos de demostrar que $\lm$ es localmente constante y como $S$ es conexa $\lm:S\to\mathbb{R}$ tiene que ser constante.\\

    \noindent
    Así, hemos conseguido ver que:
    \begin{equation*}
        (dN)_p = -\lm I_{T_pS}, \quad \lm\in \mathbb{R} \qquad \forall p\in S
    \end{equation*}
    \begin{itemize}
        \item Si $\lm = 0$, tenemos entonces que $N:S\to\mathbb{R}^3$ es constante por ser $S$ conexa, con lo que existe $a\in \mathbb{R}^3$ de manera que $N_p = a \quad \forall p\in S$.

            Así, tenemos que todas las rectas normales a $S$ son paralelas, con lo que por un ejercicio visto anteriormente, $S$ debe ser un abierto de un plano.
        \item Si $\lm\neq 0$, definimos $f:S\to\mathbb{R}^3$ dada por:
            \begin{equation*}
                f(p) = \frac{1}{\lm} Np + p
            \end{equation*}
            vemos que es diferenciable, puesto que $N$ es diferenciable. Calculamos su diferencial $(d f)_p:T_pS\to \mathbb{R}^3$:
            \begin{equation*}
                (d f)_p(v) = \frac{1}{\lm} (dN)_p + I_{\mathbb{R}^3}\big|_{T_pS} = \frac{1}{\lm}(-\lm I) + I = 0 \qquad \forall p\in S
            \end{equation*}
            como $S$ es conexa deducimos que $\exists p_0\in \mathbb{R}^3$ de manera que $f(p)=p_0\quad \forall p\in S$:
            \begin{equation*}
                \frac{1}{\lm}N(p) + p = p_0 \quad \forall p\in S
            \end{equation*}
            Así, vemos que:
            \begin{equation*}
            p-p_0 = -\frac{1}{\lm}N(p) \quad\Longrightarrow\quad |p-p_0|^2 = \frac{1}{\lm^2} \qquad \forall p\in S
            \end{equation*}
            Con lo que $S\subset S^2\left(p_0,\frac{1}{|\lm|}\right)$, $S$ es un abierto de dicha esfera.
    \end{itemize}
    Este resultado se abrevia como que ``las únicas superficies conexas orientables y totalmente umbilicales son los dominios de un plano y los dominios de una esfera''. En particular:
    \begin{itemize}
        \item Si exigimos que la superficie sea cerrada, debe ser un plano o una esfera entera.
        \item Si exigimos que la superficie sea compacta, debe ser una esfera.
    \end{itemize}
    A este resultado le llamamos \textbf{clasficación de las superficies totalmente umbilicales}. % // TODO: Este enunciado es preguntable
\end{ejercicio}

\begin{ejemplo}
    \noindent
    Estudiemos las curvaturas de una cuádrica:\\

    \noindent
    Sea $S$ una cuádrica:
    \begin{equation*}
        S = \left\{p\in \mathbb{R}^3 : (1,p^t) A \left(\begin{array}{cc}
            1 \\
            p 
        \end{array}\right) = 0\right\}
    \end{equation*}
    Para $A\in M_4(\mathbb{R})$ una matriz simétrica. Sabíamos ya que $S$ es una superficie si (no hay puntos singulares) $(1,p^t)A\neq 0\quad \forall p\in S$. Descomponemos $A$ en cajas:
    \begin{equation*}
        A = \left(\begin{array}{c|ccc}
            c &  & a^t &  \\
            \hline
              &  &  &  \\
            a &  & B &  \\
              &  &  &  
        \end{array}\right), \quad c\in \mathbb{R}, a\in \mathbb{R}^3, B\in M_3(\mathbb{R}), B=B^t
    \end{equation*}
    Así:
    \begin{equation*}
        (1, p^t) \left(\begin{array}{c|ccc}
            c &  & a^t &  \\
            \hline
              &  &  &  \\
            a &  & B &  \\
              &  &  &  
        \end{array}\right)\left(\begin{array}{cc}
            1 \\
            p 
        \end{array}\right) = (c + \langle p,a \rangle, a^t + p^tB)\left(\begin{array}{cc}
            1 \\
            p 
        \end{array}\right) = c+2\langle p,a \rangle + p^tBp
    \end{equation*}
    Con lo que:
    \begin{equation*}
        S = \{p\in \mathbb{R}^3 : c + 2\langle p,a \rangle +p^tBp = 0\}
    \end{equation*}
    Hemos descompuesto la expresión en términos de segundo grado, términos lineales y constantes. Conseguimos así:
    \begin{align*}
        T_pS &= \{v\in \mathbb{R}^3 : 2\langle v,a \rangle +v^tBp + p^tBv = 0\} \\
             &\AstIg \{v\in \mathbb{R}^3 : \langle v,a \rangle +p^tBv = 0\} \\
             &= \{v\in \mathbb{R}^3 : \langle v,a \rangle + \langle Bp,v \rangle = 0 \} \\
             &= \{v\in \mathbb{R}^3 : \langle a+Bp,v \rangle =0\} = \langle a+Bp \rangle^\perp
    \end{align*}
    donde en $(\ast)$ hemos usado que $v^tBp = p^tBv$, por ser $B$ una matriz simétrica. Como $T_pS$ es un plano, debemos tener entonces que $a+Bp \neq 0$, que es la condición para no tener puntos singulares.\\

    \noindent
    Tenemos así que $S$ es orientada, y una aplicación de Gau\ss\ es $N:S\to S^2$:
    \begin{equation*}
        N_p = \frac{a+Bp}{|a+Bp|}
    \end{equation*}
    Calculamos ahora $(dN)_p:T_pS\to T_pS$
    \begin{equation*}
        (dN)_p(v) = \frac{Bv}{|a+Bp|} + \lm(p,v)(a + Bp)
    \end{equation*}
    Calculamos la segunda forma fundamental\footnote{Vista como forma cuadrática.}, $\sigma_p:T_pS\to\mathbb{R}$ dada por:
    \begin{equation*}
        \sigma_p(v) = -\langle (dN)_p(v),v \rangle = - \left\langle \frac{Bv}{|a+Bp|}+\lm(p,v)(a+Bp),v \right\rangle  \AstIg -\frac{\langle Bv,v \rangle}{|a+Bp|} 
    \end{equation*}
    donde en $(\ast)$ hemos usado que $\lm(p,v)(a+Bp)$ es un vector perpendicular a $T_pS$ y $v\in T_pS$. Tratamos de calcular las curvaturas.
    \begin{itemize}
        \item $S$ es un elipsoide $\Longleftrightarrow B$ es definida (positiva o negativa), $\Longrightarrow \sigma_p$ es definida $\Longrightarrow K_p>0$.

            Parece ser que $K_p>0$ significa que la superficie encierra un conjunto convexo.

            Notemos además que $K_p>0$ y constante solo en las esferas.
        \item $S$ es un paraboloide $\Longleftrightarrow B$ es semidefinida no definida (es decir, que hay un valor propio cero y dos que tienen el mismo signo) $\Longrightarrow \sigma_p$ es semidefinida $\Longrightarrow K_p\geq 0$.

            Posiblemente si $K_p = 0$ en un punto $p$ entonces estamos en el vértice de un paraboloide elíptico.

        \item $S$ es un ciclindro $\Longleftrightarrow B$ tiene dos valores propios cero y uno no $\Longrightarrow K_p = 0$.
    \end{itemize}
\end{ejemplo}

\section{Curvaturas en coordenadas locales}
\begin{definicion}
    Sea $S$ una superficie orientada, si fijamos una orientación con una aplicación de Gau\ss\ $N:S\to S^2$. Una parametrización $X:U\to S$ se dice \textbf{compatible con $N$} si:
    \begin{equation*}
        N\circ X = N^X = \frac{X_u\land X_v}{|X_u\land X_v|}
    \end{equation*}
\end{definicion}

\noindent
En la situación de la definición anterior, como base de $T_XS$ tomamos $B=\{X_u,X_v\}$ y consideramos la primera forma fundamental en $T_XS$. Sea $M$ su matriz respecto de $B$: 
\begin{equation*}
    M = \left(\begin{array}{cc}
        |X_u|^2 & \langle X_u,X_v \rangle  \\
        \langle X_u,X_v \rangle  & |X_v|^2
    \end{array}\right)
\end{equation*}
Suele denotarse por:
\begin{equation*}
    M = \left(\begin{array}{cc}
        E & F \\
        F & G 
    \end{array}\right)
\end{equation*}
donde $E,F,G:U\to \mathbb{R}$ son diferenciables.\\

\noindent
Consideramos ahora la segunda forma fundamental de $T_XS$, que también es una forma bilineal simétrica en el mismo plano. Sea $\Sigma$ su matriz respecto de $B$:
\begin{equation*}
    \Sigma = \left(\begin{array}{cc}
        \sigma_X(X_u,X_u) &\sigma_X(X_u,X_v)  \\
         \sigma_X(X_u,X_v) &  \sigma_X(X_v,X_v)
    \end{array}\right)
\end{equation*}
Vemos que:
\begin{align*}
    \sigma_X(X_u,X_u) &= -\langle (dN)_XXu,Xu \rangle = - \langle (N\circ X)_u,X_u \rangle \AstIg \langle N\circ X, X_{uu} \rangle  \\
                      &= \langle N^X, X_{uu} \rangle  = \left\langle \frac{X_u\land X_v}{|X_u\land X_v|},X_{uu} \right\rangle = \frac{1}{|X_u\land X_v|}\langle X_u\land X_v,X_{uu} \rangle  \\
                      &= \frac{\det(X_u,X_v,X_{uu})}{|X_u\land X_v|} = e
\end{align*}
donde en $(\ast)$ hemos usado que $\langle N\circ X,X_u \rangle =0$, por lo que al derivar respecto $u$ obtenemos que $-\langle (N\circ X)_u,X_u \rangle = \langle N\circ X,X_{uu} \rangle $. Análogamente:
\begin{align*}
    \sigma_X(X_u,X_v) &= \frac{\det(X_u,X_v,X_{uv})}{|X_u\land X_v|} = f \\
    \sigma_X(X_v,X_v) &= \frac{\det(X_u,X_v,X_{vv})}{|X_u\land X_v|} = g
\end{align*}
Por lo que:
\begin{equation*}
    \Sigma = \left(\begin{array}{cc}
        e & f \\
        f & g 
    \end{array}\right)
\end{equation*}
con $e,f,g:U\to \mathbb{R}$ diferenciables.\\

\noindent
Sea finalmente $A_X$ la matriz del endomorfismo de Weingarten respecto de $B$, se cumple que (hágase):
\begin{equation*}
    A_X = M^{-1}\Sigma
\end{equation*}
Vamos a tratar de calcularlo. Primero tenemos que discutir si $M$ tiene inversa. Vemos que $\langle \cdot ,\cdot  \rangle $ es definida positiva, por lo que su matriz respecto cualquier base también lo será, con lo que:
\begin{equation*}
    \det M = EG-F^2 \neq 0
\end{equation*}
Podemos ahora calcular su inversa y escribirla multiplicada por $\Sigma$:
\begin{equation*}
    A_X = \frac{1}{EG-F^2}\left(\begin{array}{cc}
        G & -F \\
        -F & E 
    \end{array}\right)\left(\begin{array}{cc}
        e & f \\
        f & g 
    \end{array}\right)
\end{equation*}
Así:
\begin{equation*}
    K_X = \det A_X = \frac{1}{{(EG-F^2)}^{2}}(EG-F^2)(eg-f^2) = \frac{eg-f^2}{EG-F^2}
\end{equation*}
Como $E,F,G,e,f,g$ son aplicaciones diferenciables, tenemos también que:
\begin{equation*}
    K_X = K\circ X
\end{equation*}
Es diferenciable para toda parametrización\footnote{Habíamos exigido que $X$ fuera compatible con $N$, pero esto no cambia nada.} $X$, por lo que la aplicación $K:S\to\mathbb{R}$ es diferenciable, en el sentido $a)$.\\

\noindent
Estudiamos ahora la curvatura media de $S$:
\begin{equation*}
    H_X = \frac{1}{2}\tr A_X = \frac{1}{2} \tr(M^{-1}\Sigma) = \frac{1}{2(EG-F^2)}(eG-2fF + gE)
\end{equation*}
Así:
\begin{equation*}
    H\circ X = \frac{eG - 2fF + gE}{2(EG-F^2)}
\end{equation*}
Obtenemos de forma análoga una aplicación diferenciable para cada parametrización, por lo que $H:S\to\mathbb{R}$ es diferenciable, al igual que antes.\\

\noindent
Nos falta ver qué le pasa a las curvaturas principales, sabemos que $k_1\circ X$  y $k_2\circ X$ son los valores propios de $A_X = M^{-1}\Sigma$, que sabemos que existen, puesto que $A_X$ es simétrica y por tanto diagonalizable. Podemos escribir el polinomio característico de $A_X$:
\begin{equation*}
    p(\lm) = \lm^2 -H\lm + K
\end{equation*}
Cuyas raíces son (faltaría componer con $X$ en todo, pero llegaríamos a la misma cuenta):
\begin{align*}
    k_1 &= H- \sqrt{H^2-K} \\
    k_2 &= H+ \sqrt{H^2-K}
\end{align*}
Así, vemos que $k_1,k_2:S\to\mathbb{R}$ son diferenciables salvo quizás en los puntos umbílicos\footnote{Puesto que tenemos una raíz. El ``quizás'' se debe a que por ejemplo en un plano sí que son diferenciables.}. Sin embargo, sí que son aplicaciones continuas.\\

\noindent
Además de estudiar la diferenciabilidad de las curvaturas, las cuentas desarrolladas tienen utilidad. Notemos que si somos capaces de tapar la superficie entonces tenemos una fórmula para las curvaturas. Más aún, si somos capaces de tapar toda la superficies salvo un punto, como son aplicaciones diferenciables, tendremos calculadas las curvaturas en todos los puntos.

\begin{ejercicio}
    Calcular las curvaturas del paraboloide elíptico
    \begin{equation*}
        S = \left\{(x,y,z)\in \mathbb{R}^3 : z=\frac{1}{2}(x^2+y^2)\right\}
    \end{equation*}
    Vemos que es el grafo de la aplicación $f:\mathbb{R}^2\to \mathbb{R}$ dada por:
    \begin{equation*}
        f(x,y) = \frac{1}{2}(x^2+y^2)
    \end{equation*}
    con lo que $S = \Gr f$. Así obtenemos una parametrización que cubre todo $S$, la aplicación $X:\mathbb{R}^2\to S$ dada por:
    \begin{equation*}
        X(u,v) = (u, v, f(u,v))
    \end{equation*}
    Calculamos:
    \begin{equation*}
        X_u = (1, 0, u), \qquad X_v = (0, 1, v) 
    \end{equation*}
    Con lo que:
    \begin{equation*}
        E = |X_u|^2 = 1 + u^2 \qquad 
        G = |X_v|^2 = 1 + v^2 \qquad 
        F = \langle X_u,X_v \rangle = uv 
    \end{equation*}
    Así:
    \begin{equation*}
        EG-F^2 = (1+u^2)(1+v^2) - u^2v^2 = 1 + u^2 + v^2
    \end{equation*}
    Ahora calculamos:
    \begin{equation*}
        X_{uu} = (0, 0, 1), \qquad X_{uv} = (0,0,0), \qquad X_{vv} = (0,0,1)
    \end{equation*}
    Así:
    \begin{equation*}
        e = \frac{\left|\begin{array}{ccc}
            1 & 0 & u \\
            0 & 1 & v \\
            0 & 0 & 1
        \end{array}\right|}{|X_u\land X_v|} \AstIg \frac{1}{\sqrt{EG - F^2}} = \frac{1}{\sqrt{1+u^2+v^2}}
    \end{equation*}
    donde en $(\ast)$ hemos usado que:
    \begin{equation*}
        |X_u \land X_v|^2 = EG-F^2
    \end{equation*}
    Ahora, observemos que $g = e$ y que $f = 0$.\\

    \noindent
    Obtenemos así que:
    \begin{equation*}
        K(u,v) = \frac{1}{{(1+u^2+v^2)}^{2}} > 0
    \end{equation*}
    Observamos que si $u,v\to \infty$ vemos que $K\to 0$, por lo que el paraboloide está cada vez menos curvado.\\

    \noindent
    Ahora:
    \begin{equation*}
        H(u,v) = \frac{eG-2fF+gE}{2(EG-F^2)} = \frac{e(G+E)}{2(EG-F^2)} = \frac{1+v^2+1+u^2}{2{(EG-F^2)}^{\nicefrac{3}{2}}} = \frac{2+u^2+v^2}{2{(1+u^2+v^2)}^{\nicefrac{3}{2}}}
    \end{equation*}
    Observamos que esta también tiende a cero lejos del origen, y siempre es distinta de cero.
\end{ejercicio}
