\chapter{Segunda forma fundamental. Curvaturas}
\noindent
Sea $S$ una superficie y $p\in S$, Â¿cuÃ¡ntos vectores $n\in \mathbb{R}^3$ cumplen $n\perp T_pS$ con $|n| = 1$? Hay exactamente dos. Si elegimos uno, orientamos $T_pS$, porque dividimos las bases $\{u,v\}$ de $T_pS$ en dos clases:
\begin{itemize}
    \item Aquellas para las que $\det(u,v,n)>0$.
    \item Aquellas para las que $\det(u,v,n)<0$.
\end{itemize}
AdemÃ¡s, un cambio de $n$ a $-n$ invierte la orientaciÃ³n en $T_pS$.

\begin{definicion}
    Diremos que la superficie $S$ es \underline{orientable} si se pueden orientar todos sus planos tangentes mediante una aplicaciÃ³n diferenciable, es decir, si existe $N:S\to \mathbb{R}^3$ diferenciable tal que
    \begin{equation*}
        \forall p\in S \quad \text{se tiene que} \quad |N(p)| = 1 \quad \text{y}\quad   N(p)\perp T_pS
    \end{equation*}
    Si $S$ es orientable, diremos que estÃ¡ orientada si se ha elegido una aplicaciÃ³n $N$ con las caracterÃ­sticas anteriores.
\end{definicion}

\begin{notacion}
    Escribiremos usualmente $N(p) = N_p$ para la aplicaciÃ³n $N$ anteriormente enunciada, a la que es costumbre referirnos por ``orientaciÃ³n'' para $S$, o tambiÃ©n campo normal unitario o aplicaciÃ³n de Gau\ss.
\end{notacion}

\subsubsection{Propiedades}
\begin{enumerate}
    \item Sea $S$ una superficie conexa y orientable, entonces existen exactamente dos orientaciones sobre $S$.

        Por ser $S$ orientable, debe existir $N:S\to \mathbb{R}^3$ verificando:
        \begin{equation*}
            N_p\perp T_pS \quad \text{y}\quad |N_p| = 1 \qquad \forall q\in S
        \end{equation*}
        Fijamos dicho $N$ y supongamos que $N':S\to\mathbb{R}^3$  es otra orientaciÃ³n de $S$. Si hacemos el producto escalar de ambas obtenemos $\langle N,N' \rangle :S\to \mathbb{R}$, que:
        \begin{equation*}
            \langle N,N' \rangle (p) = \langle N_p,N_p' \rangle  = \pm 1
        \end{equation*}
        Como $\langle N,N' \rangle $ es continua (es diferenciable de hecho), toma valores en un discreto y $S$ es conexo tenemos que $\langle N,N' \rangle $ tiene que ser constante, con lo que hay dos posibilidades:
        \begin{itemize}
            \item Bien $\langle N,N' \rangle (p) = 1\quad \forall p\in S$, de donde $N_p = N_p'\quad \forall p\in S$.
            \item Bien $\langle N,N' \rangle (p) = -1\quad \forall p\in S$, de donde $N_p = -N_p'\quad \forall p\in S$.
        \end{itemize}
    \item Toda superficie $S$ es localmente orientable, es decir, cada punto $p\in S$ admite un entorno $V$ en $S$ de manera que $V$ es orientable\footnote{Observemos que $V$ es una superficie por ser un abierto de una superficie, por lo que tiene sentido hablar de orientablidad de $V$.}.

        Sea $X:U\to\mathbb{R}^3$ una parametrizaciÃ³n de $S$ con $p\in X(U) = V$. Fijado $(u,v)\in U$, podemos considerar $T_{X(u,v)}S$, plano para el que $X$ proporciona una base:
        \begin{equation*}
            \{X_u(u,v), X_v(u,v)\}
        \end{equation*}
        AsÃ­, $\{X_u,X_v\}$ forman una base de $T_XS$. Definimos ahora $N^X:U\to\mathbb{R}^3$ dado por:
        \begin{equation*}
            N^X = \frac{X_u\land X_v}{|X_u\land X_v|}
        \end{equation*}
        que tiene mÃ³dulo 1 y es perpendicular a $X_u$ y a $X_v$, luego a $T_XS$. Para orientar $V$ falta considerar finalmente $N:V\to\mathbb{R}^3$ como la composiciÃ³n $N = N^X\circ X^{-1}$, que es diferenciable. Hemos conseguido orientar $V$.

        Observemos que una vez fijemos una parametrizaciÃ³n estamos asÃ­ fijando una orientaciÃ³n.
\end{enumerate}

\begin{ejemplo}
    Veamos ejemplos de superficies orientables:
    \begin{enumerate}
        \item Para un plano afÃ­n $P\subset \mathbb{R}^3$  que pasa por un punto $p_0\in \mathbb{R}^3$ y tiene por vector normal $a\in \mathbb{R}^3$:
            \begin{equation*}
                P = \{p\in \mathbb{R}^3 : \langle p-p_0,a \rangle =0\}
            \end{equation*}
            Tenemos que $T_pP = \vec{P} = \langle a \rangle^\perp$. AsÃ­, vemos que $a\perp T_pP$ con $|a| =1$. Si tomamos $N:P\to \mathbb{R}^3$ dada por $N_p=a$, hemos orientado $P$.
        \item Para una esfera $S^2(p_0,R)$ con $p_0\in \mathbb{R}^3$ y $R>0$. Para $p\in S^2(p_0,R)$ tenÃ­amos:
            \begin{equation*}
                T_pS^2(p_0,R) = \{v\in \mathbb{R}^3 : v\perp p-p_0\} = \langle p-p_0 \rangle^\perp
            \end{equation*}
            definimos una aplicaciÃ³n de Gau\ss\ $N:S^2(p_0,R)\to \mathbb{R}^3$ dada por:
            \begin{equation*}
                N_p = \frac{p-p_0}{|p-p_0|} = \frac{1}{R}(p-p_0)
            \end{equation*}
        \item Todos los grafos son orientables, porque existe una parametrizaciÃ³n que los cubre por completo.

            Tenemos una aplicaciÃ³n diferenciable $h:U\subset\mathbb{R}^2\to \mathbb{R}^3$ que define una superficie: 
            \begin{equation*}
                \Gr g = S = \{(x,y,z)\in \mathbb{R}^3 : (x,y)\in U, \ z = h(x,y)\}
            \end{equation*}
            Podemos tomar la parametrizaciÃ³n $X:U\to \Gr h$ dada por:
            \begin{equation*}
                X(u,v) = (u,v,h(u,v))
            \end{equation*}
            que cumple $X(U)=S$. Ahora, tenemos:
            \begin{equation*}
                X_u = (1, 0, h_u), \qquad X_v = (0, 1, h_v)
            \end{equation*}
            Sea $N^X:U\to \mathbb{R}^3$ dada por
            \begin{equation*}
                N^x = \frac{(-h_u,-h_v,1)}{\sqrt{1+h_u^2+h_v^2}} = \frac{1}{\sqrt{1+|\nabla h|^2}}(-h_u,-h_v,1)
            \end{equation*}
            Con este $N^X$ definimos la correspondiente aplicaciÃ³n de Gau\ss, $N=N^X\circ X^{-1}$.
        \item ImÃ¡genes inversas de un valor regular. Tenemos $f:O\subset\mathbb{R}^3\to \mathbb{R}$ y $a\in \mathbb{R}$ un valor regular de $f$, con $f^{-1}(\{a\})\neq \emptyset $. Si tomamos:
            \begin{equation*}
                S = \{(x,y,z)\in \mathbb{R}^3 : f(x,y,z)=a\}
            \end{equation*}
            es una superficie. Para $(x,y,z)\in S$, tenemos que:
            \begin{equation*}
                T_{(x,y,z)}S = \left\langle \nabla f_{(x,y,z)} \right\rangle^\perp
            \end{equation*}
            Definimos asÃ­ una aplicaciÃ³n de Gau\ss\ $N:S\to \mathbb{R}^3$ por:
            \begin{equation*}
                N(x,y,z) = \frac{\nabla f}{|\nabla f|}(x,y,z) = \frac{1}{\sqrt{f_x^2 + f_y^2 + f_z^2}} (f_x,f_y,f_z)
            \end{equation*}
        \item \textbf{Teorema de Brower-Samelson.} Toda superficie $S\subset\mathbb{R}^3$ compacta (basta con cerrada) es orientable\footnote{Notemos que en TopologÃ­a II vimos ejemplos de superficies compactas no orientables, pero en dicha asignatura nuestra definiciÃ³n de superficie era distinta, pues admitÃ­a autointersecciones, no como en esta asignatura.}.
    \end{enumerate}
\end{ejemplo}

\section{Curvatura}
\noindent
Sea $S$ una superficie orientable, consideramos una aplicaciÃ³n de Gau\ss\ $N:S\to\mathbb{R}^3$ que:
\begin{equation*}
    N_p\perp T_pS, \quad |N_p| = 1 \qquad \forall p\in S
\end{equation*}
Podemos considerarla tambiÃ©n como $N:S\to S^2$ diferenciable que verifica:
\begin{equation*}
    N_p\perp T_pS\qquad \forall p\in S
\end{equation*}
Para $p\in S$, consideramos $(dN)_p:T_pS\to T_{N_p}S^2 = \langle N_p \rangle^\perp = T_pS$.\\

\noindent
AsÃ­, vemos que $(dN)_p$ es un endomorfismo de $T_pS$. Esto nos interesa, pues los endomorfismos presentan invariantes geomÃ©tricos, las raÃ­ces de su polinomio caracterÃ­stico. En el caso de dimensiÃ³n dos, estos se corresponden con el determinante y la traza de la matriz que representa la aplicaciÃ³n lineal. AsÃ­, notaremos:
\begin{equation*}
    K(p) = \det(dN)_p, \qquad H(p)=-\frac{1}{2}\tr(dN)_p
\end{equation*}
Llamaremos a $K(p)$ \textbf{curvatura de Gau\ss} y a $H(p)$ \textbf{curvatura media}\footnote{La $H$ proviene de \textit{half}.}.\\

\noindent
Observemos que tanto $K$ como $H$ son aplicaciones $K,H:S\to\mathbb{R}$, que a priori no sabemos si son continuas.\\

\noindent
Por ser $(dN)_p$ un endomorfismo, este presenta una forma bilineal asociada, que definiremos por $\sigma_p:T_pS\times T_pS\to\mathbb{R}$ dada por:
\begin{equation*}
    \sigma_p(v,w) = -\langle (dN)_p(v),w \rangle 
\end{equation*}
Llamaremos a esta aplicaciÃ³n por \textbf{La segunda forma fundamental de $S$ en $p$} (para la orientaciÃ³n $N$). Finalmente, es costumbre considerar tambiÃ©n al trabajar con endomorfismos la forma cuadrÃ¡tica asociada a la forma bilineal, $\sigma_p:T_pS\to\mathbb{R}$ dada por:
\begin{equation*}
    \sigma_p(v) = \sigma_p(v,v)
\end{equation*}

\begin{observacion}
    Si $S$ es una superficie conexa, solo presentarÃ¡ dos orientaciones, y si cambiamos la orientaciÃ³n $N$ a $-N$, la segunda forma fundamental y la curvatura media cambian de signo, pero la curvatura de Gau\ss\ no, pues el determinante de una matriz $2\times 2$ se conserva al cambiar la matriz por su opuesta, por tener dimensiÃ³n $2$.
\end{observacion}

\noindent
AdemÃ¡s, la \textbf{primera forma fundamental $g_p$ de $S$ en $p$} no es mÃ¡s que el producto escalar de $\mathbb{R}^3$ restringido a $T_pS$, $g_P:T_pS\times T_pS\to\mathbb{R}$ dada por:
\begin{equation*}
    g_p(v,w) = \langle v,w \rangle 
\end{equation*}
Finalmente, es costumbre trabajar con $-(dN)_p$ en lugar que con la propia diferencial en el punto $p$. AsÃ­, escribiremos normalmente:
\begin{equation*}
    A_p = -(dN)_p
\end{equation*}
y lo llamaremos \textbf{endomorfismo de Weingarten}. AsÃ­, podemos expresar la curvatura de Gau\ss, la curvatura media y la segunda forma fundamental en funciÃ³n de este nuevo endomorfismo:
\begin{gather*}
    K(p) = \det A_p, \qquad H(p) = \frac{1}{2}\tr(A_p) \\
    \sigma_p(v,w) = \langle A_p(v),w \rangle 
\end{gather*}~\\

\subsection{IntuiciÃ³n geomÃ©trica}
\noindent
Sea $S$ una superficie orientada por $N:S\to S^2$ una aplicaciÃ³n de Gau\ss, consideramos $p\in S$ y tomamos $v\in T_pS$ con $|v|=1$. Podemos considerar:
\begin{equation*}
    P_v = p + \langle v,N_p \rangle 
\end{equation*}
AsÃ­, el conjunto $\{P_v : v\in T_pS\}$ es un haz de planos con eje $N_p$.\\

\noindent
Para cada $v\in T_pS$ parece intuitivo\footnote{HabrÃ­a que demostrarlo, pero la demostraciÃ³n es larga y tediosa.} ver que $P_v\cap S$ es la traza de una curva, es decir, que $\exists \alpha:\left]-\varepsilon,\varepsilon\right[\to S\cap P_v$ con:
\begin{equation*}
    \alpha(0) = p, \qquad \alpha'(0) = v
\end{equation*}
AsÃ­, $P_v\cap S=\tr \alpha$ con $\alpha$ una curva regular\footnote{Es al menos regular en $0$, luego en un entorno de $0$ y podemos restringirnos al intervalo en el que la curva sea regular.} (por lo que podemos suponer que es p.p.a. ya que si no la reparametrizamos). Observemos que $\alpha$ es una curva plana, pues estÃ¡ contenida en el plano $P_v$.\\

\noindent
Por ser $\alpha$ una curva plana, tiene curvatura. Consideramos:
\begin{equation*}
    \langle N_{\alpha(s)},\alpha'(s) \rangle = 0 \qquad \forall s\in \left]-\varepsilon,\varepsilon\right[
\end{equation*}
derivando en el cero:
\begin{equation*}
    \langle (dN)_p(v),v \rangle + \langle N_p, \alpha''(0) \rangle = 0
\end{equation*}
Despejando:
\begin{equation*}
    \sigma_p(v,v) = \langle \alpha''(0),N_p \rangle = k_\alpha(p) \cancelto{1}{\langle N_p,N_p \rangle} 
\end{equation*}
A estas curvas $\alpha$ que han aparecido por cada $v\in T_pS$ son secciones normales a la superfice $S$ en $p$.\\

\noindent
AsÃ­, hemos visto que $\sigma_p$ mide la curvatura de las secciones normales a $S$ en $p$, vistas como curvas planas.

\subsection{Propiedades}
\noindent
Sea $N:S\to S^2$ la aplicaciÃ³n de Gau\ss\ de una superficie, sea $X:U\to S$ una parametrizaciÃ³n, para $(u,v)\in U$ consideramos $N_{X(u,v)} = (N\circ X)(u,v)\perp T_{X(u,v)}S$, pero sabemos que $X$ nos da una base de este plano, $\{X_u(u,v),X_v(u,v)\}$. AsÃ­:
\begin{equation*}
    \langle (N\circ X)(u,v),X_u(u,v) \rangle  = 0 = \langle (N\circ X)(u,v),X_v(u,v) \rangle  
\end{equation*}
Como esto vale para todo $(u,v)\in U$, seguimos escribiendo a continuaciÃ³n lo mismo pero sin indicar las variables en las que se evalÃºan las funciones:
\begin{equation*}
    \langle N\circ X, X_u \rangle  = 0 = \langle N\circ X, X_v \rangle 
\end{equation*}
Derivando la primemra respecto $v$ y la segunda respecto $u$:
\begin{align*}
    \langle (N\circ X)_v,X_u \rangle  + \langle N\circ X,X_{uv} \rangle  &= 0 \\
    \langle (N\circ X)_u,X_v \rangle + \langle N\circ X, X_{vu} \rangle  &= 0
\end{align*}
Pero la igualdad de Schwartz nos dice que los dos sumandos son idÃ©nticos, por lo que deducimos que los dos primeros son iguales tambiÃ©n:
\begin{align*}
    \sigma_X(X_v,X_u) &= \langle (dN)_XXv,X_u \rangle = \langle (N\circ X)_v,X_u \rangle  = \langle (N\circ X)_u,X_v \rangle \\ &= \langle (dN)_XX_u, X_v \rangle = \sigma_X(X_u,X_v)
\end{align*}
Pero $\{X_u,X_v\}$ es una base de $T_XS$, por lo que deducimos que $\sigma_{X(u,v)}$ es simÃ©trico $\forall (u,v)\in U$. Para cada $p\in S$ podemos encontrar una parametrizaciÃ³n $X:U\to S$ con $p\in X(U)$, por lo que deducimos que $\sigma_p$ es simÃ©trica $\forall p\in S$.\\

\noindent
En vista de la definiciÃ³n de la segunda forma fundamental deducimos que $(dN)_p$ y $A_p$ son autoadjuntos respecto de la primera forma fundamental:
\begin{equation*}
    \langle A_pv,w \rangle = \langle v,A_pw \rangle  \qquad \forall v,w\in T_pS
\end{equation*}
Tenemos asÃ­ que tanto $A_p$ como $(dN)_p$ \textbf{son diagonalizables}, por lo que existe una base ortonormal $\{e_1,e_2\}$ de $T_pS$ de forma que:
\begin{equation*}
    A_pe_i = k_i(p)e_i, \quad i \in \{1,2\}
\end{equation*}

\begin{notacion}
    A los valores propios de $A_p$, $k_1(p)\leq k_2(p)$ se les llama \textbf{curvaturas principales} de $S$ en $p$. Son aplicaciones $k_1,k_2:S\to\mathbb{R}$ que verifican:
    \begin{equation*}
        K = k_1\cdot k_2, \qquad H = \frac{1}{2}(k_1+k_2)
    \end{equation*}
    Notemos que por el trabajo que realizamos en la secciÃ³n anterior, $k_1(p)$ y $k_2(p)$ son los valores mÃ¡ximo y mÃ­nimo de la sgunda forma fundamental:
    \begin{equation*}
        k_1(p) = \min_{\substack{v\in T_pS\\|v| =1}} \sigma_p(v,v), \qquad k_2(p) = \max_{\substack{v\in T_pS\\|v| =1}}\sigma_p(v,v)
    \end{equation*}
    A $e_1$ y $e_2$ se les llama direcciones principales de $S$ en $p$.\\

    \noindent
    El polinomio caracterÃ­stico de $A_p$ es:
    \begin{equation*}
        \lm^2 - 2H\lm + K 
    \end{equation*}
    Que sabemos que tiene raÃ­ces (por ser simÃ©trica), luego:
    \begin{equation*}
        \Delta = 4H^2 - 4K \geq 0 \quad\Longleftrightarrow\quad K\leq H^2
    \end{equation*}
    Y la igualdad se alcanza si y solamente si $k_1(p) = k_2(p)$, por lo que se alcanza si y solo si $\sigma_p$ es constante.\\

    \noindent
    Si $p\in P$ verifica $k_1(p) = k_2(p) \quad\Longleftrightarrow\quad K(p) = H^2(p)$, entonces $p$ se dice un punto \textbf{umbilical} o \textbf{umbÃ­lico}.
\end{notacion}

\begin{observacion}
    Notemos que si nos encontramos en una superficie, la direcciÃ³n de mÃ¡xima curvatura es ortogonal a la direcciÃ³n de mÃ­nima curvatura.
\end{observacion}

\begin{ejemplo}
    Veamos ejemplos:
    \begin{enumerate}
        \item Un plano:
            \begin{equation*}
                P = \{p\in \mathbb{R}^3 : \langle p-p_0,u \rangle = 0\}, \qquad p_0,u\in \mathbb{R}^3, \quad |u| = 1
            \end{equation*}
            Tenemos que $N_p = u\quad \forall p\in P$, con lo que $(dN)_p = 0 = A_p$, de donde:
            \begin{equation*}
                k_1(p) = 0 = k_2(p) \qquad \forall p\in P
            \end{equation*}
            O bien $K(p) = 0 = H(p)\quad \forall p\in P$.

            AsÃ­, $P$ cumple que todos sus puntos son umbilicales, se suele decir que $P$ es totalmente umbilical.
        \item Para una esfera $S=S^2(p_0,R)$ con $p_0\in \mathbb{R}^3$ y $R>0$, sea la aplicaciÃ³n de Gau\ss\ $N_p:S^2(p_0,R)\to S^2$ dada por:
            \begin{equation*}
                N_p = \frac{1}{R}(p-p_0)
            \end{equation*}
            Cuya diferencial es para $v\in T_pS^2(p_0,R)$:
            \begin{equation*}
                (dN)_p(v) = \frac{1}{R}v
            \end{equation*}
            AsÃ­, el endomorfismo de Weingarten es:
            \begin{equation*}
                A_p = -\frac{1}{R}I_{T_pS^2(p_0,R)}
            \end{equation*}
            Por lo que podemos calcular ya nuestras curvaturas:
            \begin{equation*}
                k_1(p) = -\frac{1}{R} = k_2(p)
            \end{equation*}
            AsÃ­, $S^2(p_0,R)$ tambiÃ©n es totalmente umbilical. Podemos tambiÃ©n calcular:
            \begin{equation*}
                K(p) = \frac{1}{R^2}, \qquad H(p) = -\frac{1}{R}
            \end{equation*}
            La segunda forma fundamental es:
            \begin{equation*}
                \sigma_p(v,v) = -\frac{1}{R}|v|^2 
            \end{equation*}
            Vemos que es proporcional a la primera.
        \item Para un cilindro circular recto de radio $R>0$, trabajaremos ahora con componentes:
            \begin{equation*}
                C = \{(x,y,z)\in \mathbb{R}^3 : x^2+y^2 = R^2\}
            \end{equation*}
            Para $(x,y,z)\in C$ tenemos que (es una superficie dada en forma implÃ­cita, de las que ya sabemos calcular su plano tangente a partir de la ecuaciÃ³n implÃ­cita):
            \begin{equation*}
                T_{(x,y,z)}C = \{(v_1,v_2,v_3)\in \mathbb{R}^3 : xv_1 + yv_2 = 0\}
            \end{equation*}
            Una base de $T_{(x,y,z)}C$ es (sacada a ojo) $\{(0,0,1), (y,-x,0)\}$. AsÃ­:
            \begin{equation*}
                N_{(x,y,z)} = \frac{1}{R}(x,y,0)
            \end{equation*}
            diferenciando\footnote{Sustituimos mentalmente $x,y,z$ por una curva $x(t),y(t),z(t)$, derivamos y evaluamos en 0.} conseguimos $(dN)_{(x,y,z)}:T_{(x,y,z)}C\to T_{(x,y,z)}C$:
            \begin{equation*}
                (dN)_{(x,y,z)}(v_1,v_2,v_3) = \frac{1}{R}(v_1,v_2,0)
            \end{equation*}
            AsÃ­, vemos que $(dN)_{(x,y,z)}(0,0,1) = 0 = 0\cdot (0,0,1)$, por lo que $(0,0,1)$ es un vector propio con valor $0$. Observamos ahora que:
            \begin{equation*}
                -A_{(x,y,z)} = (dN)_{\{(x,y,z)\}(y,-x,0)} = \frac{1}{R}(y,-x,0)
            \end{equation*}
            Hemos obtenido asÃ­ que:
            \begin{equation*}
                k_1(x,y,z) = -\frac{1}{R}, \quad k_2(x,y,z) = 0\qquad \forall (x,y,z)\in C
            \end{equation*}
            El cilindro no tiene ningÃºn punto umbilical. Calculamos ahora:
            \begin{equation*}
                K(x,y,z) = 0,\quad H(x,y,z) = -\frac{1}{2R}
            \end{equation*}
    \end{enumerate}
\end{ejemplo}
