    \begin{ejercicio}
        Realice los siguientes apartados:
        \begin{enumerate}[label=\alph*)]
            \item Sea $\alpha:\left]0,1\right[\to\mathbb{R}^3$ una curva diferenciable p.p.a. y tal que su traza está contenida en la esfera centrada en el origen de radio $r>0$. Comprueba que la curvatura de $\alpha$ es siempre mayor o igual que $\nicefrac{1}{r}$.
            \item Si $\beta:\left]0,1\right[\to\mathbb{R}^3$ una curva diferenciable p.p.a. con curvatura no nula en todo instante, entonces $\gamma:\left]0,1\right[\to\mathbb{R}^3, \gamma(t) = \beta'(t)$ es una curva regular con curvatura mayor o igual que $1$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Sea $S = \left\{X(u,v):u\in \left]0,+\infty\right[, v\in \left]-\frac{\pi}{2},\frac{\pi}{2}\right[\right\}$ donde
        \begin{equation*}
            X(u,v) = (u\cos(v),u\sen(v),\ln(\cos(v))+u)
        \end{equation*}
        Se pide:
        \begin{enumerate}[label=\alph*)]
            \item Demuestra que $X$ es una parametrización global de $S$ y que $S$ es una superficie orientable.
            \item Calcula la primera y segunda formas fundamentales de $S$. Prueba que el punto $p=(1,0,1)\in S$ y determina el plano tangente de la superficie en ese punto.
            \item Calcula la curvatura de Gauss y la curvatura media de $S$ y clasifica los puntos de $S$ en elípitivos, hiperbólicos, llanos (parabólicos y planos).
            \item Prueba que $S$ es una superficie reglada. Razonar si $S$ es localmente isométrica o no a un plano.
            \item Sea $\pi:S\to \{z=0\}$ la aplicación dada por $\pi(x,y,z)=(x,y,0)$. Prueba que $\pi$ es un difeomorfismo sobre su imagen.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Razonar si son verdaderas o falsas las siguientes afirmaciones:
        \begin{enumerate}[label=\alph*)]
            \item Si $S_1,S_2$ son superficies regulares orientables con aplicaciones de Gauss $N_1:S_1\to\bb{S}^2$ y $N_2:S_2\to\bb{S}^2$ y $f:S_1\to S_2$ es un difeomorfismo local tal que $N_2\circ f = N_1$, entonces $p\in S_1$ es elíptico (hiperbólico, llano) si y sólo si $f(p)\in S_2$ es elíptico (hiperbolico, llano).
            \item Si $S_1,S_2$ son dos esferas con igual radio $r>0$, $S$ es una superficie regular orientable no contenida en la bola con borde $S_j$ para $j=1,2$ y $S_1\cap S = S_2\cap S = S_1\cap S_2 = \{p\}$ para un punto $p\in \mathbb{R}^3$, entonces las curvaturas principales $k_1(p),k_2(p)$ de $S$ en $p$ satisfacen $-\nicefrac{1}{r}\leq k_1(p) \leq k_2(p)\leq \nicefrac{1}{r}$.
            \item Si $S_1,S_2$ son superficie regulares orientables, $N_1:S_1\to\bb{S}^2$ y $N_2:S_2\to\bb{S}^2$ sus aplicacione de Gauss y $f:S_1\to S_2$ una isometría local, entonces $N_2\circ f = N_1$.
            \item Existe una parametrización $X:U\to\bb{S}^2$ tal que $E = G = 3$ y $F=0$.
        \end{enumerate}
    \end{ejercicio}
