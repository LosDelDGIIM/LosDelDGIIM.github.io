	\begin{ejercicio}[2 puntos, 0.5 cada apartado] Contesta, razonadamente, si las afirmaciones que siguen son verdaderas o falsas.
		\begin{enumerate}[label=\alph*)]
			\item $f \in \text{C}(\bb{R}^n, \, \bb{R}^m)$, $A \subset \bb{R}^n$ cerrado $\Rightarrow$ $f(A)$ cerrado en $\bb{R}^m$. \\
			\underline{Falso}. Como contraejemplo tomamos $n=m=1$ y la función $\arctan: \bb{R} \longrightarrow (\nicefrac{-\pi}{2}, \nicefrac{\pi}{2})$. Tomando el cerrado $\bb{R}$, tenemos que $\arctan(\bb{R})=(\nicefrac{-\pi}{2}, \nicefrac{\pi}{2})$. Que es un abierto en $\bb{R}$ (con la topología usual).
			
			
			\item $f \in \text{C}(\bb{R}^n, \, \bb{R}^m)$, $A \subset \bb{R}^n$ abierto $\Rightarrow$ $f(A)$ abierto en $\bb{R}^m$. \\
			\underline{Falso}. Como contraejemplo tomamos $n=m=1$ y la función $f: \bb{R} \longrightarrow \bb{R}, \, f(x)=c, \, \forall x \in \bb{R}$. Tomando el abierto $\bb{R}$, tenemos que $f(\bb{R})=\{c\}$. Que es un cerrado en $\bb{R}$ (con la topología usual).
			
			\item $f \in \text{C}(\bb{R}^n, \, \bb{R}^m)$, $A \subset \bb{R}^n$ acotado $\Rightarrow$ $f(A)$ acotado en $\bb{R}^m$. \\
			\underline{Verdadero}. $A$ acotado $\Rightarrow \overline{A}$ compacto $\Rightarrow f(\overline{A})$ compacto $\Rightarrow f(\overline{A})$ acotado. Como $f(A) \subseteq f(\overline{A}), \, f(A)$ es acotado.
			
			\item $f \in \text{C}(\bb{R}^n, \, \bb{R}^m)$, $A \subset \bb{R}^n$ arcoconexo $\Rightarrow$ $f(A)$ arcoconexo en $\bb{R}^m$. \\
			\underline{Verdadero}. $\forall y_1, y_2 \in f(A), \, \exists x_1, x_2 \in A | f(x_1)=y_1, f(x_2)=y_2$. Como $x_1, x_2 \in A$ y $A$ arcoconexo $\Rightarrow \exists \gamma:[0,1] \longrightarrow A$ continua, tal que $\gamma(0)=x_1, \gamma(1)=x_2$. Ahora
			$$[0,1]\overset{\gamma}{\longrightarrow} A \overset{f}{\longrightarrow} f(A)$$
			$$ 0 \longrightarrow x_1 \longrightarrow y_1$$
			$$1 \longrightarrow x_2 \longrightarrow y_2$$
			$f \circ \gamma:[0,1] \longrightarrow f(A)$ es un arcocontinuo, contenido en $f(A)$ que conecta $y_1$ con $y_2$.
			
		\end{enumerate}
	\end{ejercicio}
	
	\begin{ejercicio}[5 puntos] Considera la función
		$$f: \bb{R}^2 \longrightarrow \bb{R}, \qquad f(x,y)= \begin{cases}
			\displaystyle  \frac{\sen(x)\cos(y)}{x} & \text{si } x\ne 0 \\
			1 & \text{si } x= 0 
		\end{cases} $$
		Calcula razonadamente, el subconjunto $A$ de $\bb{R}^2$ donde $f$ es continua. \\
		
		Sea $(x_0, y_0) \in \bb{R}^2$. Si $x_0\ne 0$, $f$ es continua en $(x_0, y_0)$ por el carácter local de la continuidad y la expresión de $f$ cuando $x\ne0$ (producto, cociente, etc. de funciones continuas). ¿Es continua en los puntos $x_0=0, \, (0, y_0)$?
		\begin{figure}[H]
			\centering
			\begin{tikzpicture}[scale=1]
				\begin{axis}[xmin=-5, xmax=5, ymin=-5, ymax=5, axis x line = center, axis y line = center, xtick=\empty, ytick=\empty, xlabel={$x$}, ylabel={$y$},]
					\node at (1,3) {$(0,y_0)$};
					\node at (0.3,-0.4) {0};
					\filldraw[black] (0,3) circle (1pt);
				\end{axis}
			\end{tikzpicture}
		\end{figure}
		Sean 
		$$ \underset{x=0}{\lim_{(x,y) \rightarrow (0, y_0)}} f(x,y)=1, \quad \underset{x\ne0}{\lim_{(x,y) \rightarrow (0, y_0)}} f(x,y)=\cos(y_0)$$
		Luego $f$ es continua en $(0,y_0) \iff \cos(y_0)=1 \iff y_0 \in \{2\pi k: k \in \bb{Z}\} = B$. Consecuentemente $A=\{(x,y)\in \bb{R}^2:x\ne0\}\cup \{(0,y):y\in B\}$.
	
	\end{ejercicio}
	
	\begin{ejercicio}[3 puntos] Sea
		$$f: \bb{R}^2 \longrightarrow \bb{R}, \qquad f(x,y)= \begin{cases}
			x^2+y^2, & (x,y) \in \bb{Q} \times \bb{Q}\\
			0, & (x,y) \notin \bb{Q} \times \bb{Q}\\ 
		\end{cases} $$
		Demuestra que $f$ es continua en $(x_0,y_0) \iff (x_0,y_0)=(0,0)$. \\
		
		Comprobemos que es continua en $(0,0)$. Si $(x_0,y_0)=(0,0)$,
		$$\underset{(x,y) \in \bb{Q}\times\bb{Q}}{\lim_{(x,y) \rightarrow (0, 0)}} f(x,y)=0, \quad \underset{(x,y) \notin \bb{Q}\times\bb{Q}}{\lim_{(x,y) \rightarrow (0, 0)}} f(x,y)=0$$
				
		Luego $f$ es continua en $(0,0)$ pues
		$$ 0 = \lim_{(x,y) \rightarrow (0, 0)} f(x,y)=f(0,0)$$
		
		Comprobemos que es continua sólo en ese punto. Si $(x_0,y_0)\ne (0,0) \Rightarrow x_0^2 + y_0^2 > 0$
		$$\underset{(x,y) \notin \bb{Q}\times\bb{Q}}{\lim_{(x,y) \rightarrow (x_0, y_0)}} f(x,y)=0, \quad \underset{(x,y) \in \bb{Q}\times\bb{Q}}{\lim_{(x,y) \rightarrow (x_0, y_0)}} f(x,y)=x_0^2 + y_0^2$$
		
		Así $ \displaystyle \nexists \lim_{(x,y) \rightarrow (x_0, y_0)} f(x,y)\Rightarrow f$ no continua en $(x_0, y_0)$. \\
		
		
		Por lo tanto queda demostrado que $f$ es continua en $(x_0,y_0) \iff (x_0,y_0)=(0,0)$.
	\end{ejercicio}
