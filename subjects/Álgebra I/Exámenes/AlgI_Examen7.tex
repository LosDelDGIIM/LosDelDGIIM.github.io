    \begin{ejercicio}
        Se pide:
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[1 punto]} Sea $X$ un conjunto y $A,B\in \cc{P}(X)$. Demostrar
                \begin{equation*}
                    c(A\triangle B) = (A\cap B) \cup (c(A) \cap c(B))
                \end{equation*}
                Donde $A\triangle B = (A-B)\cup (B-A)$ es la diferencia simétrica.
            \item \textbf{[1 punto]} Sea $X$ un conjunto finito no vacío y $f:X\to X$ una aplicación. Demostrar:
                \begin{equation*}
                    f\text{\ es biyectiva} \Longleftrightarrow f \text{\ es inyectiva}
                \end{equation*}

                \begin{description}
                    \item [$\Longrightarrow )$] Toda aplicación biyectiva es, en particular, inyectiva.
                    \item [$\Longleftarrow )$] Si escribimos $n = |X|$ tenemos por ser $f$ inyectiva que los elementos del conjunto $\{f(x) : x\in X\}\subseteq X$ son todos distintos entre sí, y como tenemos $n$ elementos de $X$ el conjunto anterior debe tener exactamente $n$ elementos, por lo que la imagen de $X$ por $f$ es un subconjunto de $X$ con $n$ elementos, luego ha de ser $Img(f)=X$, de donde $f$ es sobreyectiva y como suponemos por hipótesis que $f$ es inyectiva tenemos que $f$ es biyectiva.
                \end{description}
            \item \textbf{[1 punto]} Sea $X$ un conjunto finito no vacío e $Y$ un subconjunto de $X$. Sea $\sim$ la relación de equivalencia sobre el conjunto $\cc{P}(X)$ definida por
                \begin{equation*}
                    A\sim B\Longleftrightarrow A\cup Y = B\cup Y
                \end{equation*}
                La afirmación ``$|\cc{P}(X)/\sim| = 1$'' es
                \begin{enumerate}
                    \item[$(i)$] siempre cierta.
                    \item[$(ii)$] siempre falsa.
                    \item[$(iii)$] a veces verdadera y a veces falsa, dependiendo de $Y$.
                \end{enumerate}
                Justifique su respuesta.\\

                La respuesta correcta es $iii)$, ya que por ejemplo si $X = \{1,2\}$ e $Y = \{1\}$ tenemos entonces que:
                \begin{equation*}
                    X \cup Y = X, \qquad Y\cup Y = Y
                \end{equation*}
                Y como $X\neq Y$ tenemos que $|\cc{P}(X)/\sim| > 1$. Sin embargo, si para el mismo $X$ tomamos $Y = X$ tenemos entonces que para todo $A\in \cc{P}(X)$ se tiene:
                \begin{equation*}
                    A\cup Y = A\cup X = X
                \end{equation*}
                Por lo que $|\cc{P}(X)/\sim| = 1$.
        \end{enumerate}
    \end{ejercicio}

    \begin{ejercicio}
        Se pide:
        \begin{enumerate}[label=\alph*)]
            \item \textbf{[1 punto]} Calcula el resto de dividir $-3425$ entre $19$ y, haciendo uso de que la aplicación $R(\cdot ;19):\mathbb{Z}\to\mathbb{Z}_{19}$ es un homomorfismo de anillos, calcular el resto de dividir $234^3 + {(-3425)}^{2}$ entre $19$.

                Dividiendo primero $3425$ entre $19$ obtenemos que:
                \begin{equation*}
                    3425 = 19\cdot 180 + 5
                \end{equation*}
                Observamos que $19\cdot (-181) = -3439$ y que $3439 - 3425 = 14$, de donde:
                \begin{equation*}
                    -3425 = 19\cdot (-181) + 14
                \end{equation*}
                Por lo que $R(-3425;19) = 14$.\\

                Para calcular el resto de $234^3 + {(-3425)}^{2}$ entre 19 calculamos primero $R(234;19)$, obteniendo:
                \begin{equation*}
                    234 = 19\cdot 12 + 6 \quad\Longrightarrow\quad R(234;19) = 6
                \end{equation*}
                Ahora:
                \begin{equation*}
                    R(234^3 + {(-3425)}^{2};19) \AstIg R(234;19)^3 + R(-3425;19)^2 = 6^3 + 14^2 = 216 + 196
                \end{equation*}
                donde en $(\ast)$ hemos aplicado que $R(\cdot ;19)$ es un homomorfismo de anillos. Si vemos ahora que:
                \begin{equation*}
                    \left\{\begin{array}{l}
                        216 = 19\cdot 11 + 7 \\
                        196 = 19\cdot 10 + 6
                    \end{array}\right.
                \end{equation*}
                    obtenemos entonces:
                    \begin{equation*}
                        R(234^3 + {(-3425)}^{2};19) = 216 + 196 = 7 + 6 = 13
                    \end{equation*}
            \item \textbf{[1 punto]} Demostrar que el elemento $3+2\sqrt{2}$ es una unidad en el anillo $\mathbb{Z}[\sqrt{2}]$ y calcular ${(3+2\sqrt{2})}^{-2}$.
            \item \textbf{[1 punto]} Sea $f:A\to B$ un homomorfismo de anillos. Demostrar que el subconjunto $Img(f)\subseteq B$ es un subanillo de $B$.

                Para ello, basta ver que:
                \begin{itemize}
                    \item $1\in Img(f)$. En efecto, tenemos que $A\ni 1 = f(1) \in Img(f)$, por ser $f$ un homomorfismo de anillos.
                    \item Si $\alpha,\beta\in Img(f)\Longrightarrow \alpha-\beta\in Img(f)$. En efecto, si $\alpha,\beta\in Img(f)$ tenemos entonces que existen $a,b\in A$ con $\alpha = f(a)$, $\beta = f(b)$, de donde:
                        \begin{equation*}
                            Img(f) \ni f(a-b) = f(a) - f(b) = \alpha - \beta 
                        \end{equation*}
                    \item Si $\alpha,\beta\in Img(f)\Longrightarrow \alpha\beta\in Img(f)$. En efecto, si $\alpha,\beta\in Img(f)$ tenemos que existen $a,b\in A$ con $\alpha= f(a)$ y $\beta=f(b)$, de donde:
                        \begin{equation*}
                            Img(f) \ni f(ab) = f(a) f(b) = \alpha \beta
                        \end{equation*}
                \end{itemize}
        \end{enumerate}
    \end{ejercicio}
