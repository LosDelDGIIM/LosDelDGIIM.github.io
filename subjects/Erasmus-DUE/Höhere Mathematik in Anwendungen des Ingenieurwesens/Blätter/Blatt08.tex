\section{Ableitungen von Funktionen mehrerer Veränderlicher}


\begin{ejercicio}[Partielle Ableitungen 1. und 2. Ordnung]
Bilden Sie alle partiellen Ableitungen erster und zweiter Ordnung der Funktion:
\Func{f}{\bb{R}^2}{\bb{R}}{(x,y)}{x^2 \cdot e^{-xy}}

The partial derivatives of first order are:
\begin{align*}
    \frac{\partial f}{\partial x}(x,y) &= 2x e^{-xy} - x^2 y e^{-xy} = e^{-xy} (2x - x^2y)\\
    \frac{\partial f}{\partial y}(x,y) &= -x^3 e^{-xy}
\end{align*}

The partial derivatives of second order are:
\begin{align*}
    \frac{\partial^2 f}{\partial x^2}(x,y) &= \frac{\partial}{\partial x} \left( e^{-xy} (2x - x^2y) \right) = -y e^{-xy} (2x - x^2y) + e^{-xy} (2 - 2xy)\\
    \frac{\partial^2 f}{\partial y^2}(x,y) &= \frac{\partial}{\partial y} \left( -x^3 e^{-xy} \right) = x^4 e^{-xy}\\
    \frac{\partial^2 f}{\partial y \partial x}(x,y) &= \frac{\partial}{\partial x} \left( -x^3 e^{-xy} \right) = -3x^2 e^{-xy} + x^3 y e^{-xy}
\end{align*}
    
\end{ejercicio}

\begin{ejercicio}[Eine Differenzialgleichung]
Zeigen Sie, dass die Funktion
\Func{f}{\bb{R}^2}{\bb{R}}{(x,t)}{e^{-\pi^2 a^2 t} \cdot \sin(\pi x)}
eine Lösung der folgenden Gleichung ist:
\[
a^2 \frac{\partial^2 f}{\partial x^2} = \frac{\partial f}{\partial t} \quad (a \in \bb{R} \text{ const.})
\]

Let's compute the partial derivatives:
\begin{align*}
    \frac{\partial f}{\partial t}(x,t) &= -\pi^2 a^2 e^{-\pi^2 a^2 t} \sin(\pi x)\\
    \frac{\partial^2 f}{\partial x^2}(x,t) &= -\pi^2 e^{-\pi^2 a^2 t} \sin(\pi x)
\end{align*}

Therefore, it is clear that it is a solution of the given equation.
\end{ejercicio}

\begin{ejercicio}[Totales Differenzial einer gebrochen rationalen Funktion]
Bestimmen Sie das totale Differenzial der Funktion
\Func{f}{\bb{R}^2}{\bb{R}}{(x,y)}{\frac{x^2 + y^2}{y - x}}

To find the total differential of the function, we first compute the partial derivatives:
\begin{align*}
    \frac{\partial f}{\partial x}(x,y) &= \frac{2x(y - x) + (x^2 + y^2)}{(y - x)^2} = \frac{-x^2 + 2xy + y^2}{(y - x)^2}\\
    \frac{\partial f}{\partial y}(x,y) &= \frac{2y(y - x) - (x^2 + y^2)}{(y - x)^2} = \frac{y^2 - 2xy - x^2}{(y - x)^2}
\end{align*}
\end{ejercicio}
