\section{Integralrechnung von Funktionen einer
Veränderlichen}

\begin{ejercicio}[Partielle Integration oder Substitution?]
Berechnen Sie die folgenden unbestimmten Integrale.
\begin{enumerate}
    \item $\displaystyle \int (x + 5) e^x \, dx$
    \item $\displaystyle \int e^{\cos x} \sin x \, dx$
\end{enumerate}
\end{ejercicio}

\begin{ejercicio}[Zwei bestimmte Integrale]
Berechnen Sie die folgenden Integrale mit partieller Integration oder Substitution.
\begin{enumerate}
    \item $\displaystyle \int_0^1 x^2 \cdot \sqrt{2x + 1} \, dx$
    \item $\displaystyle \int_0^\pi \cos^3 x \cdot \sin x \, dx$
\end{enumerate}
\end{ejercicio}

\begin{ejercicio}[Logarithmische Integration]
Berechnen Sie:
\begin{enumerate}
    \item $\displaystyle \int \frac{dx}{x \cdot \ln x}$
    \item $\displaystyle \int \frac{3x^2 - 2}{2x^3 - 4x + 2} \, dx$
\end{enumerate}
\end{ejercicio}

\begin{ejercicio}[Fläche zwischen Sinus und Kosinus]
Berechnen Sie den Inhalt der Fläche, die zwischen zwei benachbarten Schnittpunkten der Sinuskurve und der Kosinuskurve von den beiden Kurven eingeschlossen wird.
\end{ejercicio}

\begin{ejercicio}[Integral einer gebrochen-rationalen Funktion]
Berechnen Sie:
\[
    \int \frac{x + 3}{x^2 - 1} \, dx
\]
\end{ejercicio}

\begin{ejercicio}[Partialbruchzerlegung]
Berechnen Sie eine Stammfunktion für:
\[
    f(x) = \frac{5x^3 + 6x^2 - 2x + 23}{(x + 1)^2 (x^2 - 4x + 8)} \hspace{2cm} (x \neq -1)
\]
\end{ejercicio}

\begin{ejercicio}[Partialbruchzerlegung mit Polynomdivision]
Berechnen Sie eine Stammfunktion für:
\[
    f(x) = \frac{x^5 - 3x^4 + 6x^3 - 3x^2 - 2x + 10}{(x^2 - 2x + 2)^2}
\]
\end{ejercicio}
